1//! Run-time feature detection for aarch64 on Darwin (macOS/iOS/tvOS/watchOS/visionOS).
2//!
3//! <https://developer.apple.com/documentation/kernel/1387446-sysctlbyname/determining_instruction_set_characteristics>
4
5use core::ffi::CStr;
6
7use crate::detect::{Feature, cache};
8
9#[inline]
10fn _sysctlbyname(name: &CStr) -> bool {
11    use libc;
12
13    let mut enabled: i32 = 0;
14    let mut enabled_len: usize = 4;
15    let enabled_ptr = &mut enabled as *mut i32 as *mut libc::c_void;
16
17    let ret = unsafe {
18        libc::sysctlbyname(name.as_ptr(), enabled_ptr, &mut enabled_len, core::ptr::null_mut(), 0)
19    };
20
21    match ret {
22        0 => enabled != 0,
23        _ => false,
24    }
25}

27/// Try to read the features using sysctlbyname.
28pub(crate) fn detect_features() -> cache::Initializer {

30
31    let mut enable_feature = |f, enable| {
32        if enable {
33            value.set(f as u32);
34        }

36
37    // Armv8.0 features not using the standard identifiers
38    let fp = _sysctlbyname(c"hw.optional.floatingpoint");
39    let asimd = _sysctlbyname(c"hw.optional.AdvSIMD");
40    let crc = _sysctlbyname(c"hw.optional.armv8_crc32");
41
42    // Armv8 and Armv9 features using the standard identifiers
43    let aes = _sysctlbyname(c"hw.optional.arm.FEAT_AES");

46    let cssc = _sysctlbyname(c"hw.optional.arm.FEAT_CSSC");
47    let dit = _sysctlbyname(c"hw.optional.arm.FEAT_DIT");
48    let dpb = _sysctlbyname(c"hw.optional.arm.FEAT_DPB");
49    let dpb2 = _sysctlbyname(c"hw.optional.arm.FEAT_DPB2");
50    let dotprod = _sysctlbyname(c"hw.optional.arm.FEAT_DotProd");
51    let ecv = _sysctlbyname(c"hw.optional.arm.FEAT_ECV");

53    let fhm = _sysctlbyname(c"hw.optional.arm.FEAT_FHM");

58    let hbc = _sysctlbyname(c"hw.optional.arm.FEAT_HBC");
59    let i8mm = _sysctlbyname(c"hw.optional.arm.FEAT_I8MM");
60    let jsconv = _sysctlbyname(c"hw.optional.arm.FEAT_JSCVT");
61    let rcpc = _sysctlbyname(c"hw.optional.arm.FEAT_LRCPC");
62    let rcpc2 = _sysctlbyname(c"hw.optional.arm.FEAT_LRCPC2");
63    let lse = _sysctlbyname(c"hw.optional.arm.FEAT_LSE");
64    let lse2 = _sysctlbyname(c"hw.optional.arm.FEAT_LSE2");
65    let pauth = _sysctlbyname(c"hw.optional.arm.FEAT_PAuth");
66    let pmull = _sysctlbyname(c"hw.optional.arm.FEAT_PMULL");
67    let rdm = _sysctlbyname(c"hw.optional.arm.FEAT_RDM");
68    let sb = _sysctlbyname(c"hw.optional.arm.FEAT_SB");
69    let sha1 = _sysctlbyname(c"hw.optional.arm.FEAT_SHA1");
70    let sha256 = _sysctlbyname(c"hw.optional.arm.FEAT_SHA256");
71    let sha3 = _sysctlbyname(c"hw.optional.arm.FEAT_SHA3");
72    let sha512 = _sysctlbyname(c"hw.optional.arm.FEAT_SHA512");
73    let sme = _sysctlbyname(c"hw.optional.arm.FEAT_SME");
74    let sme2 = _sysctlbyname(c"hw.optional.arm.FEAT_SME2");
75    let sme_f64f64 = _sysctlbyname(c"hw.optional.arm.FEAT_SME_F64F64");

79
80    // The following features are not exposed by `is_aarch64_feature_detected`,
81    // but *are* reported by `sysctl`. They are here as documentation that they
82    // exist, and may potentially be exposed later.
83    /*
84    let afp = _sysctlbyname(c"hw.optional.arm.FEAT_AFP");
85    let csv2 = _sysctlbyname(c"hw.optional.arm.FEAT_CSV2");
86    let csv3 = _sysctlbyname(c"hw.optional.arm.FEAT_CSV3");
87    let ebf16 = _sysctlbyname(c"hw.optional.arm.FEAT_EBF16");
88    let fpac = _sysctlbyname(c"hw.optional.arm.FEAT_FPAC");
89    let fpaccombine = _sysctlbyname(c"hw.optional.arm.FEAT_FPACCOMBINE");
90    let pacimp = _sysctlbyname(c"hw.optional.arm.FEAT_PACIMP");
91    let pauth2 = _sysctlbyname(c"hw.optional.arm.FEAT_PAuth2");
92    let rpres = _sysctlbyname(c"hw.optional.arm.FEAT_RPRES");

109
110    enable_feature(Feature::aes, aes && pmull);
111    enable_feature(Feature::asimd, asimd);
112    enable_feature(Feature::bf16, bf16);
113    enable_feature(Feature::bti, bti);
114    enable_feature(Feature::crc, crc);
115    enable_feature(Feature::cssc, cssc);
116    enable_feature(Feature::dit, dit);
117    enable_feature(Feature::dotprod, dotprod);
118    enable_feature(Feature::dpb, dpb);
119    enable_feature(Feature::dpb2, dpb2);
120    enable_feature(Feature::ecv, ecv);
121    enable_feature(Feature::fcma, fcma);
122    enable_feature(Feature::fhm, fhm);
123    enable_feature(Feature::flagm, flagm);
124    enable_feature(Feature::flagm2, flagm2);
125    enable_feature(Feature::fp, fp);
126    enable_feature(Feature::fp16, fp16);
127    enable_feature(Feature::frintts, frintts);
128    enable_feature(Feature::hbc, hbc);

132    enable_feature(Feature::lse2, lse2);

139    enable_feature(Feature::sb, sb);

141    enable_feature(Feature::sha3, sha512 && sha3 && asimd);
142    enable_feature(Feature::sme, sme);
143    enable_feature(Feature::sme2, sme2);
144    enable_feature(Feature::sme_f64f64, sme_f64f64);
145    enable_feature(Feature::sme_i16i64, sme_i16i64);
146    enable_feature(Feature::ssbs, ssbs);
147    enable_feature(Feature::wfxt, wfxt);
