85/// `#[clippy::msrv]` attributes are rarely used outside of Clippy's test suite, as a basic
86/// optimization we can skip traversing the HIR in [`Msrv::meets`] if we never saw an MSRV attribute
87/// during the early lint passes
88static SEEN_MSRV_ATTR: AtomicBool = AtomicBool::new(false);
89
90/// Tracks the current MSRV from `clippy.toml`, `Cargo.toml` or set via `#[clippy::msrv]` in late
91/// lint passes, use [`MsrvStack`] for early passes
92#[derive(Copy, Clone, Debug, Default)]
93pub struct Msrv(Option<RustcVersion>);
94
95impl<'de> Deserialize<'de> for Msrv {
96    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
97    where
98        D: serde::Deserializer<'de>,
99    {
100        let v = String::deserialize(deserializer)?;
101        parse_version(Symbol::intern(&v))
102            .map(|v| Self(Some(v)))
103            .ok_or_else(|| serde::de::Error::custom("not a valid Rust version"))
104    }
105}
106
107impl Msrv {
108    /// Returns the MSRV at the current node
109    ///
110    /// If the crate being linted uses an `#[clippy::msrv]` attribute this will search the parent
111    /// nodes for that attribute, prefer to run this check after cheaper pattern matching operations
112    pub fn current(self, cx: &LateContext<'_>) -> Option<RustcVersion> {
113        if SEEN_MSRV_ATTR.load(Ordering::Relaxed) {
114            let start = cx.last_node_with_lint_attrs;
115            if let Some(msrv_attr) = once(start)
116                .chain(cx.tcx.hir_parent_id_iter(start))
117                .find_map(|id| parse_attrs(cx.tcx.sess, cx.tcx.hir_attrs(id)))
118            {
119                return Some(msrv_attr);
120            }
121        }
122
123        self.0
124    }
125
126    /// Checks if a required version from [this module](self) is met at the current node
127    ///
128    /// If the crate being linted uses an `#[clippy::msrv]` attribute this will search the parent
129    /// nodes for that attribute, prefer to run this check after cheaper pattern matching operations
130    pub fn meets(self, cx: &LateContext<'_>, required: RustcVersion) -> bool {
131        self.current(cx).is_none_or(|msrv| msrv >= required)
132    }
133
134    pub fn read_cargo(&mut self, sess: &Session) {
135        let cargo_msrv = std::env::var("CARGO_PKG_RUST_VERSION")
136            .ok()
137            .and_then(|v| parse_version(Symbol::intern(&v)));
138
139        match (self.0, cargo_msrv) {
140            (None, Some(cargo_msrv)) => self.0 = Some(cargo_msrv),
141            (Some(clippy_msrv), Some(cargo_msrv)) => {
142                if clippy_msrv != cargo_msrv {
143                    sess.dcx().warn(format!(
144                        "the MSRV in `clippy.toml` and `Cargo.toml` differ; using `{clippy_msrv}` from `clippy.toml`"
145                    ));
146                }
147            },
148            _ => {},
149        }
150    }
151}

153/// Tracks the current MSRV from `clippy.toml`, `Cargo.toml` or set via `#[clippy::msrv]` in early
154/// lint passes, use [`Msrv`] for late passes
155#[derive(Debug, Clone)]
156pub struct MsrvStack {
157    stack: SmallVec<[RustcVersion; 2]>,
158}
159
160impl MsrvStack {
161    pub fn new(initial: Msrv) -> Self {
162        Self {
163            stack: SmallVec::from_iter(initial.0),
164        }
165    }
166
167    pub fn current(&self) -> Option<RustcVersion> {
168        self.stack.last().copied()
169    }
170
171    pub fn meets(&self, required: RustcVersion) -> bool {
172        self.current().is_none_or(|msrv| msrv >= required)
173    }
174
175    pub fn check_attributes(&mut self, sess: &Session, attrs: &[Attribute]) {
176        if let Some(version) = parse_attrs(sess, attrs) {
177            SEEN_MSRV_ATTR.store(true, Ordering::Relaxed);
178            self.stack.push(version);
179        }
180    }
181
182    pub fn check_attributes_post(&mut self, sess: &Session, attrs: &[Attribute]) {
183        if parse_attrs(sess, attrs).is_some() {
184            self.stack.pop();
185        }
186    }
187}
188
189fn parse_attrs(sess: &Session, attrs: &[impl AttributeExt]) -> Option<RustcVersion> {
190    let mut msrv_attrs = attrs.iter().filter(|attr| attr.path_matches(&[sym::clippy, sym::msrv]));
191
192    let msrv_attr = msrv_attrs.next()?;
193
194    if let Some(duplicate) = msrv_attrs.next_back() {
195        sess.dcx()
196            .struct_span_err(duplicate.span(), "`clippy::msrv` is defined multiple times")
197            .with_span_note(msrv_attr.span(), "first definition found here")
198            .emit();
199    }
200
201    let Some(msrv) = msrv_attr.value_str() else {
202        sess.dcx().span_err(msrv_attr.span(), "bad clippy attribute");
203        return None;
204    };
205
206    let Some(version) = parse_version(msrv) else {
207        sess.dcx()
208            .span_err(msrv_attr.span(), format!("`{msrv}` is not a valid Rust version"));
209        return None;
210    };
211
212    Some(version)
