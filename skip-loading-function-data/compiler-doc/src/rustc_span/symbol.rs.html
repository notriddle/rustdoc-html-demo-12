2562/// The most general type to print identifiers.
2563///
2564/// AST pretty-printer is used as a fallback for turning AST structures into token streams for
2565/// proc macros. Additionally, proc macros may stringify their input and expect it survive the
2566/// stringification (especially true for proc macro derives written between Rust 1.15 and 1.30).

2568/// hygiene data, most importantly name of the crate it refers to.

2570/// and as `::other_crate_name` if it refers to some other crate.
2571/// Note, that this is only done if the ident token is printed from inside of AST pretty-printing,
2572/// but not otherwise. Pretty-printing is the only way for proc macros to discover token contents,
2573/// so we should not perform this lossy conversion if the top level call to the pretty-printer was
2574/// done for a token stream or a single token.

2576    symbol: Symbol,
2577    mode: IdentPrintMode,
2578    /// Span used for retrieving the crate name to which `$crate` refers to,
2579    /// if this field is `None` then the `$crate` conversion doesn't happen.
2580    convert_dollar_crate: Option<Span>,
2581}

2601impl fmt::Display for IdentPrinter {
2602    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
2603        let s = match self.mode {
2604            IdentPrintMode::Normal
2605                if self.symbol == kw::DollarCrate
2606                    && let Some(span) = self.convert_dollar_crate =>
2607            {
2608                let converted = span.ctxt().dollar_crate_name();
2609                if !converted.is_path_segment_keyword() {
2610                    f.write_str("::")?;
2611                }
2612                converted
2613            }
2614            IdentPrintMode::Normal => self.symbol,
2615            IdentPrintMode::RawIdent => {
2616                f.write_str("r#")?;
2617                self.symbol
2618            }
2619            IdentPrintMode::RawLifetime => {
2620                f.write_str("'r#")?;
2621                let s = self
2622                    .symbol
2623                    .as_str()
2624                    .strip_prefix("'")
2625                    .expect("only lifetime idents should be passed with RawLifetime mode");

2627            }
2628        };
2629        s.fmt(f)

2844// Interner used for both `Symbol`s and `ByteSymbol`s. If a string and a byte
2845// string with identical contents (e.g. "foo" and b"foo") are both interned,
2846// only one copy will be stored and the resulting `Symbol` and `ByteSymbol`
2847// will have the same index.
2848pub(crate) struct Interner(Lock<InternerInner>);
2849
2850// The `&'static [u8]`s in this type actually point into the arena.
2851//
2852// This type is private to prevent accidentally constructing more than one
2853// `Interner` on the same thread, which makes it easy to mix up `Symbol`s
2854// between `Interner`s.
2855struct InternerInner {
2856    arena: DroplessArena,
2857    byte_strs: FxIndexSet<&'static [u8]>,
2858}
2859
2860impl Interner {
2861    // These arguments are `&str`, but because of the sharing, we are
2862    // effectively pre-interning all these strings for both `Symbol` and
2863    // `ByteSymbol`.
2864    fn prefill(init: &[&'static str], extra: &[&'static str]) -> Self {
2865        let byte_strs = FxIndexSet::from_iter(
2866            init.iter().copied().chain(extra.iter().copied()).map(|str| str.as_bytes()),
2867        );
2868        assert_eq!(
2869            byte_strs.len(),
2870            init.len() + extra.len(),
2871            "duplicate symbols in the rustc symbol list and the extra symbols added by the driver",
2872        );
2873        Interner(Lock::new(InternerInner { arena: Default::default(), byte_strs }))
2874    }
2875
2876    fn intern_str(&self, str: &str) -> Symbol {
2877        Symbol::new(self.intern_inner(str.as_bytes()))
2878    }
2879
2880    fn intern_byte_str(&self, byte_str: &[u8]) -> ByteSymbol {
2881        ByteSymbol::new(self.intern_inner(byte_str))
2882    }
2883
2884    #[inline]
2885    fn intern_inner(&self, byte_str: &[u8]) -> u32 {
2886        let mut inner = self.0.lock();
2887        if let Some(idx) = inner.byte_strs.get_index_of(byte_str) {
2888            return idx as u32;
2889        }
2890
2891        let byte_str: &[u8] = inner.arena.alloc_slice(byte_str);
2892
2893        // SAFETY: we can extend the arena allocation to `'static` because we
2894        // only access these while the arena is still alive.
2895        let byte_str: &'static [u8] = unsafe { &*(byte_str as *const [u8]) };
2896
2897        // This second hash table lookup can be avoided by using `RawEntryMut`,
2898        // but this code path isn't hot enough for it to be worth it. See
2899        // #91445 for details.
2900        let (idx, is_new) = inner.byte_strs.insert_full(byte_str);
2901        debug_assert!(is_new); // due to the get_index_of check above
2902
2903        idx as u32
2904    }
2905
2906    /// Get the symbol as a string.
2907    ///
2908    /// [`Symbol::as_str()`] should be used in preference to this function.
2909    fn get_str(&self, symbol: Symbol) -> &str {
2910        let byte_str = self.get_inner(symbol.0.as_usize());
2911        // SAFETY: known to be a UTF8 string because it's a `Symbol`.
2912        unsafe { str::from_utf8_unchecked(byte_str) }
2913    }
2914
2915    /// Get the symbol as a string.
2916    ///
2917    /// [`ByteSymbol::as_byte_str()`] should be used in preference to this function.
2918    fn get_byte_str(&self, symbol: ByteSymbol) -> &[u8] {
2919        self.get_inner(symbol.0.as_usize())
2920    }
2921
2922    fn get_inner(&self, index: usize) -> &[u8] {
2923        self.0.lock().byte_strs.get_index(index).unwrap()
2924    }
2925}
