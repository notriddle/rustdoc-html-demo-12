13#[derive(Copy, Clone)]
14pub struct Erased<T: Copy> {
15    // We use `MaybeUninit` here so we can store any value
16    // in `data` since we aren't actually storing a `T`.
17    data: MaybeUninit<T>,
18}
19
20pub trait EraseType: Copy {
21    type Result: Copy;
22}
23
24// Allow `type_alias_bounds` since compilation will fail without `EraseType`.
25#[allow(type_alias_bounds)]
26pub type Erase<T: EraseType> = Erased<impl Copy>;
27
28#[inline(always)]
29#[define_opaque(Erase)]
30pub fn erase<T: EraseType>(src: T) -> Erase<T> {
31    // Ensure the sizes match
32    const {
33        if size_of::<T>() != size_of::<T::Result>() {
34            panic!("size of T must match erased type T::Result")
35        }
36    };
37
38    Erased::<<T as EraseType>::Result> {
39        // `transmute_unchecked` is needed here because it does not have `transmute`'s size check
40        // (and thus allows to transmute between `T` and `MaybeUninit<T::Result>`) (we do the size
41        // check ourselves in the `const` block above).
42        //
43        // `transmute_copy` is also commonly used for this (and it would work here since
44        // `EraseType: Copy`), but `transmute_unchecked` better explains the intent.
45        //
46        // SAFETY: It is safe to transmute to MaybeUninit for types with the same sizes.
47        data: unsafe { transmute_unchecked::<T, MaybeUninit<T::Result>>(src) },
48    }
49}
50
51/// Restores an erased value.
52#[inline(always)]
53#[define_opaque(Erase)]
54pub fn restore<T: EraseType>(value: Erase<T>) -> T {
55    let value: Erased<<T as EraseType>::Result> = value;
56    // See comment in `erase` for why we use `transmute_unchecked`.
57    //
58    // SAFETY: Due to the use of impl Trait in `Erase` the only way to safely create an instance
59    // of `Erase` is to call `erase`, so we know that `value.data` is a valid instance of `T` of
60    // the right size.
61    unsafe { transmute_unchecked::<MaybeUninit<T::Result>, T>(value.data) }
62}
