1//! Detecting usage of the `#[debugger_visualizer]` attribute.
2
3use rustc_ast::Attribute;
4use rustc_expand::base::resolve_path;
5use rustc_middle::middle::debugger_visualizer::{DebuggerVisualizerFile, DebuggerVisualizerType};
6use rustc_middle::query::{LocalCrate, Providers};
7use rustc_middle::ty::TyCtxt;
8use rustc_session::Session;
9use rustc_span::sym;
10
11use crate::errors::{DebugVisualizerInvalid, DebugVisualizerUnreadable};
12
13impl DebuggerVisualizerCollector<'_> {
14    fn check_for_debugger_visualizer(&mut self, attr: &Attribute) {
15        if attr.has_name(sym::debugger_visualizer) {
16            let Some(hints) = attr.meta_item_list() else {
17                self.sess.dcx().emit_err(DebugVisualizerInvalid { span: attr.span });
18                return;
19            };
20
21            let [hint] = hints.as_slice() else {
22                self.sess.dcx().emit_err(DebugVisualizerInvalid { span: attr.span });
23                return;
24            };
25
26            let Some(meta_item) = hint.meta_item() else {
27                self.sess.dcx().emit_err(DebugVisualizerInvalid { span: attr.span });
28                return;
29            };
30
31            let (visualizer_type, visualizer_path) = match (meta_item.name(), meta_item.value_str())
32            {
33                (Some(sym::natvis_file), Some(value)) => (DebuggerVisualizerType::Natvis, value),
34                (Some(sym::gdb_script_file), Some(value)) => {
35                    (DebuggerVisualizerType::GdbPrettyPrinter, value)
36                }
37                (_, _) => {
38                    self.sess.dcx().emit_err(DebugVisualizerInvalid { span: meta_item.span });
39                    return;
40                }
41            };
42
43            let file = match resolve_path(&self.sess, visualizer_path.as_str(), attr.span) {
44                Ok(file) => file,
45                Err(err) => {
46                    err.emit();
47                    return;
48                }
49            };
50
51            match self.sess.source_map().load_binary_file(&file) {
52                Ok((source, _)) => {
53                    self.visualizers.push(DebuggerVisualizerFile::new(
54                        source,
55                        visualizer_type,
56                        file,
57                    ));
58                }
59                Err(error) => {
60                    self.sess.dcx().emit_err(DebugVisualizerUnreadable {
61                        span: meta_item.span,
62                        file: &file,
63                        error,
64                    });
65                }
66            }
67        }
68    }
69}
70
71struct DebuggerVisualizerCollector<'a> {
72    sess: &'a Session,
73    visualizers: Vec<DebuggerVisualizerFile>,
74}
75
76impl<'ast> rustc_ast::visit::Visitor<'ast> for DebuggerVisualizerCollector<'_> {
77    fn visit_attribute(&mut self, attr: &'ast Attribute) {
78        self.check_for_debugger_visualizer(attr);
79        rustc_ast::visit::walk_attribute(self, attr);
80    }
81}
82
83/// Traverses and collects the debugger visualizers for a specific crate.
84fn debugger_visualizers(tcx: TyCtxt<'_>, _: LocalCrate) -> Vec<DebuggerVisualizerFile> {
85    let resolver_and_krate = tcx.resolver_for_lowering().borrow();
86    let krate = &*resolver_and_krate.1;
87
88    let mut visitor = DebuggerVisualizerCollector { sess: tcx.sess, visualizers: Vec::new() };
89    rustc_ast::visit::Visitor::visit_crate(&mut visitor, krate);
90
91    // We are collecting visualizers in AST-order, which is deterministic,
92    // so we don't need to do any explicit sorting in order to get a
93    // deterministic query result
94    visitor.visualizers
95}
96
97pub(crate) fn provide(providers: &mut Providers) {
98    providers.debugger_visualizers = debugger_visualizers;
99}
