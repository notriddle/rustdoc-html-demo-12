17impl Target {
18    /// Loads a target descriptor from a JSON object.
19    pub fn from_json(json: &str) -> Result<(Target, TargetWarnings), String> {
20        let json_deserializer = &mut serde_json::Deserializer::from_str(json);
21
22        let json: TargetSpecJson =
23            serde_path_to_error::deserialize(json_deserializer).map_err(|err| err.to_string())?;
24
25        let mut base = Target {
26            llvm_target: json.llvm_target,
27            metadata: Default::default(),
28            pointer_width: json.target_pointer_width,
29            data_layout: json.data_layout,
30            arch: json.arch,
31            options: Default::default(),
32        };
33
34        // FIXME: This doesn't properly validate anything and just ignores the data if it's invalid.
35        // That's okay for now, the only use of this is when generating docs, which we don't do for
36        // custom targets.
37        if let Some(metadata) = json.metadata {
38            base.metadata.description = metadata.description;
39            base.metadata.tier = metadata.tier.filter(|tier| (1..=3).contains(tier));
40            base.metadata.host_tools = metadata.host_tools;
41            base.metadata.std = metadata.std;
42        }
43
44        let alignment_error = |field_name: &str, error: AlignFromBytesError| -> String {
45            let msg = match error {
46                AlignFromBytesError::NotPowerOfTwo(_) => "not a power of 2 number of bytes",
47                AlignFromBytesError::TooLarge(_) => "too large",
48            };
49            format!("`{}` bits is not a valid value for {field_name}: {msg}", error.align() * 8)
50        };
51
52        macro_rules! forward {
53            ($name:ident) => {
54                if let Some($name) = json.$name {
55                    base.$name = $name;
56                }
57            };
58        }
59        macro_rules! forward_opt {
60            ($name:ident) => {
61                if let Some($name) = json.$name {
62                    base.$name = Some($name);
63                }
64            };
65        }
66
67        if let Some(target_endian) = json.target_endian {
68            base.endian = target_endian.0;
69        }
70
71        forward!(frame_pointer);
72        forward!(c_int_width);
73        forward_opt!(c_enum_min_bits); // if None, matches c_int_width
74        forward!(os);
75        forward!(env);
76        forward!(abi);
77        forward!(vendor);
78        forward_opt!(linker);
79        forward!(linker_flavor_json);
80        forward!(lld_flavor_json);
81        forward!(linker_is_gnu_json);
82        forward!(pre_link_objects);
83        forward!(post_link_objects);
84        forward!(pre_link_objects_self_contained);
85        forward!(post_link_objects_self_contained);
86
87        // Deserializes the backwards-compatible variants of `-Clink-self-contained`
88        if let Some(link_self_contained) = json.link_self_contained_backwards_compatible {
89            base.link_self_contained = link_self_contained;
90        }
91        // Deserializes the components variant of `-Clink-self-contained`
92        if let Some(link_self_contained) = json.link_self_contained {
93            let components = link_self_contained
94                .components
95                .into_iter()
96                .fold(LinkSelfContainedComponents::empty(), |a, b| a | b);
97            base.link_self_contained = LinkSelfContainedDefault::WithComponents(components);
98        }
99
100        forward!(pre_link_args_json);
101        forward!(late_link_args_json);
102        forward!(late_link_args_dynamic_json);
103        forward!(late_link_args_static_json);
104        forward!(post_link_args_json);
105        forward_opt!(link_script);
106
107        if let Some(link_env) = json.link_env {
108            for s in link_env {
109                if let [k, v] = *s.split('=').collect::<Vec<_>>() {
110                    base.link_env.to_mut().push((k.to_string().into(), v.to_string().into()))
111                } else {
112                    return Err(format!("link-env value '{s}' must be of the pattern 'KEY=VALUE'"));
113                }
114            }
115        }
116
117        forward!(link_env_remove);
118        forward!(asm_args);
119        forward!(cpu);
120        forward!(need_explicit_cpu);
121        forward!(features);
122        forward!(dynamic_linking);
123        forward_opt!(direct_access_external_data);
124        forward!(dll_tls_export);
125        forward!(only_cdylib);
126        forward!(executables);
127        forward!(relocation_model);
128        forward_opt!(code_model);
129        forward!(tls_model);
130        forward!(disable_redzone);
131        forward!(function_sections);
132        forward!(dll_prefix);
133        forward!(dll_suffix);
134        forward!(exe_suffix);
135        forward!(staticlib_prefix);
136        forward!(staticlib_suffix);
137
138        if let Some(target_family) = json.target_family {
139            match target_family {
140                TargetFamiliesJson::Array(families) => base.families = families,
141                TargetFamiliesJson::String(family) => base.families = vec![family].into(),
142            }
143        }
144
145        forward!(abi_return_struct_as_int);
146        forward!(is_like_aix);
147        forward!(is_like_darwin);
148        forward!(is_like_solaris);
149        forward!(is_like_windows);
150        forward!(is_like_msvc);
151        forward!(is_like_wasm);
152        forward!(is_like_android);
153        forward!(is_like_vexos);
154        forward!(binary_format);
155        forward!(default_dwarf_version);
156        forward!(allows_weak_linkage);
157        forward!(has_rpath);
158        forward!(no_default_libraries);
159        forward!(position_independent_executables);
160        forward!(static_position_independent_executables);
161        forward!(plt_by_default);
162        forward!(relro_level);
163        forward!(archive_format);
164        forward!(allow_asm);
165        forward!(main_needs_argc_argv);
166        forward!(has_thread_local);
167        forward!(obj_is_bitcode);
168        forward_opt!(max_atomic_width);
169        forward_opt!(min_atomic_width);
170        forward!(atomic_cas);
171        forward!(panic_strategy);
172        forward!(crt_static_allows_dylibs);
173        forward!(crt_static_default);
174        forward!(crt_static_respected);
175        forward!(stack_probes);
176
177        if let Some(min_global_align) = json.min_global_align {
178            match Align::from_bits(min_global_align) {
179                Ok(align) => base.min_global_align = Some(align),
180                Err(e) => return Err(alignment_error("min-global-align", e)),
181            }
182        }
183
184        forward_opt!(default_codegen_units);
185        forward_opt!(default_codegen_backend);
186        forward!(trap_unreachable);
187        forward!(requires_lto);
188        forward!(singlethread);
189        forward!(no_builtins);
190        forward_opt!(default_visibility);
191        forward!(emit_debug_gdb_scripts);
192        forward!(requires_uwtable);
193        forward!(default_uwtable);
194        forward!(simd_types_indirect);
195        forward!(limit_rdylib_exports);
196        forward_opt!(override_export_symbols);
197        forward!(merge_functions);
198        forward!(mcount);
199        forward_opt!(llvm_mcount_intrinsic);
200        forward!(llvm_abiname);
201        forward_opt!(llvm_floatabi);
202        forward_opt!(rustc_abi);
203        forward!(relax_elf_relocations);
204        forward!(llvm_args);
205        forward!(use_ctors_section);
206        forward!(eh_frame_header);
207        forward!(has_thumb_interworking);
208        forward!(debuginfo_kind);
209        forward!(split_debuginfo);
210        forward!(supported_split_debuginfo);
211
212        if let Some(supported_sanitizers) = json.supported_sanitizers {
213            base.supported_sanitizers =
214                supported_sanitizers.into_iter().fold(SanitizerSet::empty(), |a, b| a | b);
215        }
216
217        forward!(generate_arange_section);
218        forward!(supports_stack_protector);
219        forward!(small_data_threshold_support);
220        forward!(entry_name);
221        forward!(supports_xray);
222
223        // we're going to run `update_from_cli`, but that won't change the target's AbiMap
224        // FIXME: better factor the Target definition so we enforce this on a type level
225        let abi_map = AbiMap::from_target(&base);
226        if let Some(entry_abi) = json.entry_abi {
227            base.options.entry_abi = abi_map.canonize_abi(entry_abi.0, false).unwrap();
228        }
229
230        base.update_from_cli();
231        base.check_consistency(TargetKind::Json)?;
232
233        Ok((base, TargetWarnings { unused_fields: vec![] }))
234    }
235}

238    fn to_json(&self) -> Json {
239        let mut d = serde_json::Map::new();
240        let default: TargetOptions = Default::default();
241        let mut target = self.clone();

244        macro_rules! target_val {
245            ($attr:ident) => {
246                target_val!($attr, (stringify!($attr)).replace("_", "-"))
247            };
248            ($attr:ident, $json_name:expr) => {{
249                let name = $json_name;
250                d.insert(name.into(), target.$attr.to_json());
251            }};
252        }
253
254        macro_rules! target_option_val {
255            ($attr:ident) => {{ target_option_val!($attr, (stringify!($attr)).replace("_", "-")) }};
256            ($attr:ident, $json_name:expr) => {{
257                let name = $json_name;
258                if default.$attr != target.$attr {
259                    d.insert(name.into(), target.$attr.to_json());
260                }
261            }};
262            (link_args - $attr:ident, $json_name:expr) => {{
263                let name = $json_name;
264                if default.$attr != target.$attr {
265                    let obj = target
266                        .$attr
267                        .iter()
268                        .map(|(k, v)| (k.desc().to_string(), v.clone()))
269                        .collect::<BTreeMap<_, _>>();
270                    d.insert(name.to_string(), obj.to_json());
271                }
272            }};
273            (env - $attr:ident) => {{
274                let name = (stringify!($attr)).replace("_", "-");
275                if default.$attr != target.$attr {
276                    let obj = target
277                        .$attr
278                        .iter()
279                        .map(|&(ref k, ref v)| format!("{k}={v}"))
280                        .collect::<Vec<_>>();
281                    d.insert(name, obj.to_json());
282                }

423/// `Endian` is in `rustc_abi`, which doesn't have access to the macro and serde.
424struct EndianWrapper(rustc_abi::Endian);
425impl FromStr for EndianWrapper {
426    type Err = String;
427    fn from_str(s: &str) -> Result<Self, Self::Err> {
428        rustc_abi::Endian::from_str(s).map(Self)
429    }
430}
431crate::json::serde_deserialize_from_str!(EndianWrapper);
