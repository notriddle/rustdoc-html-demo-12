18impl<'a, 'tcx> Builder<'a, 'tcx> {
19    /// Compile `expr`, yielding a compile-time constant. Assumes that
20    /// `expr` is a valid compile-time constant!
21    pub(crate) fn as_constant(&mut self, expr: &Expr<'tcx>) -> ConstOperand<'tcx> {
22        let this = self;
23        let tcx = this.tcx;
24        let Expr { ty, temp_lifetime: _, span, ref kind } = *expr;
25        match kind {
26            ExprKind::Scope { region_scope: _, lint_level: _, value } => {
27                this.as_constant(&this.thir[*value])
28            }
29            _ => as_constant_inner(
30                expr,
31                |user_ty| {
32                    Some(this.canonical_user_type_annotations.push(CanonicalUserTypeAnnotation {
33                        span,
34                        user_ty: user_ty.clone(),
35                        inferred_ty: ty,
36                    }))
37                },
38                tcx,
39            ),
40        }
41    }
42}

44pub(crate) fn as_constant_inner<'tcx>(
45    expr: &Expr<'tcx>,
46    push_cuta: impl FnMut(&Box<CanonicalUserType<'tcx>>) -> Option<UserTypeAnnotationIndex>,
47    tcx: TyCtxt<'tcx>,
48) -> ConstOperand<'tcx> {
49    let Expr { ty, temp_lifetime: _, span, ref kind } = *expr;
50    match *kind {
51        ExprKind::Literal { lit, neg } => {
52            let const_ = lit_to_mir_constant(tcx, LitToConstInput { lit: lit.node, ty, neg });
53
54            ConstOperand { span, user_ty: None, const_ }
55        }
56        ExprKind::NonHirLiteral { lit, ref user_ty } => {
57            let user_ty = user_ty.as_ref().and_then(push_cuta);
58
59            let const_ = Const::Val(ConstValue::Scalar(Scalar::Int(lit)), ty);
60
61            ConstOperand { span, user_ty, const_ }
62        }
63        ExprKind::ZstLiteral { ref user_ty } => {
64            let user_ty = user_ty.as_ref().and_then(push_cuta);
65
66            let const_ = Const::Val(ConstValue::ZeroSized, ty);
67
68            ConstOperand { span, user_ty, const_ }
69        }
70        ExprKind::NamedConst { def_id, args, ref user_ty } => {
71            let user_ty = user_ty.as_ref().and_then(push_cuta);
72
73            let uneval = mir::UnevaluatedConst::new(def_id, args);
74            let const_ = Const::Unevaluated(uneval, ty);
75
76            ConstOperand { user_ty, span, const_ }
77        }
78        ExprKind::ConstParam { param, def_id: _ } => {
79            let const_param = ty::Const::new_param(tcx, param);
80            let const_ = Const::Ty(expr.ty, const_param);
81
82            ConstOperand { user_ty: None, span, const_ }
83        }
84        ExprKind::ConstBlock { did: def_id, args } => {
85            let uneval = mir::UnevaluatedConst::new(def_id, args);
86            let const_ = Const::Unevaluated(uneval, ty);
87
88            ConstOperand { user_ty: None, span, const_ }
89        }
90        ExprKind::StaticRef { alloc_id, ty, .. } => {
91            let const_val = ConstValue::Scalar(Scalar::from_pointer(alloc_id.into(), &tcx));
92            let const_ = Const::Val(const_val, ty);
93
94            ConstOperand { span, user_ty: None, const_ }
95        }
96        _ => span_bug!(span, "expression is not a valid constant {:?}", kind),
97    }
98}

100#[instrument(skip(tcx, lit_input))]
101fn lit_to_mir_constant<'tcx>(tcx: TyCtxt<'tcx>, lit_input: LitToConstInput<'tcx>) -> Const<'tcx> {
102    let LitToConstInput { lit, ty, neg } = lit_input;
103
104    if let Err(guar) = ty.error_reported() {
105        return Const::Ty(Ty::new_error(tcx, guar), ty::Const::new_error(tcx, guar));
106    }
107
108    let lit_ty = match *ty.kind() {
109        ty::Pat(base, _) => base,
110        _ => ty,
111    };
112
113    let trunc = |n| {
114        let width = lit_ty.primitive_size(tcx);
115        trace!("trunc {} with size {} and shift {}", n, width.bits(), 128 - width.bits());
116        let result = width.truncate(n);
117        trace!("trunc result: {}", result);
118        ConstValue::Scalar(Scalar::from_uint(result, width))
119    };
120
121    let value = match (lit, lit_ty.kind()) {
122        (ast::LitKind::Str(s, _), ty::Ref(_, inner_ty, _)) if inner_ty.is_str() => {
123            let s = s.as_str().as_bytes();
124            let len = s.len();
125            let allocation = tcx.allocate_bytes_dedup(s, CTFE_ALLOC_SALT);
126            ConstValue::Slice { alloc_id: allocation, meta: len.try_into().unwrap() }
127        }
128        (ast::LitKind::ByteStr(byte_sym, _), ty::Ref(_, inner_ty, _))
129            if matches!(inner_ty.kind(), ty::Slice(_)) =>
130        {
131            let data = byte_sym.as_byte_str();
132            let len = data.len();
133            let allocation = tcx.allocate_bytes_dedup(data, CTFE_ALLOC_SALT);
134            ConstValue::Slice { alloc_id: allocation, meta: len.try_into().unwrap() }
135        }
136        (ast::LitKind::ByteStr(byte_sym, _), ty::Ref(_, inner_ty, _)) if inner_ty.is_array() => {
137            let id = tcx.allocate_bytes_dedup(byte_sym.as_byte_str(), CTFE_ALLOC_SALT);
138            ConstValue::Scalar(Scalar::from_pointer(id.into(), &tcx))
139        }
140        (ast::LitKind::CStr(byte_sym, _), ty::Ref(_, inner_ty, _)) if matches!(inner_ty.kind(), ty::Adt(def, _) if tcx.is_lang_item(def.did(), LangItem::CStr)) =>
141        {
142            let data = byte_sym.as_byte_str();
143            let len = data.len();
144            let allocation = tcx.allocate_bytes_dedup(data, CTFE_ALLOC_SALT);
145            ConstValue::Slice { alloc_id: allocation, meta: len.try_into().unwrap() }
146        }
147        (ast::LitKind::Byte(n), ty::Uint(ty::UintTy::U8)) => {
148            ConstValue::Scalar(Scalar::from_uint(n, Size::from_bytes(1)))
149        }
150        (ast::LitKind::Int(n, _), ty::Uint(_)) if !neg => trunc(n.get()),
151        (ast::LitKind::Int(n, _), ty::Int(_)) => {
152            trunc(if neg { (n.get() as i128).overflowing_neg().0 as u128 } else { n.get() })
153        }
154        (ast::LitKind::Float(n, _), ty::Float(fty)) => {
155            parse_float_into_constval(n, *fty, neg).unwrap()
156        }
157        (ast::LitKind::Bool(b), ty::Bool) => ConstValue::Scalar(Scalar::from_bool(b)),
158        (ast::LitKind::Char(c), ty::Char) => ConstValue::Scalar(Scalar::from_char(c)),
159        (ast::LitKind::Err(guar), _) => {
160            return Const::Ty(Ty::new_error(tcx, guar), ty::Const::new_error(tcx, guar));
161        }
162        _ => bug!("invalid lit/ty combination in `lit_to_mir_constant`: {lit:?}: {ty:?}"),
163    };
164
165    Const::Val(value, ty)
166}
