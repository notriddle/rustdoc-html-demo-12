71/// Extracts coverage-relevant spans from MIR, and associates them with
72/// their corresponding BCBs.
73pub(super) fn extract_all_mapping_info_from_mir<'tcx>(
74    tcx: TyCtxt<'tcx>,
75    mir_body: &mir::Body<'tcx>,
76    hir_info: &ExtractedHirInfo,
77    graph: &CoverageGraph,
78) -> ExtractedMappings {
79    let mut code_mappings = vec![];
80    let mut branch_pairs = vec![];
81    let mut mcdc_bitmap_bits = 0;
82    let mut mcdc_degraded_branches = vec![];
83    let mut mcdc_mappings = vec![];
84
85    // Extract ordinary code mappings from MIR statement/terminator spans.
86    extract_refined_covspans(tcx, mir_body, hir_info, graph, &mut code_mappings);
87
88    branch_pairs.extend(extract_branch_pairs(mir_body, hir_info, graph));
89
90    extract_mcdc_mappings(
91        mir_body,
92        tcx,
93        hir_info.body_span,
94        graph,
95        &mut mcdc_bitmap_bits,
96        &mut mcdc_degraded_branches,
97        &mut mcdc_mappings,
98    );
99
100    ExtractedMappings {
101        code_mappings,
102        branch_pairs,
103        mcdc_bitmap_bits,
104        mcdc_degraded_branches,
105        mcdc_mappings,
106    }
107}

109fn resolve_block_markers(
110    coverage_info_hi: &CoverageInfoHi,
111    mir_body: &mir::Body<'_>,
112) -> IndexVec<BlockMarkerId, Option<BasicBlock>> {
113    let mut block_markers = IndexVec::<BlockMarkerId, Option<BasicBlock>>::from_elem_n(
114        None,
115        coverage_info_hi.num_block_markers,
116    );
117
118    // Fill out the mapping from block marker IDs to their enclosing blocks.
119    for (bb, data) in mir_body.basic_blocks.iter_enumerated() {
120        for statement in &data.statements {
121            if let StatementKind::Coverage(CoverageKind::BlockMarker { id }) = statement.kind {
122                block_markers[id] = Some(bb);
123            }
124        }
125    }
126
127    block_markers
128}

131// `extract_branch_pairs` and `extract_mcdc_mappings`. This is needed so
132// that they can each be modified without interfering with the other, but in
133// the long term we should try to bring them together again when branch coverage

136pub(super) fn extract_branch_pairs(
137    mir_body: &mir::Body<'_>,
138    hir_info: &ExtractedHirInfo,
139    graph: &CoverageGraph,
140) -> Vec<BranchPair> {
141    let Some(coverage_info_hi) = mir_body.coverage_info_hi.as_deref() else { return vec![] };
142
143    let block_markers = resolve_block_markers(coverage_info_hi, mir_body);
144
145    coverage_info_hi
146        .branch_spans
147        .iter()
148        .filter_map(|&BranchSpan { span: raw_span, true_marker, false_marker }| {
149            // For now, ignore any branch span that was introduced by
150            // expansion. This makes things like assert macros less noisy.
151            if !raw_span.ctxt().outer_expn_data().is_root() {
152                return None;
153            }
154            let span = unexpand_into_body_span(raw_span, hir_info.body_span)?;
155
156            let bcb_from_marker = |marker: BlockMarkerId| graph.bcb_from_bb(block_markers[marker]?);
157
158            let true_bcb = bcb_from_marker(true_marker)?;
159            let false_bcb = bcb_from_marker(false_marker)?;
160
161            Some(BranchPair { span, true_bcb, false_bcb })
162        })
163        .collect::<Vec<_>>()
164}

166pub(super) fn extract_mcdc_mappings(
167    mir_body: &mir::Body<'_>,
168    tcx: TyCtxt<'_>,
169    body_span: Span,
170    graph: &CoverageGraph,
171    mcdc_bitmap_bits: &mut usize,
172    mcdc_degraded_branches: &mut impl Extend<MCDCBranch>,
173    mcdc_mappings: &mut impl Extend<(MCDCDecision, Vec<MCDCBranch>)>,
174) {
175    let Some(coverage_info_hi) = mir_body.coverage_info_hi.as_deref() else { return };
176
177    let block_markers = resolve_block_markers(coverage_info_hi, mir_body);
178
179    let bcb_from_marker = |marker: BlockMarkerId| graph.bcb_from_bb(block_markers[marker]?);
180
181    let check_branch_bcb =
182        |raw_span: Span, true_marker: BlockMarkerId, false_marker: BlockMarkerId| {
183            // For now, ignore any branch span that was introduced by
184            // expansion. This makes things like assert macros less noisy.
185            if !raw_span.ctxt().outer_expn_data().is_root() {
186                return None;
187            }
188            let span = unexpand_into_body_span(raw_span, body_span)?;
189
190            let true_bcb = bcb_from_marker(true_marker)?;
191            let false_bcb = bcb_from_marker(false_marker)?;
192            Some((span, true_bcb, false_bcb))
193        };
194
195    let to_mcdc_branch = |&mir::coverage::MCDCBranchSpan {
196                              span: raw_span,
197                              condition_info,
198                              true_marker,
199                              false_marker,
200                          }| {
201        let (span, true_bcb, false_bcb) = check_branch_bcb(raw_span, true_marker, false_marker)?;
202        Some(MCDCBranch {
203            span,
204            true_bcb,
205            false_bcb,
206            condition_info,
207            true_index: usize::MAX,
208            false_index: usize::MAX,
209        })
210    };
211
212    let mut get_bitmap_idx = |num_test_vectors: usize| -> Option<usize> {
213        let bitmap_idx = *mcdc_bitmap_bits;
214        let next_bitmap_bits = bitmap_idx.saturating_add(num_test_vectors);
215        (next_bitmap_bits <= MCDC_MAX_BITMAP_SIZE).then(|| {
216            *mcdc_bitmap_bits = next_bitmap_bits;
217            bitmap_idx
218        })
219    };
220    mcdc_degraded_branches
221        .extend(coverage_info_hi.mcdc_degraded_branch_spans.iter().filter_map(to_mcdc_branch));
222
223    mcdc_mappings.extend(coverage_info_hi.mcdc_spans.iter().filter_map(|(decision, branches)| {
224        if branches.len() == 0 {
225            return None;
226        }
227        let decision_span = unexpand_into_body_span(decision.span, body_span)?;
228
229        let end_bcbs = decision
230            .end_markers
231            .iter()
232            .map(|&marker| bcb_from_marker(marker))
233            .collect::<Option<_>>()?;
234        let mut branch_mappings: Vec<_> = branches.into_iter().filter_map(to_mcdc_branch).collect();
235        if branch_mappings.len() != branches.len() {
236            mcdc_degraded_branches.extend(branch_mappings);
237            return None;
238        }
239        let num_test_vectors = calc_test_vectors_index(&mut branch_mappings);
240        let Some(bitmap_idx) = get_bitmap_idx(num_test_vectors) else {
241            tcx.dcx().emit_warn(MCDCExceedsTestVectorLimit {
242                span: decision_span,
243                max_num_test_vectors: MCDC_MAX_BITMAP_SIZE,
244            });
245            mcdc_degraded_branches.extend(branch_mappings);
246            return None;
247        };
248        // LLVM requires span of the decision contains all spans of its conditions.
249        // Usually the decision span meets the requirement well but in cases like macros it may not.
250        let span = branch_mappings
251            .iter()
252            .map(|branch| branch.span)
253            .reduce(|lhs, rhs| lhs.to(rhs))
254            .map(
255                |joint_span| {
256                    if decision_span.contains(joint_span) { decision_span } else { joint_span }
257                },
258            )
259            .expect("branch mappings are ensured to be non-empty as checked above");
260        Some((
261            MCDCDecision {
262                span,
263                end_bcbs,
264                bitmap_idx,
265                num_test_vectors,
266                decision_depth: decision.decision_depth,
267            },
268            branch_mappings,
269        ))
270    }));
271}

273// LLVM checks the executed test vector by accumulating indices of tested branches.
274// We calculate number of all possible test vectors of the decision and assign indices
275// to branches here.
276// See [the rfc](https://discourse.llvm.org/t/rfc-coverage-new-algorithm-and-file-format-for-mc-dc/76798/)
277// for more details about the algorithm.
278// This function is mostly like [`TVIdxBuilder::TvIdxBuilder`](https://github.com/llvm/llvm-project/blob/d594d9f7f4dc6eb748b3261917db689fdc348b96/llvm/lib/ProfileData/Coverage/CoverageMapping.cpp#L226)
279fn calc_test_vectors_index(conditions: &mut Vec<MCDCBranch>) -> usize {
280    let mut indegree_stats = IndexVec::<ConditionId, usize>::from_elem_n(0, conditions.len());
281    // `num_paths` is `width` described at the llvm rfc, which indicates how many paths reaching the condition node.
282    let mut num_paths_stats = IndexVec::<ConditionId, usize>::from_elem_n(0, conditions.len());
283    let mut next_conditions = conditions
284        .iter_mut()
285        .map(|branch| {
286            let ConditionInfo { condition_id, true_next_id, false_next_id } = branch.condition_info;
287            [true_next_id, false_next_id]
288                .into_iter()
289                .flatten()
290                .for_each(|next_id| indegree_stats[next_id] += 1);
291            (condition_id, branch)
292        })
293        .collect::<FxIndexMap<_, _>>();
294
295    let mut queue =
296        std::collections::VecDeque::from_iter(next_conditions.swap_remove(&ConditionId::START));
297    num_paths_stats[ConditionId::START] = 1;
298    let mut decision_end_nodes = Vec::new();
299    while let Some(branch) = queue.pop_front() {
300        let ConditionInfo { condition_id, true_next_id, false_next_id } = branch.condition_info;
301        let (false_index, true_index) = (&mut branch.false_index, &mut branch.true_index);
302        let this_paths_count = num_paths_stats[condition_id];
303        // Note. First check the false next to ensure conditions are touched in same order with llvm-cov.
304        for (next, index) in [(false_next_id, false_index), (true_next_id, true_index)] {
305            if let Some(next_id) = next {
306                let next_paths_count = &mut num_paths_stats[next_id];
307                *index = *next_paths_count;
308                *next_paths_count = next_paths_count.saturating_add(this_paths_count);
309                let next_indegree = &mut indegree_stats[next_id];
310                *next_indegree -= 1;
311                if *next_indegree == 0 {
312                    queue.push_back(next_conditions.swap_remove(&next_id).expect(
313                        "conditions with non-zero indegree before must be in next_conditions",
314                    ));
315                }
316            } else {
317                decision_end_nodes.push((this_paths_count, condition_id, index));
318            }
319        }
320    }
321    assert!(next_conditions.is_empty(), "the decision tree has untouched nodes");
322    let mut cur_idx = 0;
323    // LLVM hopes the end nodes are sorted in descending order by `num_paths` so that it can
324    // optimize bitmap size for decisions in tree form such as `a && b && c && d && ...`.
325    decision_end_nodes.sort_by_key(|(num_paths, _, _)| usize::MAX - *num_paths);
326    for (num_paths, condition_id, index) in decision_end_nodes {
327        assert_eq!(
328            num_paths, num_paths_stats[condition_id],
329            "end nodes should not be updated since they were visited"
330        );
331        assert_eq!(*index, usize::MAX, "end nodes should not be assigned index before");
332        *index = cur_idx;
333        cur_idx += num_paths;
334    }
335    cur_idx
