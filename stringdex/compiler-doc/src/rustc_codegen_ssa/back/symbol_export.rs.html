23fn threshold(tcx: TyCtxt<'_>) -> SymbolExportLevel {
24    crates_export_threshold(tcx.crate_types())
25}

27fn crate_export_threshold(crate_type: CrateType) -> SymbolExportLevel {
28    match crate_type {
29        CrateType::Executable | CrateType::Staticlib | CrateType::ProcMacro | CrateType::Cdylib => {
30            SymbolExportLevel::C
31        }
32        CrateType::Rlib | CrateType::Dylib | CrateType::Sdylib => SymbolExportLevel::Rust,
33    }
34}

36pub fn crates_export_threshold(crate_types: &[CrateType]) -> SymbolExportLevel {
37    if crate_types
38        .iter()
39        .any(|&crate_type| crate_export_threshold(crate_type) == SymbolExportLevel::Rust)
40    {
41        SymbolExportLevel::Rust
42    } else {
43        SymbolExportLevel::C
44    }
45}

47fn reachable_non_generics_provider(tcx: TyCtxt<'_>, _: LocalCrate) -> DefIdMap<SymbolExportInfo> {
48    if !tcx.sess.opts.output_types.should_codegen() && !tcx.is_sdylib_interface_build() {
49        return Default::default();
50    }
51
52    // Check to see if this crate is a "special runtime crate". These
53    // crates, implementation details of the standard library, typically
54    // have a bunch of `pub extern` and `#[no_mangle]` functions as the
55    // ABI between them. We don't want their symbols to have a `C`
56    // export level, however, as they're just implementation details.
57    // Down below we'll hardwire all of the symbols to the `Rust` export
58    // level instead.
59    let special_runtime_crate =
60        tcx.is_panic_runtime(LOCAL_CRATE) || tcx.is_compiler_builtins(LOCAL_CRATE);
61
62    let mut reachable_non_generics: DefIdMap<_> = tcx
63        .reachable_set(())
64        .items()
65        .filter_map(|&def_id| {
66            // We want to ignore some FFI functions that are not exposed from
67            // this crate. Reachable FFI functions can be lumped into two
68            // categories:
69            //
70            // 1. Those that are included statically via a static library
71            // 2. Those included otherwise (e.g., dynamically or via a framework)
72            //
73            // Although our LLVM module is not literally emitting code for the
74            // statically included symbols, it's an export of our library which
75            // needs to be passed on to the linker and encoded in the metadata.
76            //
77            // As a result, if this id is an FFI item (foreign item) then we only
78            // let it through if it's included statically.
79            if let Some(parent_id) = tcx.opt_local_parent(def_id)
80                && let DefKind::ForeignMod = tcx.def_kind(parent_id)
81            {
82                let library = tcx.native_library(def_id)?;
83                return library.kind.is_statically_included().then_some(def_id);
84            }
85
86            // Only consider nodes that actually have exported symbols.
87            match tcx.def_kind(def_id) {
88                DefKind::Fn | DefKind::Static { .. } => {}
89                DefKind::AssocFn if tcx.impl_of_assoc(def_id.to_def_id()).is_some() => {}
90                _ => return None,
91            };
92
93            let generics = tcx.generics_of(def_id);
94            if generics.requires_monomorphization(tcx) {
95                return None;
96            }
97
98            if Instance::mono(tcx, def_id.into()).def.requires_inline(tcx) {
99                return None;
100            }
101
102            if tcx.cross_crate_inlinable(def_id) { None } else { Some(def_id) }
103        })
104        .map(|def_id| {
105            // We won't link right if this symbol is stripped during LTO.
106            let name = tcx.symbol_name(Instance::mono(tcx, def_id.to_def_id())).name;
107            let used = name == "rust_eh_personality";
108
109            let export_level = if special_runtime_crate {
110                SymbolExportLevel::Rust
111            } else {
112                symbol_export_level(tcx, def_id.to_def_id())
113            };
114            let codegen_attrs = tcx.codegen_fn_attrs(def_id.to_def_id());
115            debug!(
116                "EXPORTED SYMBOL (local): {} ({:?})",
117                tcx.symbol_name(Instance::mono(tcx, def_id.to_def_id())),
118                export_level
119            );
120            let info = SymbolExportInfo {
121                level: export_level,
122                kind: if tcx.is_static(def_id.to_def_id()) {
123                    if codegen_attrs.flags.contains(CodegenFnAttrFlags::THREAD_LOCAL) {
124                        SymbolExportKind::Tls
125                    } else {
126                        SymbolExportKind::Data
127                    }
128                } else {
129                    SymbolExportKind::Text
130                },
131                used: codegen_attrs.flags.contains(CodegenFnAttrFlags::USED_COMPILER)
132                    || codegen_attrs.flags.contains(CodegenFnAttrFlags::USED_LINKER)
133                    || used,
134                rustc_std_internal_symbol: codegen_attrs
135                    .flags
136                    .contains(CodegenFnAttrFlags::RUSTC_STD_INTERNAL_SYMBOL),
137            };
138            (def_id.to_def_id(), info)
139        })
140        .into();
141
142    if let Some(id) = tcx.proc_macro_decls_static(()) {
143        reachable_non_generics.insert(
144            id.to_def_id(),
145            SymbolExportInfo {
146                level: SymbolExportLevel::C,
147                kind: SymbolExportKind::Data,
148                used: false,
149                rustc_std_internal_symbol: false,
150            },
151        );
152    }
153
154    reachable_non_generics
155}

157fn is_reachable_non_generic_provider_local(tcx: TyCtxt<'_>, def_id: LocalDefId) -> bool {
158    let export_threshold = threshold(tcx);
159
160    if let Some(&info) = tcx.reachable_non_generics(LOCAL_CRATE).get(&def_id.to_def_id()) {
161        info.level.is_below_threshold(export_threshold)
162    } else {
163        false
164    }
165}

167fn is_reachable_non_generic_provider_extern(tcx: TyCtxt<'_>, def_id: DefId) -> bool {
168    tcx.reachable_non_generics(def_id.krate).contains_key(&def_id)
169}

171fn exported_non_generic_symbols_provider_local<'tcx>(
172    tcx: TyCtxt<'tcx>,
173    _: LocalCrate,
174) -> &'tcx [(ExportedSymbol<'tcx>, SymbolExportInfo)] {
175    if !tcx.sess.opts.output_types.should_codegen() && !tcx.is_sdylib_interface_build() {
176        return &[];
177    }
178
179    // FIXME: Sorting this is unnecessary since we are sorting later anyway.
180    //        Can we skip the later sorting?
181    let sorted = tcx.with_stable_hashing_context(|hcx| {
182        tcx.reachable_non_generics(LOCAL_CRATE).to_sorted(&hcx, true)
183    });
184
185    let mut symbols: Vec<_> =
186        sorted.iter().map(|&(&def_id, &info)| (ExportedSymbol::NonGeneric(def_id), info)).collect();
187
188    // Export TLS shims
189    if !tcx.sess.target.dll_tls_export {
190        symbols.extend(sorted.iter().filter_map(|&(&def_id, &info)| {
191            tcx.needs_thread_local_shim(def_id).then(|| {
192                (
193                    ExportedSymbol::ThreadLocalShim(def_id),
194                    SymbolExportInfo {
195                        level: info.level,
196                        kind: SymbolExportKind::Text,
197                        used: info.used,
198                        rustc_std_internal_symbol: info.rustc_std_internal_symbol,
199                    },
200                )
201            })
202        }))
203    }
204
205    if tcx.entry_fn(()).is_some() {
206        let exported_symbol =
207            ExportedSymbol::NoDefId(SymbolName::new(tcx, tcx.sess.target.entry_name.as_ref()));
208
209        symbols.push((
210            exported_symbol,
211            SymbolExportInfo {
212                level: SymbolExportLevel::C,
213                kind: SymbolExportKind::Text,
214                used: false,
215                rustc_std_internal_symbol: false,
216            },
217        ));
218    }
219
220    // Mark allocator shim symbols as exported only if they were generated.
221    if allocator_kind_for_codegen(tcx).is_some() {
222        for symbol_name in ALLOCATOR_METHODS
223            .iter()
224            .map(|method| mangle_internal_symbol(tcx, global_fn_name(method.name).as_str()))
225            .chain([
226                mangle_internal_symbol(tcx, "__rust_alloc_error_handler"),
227                mangle_internal_symbol(tcx, OomStrategy::SYMBOL),
228                mangle_internal_symbol(tcx, NO_ALLOC_SHIM_IS_UNSTABLE),
229            ])
230        {
231            let exported_symbol = ExportedSymbol::NoDefId(SymbolName::new(tcx, &symbol_name));
232
233            symbols.push((
234                exported_symbol,
235                SymbolExportInfo {
236                    level: SymbolExportLevel::Rust,
237                    kind: SymbolExportKind::Text,
238                    used: false,
239                    rustc_std_internal_symbol: true,
240                },
241            ));
242        }
243    }
244
245    if tcx.sess.instrument_coverage() || tcx.sess.opts.cg.profile_generate.enabled() {
246        // These are weak symbols that point to the profile version and the
247        // profile name, which need to be treated as exported so LTO doesn't nix
248        // them.
249        const PROFILER_WEAK_SYMBOLS: [&str; 2] =
250            ["__llvm_profile_raw_version", "__llvm_profile_filename"];
251
252        symbols.extend(PROFILER_WEAK_SYMBOLS.iter().map(|sym| {
253            let exported_symbol = ExportedSymbol::NoDefId(SymbolName::new(tcx, sym));
254            (
255                exported_symbol,
256                SymbolExportInfo {
257                    level: SymbolExportLevel::C,
258                    kind: SymbolExportKind::Data,
259                    used: false,
260                    rustc_std_internal_symbol: false,
261                },
262            )
263        }));
264    }
265
266    if tcx.sess.opts.unstable_opts.sanitizer.contains(SanitizerSet::MEMORY) {
267        let mut msan_weak_symbols = Vec::new();
268
269        // Similar to profiling, preserve weak msan symbol during LTO.
270        if tcx.sess.opts.unstable_opts.sanitizer_recover.contains(SanitizerSet::MEMORY) {
271            msan_weak_symbols.push("__msan_keep_going");
272        }
273
274        if tcx.sess.opts.unstable_opts.sanitizer_memory_track_origins != 0 {
275            msan_weak_symbols.push("__msan_track_origins");
276        }
277
278        symbols.extend(msan_weak_symbols.into_iter().map(|sym| {
279            let exported_symbol = ExportedSymbol::NoDefId(SymbolName::new(tcx, sym));
280            (
281                exported_symbol,
282                SymbolExportInfo {
283                    level: SymbolExportLevel::C,
284                    kind: SymbolExportKind::Data,
285                    used: false,
286                    rustc_std_internal_symbol: false,
287                },
288            )
289        }));
290    }
291
292    // Sort so we get a stable incr. comp. hash.
293    symbols.sort_by_cached_key(|s| s.0.symbol_name_for_local_instance(tcx));
294
295    tcx.arena.alloc_from_iter(symbols)
296}

298fn exported_generic_symbols_provider_local<'tcx>(
299    tcx: TyCtxt<'tcx>,
300    _: LocalCrate,
301) -> &'tcx [(ExportedSymbol<'tcx>, SymbolExportInfo)] {
302    if !tcx.sess.opts.output_types.should_codegen() && !tcx.is_sdylib_interface_build() {
303        return &[];
304    }
305
306    let mut symbols: Vec<_> = vec![];
307
308    if tcx.local_crate_exports_generics() {
309        use rustc_middle::mir::mono::{Linkage, MonoItem, Visibility};
310        use rustc_middle::ty::InstanceKind;
311
312        // Normally, we require that shared monomorphizations are not hidden,
313        // because if we want to re-use a monomorphization from a Rust dylib, it
314        // needs to be exported.
315        // However, on platforms that don't allow for Rust dylibs, having
316        // external linkage is enough for monomorphization to be linked to.
317        let need_visibility = tcx.sess.target.dynamic_linking && !tcx.sess.target.only_cdylib;
318
319        let cgus = tcx.collect_and_partition_mono_items(()).codegen_units;
320
321        // Do not export symbols that cannot be instantiated by downstream crates.
322        let reachable_set = tcx.reachable_set(());
323        let is_local_to_current_crate = |ty: Ty<'_>| {
324            let no_refs = ty.peel_refs();
325            let root_def_id = match no_refs.kind() {
326                ty::Closure(closure, _) => *closure,
327                ty::FnDef(def_id, _) => *def_id,
328                ty::Coroutine(def_id, _) => *def_id,
329                ty::CoroutineClosure(def_id, _) => *def_id,
330                ty::CoroutineWitness(def_id, _) => *def_id,
331                _ => return false,
332            };
333            let Some(root_def_id) = root_def_id.as_local() else {
334                return false;
335            };
336
337            let is_local = !reachable_set.contains(&root_def_id);
338            is_local
339        };
340
341        let is_instantiable_downstream =
342            |did: Option<DefId>, generic_args: GenericArgsRef<'tcx>| {
343                generic_args
344                    .types()
345                    .chain(did.into_iter().map(move |did| tcx.type_of(did).skip_binder()))
346                    .all(move |arg| {
347                        arg.walk().all(|ty| {
348                            ty.as_type().map_or(true, |ty| !is_local_to_current_crate(ty))
349                        })
350                    })
351            };
352
353        // The symbols created in this loop are sorted below it
354        #[allow(rustc::potential_query_instability)]
355        for (mono_item, data) in cgus.iter().flat_map(|cgu| cgu.items().iter()) {
356            if data.linkage != Linkage::External {
357                // We can only re-use things with external linkage, otherwise
358                // we'll get a linker error
359                continue;
360            }
361
362            if need_visibility && data.visibility == Visibility::Hidden {
363                // If we potentially share things from Rust dylibs, they must
364                // not be hidden
365                continue;
366            }
367
368            if !tcx.sess.opts.share_generics() {
369                if tcx.codegen_fn_attrs(mono_item.def_id()).inline
370                    == rustc_hir::attrs::InlineAttr::Never
371                {
372                    // this is OK, we explicitly allow sharing inline(never) across crates even
373                    // without share-generics.
374                } else {
375                    continue;
376                }
377            }
378
379            // Note: These all set rustc_std_internal_symbol to false as generic functions must not
380            // be marked with this attribute and we are only handling generic functions here.
381            match *mono_item {
382                MonoItem::Fn(Instance { def: InstanceKind::Item(def), args }) => {
383                    let has_generics = args.non_erasable_generics().next().is_some();
384
385                    let should_export =
386                        has_generics && is_instantiable_downstream(Some(def), &args);
387
388                    if should_export {
389                        let symbol = ExportedSymbol::Generic(def, args);
390                        symbols.push((
391                            symbol,
392                            SymbolExportInfo {
393                                level: SymbolExportLevel::Rust,
394                                kind: SymbolExportKind::Text,
395                                used: false,
396                                rustc_std_internal_symbol: false,
397                            },
398                        ));
399                    }
400                }
401                MonoItem::Fn(Instance { def: InstanceKind::DropGlue(_, Some(ty)), args }) => {
402                    // A little sanity-check
403                    assert_eq!(args.non_erasable_generics().next(), Some(GenericArgKind::Type(ty)));
404
405                    // Drop glue did is always going to be non-local outside of libcore, thus we don't need to check it's locality (which includes invoking `type_of` query).
406                    let should_export = match ty.kind() {
407                        ty::Adt(_, args) => is_instantiable_downstream(None, args),
408                        ty::Closure(_, args) => is_instantiable_downstream(None, args),
409                        _ => true,
410                    };
411
412                    if should_export {
413                        symbols.push((
414                            ExportedSymbol::DropGlue(ty),
415                            SymbolExportInfo {
416                                level: SymbolExportLevel::Rust,
417                                kind: SymbolExportKind::Text,
418                                used: false,
419                                rustc_std_internal_symbol: false,
420                            },
421                        ));
422                    }
423                }
424                MonoItem::Fn(Instance {
425                    def: InstanceKind::AsyncDropGlueCtorShim(_, ty),
426                    args,
427                }) => {
428                    // A little sanity-check
429                    assert_eq!(args.non_erasable_generics().next(), Some(GenericArgKind::Type(ty)));
430                    symbols.push((
431                        ExportedSymbol::AsyncDropGlueCtorShim(ty),
432                        SymbolExportInfo {
433                            level: SymbolExportLevel::Rust,
434                            kind: SymbolExportKind::Text,
435                            used: false,
436                            rustc_std_internal_symbol: false,
437                        },
438                    ));
439                }
440                MonoItem::Fn(Instance { def: InstanceKind::AsyncDropGlue(def, ty), args: _ }) => {
441                    symbols.push((
442                        ExportedSymbol::AsyncDropGlue(def, ty),
443                        SymbolExportInfo {
444                            level: SymbolExportLevel::Rust,
445                            kind: SymbolExportKind::Text,
446                            used: false,
447                            rustc_std_internal_symbol: false,
448                        },
449                    ));
450                }
451                _ => {
452                    // Any other symbols don't qualify for sharing
453                }
454            }
455        }
456    }
457
458    // Sort so we get a stable incr. comp. hash.
459    symbols.sort_by_cached_key(|s| s.0.symbol_name_for_local_instance(tcx));
460
461    tcx.arena.alloc_from_iter(symbols)
462}

464fn upstream_monomorphizations_provider(
465    tcx: TyCtxt<'_>,
466    (): (),
467) -> DefIdMap<UnordMap<GenericArgsRef<'_>, CrateNum>> {
468    let cnums = tcx.crates(());
469
470    let mut instances: DefIdMap<UnordMap<_, _>> = Default::default();
471
472    let drop_in_place_fn_def_id = tcx.lang_items().drop_in_place_fn();
473    let async_drop_in_place_fn_def_id = tcx.lang_items().async_drop_in_place_fn();
474
475    for &cnum in cnums.iter() {
476        for (exported_symbol, _) in tcx.exported_generic_symbols(cnum).iter() {
477            let (def_id, args) = match *exported_symbol {
478                ExportedSymbol::Generic(def_id, args) => (def_id, args),
479                ExportedSymbol::DropGlue(ty) => {
480                    if let Some(drop_in_place_fn_def_id) = drop_in_place_fn_def_id {
481                        (drop_in_place_fn_def_id, tcx.mk_args(&[ty.into()]))
482                    } else {
483                        // `drop_in_place` in place does not exist, don't try
484                        // to use it.
485                        continue;
486                    }
487                }
488                ExportedSymbol::AsyncDropGlueCtorShim(ty) => {
489                    if let Some(async_drop_in_place_fn_def_id) = async_drop_in_place_fn_def_id {
490                        (async_drop_in_place_fn_def_id, tcx.mk_args(&[ty.into()]))
491                    } else {
492                        continue;
493                    }
494                }
495                ExportedSymbol::AsyncDropGlue(def_id, ty) => (def_id, tcx.mk_args(&[ty.into()])),
496                ExportedSymbol::NonGeneric(..)
497                | ExportedSymbol::ThreadLocalShim(..)
498                | ExportedSymbol::NoDefId(..) => unreachable!("{exported_symbol:?}"),
499            };
500
501            let args_map = instances.entry(def_id).or_default();
502
503            match args_map.entry(args) {
504                Occupied(mut e) => {
505                    // If there are multiple monomorphizations available,
506                    // we select one deterministically.
507                    let other_cnum = *e.get();
508                    if tcx.stable_crate_id(other_cnum) > tcx.stable_crate_id(cnum) {
509                        e.insert(cnum);
510                    }
511                }
512                Vacant(e) => {
513                    e.insert(cnum);
514                }
515            }
516        }
517    }
518
519    instances
520}

522fn upstream_monomorphizations_for_provider(
523    tcx: TyCtxt<'_>,

551    providers.reachable_non_generics = reachable_non_generics_provider;

553    providers.exported_non_generic_symbols = exported_non_generic_symbols_provider_local;
554    providers.exported_generic_symbols = exported_generic_symbols_provider_local;
555    providers.upstream_monomorphizations = upstream_monomorphizations_provider;

557    providers.upstream_drop_glue_for = upstream_drop_glue_for_provider;
558    providers.upstream_async_drop_glue_for = upstream_async_drop_glue_for_provider;
559    providers.wasm_import_module_map = wasm_import_module_map;
560    providers.extern_queries.is_reachable_non_generic = is_reachable_non_generic_provider_extern;
561    providers.extern_queries.upstream_monomorphizations_for =
562        upstream_monomorphizations_for_provider;
563}

565fn symbol_export_level(tcx: TyCtxt<'_>, sym_def_id: DefId) -> SymbolExportLevel {
566    // We export anything that's not mangled at the "C" layer as it probably has
567    // to do with ABI concerns. We do not, however, apply such treatment to
568    // special symbols in the standard library for various plumbing between
569    // core/std/allocators/etc. For example symbols used to hook up allocation
570    // are not considered for export
571    let codegen_fn_attrs = tcx.codegen_fn_attrs(sym_def_id);
572    let is_extern = codegen_fn_attrs.contains_extern_indicator();
573    let std_internal =
574        codegen_fn_attrs.flags.contains(CodegenFnAttrFlags::RUSTC_STD_INTERNAL_SYMBOL);
575
576    if is_extern && !std_internal {
577        let target = &tcx.sess.target.llvm_target;
578        // WebAssembly cannot export data symbols, so reduce their export level
579        if target.contains("emscripten") {
580            if let DefKind::Static { .. } = tcx.def_kind(sym_def_id) {
581                return SymbolExportLevel::Rust;
582            }
583        }
584
585        SymbolExportLevel::C
586    } else {
587        SymbolExportLevel::Rust
588    }
589}

591/// This is the symbol name of the given instance instantiated in a specific crate.
592pub(crate) fn symbol_name_for_instance_in_crate<'tcx>(
593    tcx: TyCtxt<'tcx>,
594    symbol: ExportedSymbol<'tcx>,
595    instantiating_crate: CrateNum,
596) -> String {
597    // If this is something instantiated in the local crate then we might
598    // already have cached the name as a query result.
599    if instantiating_crate == LOCAL_CRATE {
600        return symbol.symbol_name_for_local_instance(tcx).to_string();
601    }
602
603    // This is something instantiated in an upstream crate, so we have to use
604    // the slower (because uncached) version of computing the symbol name.
605    match symbol {
606        ExportedSymbol::NonGeneric(def_id) => {
607            rustc_symbol_mangling::symbol_name_for_instance_in_crate(
608                tcx,
609                Instance::mono(tcx, def_id),
610                instantiating_crate,
611            )
612        }
613        ExportedSymbol::Generic(def_id, args) => {
614            rustc_symbol_mangling::symbol_name_for_instance_in_crate(
615                tcx,
616                Instance::new_raw(def_id, args),
617                instantiating_crate,
618            )
619        }
620        ExportedSymbol::ThreadLocalShim(def_id) => {
621            rustc_symbol_mangling::symbol_name_for_instance_in_crate(
622                tcx,
623                ty::Instance {
624                    def: ty::InstanceKind::ThreadLocalShim(def_id),
625                    args: ty::GenericArgs::empty(),
626                },
627                instantiating_crate,
628            )
629        }
630        ExportedSymbol::DropGlue(ty) => rustc_symbol_mangling::symbol_name_for_instance_in_crate(
631            tcx,
632            Instance::resolve_drop_in_place(tcx, ty),
633            instantiating_crate,
634        ),
635        ExportedSymbol::AsyncDropGlueCtorShim(ty) => {
636            rustc_symbol_mangling::symbol_name_for_instance_in_crate(
637                tcx,
638                Instance::resolve_async_drop_in_place(tcx, ty),
639                instantiating_crate,
640            )
641        }
642        ExportedSymbol::AsyncDropGlue(def_id, ty) => {
643            rustc_symbol_mangling::symbol_name_for_instance_in_crate(
644                tcx,
645                Instance::resolve_async_drop_in_place_poll(tcx, def_id, ty),
646                instantiating_crate,
647            )
648        }
649        ExportedSymbol::NoDefId(symbol_name) => symbol_name.to_string(),
650    }
651}
652
653fn calling_convention_for_symbol<'tcx>(
654    tcx: TyCtxt<'tcx>,
655    symbol: ExportedSymbol<'tcx>,
656) -> (CanonAbi, &'tcx [rustc_target::callconv::ArgAbi<'tcx, Ty<'tcx>>]) {
657    let instance = match symbol {
658        ExportedSymbol::NonGeneric(def_id) | ExportedSymbol::Generic(def_id, _)
659            if tcx.is_static(def_id) =>
660        {
661            None
662        }
663        ExportedSymbol::NonGeneric(def_id) => Some(Instance::mono(tcx, def_id)),
664        ExportedSymbol::Generic(def_id, args) => Some(Instance::new_raw(def_id, args)),
665        // DropGlue always use the Rust calling convention and thus follow the target's default
666        // symbol decoration scheme.
667        ExportedSymbol::DropGlue(..) => None,
668        // AsyncDropGlueCtorShim always use the Rust calling convention and thus follow the
669        // target's default symbol decoration scheme.
670        ExportedSymbol::AsyncDropGlueCtorShim(..) => None,
671        ExportedSymbol::AsyncDropGlue(..) => None,
672        // NoDefId always follow the target's default symbol decoration scheme.
673        ExportedSymbol::NoDefId(..) => None,
674        // ThreadLocalShim always follow the target's default symbol decoration scheme.
675        ExportedSymbol::ThreadLocalShim(..) => None,
676    };
677
678    instance
679        .map(|i| {
680            tcx.fn_abi_of_instance(
681                ty::TypingEnv::fully_monomorphized().as_query_input((i, ty::List::empty())),
682            )
683            .unwrap_or_else(|_| bug!("fn_abi_of_instance({i:?}) failed"))
684        })
685        .map(|fnabi| (fnabi.conv, &fnabi.args[..]))
686        // FIXME(workingjubilee): why don't we know the convention here?
687        .unwrap_or((CanonAbi::Rust, &[]))
688}

690/// This is the symbol name of the given instance as seen by the linker.
691///
692/// On 32-bit Windows symbols are decorated according to their calling conventions.
693pub(crate) fn linking_symbol_name_for_instance_in_crate<'tcx>(
694    tcx: TyCtxt<'tcx>,
695    symbol: ExportedSymbol<'tcx>,
696    export_kind: SymbolExportKind,
697    instantiating_crate: CrateNum,
698) -> String {
699    let mut undecorated = symbol_name_for_instance_in_crate(tcx, symbol, instantiating_crate);
700
701    // thread local will not be a function call,
702    // so it is safe to return before windows symbol decoration check.
703    if let Some(name) = maybe_emutls_symbol_name(tcx, symbol, &undecorated) {
704        return name;
705    }
706
707    let target = &tcx.sess.target;
708    if !target.is_like_windows {
709        // Mach-O has a global "_" suffix and `object` crate will handle it.
710        // ELF does not have any symbol decorations.
711        return undecorated;
712    }
713
714    let prefix = match &target.arch[..] {
715        "x86" => Some('_'),
716        "x86_64" => None,
717        // Only functions are decorated for arm64ec.
718        "arm64ec" if export_kind == SymbolExportKind::Text => Some('#'),
719        // Only x86/64 and arm64ec use symbol decorations.
720        _ => return undecorated,
721    };
722
723    let (callconv, args) = calling_convention_for_symbol(tcx, symbol);
724
725    // Decorate symbols with prefixes, suffixes and total number of bytes of arguments.
726    // Reference: https://docs.microsoft.com/en-us/cpp/build/reference/decorated-names?view=msvc-170
727    let (prefix, suffix) = match callconv {
728        CanonAbi::X86(X86Call::Fastcall) => ("@", "@"),
729        CanonAbi::X86(X86Call::Stdcall) => ("_", "@"),
730        CanonAbi::X86(X86Call::Vectorcall) => ("", "@@"),
731        _ => {
732            if let Some(prefix) = prefix {
733                undecorated.insert(0, prefix);
734            }
735            return undecorated;
736        }
737    };
738
739    let args_in_bytes: u64 = args
740        .iter()
741        .map(|abi| abi.layout.size.bytes().next_multiple_of(target.pointer_width as u64 / 8))
742        .sum();
743    format!("{prefix}{undecorated}{suffix}{args_in_bytes}")
744}
745
746pub(crate) fn exporting_symbol_name_for_instance_in_crate<'tcx>(
747    tcx: TyCtxt<'tcx>,
748    symbol: ExportedSymbol<'tcx>,
749    cnum: CrateNum,
750) -> String {
751    let undecorated = symbol_name_for_instance_in_crate(tcx, symbol, cnum);
752    maybe_emutls_symbol_name(tcx, symbol, &undecorated).unwrap_or(undecorated)
753}

755/// On amdhsa, `gpu-kernel` functions have an associated metadata object with a `.kd` suffix.
756/// Add it to the symbols list for all kernel functions, so that it is exported in the linked
757/// object.
758pub(crate) fn extend_exported_symbols<'tcx>(
759    symbols: &mut Vec<(String, SymbolExportKind)>,
760    tcx: TyCtxt<'tcx>,
761    symbol: ExportedSymbol<'tcx>,
762    instantiating_crate: CrateNum,
763) {
764    let (callconv, _) = calling_convention_for_symbol(tcx, symbol);
765
766    if callconv != CanonAbi::GpuKernel || tcx.sess.target.os != "amdhsa" {
767        return;
768    }
769
770    let undecorated = symbol_name_for_instance_in_crate(tcx, symbol, instantiating_crate);
771
772    // Add the symbol for the kernel descriptor (with .kd suffix)
773    // Per https://llvm.org/docs/AMDGPUUsage.html#symbols these will always be `STT_OBJECT` so
774    // export as data.
775    symbols.push((format!("{undecorated}.kd"), SymbolExportKind::Data));
776}
777
778fn maybe_emutls_symbol_name<'tcx>(
779    tcx: TyCtxt<'tcx>,
780    symbol: ExportedSymbol<'tcx>,
781    undecorated: &str,
782) -> Option<String> {
783    if matches!(tcx.sess.tls_model(), TlsModel::Emulated)
784        && let ExportedSymbol::NonGeneric(def_id) = symbol
785        && tcx.is_thread_local_static(def_id)
786    {
787        // When using emutls, LLVM will add the `__emutls_v.` prefix to thread local symbols,
788        // and exported symbol name need to match this.
789        Some(format!("__emutls_v.{undecorated}"))
790    } else {
791        None
792    }
793}

795fn wasm_import_module_map(tcx: TyCtxt<'_>, cnum: CrateNum) -> DefIdMap<String> {
796    // Build up a map from DefId to a `NativeLib` structure, where
797    // `NativeLib` internally contains information about
798    // `#[link(wasm_import_module = "...")]` for example.
799    let native_libs = tcx.native_libraries(cnum);
800
801    let def_id_to_native_lib = native_libs
802        .iter()
803        .filter_map(|lib| lib.foreign_module.map(|id| (id, lib)))
804        .collect::<DefIdMap<_>>();
805
806    let mut ret = DefIdMap::default();
807    for (def_id, lib) in tcx.foreign_modules(cnum).iter() {
808        let module = def_id_to_native_lib.get(def_id).and_then(|s| s.wasm_import_module());
809        let Some(module) = module else { continue };
810        ret.extend(lib.foreign_items.iter().map(|id| {
811            assert_eq!(id.krate, cnum);
812            (*id, module.to_string())
813        }));
814    }
815
816    ret
817}
