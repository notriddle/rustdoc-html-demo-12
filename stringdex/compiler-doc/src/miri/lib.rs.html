164/// Insert rustc arguments at the beginning of the argument list that Miri wants to be
165/// set per default, for maximal validation power.
166/// Also disable the MIR pass that inserts an alignment check on every pointer dereference. Miri
167/// does that too, and with a better error message.
168pub const MIRI_DEFAULT_ARGS: &[&str] = &[
169    "--cfg=miri",
170    "-Zalways-encode-mir",
171    "-Zextra-const-ub-checks",
172    "-Zmir-emit-retag",
173    "-Zmir-preserve-ub",
174    "-Zmir-opt-level=0",
175    "-Zmir-enable-passes=-CheckAlignment,-CheckNull,-CheckEnums",
176    // Deduplicating diagnostics means we miss events when tracking what happens during an
177    // execution. Let's not do that.
178    "-Zdeduplicate-diagnostics=no",
179];
