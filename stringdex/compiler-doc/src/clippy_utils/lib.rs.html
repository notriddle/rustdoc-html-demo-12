149/// If the given expression is a local binding, find the initializer expression.
150/// If that initializer expression is another local binding, find its initializer again.
151///
152/// This process repeats as long as possible (but usually no more than once). Initializer
153/// expressions with adjustments are ignored. If this is not desired, use [`find_binding_init`]
154/// instead.
155///
156/// Examples:
157/// ```no_run
158/// let abc = 1;
159/// //        ^ output
160/// let def = abc;
161/// dbg!(def);
162/// //   ^^^ input
163///
164/// // or...
165/// let abc = 1;
166/// let def = abc + 2;
167/// //        ^^^^^^^ output
168/// dbg!(def);
169/// //   ^^^ input
170/// ```
171pub fn expr_or_init<'a, 'b, 'tcx: 'b>(cx: &LateContext<'tcx>, mut expr: &'a Expr<'b>) -> &'a Expr<'b> {
172    while let Some(init) = path_to_local(expr)
173        .and_then(|id| find_binding_init(cx, id))
174        .filter(|init| cx.typeck_results().expr_adjustments(init).is_empty())
175    {
176        expr = init;
177    }
178    expr
179}

181/// Finds the initializer expression for a local binding. Returns `None` if the binding is mutable.
182///
183/// By only considering immutable bindings, we guarantee that the returned expression represents the
184/// value of the binding wherever it is referenced.
185///
186/// Example: For `let x = 1`, if the `HirId` of `x` is provided, the `Expr` `1` is returned.
187/// Note: If you have an expression that references a binding `x`, use `path_to_local` to get the
188/// canonical binding `HirId`.
189pub fn find_binding_init<'tcx>(cx: &LateContext<'tcx>, hir_id: HirId) -> Option<&'tcx Expr<'tcx>> {
190    if let Node::Pat(pat) = cx.tcx.hir_node(hir_id)
191        && matches!(pat.kind, PatKind::Binding(BindingMode::NONE, ..))
192        && let Node::LetStmt(local) = cx.tcx.parent_hir_node(hir_id)
193    {
194        return local.init;
195    }
196    None
197}

200///
201/// e.g. returns true for `x` in `fn f(x: usize) { .. }` and `let x = 1;` but false for `let x;`
202pub fn local_is_initialized(cx: &LateContext<'_>, local: HirId) -> bool {
203    for (_, node) in cx.tcx.hir_parent_iter(local) {
204        match node {
205            Node::Pat(..) | Node::PatField(..) => {},
206            Node::LetStmt(let_stmt) => return let_stmt.init.is_some(),
207            _ => return true,
208        }
209    }

214/// Checks if we are currently in a const context (e.g. `const fn`, `static`/`const` initializer).
215///
216/// The current context is determined based on the current body which is set before calling a lint's
217/// entry point (any function on `LateLintPass`). If you need to check in a different context use
218/// `tcx.hir_is_inside_const_context(_)`.
219///
220/// Do not call this unless the `LateContext` has an enclosing body. For release build this case
221/// will safely return `false`, but debug builds will ICE. Note that `check_expr`, `check_block`,
222/// `check_pat` and a few other entry points will always have an enclosing body. Some entry points
223/// like `check_path` or `check_ty` may or may not have one.
224pub fn is_in_const_context(cx: &LateContext<'_>) -> bool {
225    debug_assert!(cx.enclosing_body.is_some(), "`LateContext` has no enclosing body");
226    cx.enclosing_body.is_some_and(|id| {
227        cx.tcx
228            .hir_body_const_context(cx.tcx.hir_body_owner_def_id(id))
229            .is_some()
230    })
231}

537/// This method will return tuple of projection stack and root of the expression,
538/// used in `can_mut_borrow_both`.
539///
540/// For example, if `e` represents the `v[0].a.b[x]`
541/// this method will return a tuple, composed of a `Vec`
542/// containing the `Expr`s for `v[0], v[0].a, v[0].a.b, v[0].a.b[x]`
543/// and an `Expr` for root of them, `v`
544fn projection_stack<'a, 'hir>(mut e: &'a Expr<'hir>) -> (Vec<&'a Expr<'hir>>, &'a Expr<'hir>) {
545    let mut result = vec![];
546    let root = loop {
547        match e.kind {
548            ExprKind::Index(ep, _, _) | ExprKind::Field(ep, _) => {
549                result.push(e);
550                e = ep;
551            },
552            _ => break e,
553        }
554    };
555    result.reverse();
556    (result, root)
557}

573/// and they aren't dependent on borrowing same thing twice
574pub fn can_mut_borrow_both(cx: &LateContext<'_>, e1: &Expr<'_>, e2: &Expr<'_>) -> bool {
575    let (s1, r1) = projection_stack(e1);
576    let (s2, r2) = projection_stack(e2);
577    if !eq_expr_value(cx, r1, r2) {
578        return true;
579    }
580    if expr_custom_deref_adjustment(cx, r1).is_some() || expr_custom_deref_adjustment(cx, r2).is_some() {
581        return false;
582    }
583
584    for (x1, x2) in zip(&s1, &s2) {
585        if expr_custom_deref_adjustment(cx, x1).is_some() || expr_custom_deref_adjustment(cx, x2).is_some() {
586            return false;
587        }
588
589        match (&x1.kind, &x2.kind) {
590            (ExprKind::Field(_, i1), ExprKind::Field(_, i2)) => {
591                if i1 != i2 {
592                    return true;
593                }
594            },
595            (ExprKind::Index(_, i1, _), ExprKind::Index(_, i2, _)) => {
596                if !eq_expr_value(cx, i1, i2) {
597                    return false;
598                }
599            },
600            _ => return false,
601        }
602    }
603    false
604}

607/// constructor from the std library

611        sym::VecDeque,

616        sym::BTreeSet,

632/// Returns true if the expr is equal to `Default::default` when evaluated.
633pub fn is_default_equivalent_call(
634    cx: &LateContext<'_>,
635    repl_func: &Expr<'_>,
636    whole_call_expr: Option<&Expr<'_>>,
637) -> bool {
638    if let ExprKind::Path(ref repl_func_qpath) = repl_func.kind
639        && let Some(repl_def_id) = cx.qpath_res(repl_func_qpath, repl_func.hir_id).opt_def_id()
640        && (is_diag_trait_item(cx, repl_def_id, sym::Default)
641            || is_default_equivalent_ctor(cx, repl_def_id, repl_func_qpath))
642    {
643        return true;
644    }
645
646    // Get the type of the whole method call expression, find the exact method definition, look at
647    // its body and check if it is similar to the corresponding `Default::default()` body.
648    let Some(e) = whole_call_expr else { return false };
649    let Some(default_fn_def_id) = cx.tcx.get_diagnostic_item(sym::default_fn) else {
650        return false;
651    };
652    let Some(ty) = cx.tcx.typeck(e.hir_id.owner.def_id).expr_ty_adjusted_opt(e) else {
653        return false;
654    };
655    let args = rustc_ty::GenericArgs::for_item(cx.tcx, default_fn_def_id, |param, _| {
656        if let rustc_ty::GenericParamDefKind::Lifetime = param.kind {
657            cx.tcx.lifetimes.re_erased.into()
658        } else if param.index == 0 && param.name == kw::SelfUpper {
659            ty.into()
660        } else {
661            param.to_error(cx.tcx)
662        }
663    });
664    let instance = rustc_ty::Instance::try_resolve(cx.tcx, cx.typing_env(), default_fn_def_id, args);
665
666    let Ok(Some(instance)) = instance else { return false };
667    if let rustc_ty::InstanceKind::Item(def) = instance.def
668        && !cx.tcx.is_mir_available(def)
669    {
670        return false;
671    }
672    let ExprKind::Path(ref repl_func_qpath) = repl_func.kind else {
673        return false;
674    };
675    let Some(repl_def_id) = cx.qpath_res(repl_func_qpath, repl_func.hir_id).opt_def_id() else {
676        return false;
677    };
678
679    // Get the MIR Body for the `<Ty as Default>::default()` function.
680    // If it is a value or call (either fn or ctor), we compare its `DefId` against the one for the
681    // resolution of the expression we had in the path. This lets us identify, for example, that
682    // the body of `<Vec<T> as Default>::default()` is a `Vec::new()`, and the field was being
683    // initialized to `Vec::new()` as well.
684    let body = cx.tcx.instance_mir(instance.def);
685    for block_data in body.basic_blocks.iter() {
686        if block_data.statements.len() == 1
687            && let StatementKind::Assign(assign) = &block_data.statements[0].kind
688            && assign.0.local == RETURN_PLACE
689            && let Rvalue::Aggregate(kind, _places) = &assign.1
690            && let AggregateKind::Adt(did, variant_index, _, _, _) = &**kind
691            && let def = cx.tcx.adt_def(did)
692            && let variant = &def.variant(*variant_index)
693            && variant.fields.is_empty()
694            && let Some((_, did)) = variant.ctor
695            && did == repl_def_id
696        {
697            return true;
698        } else if block_data.statements.is_empty()
699            && let Some(term) = &block_data.terminator
700        {
701            match &term.kind {
702                TerminatorKind::Call {
703                    func: Operand::Constant(c),
704                    ..
705                } if let rustc_ty::FnDef(did, _args) = c.ty().kind()
706                    && *did == repl_def_id =>
707                {
708                    return true;
709                },
710                TerminatorKind::TailCall {
711                    func: Operand::Constant(c),
712                    ..
713                } if let rustc_ty::FnDef(did, _args) = c.ty().kind()
714                    && *did == repl_def_id =>
715                {
716                    return true;
717                },
718                _ => {},
719            }
720        }
721    }
722    false
723}

725/// Returns true if the expr is equal to `Default::default()` of its type when evaluated.
726///
727/// It doesn't cover all cases, like struct literals, but it is a close approximation.
728pub fn is_default_equivalent(cx: &LateContext<'_>, e: &Expr<'_>) -> bool {
729    match &e.kind {
730        ExprKind::Lit(lit) => match lit.node {
731            LitKind::Bool(false) | LitKind::Int(Pu128(0), _) => true,
732            LitKind::Str(s, _) => s.is_empty(),
733            _ => false,
734        },
735        ExprKind::Tup(items) | ExprKind::Array(items) => items.iter().all(|x| is_default_equivalent(cx, x)),
736        ExprKind::Repeat(x, len) => {
737            if let ConstArgKind::Anon(anon_const) = len.kind
738                && let ExprKind::Lit(const_lit) = cx.tcx.hir_body(anon_const.body).value.kind
739                && let LitKind::Int(v, _) = const_lit.node
740                && v <= 32
741                && is_default_equivalent(cx, x)
742            {
743                true
744            } else {
745                false
746            }
747        },
748        ExprKind::Call(repl_func, []) => is_default_equivalent_call(cx, repl_func, Some(e)),
749        ExprKind::Call(from_func, [arg]) => is_default_equivalent_from(cx, from_func, arg),
750        ExprKind::Path(qpath) => is_res_lang_ctor(cx, cx.qpath_res(qpath, e.hir_id), OptionNone),
751        ExprKind::AddrOf(rustc_hir::BorrowKind::Ref, _, expr) => matches!(expr.kind, ExprKind::Array([])),
752        ExprKind::Block(Block { stmts: [], expr, .. }, _) => expr.is_some_and(|e| is_default_equivalent(cx, e)),
753        _ => false,
754    }
755}

775            _ => (),

781/// Checks if the top level expression can be moved into a closure as is.
782/// Currently checks for:
783/// * Break/Continue outside the given loop HIR ids.
784/// * Yield/Return statements.
785/// * Inline assembly.
786/// * Usages of a field of a local where the type of the local can be partially moved.
787///
788/// For example, given the following function:
789///
790/// ```no_run
791/// fn f<'a>(iter: &mut impl Iterator<Item = (usize, &'a mut String)>) {
792///     for item in iter {
793///         let s = item.1;
794///         if item.0 > 10 {
795///             continue;
796///         } else {
797///             s.clear();
798///         }
799///     }
800/// }
801/// ```
802///
803/// When called on the expression `item.0` this will return false unless the local `item` is in the
804/// `ignore_locals` set. The type `(usize, &mut String)` can have the second element moved, so it
805/// isn't always safe to move into a closure when only a single field is needed.
806///
807/// When called on the `continue` expression this will return false unless the outer loop expression
808/// is in the `loop_ids` set.
809///
810/// Note that this check is not recursive, so passing the `if` expression will always return true
811/// even though sub-expressions might return false.
812pub fn can_move_expr_to_closure_no_visit<'tcx>(
813    cx: &LateContext<'tcx>,
814    expr: &'tcx Expr<'_>,
815    loop_ids: &[HirId],
816    ignore_locals: &HirIdSet,
817) -> bool {
818    match expr.kind {
819        ExprKind::Break(Destination { target_id: Ok(id), .. }, _)
820        | ExprKind::Continue(Destination { target_id: Ok(id), .. })
821            if loop_ids.contains(&id) =>
822        {
823            true
824        },
825        ExprKind::Break(..)
826        | ExprKind::Continue(_)
827        | ExprKind::Ret(_)
828        | ExprKind::Yield(..)
829        | ExprKind::InlineAsm(_) => false,
830        // Accessing a field of a local value can only be done if the type isn't
831        // partially moved.
832        ExprKind::Field(
833            &Expr {
834                hir_id,
835                kind:
836                    ExprKind::Path(QPath::Resolved(
837                        _,
838                        Path {
839                            res: Res::Local(local_id),
840                            ..
841                        },
842                    )),
843                ..
844            },
845            _,
846        ) if !ignore_locals.contains(local_id) && can_partially_move_ty(cx, cx.typeck_results().node_type(hir_id)) => {
847            // TODO: check if the local has been partially moved. Assume it has for now.
848            false
849        },
850        _ => true,
851    }
852}

854/// How a local is captured by a closure
855#[derive(Debug, Clone, Copy, PartialEq, Eq)]
856pub enum CaptureKind {
857    Value,
858    Use,
859    Ref(Mutability),
860}
861impl CaptureKind {
862    pub fn is_imm_ref(self) -> bool {
863        self == Self::Ref(Mutability::Not)
864    }
865}
866impl std::ops::BitOr for CaptureKind {
867    type Output = Self;
868    fn bitor(self, rhs: Self) -> Self::Output {
869        match (self, rhs) {
870            (CaptureKind::Value, _) | (_, CaptureKind::Value) => CaptureKind::Value,
871            (CaptureKind::Use, _) | (_, CaptureKind::Use) => CaptureKind::Use,
872            (CaptureKind::Ref(Mutability::Mut), CaptureKind::Ref(_))
873            | (CaptureKind::Ref(_), CaptureKind::Ref(Mutability::Mut)) => CaptureKind::Ref(Mutability::Mut),
874            (CaptureKind::Ref(Mutability::Not), CaptureKind::Ref(Mutability::Not)) => CaptureKind::Ref(Mutability::Not),
875        }
876    }
877}
878impl std::ops::BitOrAssign for CaptureKind {
879    fn bitor_assign(&mut self, rhs: Self) {
880        *self = *self | rhs;
881    }

884/// Given an expression referencing a local, determines how it would be captured in a closure.
885///
886/// Note as this will walk up to parent expressions until the capture can be determined it should
887/// only be used while making a closure somewhere a value is consumed. e.g. a block, match arm, or
888/// function argument (other than a receiver).
889pub fn capture_local_usage(cx: &LateContext<'_>, e: &Expr<'_>) -> CaptureKind {
890    fn pat_capture_kind(cx: &LateContext<'_>, pat: &Pat<'_>) -> CaptureKind {
891        let mut capture = CaptureKind::Ref(Mutability::Not);
892        pat.each_binding_or_first(&mut |_, id, span, _| match cx
893            .typeck_results()
894            .extract_binding_mode(cx.sess(), id, span)
895            .0
896        {
897            ByRef::No if !is_copy(cx, cx.typeck_results().node_type(id)) => {
898                capture = CaptureKind::Value;
899            },
900            ByRef::Yes(Mutability::Mut) if capture != CaptureKind::Value => {
901                capture = CaptureKind::Ref(Mutability::Mut);
902            },
903            _ => (),
904        });
905        capture
906    }
907
908    debug_assert!(matches!(
909        e.kind,
910        ExprKind::Path(QPath::Resolved(None, Path { res: Res::Local(_), .. }))
911    ));
912
913    let mut child_id = e.hir_id;
914    let mut capture = CaptureKind::Value;
915    let mut capture_expr_ty = e;
916
917    for (parent_id, parent) in cx.tcx.hir_parent_iter(e.hir_id) {
918        if let [
919            Adjustment {
920                kind: Adjust::Deref(_) | Adjust::Borrow(AutoBorrow::Ref(..)),
921                target,
922            },
923            ref adjust @ ..,
924        ] = *cx
925            .typeck_results()
926            .adjustments()
927            .get(child_id)
928            .map_or(&[][..], |x| &**x)
929            && let rustc_ty::RawPtr(_, mutability) | rustc_ty::Ref(_, _, mutability) =
930                *adjust.last().map_or(target, |a| a.target).kind()
931        {
932            return CaptureKind::Ref(mutability);
933        }
934
935        match parent {
936            Node::Expr(e) => match e.kind {
937                ExprKind::AddrOf(_, mutability, _) => return CaptureKind::Ref(mutability),
938                ExprKind::Index(..) | ExprKind::Unary(UnOp::Deref, _) => capture = CaptureKind::Ref(Mutability::Not),
939                ExprKind::Assign(lhs, ..) | ExprKind::AssignOp(_, lhs, _) if lhs.hir_id == child_id => {
940                    return CaptureKind::Ref(Mutability::Mut);
941                },
942                ExprKind::Field(..) => {
943                    if capture == CaptureKind::Value {
944                        capture_expr_ty = e;
945                    }
946                },
947                ExprKind::Let(let_expr) => {
948                    let mutability = match pat_capture_kind(cx, let_expr.pat) {
949                        CaptureKind::Value | CaptureKind::Use => Mutability::Not,
950                        CaptureKind::Ref(m) => m,
951                    };
952                    return CaptureKind::Ref(mutability);
953                },
954                ExprKind::Match(_, arms, _) => {
955                    let mut mutability = Mutability::Not;
956                    for capture in arms.iter().map(|arm| pat_capture_kind(cx, arm.pat)) {
957                        match capture {
958                            CaptureKind::Value | CaptureKind::Use => break,
959                            CaptureKind::Ref(Mutability::Mut) => mutability = Mutability::Mut,
960                            CaptureKind::Ref(Mutability::Not) => (),
961                        }
962                    }
963                    return CaptureKind::Ref(mutability);
964                },
965                _ => break,
966            },
967            Node::LetStmt(l) => match pat_capture_kind(cx, l.pat) {
968                CaptureKind::Value | CaptureKind::Use => break,
969                capture @ CaptureKind::Ref(_) => return capture,
970            },
971            _ => break,
972        }
973
974        child_id = parent_id;
975    }
976
977    if capture == CaptureKind::Value && is_copy(cx, cx.typeck_results().expr_ty(capture_expr_ty)) {
978        // Copy types are never automatically captured by value.
979        CaptureKind::Ref(Mutability::Not)
980    } else {
981        capture
982    }
983}

985/// Checks if the expression can be moved into a closure as is. This will return a list of captures
986/// if so, otherwise, `None`.
987pub fn can_move_expr_to_closure<'tcx>(cx: &LateContext<'tcx>, expr: &'tcx Expr<'_>) -> Option<HirIdMap<CaptureKind>> {
988    struct V<'cx, 'tcx> {
989        cx: &'cx LateContext<'tcx>,
990        // Stack of potential break targets contained in the expression.
991        loops: Vec<HirId>,
992        /// Local variables created in the expression. These don't need to be captured.
993        locals: HirIdSet,
994        /// Whether this expression can be turned into a closure.
995        allow_closure: bool,
996        /// Locals which need to be captured, and whether they need to be by value, reference, or
997        /// mutable reference.
998        captures: HirIdMap<CaptureKind>,
999    }
1000    impl<'tcx> Visitor<'tcx> for V<'_, 'tcx> {
1001        fn visit_expr(&mut self, e: &'tcx Expr<'_>) {
1002            if !self.allow_closure {
1003                return;
1004            }
1005
1006            match e.kind {
1007                ExprKind::Path(QPath::Resolved(None, &Path { res: Res::Local(l), .. })) => {
1008                    if !self.locals.contains(&l) {
1009                        let cap = capture_local_usage(self.cx, e);
1010                        self.captures.entry(l).and_modify(|e| *e |= cap).or_insert(cap);
1011                    }
1012                },
1013                ExprKind::Closure(closure) => {
1014                    for capture in self.cx.typeck_results().closure_min_captures_flattened(closure.def_id) {
1015                        let local_id = match capture.place.base {
1016                            PlaceBase::Local(id) => id,
1017                            PlaceBase::Upvar(var) => var.var_path.hir_id,
1018                            _ => continue,
1019                        };
1020                        if !self.locals.contains(&local_id) {
1021                            let capture = match capture.info.capture_kind {
1022                                UpvarCapture::ByValue => CaptureKind::Value,
1023                                UpvarCapture::ByUse => CaptureKind::Use,
1024                                UpvarCapture::ByRef(kind) => match kind {
1025                                    BorrowKind::Immutable => CaptureKind::Ref(Mutability::Not),
1026                                    BorrowKind::UniqueImmutable | BorrowKind::Mutable => {
1027                                        CaptureKind::Ref(Mutability::Mut)
1028                                    },
1029                                },
1030                            };
1031                            self.captures
1032                                .entry(local_id)
1033                                .and_modify(|e| *e |= capture)
1034                                .or_insert(capture);
1035                        }
1036                    }
1037                },
1038                ExprKind::Loop(b, ..) => {
1039                    self.loops.push(e.hir_id);
1040                    self.visit_block(b);
1041                    self.loops.pop();
1042                },
1043                _ => {
1044                    self.allow_closure &= can_move_expr_to_closure_no_visit(self.cx, e, &self.loops, &self.locals);
1045                    walk_expr(self, e);
1046                },
1047            }
1048        }
1049
1050        fn visit_pat(&mut self, p: &'tcx Pat<'tcx>) {
1051            p.each_binding_or_first(&mut |_, id, _, _| {
1052                self.locals.insert(id);
1053            });
1054        }
1055    }
1056
1057    let mut v = V {
1058        cx,
1059        loops: Vec::new(),
1060        locals: HirIdSet::default(),
1061        allow_closure: true,
1062        captures: HirIdMap::default(),
1063    };
1064    v.visit_expr(expr);
1065    v.allow_closure.then_some(v.captures)
1066}

1071/// Returns the method names and argument list of nested method call expressions that make up
1072/// `expr`. method/span lists are sorted with the most recent call first.
1073pub fn method_calls<'tcx>(expr: &'tcx Expr<'tcx>, max_depth: usize) -> (Vec<Symbol>, MethodArguments<'tcx>, Vec<Span>) {
1074    let mut method_names = Vec::with_capacity(max_depth);
1075    let mut arg_lists = Vec::with_capacity(max_depth);
1076    let mut spans = Vec::with_capacity(max_depth);
1077
1078    let mut current = expr;
1079    for _ in 0..max_depth {
1080        if let ExprKind::MethodCall(path, receiver, args, _) = &current.kind {
1081            if receiver.span.from_expansion() || args.iter().any(|e| e.span.from_expansion()) {
1082                break;
1083            }
1084            method_names.push(path.ident.name);
1085            arg_lists.push((*receiver, &**args));
1086            spans.push(path.ident.span);
1087            current = receiver;
1088        } else {
1089            break;
1090        }
1091    }
1092
1093    (method_names, arg_lists, spans)
1094}

1096/// Matches an `Expr` against a chain of methods, and return the matched `Expr`s.
1097///
1098/// For example, if `expr` represents the `.baz()` in `foo.bar().baz()`,
1099/// `method_chain_args(expr, &["bar", "baz"])` will return a `Vec`
1100/// containing the `Expr`s for
1101/// `.bar()` and `.baz()`
1102pub fn method_chain_args<'a>(expr: &'a Expr<'_>, methods: &[Symbol]) -> Option<Vec<(&'a Expr<'a>, &'a [Expr<'a>])>> {
1103    let mut current = expr;
1104    let mut matched = Vec::with_capacity(methods.len());
1105    for method_name in methods.iter().rev() {
1106        // method chains are stored last -> first
1107        if let ExprKind::MethodCall(path, receiver, args, _) = current.kind {
1108            if path.ident.name == *method_name {
1109                if receiver.span.from_expansion() || args.iter().any(|e| e.span.from_expansion()) {
1110                    return None;
1111                }
1112                matched.push((receiver, args)); // build up `matched` backwards
1113                current = receiver; // go to parent expression
1114            } else {
1115                return None;
1116            }
1117        } else {
1118            return None;
1119        }
1120    }
1121    // Reverse `matched` so that it is in the same order as `methods`.
1122    matched.reverse();
1123    Some(matched)
1124}

1268/// and no statements. Unsafe blocks are not removed.
1269///
1270/// Examples:
1271///  * `{}`               -> `{}`
1272///  * `{ x }`            -> `x`
1273///  * `{{ x }}`          -> `x`
1274///  * `{ x; }`           -> `{ x; }`
1275///  * `{ x; y }`         -> `{ x; y }`
1276///  * `{ unsafe { x } }` -> `unsafe { x }`

1279        Block {
1280            stmts: [],
1281            expr: Some(inner),
1282            rules: BlockCheckMode::DefaultBlock,
1283            ..
1284        },
1285        _,
1286    ) = expr.kind
1287    {
1288        expr = inner;
1289    }
1290    expr
1291}

1383/// Checks whether the given `Expr` is a range equivalent to a `RangeFull`.
1384///
1385/// For the lower bound, this means that:
1386/// - either there is none
1387/// - or it is the smallest value that can be represented by the range's integer type
1388///
1389/// For the upper bound, this means that:
1390/// - either there is none
1391/// - or it is the largest value that can be represented by the range's integer type and is
1392///   inclusive
1393/// - or it is a call to some container's `len` method and is exclusive, and the range is passed to
1394///   a method call on that same container (e.g. `v.drain(..v.len())`)
1395///
1396/// If the given `Expr` is not some kind of range, the function returns `false`.
1397pub fn is_range_full(cx: &LateContext<'_>, expr: &Expr<'_>, container_path: Option<&Path<'_>>) -> bool {
1398    let ty = cx.typeck_results().expr_ty(expr);
1399    if let Some(Range { start, end, limits }) = Range::hir(expr) {
1400        let start_is_none_or_min = start.is_none_or(|start| {
1401            if let rustc_ty::Adt(_, subst) = ty.kind()
1402                && let bnd_ty = subst.type_at(0)
1403                && let Some(min_const) = bnd_ty.numeric_min_val(cx.tcx)
1404                && let Some(min_const) = mir_to_const(cx.tcx, min_const)
1405                && let Some(start_const) = ConstEvalCtxt::new(cx).eval(start)
1406            {
1407                start_const == min_const
1408            } else {
1409                false
1410            }
1411        });
1412        let end_is_none_or_max = end.is_none_or(|end| match limits {
1413            RangeLimits::Closed => {
1414                if let rustc_ty::Adt(_, subst) = ty.kind()
1415                    && let bnd_ty = subst.type_at(0)
1416                    && let Some(max_const) = bnd_ty.numeric_max_val(cx.tcx)
1417                    && let Some(max_const) = mir_to_const(cx.tcx, max_const)
1418                    && let Some(end_const) = ConstEvalCtxt::new(cx).eval(end)
1419                {
1420                    end_const == max_const
1421                } else {
1422                    false
1423                }
1424            },
1425            RangeLimits::HalfOpen => {
1426                if let Some(container_path) = container_path
1427                    && let ExprKind::MethodCall(name, self_arg, [], _) = end.kind
1428                    && name.ident.name == sym::len
1429                    && let ExprKind::Path(QPath::Resolved(None, path)) = self_arg.kind
1430                {
1431                    container_path.res == path.res
1432                } else {
1433                    false
1434                }
1435            },
1436        });
1437        return start_is_none_or_min && end_is_none_or_max;
1438    }
1439    false
1440}

1566/// Returns `true` if a pattern is refutable.
1567// TODO: should be implemented using rustc/mir_build/thir machinery
1568pub fn is_refutable(cx: &LateContext<'_>, pat: &Pat<'_>) -> bool {
1569    fn is_qpath_refutable(cx: &LateContext<'_>, qpath: &QPath<'_>, id: HirId) -> bool {
1570        !matches!(
1571            cx.qpath_res(qpath, id),
1572            Res::Def(DefKind::Struct, ..) | Res::Def(DefKind::Ctor(def::CtorOf::Struct, _), _)
1573        )
1574    }
1575
1576    fn are_refutable<'a, I: IntoIterator<Item = &'a Pat<'a>>>(cx: &LateContext<'_>, i: I) -> bool {
1577        i.into_iter().any(|pat| is_refutable(cx, pat))
1578    }
1579
1580    match pat.kind {
1581        PatKind::Missing => unreachable!(),
1582        PatKind::Wild | PatKind::Never => false, // If `!` typechecked then the type is empty, so not refutable.
1583        PatKind::Binding(_, _, _, pat) => pat.is_some_and(|pat| is_refutable(cx, pat)),
1584        PatKind::Box(pat) | PatKind::Ref(pat, _) => is_refutable(cx, pat),
1585        PatKind::Expr(PatExpr {
1586            kind: PatExprKind::Path(qpath),
1587            hir_id,
1588            ..
1589        }) => is_qpath_refutable(cx, qpath, *hir_id),
1590        PatKind::Or(pats) => {
1591            // TODO: should be the honest check, that pats is exhaustive set
1592            are_refutable(cx, pats)
1593        },
1594        PatKind::Tuple(pats, _) => are_refutable(cx, pats),
1595        PatKind::Struct(ref qpath, fields, _) => {
1596            is_qpath_refutable(cx, qpath, pat.hir_id) || are_refutable(cx, fields.iter().map(|field| field.pat))
1597        },
1598        PatKind::TupleStruct(ref qpath, pats, _) => {
1599            is_qpath_refutable(cx, qpath, pat.hir_id) || are_refutable(cx, pats)
1600        },
1601        PatKind::Slice(head, middle, tail) => {
1602            match &cx.typeck_results().node_type(pat.hir_id).kind() {
1603                rustc_ty::Slice(..) => {
1604                    // [..] is the only irrefutable slice pattern.
1605                    !head.is_empty() || middle.is_none() || !tail.is_empty()
1606                },
1607                rustc_ty::Array(..) => are_refutable(cx, head.iter().chain(middle).chain(tail.iter())),
1608                _ => {
1609                    // unreachable!()
1610                    true
1611                },
1612            }
1613        },
1614        PatKind::Expr(..) | PatKind::Range(..) | PatKind::Err(_) | PatKind::Deref(_) | PatKind::Guard(..) => true,
1615    }
1616}

1649/// Checks if a given expression is a match expression expanded from the `?`
1650/// operator or the `try` macro.
1651pub fn is_try<'tcx>(cx: &LateContext<'_>, expr: &'tcx Expr<'tcx>) -> Option<&'tcx Expr<'tcx>> {
1652    fn is_ok(cx: &LateContext<'_>, arm: &Arm<'_>) -> bool {
1653        if let PatKind::TupleStruct(ref path, pat, ddpos) = arm.pat.kind
1654            && ddpos.as_opt_usize().is_none()
1655            && is_res_lang_ctor(cx, cx.qpath_res(path, arm.pat.hir_id), ResultOk)
1656            && let PatKind::Binding(_, hir_id, _, None) = pat[0].kind
1657            && path_to_local_id(arm.body, hir_id)
1658        {
1659            return true;
1660        }
1661        false
1662    }
1663
1664    fn is_err(cx: &LateContext<'_>, arm: &Arm<'_>) -> bool {
1665        if let PatKind::TupleStruct(ref path, _, _) = arm.pat.kind {
1666            is_res_lang_ctor(cx, cx.qpath_res(path, arm.pat.hir_id), ResultErr)
1667        } else {
1668            false
1669        }
1670    }
1671
1672    if let ExprKind::Match(_, arms, ref source) = expr.kind {
1673        // desugared from a `?` operator
1674        if let MatchSource::TryDesugar(_) = *source {
1675            return Some(expr);
1676        }
1677
1678        if arms.len() == 2
1679            && arms[0].guard.is_none()
1680            && arms[1].guard.is_none()
1681            && ((is_ok(cx, &arms[0]) && is_err(cx, &arms[1])) || (is_ok(cx, &arms[1]) && is_err(cx, &arms[0])))
1682        {
1683            return Some(expr);
1684        }
1685    }
1686
1687    None
1688}

1802/// Returns the list of condition expressions and the list of blocks in a
1803/// sequence of `if/else`.
1804/// E.g., this returns `([a, b], [c, d, e])` for the expression
1805/// `if a { c } else if b { d } else { e }`.
1806pub fn if_sequence<'tcx>(mut expr: &'tcx Expr<'tcx>) -> (Vec<&'tcx Expr<'tcx>>, Vec<&'tcx Block<'tcx>>) {
1807    let mut conds = Vec::new();
1808    let mut blocks: Vec<&Block<'_>> = Vec::new();
1809
1810    while let Some(higher::IfOrIfLet { cond, then, r#else }) = higher::IfOrIfLet::hir(expr) {
1811        conds.push(cond);
1812        if let ExprKind::Block(block, _) = then.kind {
1813            blocks.push(block);
1814        } else {
1815            panic!("ExprKind::If node is not an ExprKind::Block");
1816        }
1817
1818        if let Some(else_expr) = r#else {
1819            expr = else_expr;
1820        } else {
1821            break;
1822        }
1823    }
1824
1825    // final `else {..}`
1826    if !blocks.is_empty()
1827        && let ExprKind::Block(block, _) = expr.kind
1828    {
1829        blocks.push(block);
1830    }
1831
1832    (conds, blocks)
1833}

1893/// Checks if a function's body represents the identity function. Looks for bodies of the form:
1894/// * `|x| x`
1895/// * `|x| return x`
1896/// * `|x| { return x }`
1897/// * `|x| { return x; }`
1898/// * `|(x, y)| (x, y)`
1899/// * `|[x, y]| [x, y]`
1900/// * `|Foo(bar, baz)| Foo(bar, baz)`
1901/// * `|Foo { bar, baz }| Foo { bar, baz }`
1902///
1903/// Consider calling [`is_expr_untyped_identity_function`] or [`is_expr_identity_function`] instead.
1904fn is_body_identity_function(cx: &LateContext<'_>, func: &Body<'_>) -> bool {
1905    let [param] = func.params else {
1906        return false;
1907    };
1908
1909    let mut expr = func.value;
1910    loop {
1911        match expr.kind {
1912            ExprKind::Block(
1913                &Block {
1914                    stmts: [],
1915                    expr: Some(e),
1916                    ..
1917                },
1918                _,
1919            )
1920            | ExprKind::Ret(Some(e)) => expr = e,
1921            ExprKind::Block(
1922                &Block {
1923                    stmts: [stmt],
1924                    expr: None,
1925                    ..
1926                },
1927                _,
1928            ) => {
1929                if let StmtKind::Semi(e) | StmtKind::Expr(e) = stmt.kind
1930                    && let ExprKind::Ret(Some(ret_val)) = e.kind
1931                {
1932                    expr = ret_val;
1933                } else {
1934                    return false;
1935                }
1936            },
1937            _ => return is_expr_identity_of_pat(cx, param.pat, expr, true),
1938        }
1939    }
1940}

1942/// Checks if the given expression is an identity representation of the given pattern:
1943/// * `x` is the identity representation of `x`
1944/// * `(x, y)` is the identity representation of `(x, y)`
1945/// * `[x, y]` is the identity representation of `[x, y]`
1946/// * `Foo(bar, baz)` is the identity representation of `Foo(bar, baz)`
1947/// * `Foo { bar, baz }` is the identity representation of `Foo { bar, baz }`
1948///
1949/// Note that `by_hir` is used to determine bindings are checked by their `HirId` or by their name.
1950/// This can be useful when checking patterns in `let` bindings or `match` arms.
1951pub fn is_expr_identity_of_pat(cx: &LateContext<'_>, pat: &Pat<'_>, expr: &Expr<'_>, by_hir: bool) -> bool {
1952    if cx
1953        .typeck_results()
1954        .pat_binding_modes()
1955        .get(pat.hir_id)
1956        .is_some_and(|mode| matches!(mode.0, ByRef::Yes(_)))
1957    {
1958        // If the parameter is `(x, y)` of type `&(T, T)`, or `[x, y]` of type `&[T; 2]`, then
1959        // due to match ergonomics, the inner patterns become references. Don't consider this
1960        // the identity function as that changes types.
1961        return false;
1962    }
1963
1964    // NOTE: we're inside a (function) body, so this won't ICE
1965    let qpath_res = |qpath, hir| cx.typeck_results().qpath_res(qpath, hir);
1966
1967    match (pat.kind, expr.kind) {
1968        (PatKind::Binding(_, id, _, _), _) if by_hir => {
1969            path_to_local_id(expr, id) && cx.typeck_results().expr_adjustments(expr).is_empty()
1970        },
1971        (PatKind::Binding(_, _, ident, _), ExprKind::Path(QPath::Resolved(_, path))) => {
1972            matches!(path.segments, [ segment] if segment.ident.name == ident.name)
1973        },
1974        (PatKind::Tuple(pats, dotdot), ExprKind::Tup(tup))
1975            if dotdot.as_opt_usize().is_none() && pats.len() == tup.len() =>
1976        {
1977            zip(pats, tup).all(|(pat, expr)| is_expr_identity_of_pat(cx, pat, expr, by_hir))
1978        },
1979        (PatKind::Slice(before, None, after), ExprKind::Array(arr)) if before.len() + after.len() == arr.len() => {
1980            zip(before.iter().chain(after), arr).all(|(pat, expr)| is_expr_identity_of_pat(cx, pat, expr, by_hir))
1981        },
1982        (PatKind::TupleStruct(pat_ident, field_pats, dotdot), ExprKind::Call(ident, fields))
1983            if dotdot.as_opt_usize().is_none() && field_pats.len() == fields.len() =>
1984        {
1985            // check ident
1986            if let ExprKind::Path(ident) = &ident.kind
1987                && qpath_res(&pat_ident, pat.hir_id) == qpath_res(ident, expr.hir_id)
1988                // check fields
1989                && zip(field_pats, fields).all(|(pat, expr)| is_expr_identity_of_pat(cx, pat, expr,by_hir))
1990            {
1991                true
1992            } else {
1993                false
1994            }
1995        },
1996        (PatKind::Struct(pat_ident, field_pats, false), ExprKind::Struct(ident, fields, hir::StructTailExpr::None))
1997            if field_pats.len() == fields.len() =>
1998        {
1999            // check ident
2000            qpath_res(&pat_ident, pat.hir_id) == qpath_res(ident, expr.hir_id)
2001                // check fields
2002                && field_pats.iter().all(|field_pat| {
2003                    fields.iter().any(|field| {
2004                        field_pat.ident == field.ident && is_expr_identity_of_pat(cx, field_pat.pat, field.expr, by_hir)
2005                    })
2006                })
2007        },
2008        _ => false,
2009    }
2010}

2245/// Returns a list of groups where elements in each group are equal according to `eq`
2246///
2247/// - Within each group the elements are sorted by the order they appear in `exprs`
2248/// - The groups themselves are sorted by their first element's appearence in `exprs`
2249///
2250/// Given functions `eq` and `hash` such that `eq(a, b) == true`
2251/// implies `hash(a) == hash(b)`
2252pub fn search_same<T, Hash, Eq>(exprs: &[T], mut hash: Hash, mut eq: Eq) -> Vec<Vec<&T>>
2253where
2254    Hash: FnMut(&T) -> u64,
2255    Eq: FnMut(&T, &T) -> bool,
2256{
2257    match exprs {
2258        [a, b] if eq(a, b) => return vec![vec![a, b]],
2259        _ if exprs.len() <= 2 => return vec![],
2260        _ => {},
2261    }
2262
2263    let mut buckets: UnindexMap<u64, Vec<Vec<&T>>> = UnindexMap::default();
2264
2265    for expr in exprs {
2266        match buckets.entry(hash(expr)) {
2267            indexmap::map::Entry::Occupied(mut o) => {
2268                let bucket = o.get_mut();
2269                match bucket.iter_mut().find(|group| eq(expr, group[0])) {
2270                    Some(group) => group.push(expr),
2271                    None => bucket.push(vec![expr]),
2272                }
2273            },
2274            indexmap::map::Entry::Vacant(v) => {
2275                v.insert(vec![vec![expr]]);
2276            },
2277        }
2278    }
2279
2280    buckets
2281        .into_values()
2282        .flatten()
2283        .filter(|group| group.len() > 1)
2284        .collect()
2285}

2513/// Walks up the HIR tree from the given expression in an attempt to find where the value is
2514/// consumed.
2515///
2516/// Termination has three conditions:
2517/// - The given function returns `Break`. This function will return the value.
2518/// - The consuming node is found. This function will return `Continue(use_node, child_id)`.
2519/// - No further parent nodes are found. This will trigger a debug assert or return `None`.
2520///
2521/// This allows walking through `if`, `match`, `break`, and block expressions to find where the
2522/// value produced by the expression is consumed.
2523pub fn walk_to_expr_usage<'tcx, T>(
2524    cx: &LateContext<'tcx>,
2525    e: &Expr<'tcx>,
2526    mut f: impl FnMut(HirId, Node<'tcx>, HirId) -> ControlFlow<T>,
2527) -> Option<ControlFlow<T, (Node<'tcx>, HirId)>> {
2528    let mut iter = cx.tcx.hir_parent_iter(e.hir_id);
2529    let mut child_id = e.hir_id;
2530
2531    while let Some((parent_id, parent)) = iter.next() {
2532        if let ControlFlow::Break(x) = f(parent_id, parent, child_id) {
2533            return Some(ControlFlow::Break(x));
2534        }
2535        let parent_expr = match parent {
2536            Node::Expr(e) => e,
2537            Node::Block(Block { expr: Some(body), .. }) | Node::Arm(Arm { body, .. }) if body.hir_id == child_id => {
2538                child_id = parent_id;
2539                continue;
2540            },
2541            Node::Arm(a) if a.body.hir_id == child_id => {
2542                child_id = parent_id;
2543                continue;
2544            },
2545            _ => return Some(ControlFlow::Continue((parent, child_id))),
2546        };
2547        match parent_expr.kind {
2548            ExprKind::If(child, ..) | ExprKind::Match(child, ..) if child.hir_id != child_id => child_id = parent_id,
2549            ExprKind::Break(Destination { target_id: Ok(id), .. }, _) => {
2550                child_id = id;
2551                iter = cx.tcx.hir_parent_iter(id);
2552            },
2553            ExprKind::Block(..) | ExprKind::DropTemps(_) => child_id = parent_id,
2554            _ => return Some(ControlFlow::Continue((parent, child_id))),
2555        }
2556    }
2557    debug_assert!(false, "no parent node found for `{child_id:?}`");
2558    None
2559}

2785                _ => {},
2786            }
2787        }
2788        ControlFlow::Continue(())
2789    });
2790    match node {
2791        Some(ControlFlow::Continue((node, child_id))) => ExprUseCtxt {
2792            node,
2793            child_id,
2794            adjustments,
2795            is_ty_unified,
2796            moved_before_use,
2797            same_ctxt,
2798        },

2801        None => ExprUseCtxt {
2802            node: Node::Crate(cx.tcx.hir_root_module()),
2803            child_id: HirId::INVALID,
2804            adjustments: &[],
2805            is_ty_unified: true,
2806            moved_before_use: true,
2807            same_ctxt: false,
2808        },
2809    }

3254/// `max_super` `super` items, produces an absolute path instead. Both `from` and `to` should be in
3255/// the local crate.

3258///

3260/// path (which _should_ be a reasonable assumption since we in order to be able to use an object of
3261/// certain type T, T is required to be visible).

3271        .iter()

3273        .skip_while(|el| matches!(el, Both(l, r) if l == r))
3274        .map(|el| match el {

3276            Left(l) => Left(l.data),
3277            Right(r) => Right(r.data),
3278        });

3281    let mut go_up_by = 0;

3285            Both(l, r) => {

3288                // c::d::e  ::f::sym

3290                //
3291                // alternatively:
3292                // a::b::c  ::d::sym refers to
3293                // e::f::sym:: ::
3294                // result should be super::super::super::super::e::f
3295                if let DefPathData::TypeNs(sym) = l {
3296                    path.push(sym);
3297                }
3298                if let DefPathData::TypeNs(_) = r {
3299                    go_up_by += 1;
3300                }
3301            },

3303            // a::b::sym:: ::    refers to
3304            // c::d::e  ::f::sym
3305            // when looking at `f`
3306            Left(DefPathData::TypeNs(sym)) => path.push(sym),
3307            // consider:
3308            // a::b::c  ::d::sym refers to
3309            // e::f::sym:: ::
3310            // when looking at `d`
3311            Right(DefPathData::TypeNs(_)) => go_up_by += 1,
3312            _ => {},
3313        }
3314    }

3320                Some(sym)

3323            }
3324        })))

3389/// Returns true if the specified `expr` requires coercion,
3390/// meaning that it either has a coercion or propagates a coercion from one of its sub expressions.
3391///
3392/// Similar to [`is_adjusted`], this not only checks if an expression's type was adjusted,
3393/// but also going through extra steps to see if it fits the description of [coercion sites].
3394///
3395/// You should used this when you want to avoid suggesting replacing an expression that is currently
3396/// a coercion site or coercion propagating expression with one that is not.
3397///
3398/// [coercion sites]: https://doc.rust-lang.org/stable/reference/type-coercions.html#coercion-sites
3399pub fn expr_requires_coercion<'tcx>(cx: &LateContext<'tcx>, expr: &Expr<'tcx>) -> bool {
3400    let expr_ty_is_adjusted = cx
3401        .typeck_results()
3402        .expr_adjustments(expr)
3403        .iter()
3404        // ignore `NeverToAny` adjustments, such as `panic!` call.
3405        .any(|adj| !matches!(adj.kind, Adjust::NeverToAny));
3406    if expr_ty_is_adjusted {
3407        return true;
3408    }
3409
3410    // Identify coercion sites and recursively check if those sites
3411    // actually have type adjustments.
3412    match expr.kind {
3413        ExprKind::Call(_, args) | ExprKind::MethodCall(_, _, args, _) if let Some(def_id) = fn_def_id(cx, expr) => {
3414            let fn_sig = cx.tcx.fn_sig(def_id).instantiate_identity();
3415
3416            if !fn_sig.output().skip_binder().has_type_flags(TypeFlags::HAS_TY_PARAM) {
3417                return false;
3418            }
3419
3420            let self_arg_count = usize::from(matches!(expr.kind, ExprKind::MethodCall(..)));
3421            let mut args_with_ty_param = {
3422                fn_sig
3423                    .inputs()
3424                    .skip_binder()
3425                    .iter()
3426                    .skip(self_arg_count)
3427                    .zip(args)
3428                    .filter_map(|(arg_ty, arg)| {
3429                        if arg_ty.has_type_flags(TypeFlags::HAS_TY_PARAM) {
3430                            Some(arg)
3431                        } else {
3432                            None
3433                        }
3434                    })
3435            };
3436            args_with_ty_param.any(|arg| expr_requires_coercion(cx, arg))
3437        },
3438        // Struct/union initialization.
3439        ExprKind::Struct(qpath, _, _) => {
3440            let res = cx.typeck_results().qpath_res(qpath, expr.hir_id);
3441            if let Some((_, v_def)) = adt_and_variant_of_res(cx, res) {
3442                let rustc_ty::Adt(_, generic_args) = cx.typeck_results().expr_ty_adjusted(expr).kind() else {
3443                    // This should never happen, but when it does, not linting is the better option.
3444                    return true;
3445                };
3446                v_def
3447                    .fields
3448                    .iter()
3449                    .any(|field| field.ty(cx.tcx, generic_args).has_type_flags(TypeFlags::HAS_TY_PARAM))
3450            } else {
3451                false
3452            }
3453        },
3454        // Function results, including the final line of a block or a `return` expression.
3455        ExprKind::Block(
3456            &Block {
3457                expr: Some(ret_expr), ..
3458            },
3459            _,
3460        )
3461        | ExprKind::Ret(Some(ret_expr)) => expr_requires_coercion(cx, ret_expr),
3462
3463        // ===== Coercion-propagation expressions =====
3464        ExprKind::Array(elems) | ExprKind::Tup(elems) => elems.iter().any(|elem| expr_requires_coercion(cx, elem)),
3465        // Array but with repeating syntax.
3466        ExprKind::Repeat(rep_elem, _) => expr_requires_coercion(cx, rep_elem),
3467        // Others that may contain coercion sites.
3468        ExprKind::If(_, then, maybe_else) => {
3469            expr_requires_coercion(cx, then) || maybe_else.is_some_and(|e| expr_requires_coercion(cx, e))
3470        },
3471        ExprKind::Match(_, arms, _) => arms
3472            .iter()
3473            .map(|arm| arm.body)
3474            .any(|body| expr_requires_coercion(cx, body)),
3475        _ => false,
3476    }
3477}

3547/// Checks if `expr` may be directly used as the return value of its enclosing body.
3548/// The following cases are covered:
3549/// - `expr` as the last expression of the body, or of a block that can be used as the return value
3550/// - `return expr`
3551/// - then or else part of a `if` in return position
3552/// - arm body of a `match` in a return position
3553/// - `break expr` or `break 'label expr` if the loop or block being exited is used as a return
3554///   value
3555///
3556/// Contrary to [`TyCtxt::hir_get_fn_id_for_return_block()`], if `expr` is part of a
3557/// larger expression, for example a field expression of a `struct`, it will not be
3558/// considered as matching the condition and will return `false`.
3559///
3560/// Also, even if `expr` is assigned to a variable which is later returned, this function
3561/// will still return `false` because `expr` is not used *directly* as the return value
3562/// as it goes through the intermediate variable.
3563pub fn potential_return_of_enclosing_body(cx: &LateContext<'_>, expr: &Expr<'_>) -> bool {
3564    let enclosing_body_owner = cx
3565        .tcx
3566        .local_def_id_to_hir_id(cx.tcx.hir_enclosing_body_owner(expr.hir_id));
3567    let mut prev_id = expr.hir_id;
3568    let mut skip_until_id = None;
3569    for (hir_id, node) in cx.tcx.hir_parent_iter(expr.hir_id) {
3570        if hir_id == enclosing_body_owner {
3571            return true;
3572        }
3573        if let Some(id) = skip_until_id {
3574            prev_id = hir_id;
3575            if id == hir_id {
3576                skip_until_id = None;
3577            }
3578            continue;
3579        }
3580        match node {
3581            Node::Block(Block { expr, .. }) if expr.is_some_and(|expr| expr.hir_id == prev_id) => {},
3582            Node::Arm(arm) if arm.body.hir_id == prev_id => {},
3583            Node::Expr(expr) => match expr.kind {
3584                ExprKind::Ret(_) => return true,
3585                ExprKind::If(_, then, opt_else)
3586                    if then.hir_id == prev_id || opt_else.is_some_and(|els| els.hir_id == prev_id) => {},
3587                ExprKind::Match(_, arms, _) if arms.iter().any(|arm| arm.hir_id == prev_id) => {},
3588                ExprKind::Block(block, _) if block.hir_id == prev_id => {},
3589                ExprKind::Break(
3590                    Destination {
3591                        target_id: Ok(target_id),
3592                        ..
3593                    },
3594                    _,
3595                ) => skip_until_id = Some(target_id),
3596                _ => break,
3597            },
3598            _ => break,
3599        }
3600        prev_id = hir_id;
3601    }
3602
3603    // `expr` is used as part of "something" and is not returned directly from its
3604    // enclosing body.
3605    false
3606}
