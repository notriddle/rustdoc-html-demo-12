8macro_rules! val {

19        /// To be supplied to `rustc_interface::Config`

21            $(
22                val!($name $($value)?),
23            )*
24        ];

26        $(
27            pub const $name: rustc_span::Symbol = rustc_span::Symbol::new(PREDEFINED_SYMBOLS_COUNT + ${index()});

29    };
30}
