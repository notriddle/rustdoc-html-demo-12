20/// Specifies whether to resolve a path in the [`TypeNS`], [`ValueNS`], [`MacroNS`] or in an
21/// arbitrary namespace
22#[derive(Clone, Copy, PartialEq, Debug)]
23pub enum PathNS {
24    Type,
25    Value,
26    Macro,
27
28    /// Resolves to the name in the first available namespace, e.g. for `std::vec` this would return
29    /// either the macro or the module but **not** both
30    ///
31    /// Must only be used when the specific resolution is unimportant such as in
32    /// `missing_enforced_import_renames`
33    Arbitrary,
34}
35
36impl PathNS {
37    fn matches(self, ns: Option<Namespace>) -> bool {
38        let required = match self {
39            PathNS::Type => TypeNS,
40            PathNS::Value => ValueNS,
41            PathNS::Macro => MacroNS,
42            PathNS::Arbitrary => return true,
43        };
44
45        ns == Some(required)
46    }
47}

49/// Lazily resolves a path into a list of [`DefId`]s using [`lookup_path`].
50///
51/// Typically it will contain one [`DefId`] or none, but in some situations there can be multiple:
52/// - `memchr::memchr` could return the functions from both memchr 1.0 and memchr 2.0
53/// - `alloc::boxed::Box::downcast` would return a function for each of the different inherent impls
54///   ([1], [2], [3])
55///
56/// [1]: https://doc.rust-lang.org/std/boxed/struct.Box.html#method.downcast
57/// [2]: https://doc.rust-lang.org/std/boxed/struct.Box.html#method.downcast-1
58/// [3]: https://doc.rust-lang.org/std/boxed/struct.Box.html#method.downcast-2
59pub struct PathLookup {
60    ns: PathNS,
61    path: &'static [Symbol],
62    once: OnceLock<Vec<DefId>>,
63}
64
65impl PathLookup {
66    /// Only exported for tests and `clippy_lints_internal`
67    #[doc(hidden)]
68    pub const fn new(ns: PathNS, path: &'static [Symbol]) -> Self {
69        Self {
70            ns,
71            path,
72            once: OnceLock::new(),
73        }
74    }
75
76    /// Returns the list of [`DefId`]s that the path resolves to
77    pub fn get(&self, cx: &LateContext<'_>) -> &[DefId] {
78        self.once.get_or_init(|| lookup_path(cx.tcx, self.ns, self.path))
79    }
80
81    /// Returns the single [`DefId`] that the path resolves to, this can only be used for paths into
82    /// stdlib crates to avoid the issue of multiple [`DefId`]s being returned
83    ///
84    /// May return [`None`] in `no_std`/`no_core` environments
85    pub fn only(&self, cx: &LateContext<'_>) -> Option<DefId> {
86        let ids = self.get(cx);
87        debug_assert!(STDLIB_STABLE_CRATES.contains(&self.path[0]));
88        debug_assert!(ids.len() <= 1, "{ids:?}");
89        ids.first().copied()
90    }
91
92    /// Checks if the path resolves to the given `def_id`
93    pub fn matches(&self, cx: &LateContext<'_>, def_id: DefId) -> bool {
94        self.get(cx).contains(&def_id)
95    }
96
97    /// Resolves `maybe_path` to a [`DefId`] and checks if the [`PathLookup`] matches it
98    pub fn matches_path<'tcx>(&self, cx: &LateContext<'_>, maybe_path: &impl MaybePath<'tcx>) -> bool {
99        path_def_id(cx, maybe_path).is_some_and(|def_id| self.matches(cx, def_id))
100    }
101
102    /// Checks if the path resolves to `ty`'s definition, must be an `Adt`
103    pub fn matches_ty(&self, cx: &LateContext<'_>, ty: Ty<'_>) -> bool {
104        ty.ty_adt_def().is_some_and(|adt| self.matches(cx, adt.did()))
105    }
106}

108macro_rules! path_macros {
109    ($($name:ident: $ns:expr,)*) => {
110        $(
111            /// Only exported for tests and `clippy_lints_internal`
112            #[doc(hidden)]
113            #[macro_export]
114            macro_rules! $name {
115                ($$($$seg:ident $$(::)?)*) => {
116                    PathLookup::new($ns, &[$$(sym::$$seg,)*])
117                };
118            }
119        )*
120    };
121}

157/// Equivalent to a [`lookup_path`] after splitting the input string on `::`
158///
159/// This function is expensive and should be used sparingly.
160pub fn lookup_path_str(tcx: TyCtxt<'_>, ns: PathNS, path: &str) -> Vec<DefId> {
161    let path: Vec<Symbol> = path.split("::").map(Symbol::intern).collect();
162    lookup_path(tcx, ns, &path)
163}

165/// Resolves a def path like `std::vec::Vec`.
166///
167/// Typically it will return one [`DefId`] or none, but in some situations there can be multiple:
168/// - `memchr::memchr` could return the functions from both memchr 1.0 and memchr 2.0
169/// - `alloc::boxed::Box::downcast` would return a function for each of the different inherent impls
170///   ([1], [2], [3])
171///
172/// This function is expensive and should be used sparingly.
173///
174/// [1]: https://doc.rust-lang.org/std/boxed/struct.Box.html#method.downcast
175/// [2]: https://doc.rust-lang.org/std/boxed/struct.Box.html#method.downcast-1
176/// [3]: https://doc.rust-lang.org/std/boxed/struct.Box.html#method.downcast-2
177pub fn lookup_path(tcx: TyCtxt<'_>, ns: PathNS, path: &[Symbol]) -> Vec<DefId> {
178    let (root, rest) = match *path {
179        [] | [_] => return Vec::new(),
180        [root, ref rest @ ..] => (root, rest),
181    };
182
183    let mut out = Vec::new();
184    for &base in find_crates(tcx, root).iter().chain(find_primitive_impls(tcx, root)) {
185        lookup_with_base(tcx, base, ns, rest, &mut out);
186    }
187    out
188}

190/// Finds the crates called `name`, may be multiple due to multiple major versions.
191pub fn find_crates(tcx: TyCtxt<'_>, name: Symbol) -> &'static [DefId] {
192    static BY_NAME: OnceLock<FxHashMap<Symbol, Vec<DefId>>> = OnceLock::new();
193    let map = BY_NAME.get_or_init(|| {
194        let mut map = FxHashMap::default();
195        map.insert(tcx.crate_name(LOCAL_CRATE), vec![LOCAL_CRATE.as_def_id()]);
196        for &num in tcx.crates(()) {
197            map.entry(tcx.crate_name(num)).or_default().push(num.as_def_id());
198        }
199        map
200    });
201    match map.get(&name) {
202        Some(def_ids) => def_ids,
203        None => &[],
204    }
205}

207fn find_primitive_impls(tcx: TyCtxt<'_>, name: Symbol) -> &[DefId] {
208    let ty = match name {
209        sym::bool => SimplifiedType::Bool,
210        sym::char => SimplifiedType::Char,
211        sym::str => SimplifiedType::Str,
212        sym::array => SimplifiedType::Array,
213        sym::slice => SimplifiedType::Slice,
214        // FIXME: rustdoc documents these two using just `pointer`.
215        //
216        // Maybe this is something we should do here too.
217        sym::const_ptr => SimplifiedType::Ptr(Mutability::Not),
218        sym::mut_ptr => SimplifiedType::Ptr(Mutability::Mut),
219        sym::isize => SimplifiedType::Int(IntTy::Isize),
220        sym::i8 => SimplifiedType::Int(IntTy::I8),
221        sym::i16 => SimplifiedType::Int(IntTy::I16),
222        sym::i32 => SimplifiedType::Int(IntTy::I32),
223        sym::i64 => SimplifiedType::Int(IntTy::I64),
224        sym::i128 => SimplifiedType::Int(IntTy::I128),
225        sym::usize => SimplifiedType::Uint(UintTy::Usize),
226        sym::u8 => SimplifiedType::Uint(UintTy::U8),
227        sym::u16 => SimplifiedType::Uint(UintTy::U16),
228        sym::u32 => SimplifiedType::Uint(UintTy::U32),
229        sym::u64 => SimplifiedType::Uint(UintTy::U64),
230        sym::u128 => SimplifiedType::Uint(UintTy::U128),
231        sym::f32 => SimplifiedType::Float(FloatTy::F32),
232        sym::f64 => SimplifiedType::Float(FloatTy::F64),
233        _ => return &[],
234    };
235
236    tcx.incoherent_impls(ty)
237}

239/// Resolves a def path like `vec::Vec` with the base `std`.
240fn lookup_with_base(tcx: TyCtxt<'_>, mut base: DefId, ns: PathNS, mut path: &[Symbol], out: &mut Vec<DefId>) {
241    loop {
242        match *path {
243            [segment] => {
244                out.extend(item_child_by_name(tcx, base, ns, segment));
245
246                // When the current def_id is e.g. `struct S`, check the impl items in
247                // `impl S { ... }`
248                let inherent_impl_children = tcx
249                    .inherent_impls(base)
250                    .iter()
251                    .filter_map(|&impl_def_id| item_child_by_name(tcx, impl_def_id, ns, segment));
252                out.extend(inherent_impl_children);
253
254                return;
255            },
256            [segment, ref rest @ ..] => {
257                path = rest;
258                let Some(child) = item_child_by_name(tcx, base, PathNS::Type, segment) else {
259                    return;
260                };
261                base = child;
262            },
263            [] => unreachable!(),
264        }
265    }
266}

268fn item_child_by_name(tcx: TyCtxt<'_>, def_id: DefId, ns: PathNS, name: Symbol) -> Option<DefId> {
269    if let Some(local_id) = def_id.as_local() {
270        local_item_child_by_name(tcx, local_id, ns, name)
271    } else {
272        non_local_item_child_by_name(tcx, def_id, ns, name)
273    }
274}

276fn local_item_child_by_name(tcx: TyCtxt<'_>, local_id: LocalDefId, ns: PathNS, name: Symbol) -> Option<DefId> {
277    let root_mod;
278    let item_kind = match tcx.hir_node_by_def_id(local_id) {
279        Node::Crate(r#mod) => {
280            root_mod = ItemKind::Mod(Ident::dummy(), r#mod);
281            &root_mod
282        },
283        Node::Item(item) => &item.kind,
284        _ => return None,
285    };
286
287    match item_kind {
288        ItemKind::Mod(_, r#mod) => r#mod.item_ids.iter().find_map(|&item_id| {
289            let item = tcx.hir_item(item_id);
290            if let ItemKind::Use(path, UseKind::Single(ident)) = item.kind {
291                if ident.name == name {
292                    let opt_def_id = |ns: Option<Res>| ns.and_then(|res| res.opt_def_id());
293                    match ns {
294                        PathNS::Type => opt_def_id(path.res.type_ns),
295                        PathNS::Value => opt_def_id(path.res.value_ns),
296                        PathNS::Macro => opt_def_id(path.res.macro_ns),
297                        PathNS::Arbitrary => unreachable!(),
298                    }
299                } else {
300                    None
301                }
302            } else if let Some(ident) = item.kind.ident()
303                && ident.name == name
304                && ns.matches(tcx.def_kind(item.owner_id).ns())
305            {
306                Some(item.owner_id.to_def_id())
307            } else {
308                None
309            }
310        }),
311        ItemKind::Impl(..) | ItemKind::Trait(..) => tcx
312            .associated_items(local_id)
313            .filter_by_name_unhygienic(name)
314            .find(|assoc_item| ns.matches(Some(assoc_item.namespace())))
315            .map(|assoc_item| assoc_item.def_id),
316        _ => None,
317    }
318}

320fn non_local_item_child_by_name(tcx: TyCtxt<'_>, def_id: DefId, ns: PathNS, name: Symbol) -> Option<DefId> {
321    match tcx.def_kind(def_id) {
322        DefKind::Mod | DefKind::Enum | DefKind::Trait => tcx.module_children(def_id).iter().find_map(|child| {
323            if child.ident.name == name && ns.matches(child.res.ns()) {
324                child.res.opt_def_id()
325            } else {
326                None
327            }
328        }),
329        DefKind::Impl { .. } => tcx
330            .associated_item_def_ids(def_id)
331            .iter()
332            .copied()
333            .find(|assoc_def_id| tcx.item_name(*assoc_def_id) == name && ns.matches(tcx.def_kind(assoc_def_id).ns())),
334        _ => None,
335    }
