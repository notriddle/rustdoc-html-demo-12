40pub(crate) fn is_same_visibility(a: &Visibility, b: &Visibility) -> bool {
41    match (&a.kind, &b.kind) {
42        (
43            VisibilityKind::Restricted { path: p, .. },
44            VisibilityKind::Restricted { path: q, .. },
45        ) => pprust::path_to_string(p) == pprust::path_to_string(q),
46        (VisibilityKind::Public, VisibilityKind::Public)
47        | (VisibilityKind::Inherited, VisibilityKind::Inherited) => true,
48        _ => false,
49    }

52// Uses Cow to avoid allocating in the common cases.
53pub(crate) fn format_visibility(
54    context: &RewriteContext<'_>,
55    vis: &Visibility,
56) -> Cow<'static, str> {
57    match vis.kind {
58        VisibilityKind::Public => Cow::from("pub "),
59        VisibilityKind::Inherited => Cow::from(""),
60        VisibilityKind::Restricted { ref path, .. } => {
61            let Path { ref segments, .. } = **path;
62            let mut segments_iter = segments.iter().map(|seg| rewrite_ident(context, seg.ident));
63            if path.is_global() {
64                segments_iter
65                    .next()
66                    .expect("Non-global path in pub(restricted)?");
67            }
68            let is_keyword = |s: &str| s == "crate" || s == "self" || s == "super";
69            let path = segments_iter.collect::<Vec<_>>().join("::");
70            let in_str = if is_keyword(&path) { "" } else { "in " };
71
72            Cow::from(format!("pub({in_str}{path}) "))
73        }
74    }
75}

137    match ext {
138        ast::Extern::None => Cow::from(""),

230pub(crate) fn last_line_extendable(s: &str) -> bool {
231    if s.ends_with("\"#") {
232        return true;
233    }
234    for c in s.chars().rev() {
235        match c {
236            '(' | ')' | ']' | '}' | '?' | '>' => continue,
237            '\n' => break,
238            _ if c.is_whitespace() => continue,
239            _ => return false,
240        }
241    }

247    match meta_item.kind {
248        MetaItemKind::Word => {

250            path_str == skip_annotation().as_str() || path_str == depr_skip_annotation().as_str()

275pub(crate) fn semicolon_for_expr(context: &RewriteContext<'_>, expr: &ast::Expr) -> bool {
276    // Never try to insert semicolons on expressions when we're inside
277    // a macro definition - this can prevent the macro from compiling
278    // when used in expression position
279    if context.is_macro_def {
280        return false;
281    }
282
283    match expr.kind {
284        ast::ExprKind::Ret(..) | ast::ExprKind::Continue(..) | ast::ExprKind::Break(..) => {
285            context.config.trailing_semicolon()
286        }
287        _ => false,
288    }
289}

291#[inline]
292pub(crate) fn semicolon_for_stmt(
293    context: &RewriteContext<'_>,
294    stmt: &ast::Stmt,
295    is_last_expr: bool,
296) -> bool {
297    match stmt.kind {
298        ast::StmtKind::Semi(ref expr) => match expr.kind {
299            ast::ExprKind::While(..) | ast::ExprKind::Loop(..) | ast::ExprKind::ForLoop { .. } => {
300                false
301            }
302            ast::ExprKind::Break(..) | ast::ExprKind::Continue(..) | ast::ExprKind::Ret(..) => {
303                // The only time we can skip the semi-colon is if the config option is set to false
304                // **and** this is the last expr (even though any following exprs are unreachable)
305                context.config.trailing_semicolon() || !is_last_expr
306            }
307            _ => true,
308        },
309        ast::StmtKind::Expr(..) => false,
310        _ => true,
311    }
312}

322/// Returns the number of LF and CRLF respectively.
323pub(crate) fn count_lf_crlf(input: &str) -> (usize, usize) {
324    let mut lf = 0;
325    let mut crlf = 0;
326    let mut is_crlf = false;
327    for c in input.as_bytes() {
328        match c {
329            b'\r' => is_crlf = true,
330            b'\n' if is_crlf => crlf += 1,
331            b'\n' => lf += 1,
332            _ => is_crlf = false,
333        }
334    }
335    (lf, crlf)
336}

387// Wraps String in an Option. Returns Some when the string adheres to the
388// Rewrite constraints defined for the Rewrite trait and None otherwise.
389pub(crate) fn wrap_str(s: String, max_width: usize, shape: Shape) -> Option<String> {
390    if filtered_str_fits(&s, max_width, shape) {
391        Some(s)
392    } else {
393        None
394    }
395}
396
397pub(crate) fn filtered_str_fits(snippet: &str, max_width: usize, shape: Shape) -> bool {
398    let snippet = &filter_normal_code(snippet);
399    if !snippet.is_empty() {
400        // First line must fits with `shape.width`.
401        if first_line_width(snippet) > shape.width {
402            return false;
403        }
404        // If the snippet does not include newline, we are done.
405        if is_single_line(snippet) {
406            return true;
407        }
408        // The other lines must fit within the maximum width.
409        if snippet
410            .lines()
411            .skip(1)
412            .any(|line| unicode_str_width(line) > max_width)
413        {
414            return false;
415        }
416        // A special check for the last line, since the caller may
417        // place trailing characters on this line.
418        if last_line_width(snippet) > shape.used_width() + shape.width {
419            return false;
420        }
421    }
422    true
423}

464// States whether an expression's last line exclusively consists of closing
465// parens, braces, and brackets in its idiomatic formatting.
466pub(crate) fn is_block_expr(context: &RewriteContext<'_>, expr: &ast::Expr, repr: &str) -> bool {
467    match expr.kind {
468        ast::ExprKind::MacCall(..)
469        | ast::ExprKind::FormatArgs(..)
470        | ast::ExprKind::Call(..)
471        | ast::ExprKind::MethodCall(..)
472        | ast::ExprKind::Array(..)
473        | ast::ExprKind::Struct(..)
474        | ast::ExprKind::While(..)
475        | ast::ExprKind::If(..)
476        | ast::ExprKind::Block(..)
477        | ast::ExprKind::ConstBlock(..)
478        | ast::ExprKind::Gen(..)
479        | ast::ExprKind::Loop(..)
480        | ast::ExprKind::ForLoop { .. }
481        | ast::ExprKind::TryBlock(..)
482        | ast::ExprKind::Match(..) => repr.contains('\n'),
483        ast::ExprKind::Paren(ref expr)
484        | ast::ExprKind::Binary(_, _, ref expr)
485        | ast::ExprKind::Index(_, ref expr, _)
486        | ast::ExprKind::Unary(_, ref expr)
487        | ast::ExprKind::Try(ref expr)
488        | ast::ExprKind::Yield(YieldKind::Prefix(Some(ref expr))) => {
489            is_block_expr(context, expr, repr)
490        }
491        ast::ExprKind::Closure(ref closure) => is_block_expr(context, &closure.body, repr),
492        // This can only be a string lit
493        ast::ExprKind::Lit(_) => {
494            repr.contains('\n') && trimmed_last_line_width(repr) <= context.config.tab_spaces()
495        }
496        ast::ExprKind::AddrOf(..)
497        | ast::ExprKind::Assign(..)
498        | ast::ExprKind::AssignOp(..)
499        | ast::ExprKind::Await(..)
500        | ast::ExprKind::Break(..)
501        | ast::ExprKind::Cast(..)
502        | ast::ExprKind::Continue(..)
503        | ast::ExprKind::Dummy
504        | ast::ExprKind::Err(_)
505        | ast::ExprKind::Field(..)
506        | ast::ExprKind::IncludedBytes(..)
507        | ast::ExprKind::InlineAsm(..)
508        | ast::ExprKind::OffsetOf(..)
509        | ast::ExprKind::UnsafeBinderCast(..)
510        | ast::ExprKind::Let(..)
511        | ast::ExprKind::Path(..)
512        | ast::ExprKind::Range(..)
513        | ast::ExprKind::Repeat(..)
514        | ast::ExprKind::Ret(..)
515        | ast::ExprKind::Become(..)
516        | ast::ExprKind::Yeet(..)
517        | ast::ExprKind::Tup(..)
518        | ast::ExprKind::Use(..)
519        | ast::ExprKind::Type(..)
520        | ast::ExprKind::Yield(..)
521        | ast::ExprKind::Underscore => false,
522    }
523}

525/// Removes trailing spaces from the specified snippet. We do not remove spaces
526/// inside strings or comments.
527pub(crate) fn remove_trailing_white_spaces(text: &str) -> String {
528    let mut buffer = String::with_capacity(text.len());
529    let mut space_buffer = String::with_capacity(128);
530    for (char_kind, c) in CharClasses::new(text.chars()) {
531        match c {
532            '\n' => {
533                if char_kind == FullCodeCharKind::InString {
534                    buffer.push_str(&space_buffer);
535                }
536                space_buffer.clear();
537                buffer.push('\n');
538            }
539            _ if c.is_whitespace() => {
540                space_buffer.push(c);
541            }
542            _ => {
543                if !space_buffer.is_empty() {
544                    buffer.push_str(&space_buffer);
545                    space_buffer.clear();
546                }
547                buffer.push(c);
548            }
549        }
550    }
551    buffer
552}

554/// Indent each line according to the specified `indent`.
555/// e.g.
556///
557/// ```rust,compile_fail
558/// foo!{
559/// x,
560/// y,
561/// foo(
562///     a,
563///     b,
564///     c,
565/// ),
566/// }
567/// ```
568///
569/// will become
570///
571/// ```rust,compile_fail
572/// foo!{
573///     x,
574///     y,
575///     foo(
576///         a,
577///         b,
578///         c,
579///     ),
580/// }
581/// ```
582pub(crate) fn trim_left_preserve_layout(
583    orig: &str,
584    indent: Indent,
585    config: &Config,
586) -> Option<String> {
587    let mut lines = LineClasses::new(orig);
588    let first_line = lines.next().map(|(_, s)| s.trim_end().to_owned())?;
589    let mut trimmed_lines = Vec::with_capacity(16);
590
591    let mut veto_trim = false;
592    let min_prefix_space_width = lines
593        .filter_map(|(kind, line)| {
594            let mut trimmed = true;
595            let prefix_space_width = if is_empty_line(&line) {
596                None
597            } else {
598                Some(get_prefix_space_width(config, &line))
599            };
600
601            // just InString{Commented} in order to allow the start of a string to be indented
602            let new_veto_trim_value = (kind == FullCodeCharKind::InString
603                || (config.style_edition() >= StyleEdition::Edition2024
604                    && kind == FullCodeCharKind::InStringCommented))
605                && !line.ends_with('\\');
606            let line = if veto_trim || new_veto_trim_value {
607                veto_trim = new_veto_trim_value;
608                trimmed = false;
609                line
610            } else {
611                line.trim().to_owned()
612            };
613            trimmed_lines.push((trimmed, line, prefix_space_width));
614
615            // Because there is a veto against trimming and indenting lines within a string,
616            // such lines should not be taken into account when computing the minimum.
617            match kind {
618                FullCodeCharKind::InStringCommented | FullCodeCharKind::EndStringCommented
619                    if config.style_edition() >= StyleEdition::Edition2024 =>
620                {
621                    None
622                }
623                FullCodeCharKind::InString | FullCodeCharKind::EndString => None,
624                _ => prefix_space_width,
625            }
626        })
627        .min()?;
628
629    Some(
630        first_line
631            + "\n"
632            + &trimmed_lines
633                .iter()
634                .map(
635                    |&(trimmed, ref line, prefix_space_width)| match prefix_space_width {
636                        _ if !trimmed => line.to_owned(),
637                        Some(original_indent_width) => {
638                            let new_indent_width = indent.width()
639                                + original_indent_width.saturating_sub(min_prefix_space_width);
640                            let new_indent = Indent::from_width(config, new_indent_width);
641                            format!("{}{}", new_indent.to_string(config), line)
642                        }
643                        None => String::new(),
644                    },
645                )
646                .collect::<Vec<_>>()
647                .join("\n"),
648    )
649}

651/// Based on the given line, determine if the next line can be indented or not.
652/// This allows to preserve the indentation of multi-line literals when
653/// re-inserted a code block that has been formatted separately from the rest
654/// of the code, such as code in macro defs or code blocks doc comments.
655pub(crate) fn indent_next_line(kind: FullCodeCharKind, line: &str, config: &Config) -> bool {
656    if kind.is_string() {
657        // If the string ends with '\', the string has been wrapped over
658        // multiple lines. If `format_strings = true`, then the indentation of
659        // strings wrapped over multiple lines will have been adjusted while
660        // formatting the code block, therefore the string's indentation needs
661        // to be adjusted for the code surrounding the code block.
662        config.format_strings() && line.ends_with('\\')
663    } else if config.style_edition() >= StyleEdition::Edition2024 {
664        !kind.is_commented_string()
665    } else {
666        true
667    }
668}

670pub(crate) fn is_empty_line(s: &str) -> bool {
671    s.is_empty() || s.chars().all(char::is_whitespace)
672}
673
674fn get_prefix_space_width(config: &Config, s: &str) -> usize {
675    let mut width = 0;
676    for c in s.chars() {
677        match c {
678            ' ' => width += 1,
679            '\t' => width += config.tab_spaces(),
680            _ => return width,
681        }
682    }
683    width
684}
