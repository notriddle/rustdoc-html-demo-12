7macro_rules! declare_features {

10    )+) => {
11        /// Formerly unstable features that have now been accepted (stabilized).
12        pub static ACCEPTED_LANG_FEATURES: &[Feature] = &[
13            $(Feature {
14                name: sym::$feature,
15                since: $ver,
16                issue: to_nonzero($issue),
17            }),+
18        ];
19    }
20}
