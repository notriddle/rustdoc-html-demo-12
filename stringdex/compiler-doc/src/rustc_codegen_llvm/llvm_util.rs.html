26pub(crate) fn init(sess: &Session) {
27    unsafe {
28        // Before we touch LLVM, make sure that multithreading is enabled.
29        if llvm::LLVMIsMultithreaded() != 1 {
30            bug!("LLVM compiled without support for threads");
31        }
32        INIT.call_once(|| {
33            configure_llvm(sess);
34        });
35    }
36}
37
38fn require_inited() {
39    if !INIT.is_completed() {
40        bug!("LLVM is not initialized");
41    }
42}

44unsafe fn configure_llvm(sess: &Session) {
45    let n_args = sess.opts.cg.llvm_args.len() + sess.target.llvm_args.len();
46    let mut llvm_c_strs = Vec::with_capacity(n_args + 1);
47    let mut llvm_args = Vec::with_capacity(n_args + 1);
48
49    unsafe {
50        llvm::LLVMRustInstallErrorHandlers();
51    }
52    // On Windows, an LLVM assertion will open an Abort/Retry/Ignore dialog
53    // box for the purpose of launching a debugger. However, on CI this will
54    // cause it to hang until it times out, which can take several hours.
55    if std::env::var_os("CI").is_some() {
56        unsafe {
57            llvm::LLVMRustDisableSystemDialogsOnCrash();
58        }
59    }
60
61    fn llvm_arg_to_arg_name(full_arg: &str) -> &str {
62        full_arg.trim().split(|c: char| c == '=' || c.is_whitespace()).next().unwrap_or("")
63    }
64
65    let cg_opts = sess.opts.cg.llvm_args.iter().map(AsRef::as_ref);
66    let tg_opts = sess.target.llvm_args.iter().map(AsRef::as_ref);
67    let sess_args = cg_opts.chain(tg_opts);
68
69    let user_specified_args: FxHashSet<_> =
70        sess_args.clone().map(|s| llvm_arg_to_arg_name(s)).filter(|s| !s.is_empty()).collect();
71
72    {
73        // This adds the given argument to LLVM. Unless `force` is true
74        // user specified arguments are *not* overridden.
75        let mut add = |arg: &str, force: bool| {
76            if force || !user_specified_args.contains(llvm_arg_to_arg_name(arg)) {
77                let s = CString::new(arg).unwrap();
78                llvm_args.push(s.as_ptr());
79                llvm_c_strs.push(s);
80            }
81        };
82        // Set the llvm "program name" to make usage and invalid argument messages more clear.
83        add("rustc -Cllvm-args=\"...\" with", true);
84        if sess.opts.unstable_opts.time_llvm_passes {
85            add("-time-passes", false);
86        }
87        if sess.opts.unstable_opts.print_llvm_passes {
88            add("-debug-pass=Structure", false);
89        }
90        if sess.target.generate_arange_section
91            && !sess.opts.unstable_opts.no_generate_arange_section
92        {
93            add("-generate-arange-section", false);
94        }
95
96        match sess.opts.unstable_opts.merge_functions.unwrap_or(sess.target.merge_functions) {
97            MergeFunctions::Disabled | MergeFunctions::Trampolines => {}
98            MergeFunctions::Aliases => {
99                add("-mergefunc-use-aliases", false);
100            }
101        }
102
103        if wants_wasm_eh(sess) {
104            add("-wasm-enable-eh", false);
105        }
106
107        if sess.target.os == "emscripten"
108            && !sess.opts.unstable_opts.emscripten_wasm_eh
109            && sess.panic_strategy() == PanicStrategy::Unwind
110        {
111            add("-enable-emscripten-cxx-exceptions", false);
112        }
113
114        // HACK(eddyb) LLVM inserts `llvm.assume` calls to preserve align attributes
115        // during inlining. Unfortunately these may block other optimizations.
116        add("-preserve-alignment-assumptions-during-inlining=false", false);
117
118        // Use non-zero `import-instr-limit` multiplier for cold callsites.
119        add("-import-cold-multiplier=0.1", false);
120
121        if sess.print_llvm_stats() {
122            add("-stats", false);
123        }
124
125        for arg in sess_args {
126            add(&(*arg), true);
127        }
128
129        match (
130            sess.opts.unstable_opts.small_data_threshold,
131            sess.target.small_data_threshold_support(),
132        ) {
133            // Set up the small-data optimization limit for architectures that use
134            // an LLVM argument to control this.
135            (Some(threshold), SmallDataThresholdSupport::LlvmArg(arg)) => {
136                add(&format!("--{arg}={threshold}"), false)
137            }
138            _ => (),
139        };
140    }
141
142    if sess.opts.unstable_opts.llvm_time_trace {
143        unsafe { llvm::LLVMRustTimeTraceProfilerInitialize() };
144    }
145
146    rustc_llvm::initialize_available_targets();
147
148    unsafe { llvm::LLVMRustSetLLVMOptions(llvm_args.len() as c_int, llvm_args.as_ptr()) };
149}

158enum TargetFeatureFoldStrength<'a> {
159    // The feature is only tied when enabling the feature, disabling
160    // this feature shouldn't disable the tied feature.
161    EnableOnly(&'a str),
162    // The feature is tied for both enabling and disabling this feature.
163    Both(&'a str),
164}
165
166impl<'a> TargetFeatureFoldStrength<'a> {
167    fn as_str(&self) -> &'a str {
168        match self {
169            TargetFeatureFoldStrength::EnableOnly(feat) => feat,
170            TargetFeatureFoldStrength::Both(feat) => feat,
171        }
172    }
173}

175pub(crate) struct LLVMFeature<'a> {
176    llvm_feature_name: &'a str,
177    dependencies: SmallVec<[TargetFeatureFoldStrength<'a>; 1]>,
178}
179
180impl<'a> LLVMFeature<'a> {
181    fn new(llvm_feature_name: &'a str) -> Self {
182        Self { llvm_feature_name, dependencies: SmallVec::new() }
183    }
184
185    fn with_dependencies(
186        llvm_feature_name: &'a str,
187        dependencies: SmallVec<[TargetFeatureFoldStrength<'a>; 1]>,
188    ) -> Self {
189        Self { llvm_feature_name, dependencies }
190    }
191}

193impl<'a> IntoIterator for LLVMFeature<'a> {
194    type Item = &'a str;
195    type IntoIter = impl Iterator<Item = &'a str>;
196
197    fn into_iter(self) -> Self::IntoIter {
198        let dependencies = self.dependencies.into_iter().map(|feat| feat.as_str());
199        std::iter::once(self.llvm_feature_name).chain(dependencies)
200    }
201}

203/// Convert a Rust feature name to an LLVM feature name. Returning `None` means the
204/// feature should be skipped, usually because it is not supported by the current
205/// LLVM version.
206///
207/// WARNING: the features after applying `to_llvm_features` must be known
208/// to LLVM or the feature detection code will walk past the end of the feature
209/// array, leading to crashes.
210///
211/// To find a list of LLVM's names, see llvm-project/llvm/lib/Target/{ARCH}/*.td
212/// where `{ARCH}` is the architecture name. Look for instances of `SubtargetFeature`.
213///
214/// Check the current rustc fork of LLVM in the repo at
215/// <https://github.com/rust-lang/llvm-project/>. The commit in use can be found via the
216/// `llvm-project` submodule in <https://github.com/rust-lang/rust/tree/master/src> Though note that
217/// Rust can also be build with an external precompiled version of LLVM which might lead to failures
218/// if the oldest tested / supported LLVM version doesn't yet support the relevant intrinsics.
219pub(crate) fn to_llvm_features<'a>(sess: &Session, s: &'a str) -> Option<LLVMFeature<'a>> {
220    let arch = if sess.target.arch == "x86_64" {
221        "x86"
222    } else if sess.target.arch == "arm64ec" {
223        "aarch64"
224    } else if sess.target.arch == "sparc64" {
225        "sparc"
226    } else if sess.target.arch == "powerpc64" {
227        "powerpc"
228    } else {
229        &*sess.target.arch
230    };
231    match (arch, s) {
232        ("x86", "sse4.2") => Some(LLVMFeature::with_dependencies(
233            "sse4.2",
234            smallvec![TargetFeatureFoldStrength::EnableOnly("crc32")],
235        )),
236        ("x86", "pclmulqdq") => Some(LLVMFeature::new("pclmul")),
237        ("x86", "rdrand") => Some(LLVMFeature::new("rdrnd")),
238        ("x86", "bmi1") => Some(LLVMFeature::new("bmi")),
239        ("x86", "cmpxchg16b") => Some(LLVMFeature::new("cx16")),
240        ("x86", "lahfsahf") => Some(LLVMFeature::new("sahf")),
241        ("aarch64", "rcpc2") => Some(LLVMFeature::new("rcpc-immo")),
242        ("aarch64", "dpb") => Some(LLVMFeature::new("ccpp")),
243        ("aarch64", "dpb2") => Some(LLVMFeature::new("ccdp")),
244        ("aarch64", "frintts") => Some(LLVMFeature::new("fptoint")),
245        ("aarch64", "fcma") => Some(LLVMFeature::new("complxnum")),
246        ("aarch64", "pmuv3") => Some(LLVMFeature::new("perfmon")),
247        ("aarch64", "paca") => Some(LLVMFeature::new("pauth")),
248        ("aarch64", "pacg") => Some(LLVMFeature::new("pauth")),
249        // Before LLVM 20 those two features were packaged together as b16b16
250        ("aarch64", "sve-b16b16") if get_version().0 < 20 => Some(LLVMFeature::new("b16b16")),
251        ("aarch64", "sme-b16b16") if get_version().0 < 20 => Some(LLVMFeature::new("b16b16")),
252        ("aarch64", "flagm2") => Some(LLVMFeature::new("altnzcv")),
253        // Rust ties fp and neon together.
254        ("aarch64", "neon") => Some(LLVMFeature::with_dependencies(
255            "neon",
256            smallvec![TargetFeatureFoldStrength::Both("fp-armv8")],
257        )),
258        // In LLVM neon implicitly enables fp, but we manually enable
259        // neon when a feature only implicitly enables fp
260        ("aarch64", "fhm") => Some(LLVMFeature::new("fp16fml")),
261        ("aarch64", "fp16") => Some(LLVMFeature::new("fullfp16")),
262        // Filter out features that are not supported by the current LLVM version
263        ("aarch64", "fpmr") => None, // only existed in 18
264        ("arm", "fp16") => Some(LLVMFeature::new("fullfp16")),
265        // NVPTX targets added in LLVM 20
266        ("nvptx64", "sm_100") if get_version().0 < 20 => None,
267        ("nvptx64", "sm_100a") if get_version().0 < 20 => None,
268        ("nvptx64", "sm_101") if get_version().0 < 20 => None,
269        ("nvptx64", "sm_101a") if get_version().0 < 20 => None,
270        ("nvptx64", "sm_120") if get_version().0 < 20 => None,
271        ("nvptx64", "sm_120a") if get_version().0 < 20 => None,
272        ("nvptx64", "ptx86") if get_version().0 < 20 => None,
273        ("nvptx64", "ptx87") if get_version().0 < 20 => None,
274        // Filter out features that are not supported by the current LLVM version
275        ("loongarch64", "div32" | "lam-bh" | "lamcas" | "ld-seq-sa" | "scq")
276            if get_version().0 < 20 =>
277        {
278            None
279        }
280        // Filter out features that are not supported by the current LLVM version
281        ("riscv32" | "riscv64", "zacas") if get_version().0 < 20 => None,
282        (
283            "s390x",
284            "message-security-assist-extension12"
285            | "concurrent-functions"
286            | "miscellaneous-extensions-4"
287            | "vector-enhancements-3"
288            | "vector-packed-decimal-enhancement-3",
289        ) if get_version().0 < 20 => None,
290        // Enable the evex512 target feature if an avx512 target feature is enabled.
291        ("x86", s) if s.starts_with("avx512") => Some(LLVMFeature::with_dependencies(
292            s,
293            smallvec![TargetFeatureFoldStrength::EnableOnly("evex512")],
294        )),
295        // Support for `wide-arithmetic` will first land in LLVM 20 as part of
296        // llvm/llvm-project#111598
297        ("wasm32" | "wasm64", "wide-arithmetic") if get_version() < (20, 0, 0) => None,
298        ("sparc", "leoncasa") => Some(LLVMFeature::new("hasleoncasa")),
299        // In LLVM 19, there is no `v8plus` feature and `v9` means "SPARC-V9 instruction available and SPARC-V8+ ABI used".
300        // https://github.com/llvm/llvm-project/blob/llvmorg-19.1.0/llvm/lib/Target/Sparc/MCTargetDesc/SparcELFObjectWriter.cpp#L27-L28
301        // Before LLVM 19, there was no `v8plus` feature and `v9` means "SPARC-V9 instruction available".
302        // https://github.com/llvm/llvm-project/blob/llvmorg-18.1.0/llvm/lib/Target/Sparc/MCTargetDesc/SparcELFObjectWriter.cpp#L26
303        ("sparc", "v8plus") if get_version().0 == 19 => Some(LLVMFeature::new("v9")),
304        ("powerpc", "power8-crypto") => Some(LLVMFeature::new("crypto")),
305        // These new `amx` variants and `movrs` were introduced in LLVM20
306        ("x86", "amx-avx512" | "amx-fp8" | "amx-movrs" | "amx-tf32" | "amx-transpose")
307            if get_version().0 < 20 =>
308        {
309            None
310        }
311        ("x86", "movrs") if get_version().0 < 20 => None,
312        ("x86", "avx10.1") => Some(LLVMFeature::new("avx10.1-512")),
313        ("x86", "avx10.2") if get_version().0 < 20 => None,
314        ("x86", "avx10.2") if get_version().0 >= 20 => Some(LLVMFeature::new("avx10.2-512")),
315        ("x86", "apxf") => Some(LLVMFeature::with_dependencies(
316            "egpr",
317            smallvec![
318                TargetFeatureFoldStrength::Both("push2pop2"),
319                TargetFeatureFoldStrength::Both("ppx"),
320                TargetFeatureFoldStrength::Both("ndd"),
321                TargetFeatureFoldStrength::Both("ccmp"),
322                TargetFeatureFoldStrength::Both("cf"),
323                TargetFeatureFoldStrength::Both("nf"),
324                TargetFeatureFoldStrength::Both("zu"),
325            ],
326        )),
327        (_, s) => Some(LLVMFeature::new(s)),
328    }
329}

331/// Used to generate cfg variables and apply features.
332/// Must express features in the way Rust understands them.
333///
334/// We do not have to worry about RUSTC_SPECIFIC_FEATURES here, those are handled outside codegen.
335pub(crate) fn target_config(sess: &Session) -> TargetConfig {
336    let target_machine = create_informational_target_machine(sess, true);
337
338    let (unstable_target_features, target_features) = cfg_target_feature(sess, |feature| {
339        // This closure determines whether the target CPU has the feature according to LLVM. We do
340        // *not* consider the `-Ctarget-feature`s here, as that will be handled later in
341        // `cfg_target_feature`.
342        if let Some(feat) = to_llvm_features(sess, feature) {
343            // All the LLVM features this expands to must be enabled.
344            for llvm_feature in feat {
345                let cstr = SmallCStr::new(llvm_feature);
346                // `LLVMRustHasFeature` is moderately expensive. On targets with many
347                // features (e.g. x86) these calls take a non-trivial fraction of runtime
348                // when compiling very small programs.
349                if !unsafe { llvm::LLVMRustHasFeature(target_machine.raw(), cstr.as_ptr()) } {
350                    return false;
351                }
352            }
353            true
354        } else {
355            false
356        }
357    });
358
359    let mut cfg = TargetConfig {
360        target_features,
361        unstable_target_features,
362        has_reliable_f16: true,
363        has_reliable_f16_math: true,
364        has_reliable_f128: true,
365        has_reliable_f128_math: true,
366    };
367
368    update_target_reliable_float_cfg(sess, &mut cfg);
369    cfg
370}

372/// Determine whether or not experimental float types are reliable based on known bugs.
373fn update_target_reliable_float_cfg(sess: &Session, cfg: &mut TargetConfig) {
374    let target_arch = sess.target.arch.as_ref();
375    let target_os = sess.target.options.os.as_ref();
376    let target_env = sess.target.options.env.as_ref();
377    let target_abi = sess.target.options.abi.as_ref();
378    let target_pointer_width = sess.target.pointer_width;
379    let version = get_version();
380    let lt_20_1_1 = version < (20, 1, 1);
381    let lt_21_0_0 = version < (21, 0, 0);
382
383    cfg.has_reliable_f16 = match (target_arch, target_os) {
384        // LLVM crash without neon <https://github.com/llvm/llvm-project/issues/129394> (fixed in llvm20)
385        ("aarch64", _)
386            if !cfg.target_features.iter().any(|f| f.as_str() == "neon") && lt_20_1_1 =>
387        {
388            false
389        }
390        // Unsupported <https://github.com/llvm/llvm-project/issues/94434>
391        ("arm64ec", _) => false,
392        // Selection failure <https://github.com/llvm/llvm-project/issues/50374> (fixed in llvm21)
393        ("s390x", _) if lt_21_0_0 => false,
394        // MinGW ABI bugs <https://gcc.gnu.org/bugzilla/show_bug.cgi?id=115054>
395        ("x86_64", "windows") if target_env == "gnu" && target_abi != "llvm" => false,
396        // Infinite recursion <https://github.com/llvm/llvm-project/issues/97981>
397        ("csky", _) => false,
398        ("hexagon", _) if lt_21_0_0 => false, // (fixed in llvm21)
399        ("powerpc" | "powerpc64", _) => false,
400        ("sparc" | "sparc64", _) => false,
401        ("wasm32" | "wasm64", _) => false,
402        // `f16` support only requires that symbols converting to and from `f32` are available. We
403        // provide these in `compiler-builtins`, so `f16` should be available on all platforms that
404        // do not have other ABI issues or LLVM crashes.
405        _ => true,
406    };
407
408    cfg.has_reliable_f128 = match (target_arch, target_os) {
409        // Unsupported <https://github.com/llvm/llvm-project/issues/94434>
410        ("arm64ec", _) => false,
411        // Selection bug <https://github.com/llvm/llvm-project/issues/96432> (fixed in llvm20)
412        ("mips64" | "mips64r6", _) if lt_20_1_1 => false,
413        // Selection bug <https://github.com/llvm/llvm-project/issues/95471>. This issue is closed
414        // but basic math still does not work.
415        ("nvptx64", _) => false,
416        // Unsupported https://github.com/llvm/llvm-project/issues/121122
417        ("amdgpu", _) => false,
418        // ABI bugs <https://github.com/rust-lang/rust/issues/125109> et al. (full
419        // list at <https://github.com/rust-lang/rust/issues/116909>)
420        ("powerpc" | "powerpc64", _) => false,
421        // ABI unsupported  <https://github.com/llvm/llvm-project/issues/41838>
422        ("sparc", _) => false,
423        // Stack alignment bug <https://github.com/llvm/llvm-project/issues/77401>. NB: tests may
424        // not fail if our compiler-builtins is linked. (fixed in llvm21)
425        ("x86", _) if lt_21_0_0 => false,
426        // MinGW ABI bugs <https://gcc.gnu.org/bugzilla/show_bug.cgi?id=115054>
427        ("x86_64", "windows") if target_env == "gnu" && target_abi != "llvm" => false,
428        // There are no known problems on other platforms, so the only requirement is that symbols
429        // are available. `compiler-builtins` provides all symbols required for core `f128`
430        // support, so this should work for everything else.
431        _ => true,
432    };
433
434    // Assume that working `f16` means working `f16` math for most platforms, since
435    // operations just go through `f32`.
436    cfg.has_reliable_f16_math = cfg.has_reliable_f16;
437
438    cfg.has_reliable_f128_math = match (target_arch, target_os) {
439        // LLVM lowers `fp128` math to `long double` symbols even on platforms where
440        // `long double` is not IEEE binary128. See
441        // <https://github.com/llvm/llvm-project/issues/44744>.
442        //
443        // This rules out anything that doesn't have `long double` = `binary128`; <= 32 bits
444        // (ld is `f64`), anything other than Linux (Windows and MacOS use `f64`), and `x86`
445        // (ld is 80-bit extended precision).
446        //
447        // musl does not implement the symbols required for f128 math at all.
448        _ if target_env == "musl" => false,
449        ("x86_64", _) => false,
450        (_, "linux") if target_pointer_width == 64 => true,
451        _ => false,
452    } && cfg.has_reliable_f128;
453}

474fn llvm_target_features(tm: &llvm::TargetMachine) -> Vec<(&str, &str)> {
475    let len = unsafe { llvm::LLVMRustGetTargetFeaturesCount(tm) };
476    let mut ret = Vec::with_capacity(len);
477    for i in 0..len {
478        unsafe {
479            let mut feature = ptr::null();
480            let mut desc = ptr::null();
481            llvm::LLVMRustGetTargetFeature(tm, i, &mut feature, &mut desc);
482            if feature.is_null() || desc.is_null() {
483                bug!("LLVM returned a `null` target feature string");
484            }
485            let feature = CStr::from_ptr(feature).to_str().unwrap_or_else(|e| {
486                bug!("LLVM returned a non-utf8 feature string: {}", e);
487            });
488            let desc = CStr::from_ptr(desc).to_str().unwrap_or_else(|e| {
489                bug!("LLVM returned a non-utf8 feature string: {}", e);
490            });
491            ret.push((feature, desc));
492        }
493    }
494    ret
495}

497pub(crate) fn print(req: &PrintRequest, out: &mut String, sess: &Session) {
498    require_inited();
499    let tm = create_informational_target_machine(sess, false);
500    match req.kind {
501        PrintKind::TargetCPUs => print_target_cpus(sess, tm.raw(), out),
502        PrintKind::TargetFeatures => print_target_features(sess, tm.raw(), out),
503        _ => bug!("rustc_codegen_llvm can't handle print request: {:?}", req),
504    }
505}

507fn print_target_cpus(sess: &Session, tm: &llvm::TargetMachine, out: &mut String) {
508    let cpu_names = llvm::build_string(|s| unsafe {
509        llvm::LLVMRustPrintTargetCPUs(&tm, s);
510    })
511    .unwrap();
512
513    struct Cpu<'a> {
514        cpu_name: &'a str,
515        remark: String,
516    }
517    // Compare CPU against current target to label the default.
518    let target_cpu = handle_native(&sess.target.cpu);
519    let make_remark = |cpu_name| {
520        if cpu_name == target_cpu {
521            // FIXME(#132514): This prints the LLVM target string, which can be
522            // different from the Rust target string. Is that intended?
523            let target = &sess.target.llvm_target;
524            format!(
525                " - This is the default target CPU for the current build target (currently {target})."
526            )
527        } else {
528            "".to_owned()
529        }
530    };
531    let mut cpus = cpu_names
532        .lines()
533        .map(|cpu_name| Cpu { cpu_name, remark: make_remark(cpu_name) })
534        .collect::<VecDeque<_>>();
535
536    // Only print the "native" entry when host and target are the same arch,
537    // since otherwise it could be wrong or misleading.
538    if sess.host.arch == sess.target.arch {
539        let host = get_host_cpu_name();
540        cpus.push_front(Cpu {
541            cpu_name: "native",
542            remark: format!(" - Select the CPU of the current host (currently {host})."),
543        });
544    }
545
546    let max_name_width = cpus.iter().map(|cpu| cpu.cpu_name.len()).max().unwrap_or(0);
547    writeln!(out, "Available CPUs for this target:").unwrap();
548    for Cpu { cpu_name, remark } in cpus {
549        // Only pad the CPU name if there's a remark to print after it.
550        let width = if remark.is_empty() { 0 } else { max_name_width };
551        writeln!(out, "    {cpu_name:<width$}{remark}").unwrap();
552    }
553}

555fn print_target_features(sess: &Session, tm: &llvm::TargetMachine, out: &mut String) {
556    let mut llvm_target_features = llvm_target_features(tm);
557    let mut known_llvm_target_features = FxHashSet::<&'static str>::default();
558    let mut rustc_target_features = sess
559        .target
560        .rust_target_features()
561        .iter()
562        .filter_map(|(feature, gate, _implied)| {
563            if !gate.in_cfg() {
564                // Only list (experimentally) supported features.
565                return None;
566            }
567            // LLVM asserts that these are sorted. LLVM and Rust both use byte comparison for these
568            // strings.
569            let llvm_feature = to_llvm_features(sess, *feature)?.llvm_feature_name;
570            let desc =
571                match llvm_target_features.binary_search_by_key(&llvm_feature, |(f, _d)| f).ok() {
572                    Some(index) => {
573                        known_llvm_target_features.insert(llvm_feature);
574                        llvm_target_features[index].1
575                    }
576                    None => "",
577                };
578
579            Some((*feature, desc))
580        })
581        .collect::<Vec<_>>();
582
583    // Since we add this at the end ...
584    rustc_target_features.extend_from_slice(&[(
585        "crt-static",
586        "Enables C Run-time Libraries to be statically linked",
587    )]);
588    // ... we need to sort the list again.
589    rustc_target_features.sort();
590
591    llvm_target_features.retain(|(f, _d)| !known_llvm_target_features.contains(f));
592
593    let max_feature_len = llvm_target_features
594        .iter()
595        .chain(rustc_target_features.iter())
596        .map(|(feature, _desc)| feature.len())
597        .max()
598        .unwrap_or(0);
599
600    writeln!(out, "Features supported by rustc for this target:").unwrap();
601    for (feature, desc) in &rustc_target_features {
602        writeln!(out, "    {feature:max_feature_len$} - {desc}.").unwrap();
603    }
604    writeln!(out, "\nCode-generation features supported by LLVM for this target:").unwrap();
605    for (feature, desc) in &llvm_target_features {
606        writeln!(out, "    {feature:max_feature_len$} - {desc}.").unwrap();
607    }
608    if llvm_target_features.is_empty() {
609        writeln!(out, "    Target features listing is not supported by this LLVM version.")
610            .unwrap();
611    }
612    writeln!(out, "\nUse +feature to enable a feature, or -feature to disable it.").unwrap();
613    writeln!(out, "For example, rustc -C target-cpu=mycpu -C target-feature=+feature1,-feature2\n")
614        .unwrap();
615    writeln!(out, "Code-generation features cannot be used in cfg or #[target_feature],").unwrap();
616    writeln!(out, "and may be renamed or removed in a future version of LLVM or rustc.\n").unwrap();
617}

619/// Returns the host CPU name, according to LLVM.
620fn get_host_cpu_name() -> &'static str {
621    let mut len = 0;
622    // SAFETY: The underlying C++ global function returns a `StringRef` that
623    // isn't tied to any particular backing buffer, so it must be 'static.
624    let slice: &'static [u8] = unsafe {
625        let ptr = llvm::LLVMRustGetHostCPUName(&mut len);
626        assert!(!ptr.is_null());
627        slice::from_raw_parts(ptr, len)
628    };
629    str::from_utf8(slice).expect("host CPU name should be UTF-8")
630}
631
632/// If the given string is `"native"`, returns the host CPU name according to
633/// LLVM. Otherwise, the string is returned as-is.
634fn handle_native(cpu_name: &str) -> &str {
635    match cpu_name {
636        "native" => get_host_cpu_name(),
637        _ => cpu_name,
638    }
639}

641pub(crate) fn target_cpu(sess: &Session) -> &str {
642    let cpu_name = sess.opts.cg.target_cpu.as_deref().unwrap_or_else(|| &sess.target.cpu);
643    handle_native(cpu_name)
644}

646/// The target features for compiler flags other than `-Ctarget-features`.
647fn llvm_features_by_flags(sess: &Session, features: &mut Vec<String>) {
648    target_features::retpoline_features_by_flags(sess, features);
649
650    // -Zfixed-x18
651    if sess.opts.unstable_opts.fixed_x18 {
652        if sess.target.arch != "aarch64" {
653            sess.dcx().emit_fatal(errors::FixedX18InvalidArch { arch: &sess.target.arch });
654        } else {
655            features.push("+reserve-x18".into());
656        }
657    }

660/// The list of LLVM features computed from CLI flags (`-Ctarget-cpu`, `-Ctarget-feature`,
661/// `--target` and similar).
662pub(crate) fn global_llvm_features(
663    sess: &Session,
664    diagnostics: bool,
665    only_base_features: bool,
666) -> Vec<String> {
667    // Features that come earlier are overridden by conflicting features later in the string.
668    // Typically we'll want more explicit settings to override the implicit ones, so:
669    //
670    // * Features from -Ctarget-cpu=*; are overridden by [^1]
671    // * Features implied by --target; are overridden by
672    // * Features from -Ctarget-feature; are overridden by
673    // * function specific features.
674    //
675    // [^1]: target-cpu=native is handled here, other target-cpu values are handled implicitly
676    // through LLVM TargetMachine implementation.
677    //
678    // FIXME(nagisa): it isn't clear what's the best interaction between features implied by
679    // `-Ctarget-cpu` and `--target` are. On one hand, you'd expect CLI arguments to always
680    // override anything that's implicit, so e.g. when there's no `--target` flag, features implied
681    // the host target are overridden by `-Ctarget-cpu=*`. On the other hand, what about when both
682    // `--target` and `-Ctarget-cpu=*` are specified? Both then imply some target features and both
683    // flags are specified by the user on the CLI. It isn't as clear-cut which order of precedence
684    // should be taken in cases like these.
685    let mut features = vec![];
686
687    // -Ctarget-cpu=native
688    match sess.opts.cg.target_cpu {
689        Some(ref s) if s == "native" => {
690            // We have already figured out the actual CPU name with `LLVMRustGetHostCPUName` and set
691            // that for LLVM, so the features implied by that CPU name will be available everywhere.
692            // However, that is not sufficient: e.g. `skylake` alone is not sufficient to tell if
693            // some of the instructions are available or not. So we have to also explicitly ask for
694            // the exact set of features available on the host, and enable all of them.
695            let features_string = unsafe {
696                let ptr = llvm::LLVMGetHostCPUFeatures();
697                let features_string = if !ptr.is_null() {
698                    CStr::from_ptr(ptr)
699                        .to_str()
700                        .unwrap_or_else(|e| {
701                            bug!("LLVM returned a non-utf8 features string: {}", e);
702                        })
703                        .to_owned()
704                } else {
705                    bug!("could not allocate host CPU features, LLVM returned a `null` string");
706                };
707
708                llvm::LLVMDisposeMessage(ptr);
709
710                features_string
711            };
712            features.extend(features_string.split(',').map(String::from));
713        }
714        Some(_) | None => {}
715    };
716
717    // Features implied by an implicit or explicit `--target`.
718    features.extend(
719        sess.target
720            .features
721            .split(',')
722            .filter(|v| !v.is_empty())
723            // Drop +v8plus feature introduced in LLVM 20.
724            // (Hard-coded target features do not go through `to_llvm_feature` since they already
725            // are LLVM feature names, hence we need a special case here.)
726            .filter(|v| *v != "+v8plus" || get_version() >= (20, 0, 0))
727            .map(String::from),
728    );
729
730    if wants_wasm_eh(sess) && sess.panic_strategy() == PanicStrategy::Unwind {
731        features.push("+exception-handling".into());
732    }
733
734    // -Ctarget-features
735    if !only_base_features {
736        target_features::flag_to_backend_features(
737            sess,
738            diagnostics,
739            |feature| {
740                to_llvm_features(sess, feature)
741                    .map(|f| SmallVec::<[&str; 2]>::from_iter(f.into_iter()))
742                    .unwrap_or_default()
743            },
744            |feature, enable| {
745                let enable_disable = if enable { '+' } else { '-' };
746                // We run through `to_llvm_features` when
747                // passing requests down to LLVM. This means that all in-language
748                // features also work on the command line instead of having two
749                // different names when the LLVM name and the Rust name differ.
750                let Some(llvm_feature) = to_llvm_features(sess, feature) else { return };
751
752                features.extend(
753                    std::iter::once(format!(
754                        "{}{}",
755                        enable_disable, llvm_feature.llvm_feature_name
756                    ))
757                    .chain(llvm_feature.dependencies.into_iter().filter_map(
758                        move |feat| match (enable, feat) {
759                            (_, TargetFeatureFoldStrength::Both(f))
760                            | (true, TargetFeatureFoldStrength::EnableOnly(f)) => {
761                                Some(format!("{enable_disable}{f}"))
762                            }
763                            _ => None,
764                        },
765                    )),
766                )
767            },
768        );
769    }
770
771    // We add this in the "base target" so that these show up in `sess.unstable_target_features`.
772    llvm_features_by_flags(sess, &mut features);
773
774    features
775}

777pub(crate) fn tune_cpu(sess: &Session) -> Option<&str> {
778    let name = sess.opts.unstable_opts.tune_cpu.as_ref()?;
779    Some(handle_native(name))
