27pub(crate) fn const_alloc_to_llvm<'ll>(
28    cx: &CodegenCx<'ll, '_>,
29    alloc: &Allocation,
30    is_static: bool,
31) -> &'ll Value {
32    // We expect that callers of const_alloc_to_llvm will instead directly codegen a pointer or
33    // integer for any &ZST where the ZST is a constant (i.e. not a static). We should never be
34    // producing empty LLVM allocations as they're just adding noise to binaries and forcing less
35    // optimal codegen.
36    //
37    // Statics have a guaranteed meaningful address so it's less clear that we want to do
38    // something like this; it's also harder.
39    if !is_static {
40        assert!(alloc.len() != 0);
41    }
42    let mut llvals = Vec::with_capacity(alloc.provenance().ptrs().len() + 1);
43    let dl = cx.data_layout();
44    let pointer_size = dl.pointer_size();
45    let pointer_size_bytes = pointer_size.bytes() as usize;
46
47    // Note: this function may call `inspect_with_uninit_and_ptr_outside_interpreter`, so `range`
48    // must be within the bounds of `alloc` and not contain or overlap a pointer provenance.
49    fn append_chunks_of_init_and_uninit_bytes<'ll, 'a, 'b>(
50        llvals: &mut Vec<&'ll Value>,
51        cx: &'a CodegenCx<'ll, 'b>,
52        alloc: &'a Allocation,
53        range: Range<usize>,
54    ) {
55        let chunks = alloc.init_mask().range_as_init_chunks(range.clone().into());
56
57        let chunk_to_llval = move |chunk| match chunk {
58            InitChunk::Init(range) => {
59                let range = (range.start.bytes() as usize)..(range.end.bytes() as usize);
60                let bytes = alloc.inspect_with_uninit_and_ptr_outside_interpreter(range);
61                cx.const_bytes(bytes)
62            }
63            InitChunk::Uninit(range) => {
64                let len = range.end.bytes() - range.start.bytes();
65                cx.const_undef(cx.type_array(cx.type_i8(), len))
66            }
67        };
68
69        // Generating partially-uninit consts is limited to small numbers of chunks,
70        // to avoid the cost of generating large complex const expressions.
71        // For example, `[(u32, u8); 1024 * 1024]` contains uninit padding in each element, and
72        // would result in `{ [5 x i8] zeroinitializer, [3 x i8] undef, ...repeat 1M times... }`.
73        let max = cx.sess().opts.unstable_opts.uninit_const_chunk_threshold;
74        let allow_uninit_chunks = chunks.clone().take(max.saturating_add(1)).count() <= max;
75
76        if allow_uninit_chunks {
77            llvals.extend(chunks.map(chunk_to_llval));
78        } else {
79            // If this allocation contains any uninit bytes, codegen as if it was initialized
80            // (using some arbitrary value for uninit bytes).
81            let bytes = alloc.inspect_with_uninit_and_ptr_outside_interpreter(range);
82            llvals.push(cx.const_bytes(bytes));
83        }
84    }
85
86    let mut next_offset = 0;
87    for &(offset, prov) in alloc.provenance().ptrs().iter() {
88        let offset = offset.bytes();
89        assert_eq!(offset as usize as u64, offset);
90        let offset = offset as usize;
91        if offset > next_offset {
92            // This `inspect` is okay since we have checked that there is no provenance, it
93            // is within the bounds of the allocation, and it doesn't affect interpreter execution
94            // (we inspect the result after interpreter execution).
95            append_chunks_of_init_and_uninit_bytes(&mut llvals, cx, alloc, next_offset..offset);
96        }
97        let ptr_offset = read_target_uint(
98            dl.endian,
99            // This `inspect` is okay since it is within the bounds of the allocation, it doesn't
100            // affect interpreter execution (we inspect the result after interpreter execution),
101            // and we properly interpret the provenance as a relocation pointer offset.
102            alloc.inspect_with_uninit_and_ptr_outside_interpreter(
103                offset..(offset + pointer_size_bytes),
104            ),
105        )
106        .expect("const_alloc_to_llvm: could not read relocation pointer")
107            as u64;
108
109        let address_space = cx.tcx.global_alloc(prov.alloc_id()).address_space(cx);
110
111        llvals.push(cx.scalar_to_backend(
112            InterpScalar::from_pointer(Pointer::new(prov, Size::from_bytes(ptr_offset)), &cx.tcx),
113            Scalar::Initialized {
114                value: Primitive::Pointer(address_space),
115                valid_range: WrappingRange::full(pointer_size),
116            },
117            cx.type_ptr_ext(address_space),
118        ));
119        next_offset = offset + pointer_size_bytes;
120    }
121    if alloc.len() >= next_offset {
122        let range = next_offset..alloc.len();
123        // This `inspect` is okay since we have check that it is after all provenance, it is
124        // within the bounds of the allocation, and it doesn't affect interpreter execution (we
125        // inspect the result after interpreter execution).
126        append_chunks_of_init_and_uninit_bytes(&mut llvals, cx, alloc, range);
127    }
128
129    // Avoid wrapping in a struct if there is only a single value. This ensures
130    // that LLVM is able to perform the string merging optimization if the constant
131    // is a valid C string. LLVM only considers bare arrays for this optimization,
132    // not arrays wrapped in a struct. LLVM handles this at:
133    // https://github.com/rust-lang/llvm-project/blob/acaea3d2bb8f351b740db7ebce7d7a40b9e21488/llvm/lib/Target/TargetLoweringObjectFile.cpp#L249-L280
134    if let &[data] = &*llvals { data } else { cx.const_struct(&llvals, true) }
135}

137fn codegen_static_initializer<'ll, 'tcx>(
138    cx: &CodegenCx<'ll, 'tcx>,

142    Ok((const_alloc_to_llvm(cx, alloc.inner(), /*static*/ true), alloc))
143}

145fn set_global_alignment<'ll>(cx: &CodegenCx<'ll, '_>, gv: &'ll Value, mut align: Align) {
146    // The target may require greater alignment for globals than the type does.
147    // Note: GCC and Clang also allow `__attribute__((aligned))` on variables,
148    // which can force it to be smaller. Rust doesn't support this yet.
149    if let Some(min_global) = cx.sess().target.min_global_align {
150        align = Ord::max(align, min_global);
151    }
152    llvm::set_alignment(gv, align);
153}

155fn check_and_apply_linkage<'ll, 'tcx>(
156    cx: &CodegenCx<'ll, 'tcx>,
157    attrs: &CodegenFnAttrs,
158    llty: &'ll Type,
159    sym: &str,
160    def_id: DefId,
161) -> &'ll Value {
162    if let Some(linkage) = attrs.import_linkage {
163        debug!("get_static: sym={} linkage={:?}", sym, linkage);
164
165        // Declare a symbol `foo`. If `foo` is an extern_weak symbol, we declare
166        // an extern_weak function, otherwise a global with the desired linkage.
167        let g1 = if matches!(attrs.import_linkage, Some(Linkage::ExternalWeak)) {
168            // An `extern_weak` function is represented as an `Option<unsafe extern ...>`,
169            // we extract the function signature and declare it as an extern_weak function
170            // instead of an extern_weak i8.
171            let instance = Instance::mono(cx.tcx, def_id);
172            if let ty::Adt(struct_def, args) = instance.ty(cx.tcx, cx.typing_env()).kind()
173                && cx.tcx.is_lang_item(struct_def.did(), LangItem::Option)
174                && let ty::FnPtr(sig, header) = args.type_at(0).kind()
175            {
176                let fn_sig = sig.with(*header);
177
178                let fn_abi = cx.fn_abi_of_fn_ptr(fn_sig, ty::List::empty());
179                cx.declare_fn(sym, &fn_abi, None)
180            } else {
181                cx.declare_global(sym, cx.type_i8())
182            }
183        } else {
184            cx.declare_global(sym, cx.type_i8())
185        };
186        llvm::set_linkage(g1, base::linkage_to_llvm(linkage));
187
188        // Declare an internal global `extern_with_linkage_foo` which
189        // is initialized with the address of `foo`. If `foo` is
190        // discarded during linking (for example, if `foo` has weak
191        // linkage and there are no definitions), then
192        // `extern_with_linkage_foo` will instead be initialized to
193        // zero.
194        let real_name =
195            format!("_rust_extern_with_linkage_{:016x}_{sym}", cx.tcx.stable_crate_id(LOCAL_CRATE));
196        let g2 = cx.define_global(&real_name, llty).unwrap_or_else(|| {
197            cx.sess().dcx().emit_fatal(SymbolAlreadyDefined {
198                span: cx.tcx.def_span(def_id),
199                symbol_name: sym,
200            })
201        });
202        llvm::set_linkage(g2, llvm::Linkage::InternalLinkage);
203        llvm::set_initializer(g2, g1);
204        g2
205    } else if cx.tcx.sess.target.arch == "x86"
206        && common::is_mingw_gnu_toolchain(&cx.tcx.sess.target)
207        && let Some(dllimport) = crate::common::get_dllimport(cx.tcx, def_id, sym)
208    {
209        cx.declare_global(&common::i686_decorated_name(dllimport, true, true, false), llty)
210    } else {
211        // Generate an external declaration.
212        // FIXME(nagisa): investigate whether it can be changed into define_global
213        cx.declare_global(sym, llty)
214    }
215}

226    /// Create a global variable.
227    ///
228    /// The returned global variable is a pointer in the default address space for globals.
229    /// Fails if a symbol with the given name already exists.
230    pub(crate) fn static_addr_of_mut(
231        &self,
232        cv: &'ll Value,
233        align: Align,
234        kind: Option<&str>,
235    ) -> &'ll Value {
236        let gv = match kind {
237            Some(kind) if !self.tcx.sess.fewer_names() => {
238                let name = self.generate_local_symbol_name(kind);
239                let gv = self.define_global(&name, self.val_ty(cv)).unwrap_or_else(|| {
240                    bug!("symbol `{}` is already defined", name);
241                });
242                llvm::set_linkage(gv, llvm::Linkage::PrivateLinkage);
243                gv
244            }
245            _ => self.define_private_global(self.val_ty(cv)),
246        };
247        llvm::set_initializer(gv, cv);
248        set_global_alignment(self, gv, align);
249        llvm::set_unnamed_address(gv, llvm::UnnamedAddr::Global);
250        gv
251    }

253    /// Create a global constant.

255    /// The returned global variable is a pointer in the default address space for globals.
256    pub(crate) fn static_addr_of_impl(
257        &self,
258        cv: &'ll Value,
259        align: Align,
260        kind: Option<&str>,
261    ) -> &'ll Value {
262        if let Some(&gv) = self.const_globals.borrow().get(&cv) {
263            unsafe {
264                // Upgrade the alignment in cases where the same constant is used with different
265                // alignment requirements
266                let llalign = align.bytes() as u32;
267                if llalign > llvm::LLVMGetAlignment(gv) {
268                    llvm::LLVMSetAlignment(gv, llalign);
269                }
270            }
271            return gv;
272        }
273        let gv = self.static_addr_of_mut(cv, align, kind);
274        llvm::set_global_constant(gv, true);
275
276        self.const_globals.borrow_mut().insert(cv, gv);
277        gv
278    }

281    pub(crate) fn get_static(&self, def_id: DefId) -> &'ll Value {
282        let instance = Instance::mono(self.tcx, def_id);
283        trace!(?instance);
284
285        let DefKind::Static { nested, .. } = self.tcx.def_kind(def_id) else { bug!() };
286        // Nested statics do not have a type, so pick a dummy type and let `codegen_static` figure
287        // out the llvm type from the actual evaluated initializer.
288        let llty = if nested {
289            self.type_i8()
290        } else {
291            let ty = instance.ty(self.tcx, self.typing_env());
292            trace!(?ty);
293            self.layout_of(ty).llvm_type(self)
294        };
295        self.get_static_inner(def_id, llty)
296    }

298    #[instrument(level = "debug", skip(self, llty))]
299    fn get_static_inner(&self, def_id: DefId, llty: &'ll Type) -> &'ll Value {
300        let instance = Instance::mono(self.tcx, def_id);
301        if let Some(&g) = self.instances.borrow().get(&instance) {
302            trace!("used cached value");
303            return g;
304        }
305
306        let defined_in_current_codegen_unit =
307            self.codegen_unit.items().contains_key(&MonoItem::Static(def_id));
308        assert!(
309            !defined_in_current_codegen_unit,
310            "consts::get_static() should always hit the cache for \
311                 statics defined in the same CGU, but did not for `{def_id:?}`"
312        );
313
314        let sym = self.tcx.symbol_name(instance).name;
315        let fn_attrs = self.tcx.codegen_fn_attrs(def_id);
316
317        debug!(?sym, ?fn_attrs);
318
319        let g = if def_id.is_local() && !self.tcx.is_foreign_item(def_id) {
320            if let Some(g) = self.get_declared_value(sym) {
321                if self.val_ty(g) != self.type_ptr() {
322                    span_bug!(self.tcx.def_span(def_id), "Conflicting types for static");
323                }
324            }
325
326            let g = self.declare_global(sym, llty);
327
328            if !self.tcx.is_reachable_non_generic(def_id) {
329                llvm::set_visibility(g, llvm::Visibility::Hidden);
330            }
331
332            g
333        } else {
334            check_and_apply_linkage(self, fn_attrs, llty, sym, def_id)
335        };
336
337        // Thread-local statics in some other crate need to *always* be linked
338        // against in a thread-local fashion, so we need to be sure to apply the
339        // thread-local attribute locally if it was present remotely. If we
340        // don't do this then linker errors can be generated where the linker
341        // complains that one object files has a thread local version of the
342        // symbol and another one doesn't.
343        if fn_attrs.flags.contains(CodegenFnAttrFlags::THREAD_LOCAL) {
344            llvm::set_thread_local_mode(g, self.tls_model);
345        }
346
347        let dso_local = self.assume_dso_local(g, true);
348
349        if !def_id.is_local() {
350            let needs_dll_storage_attr = self.use_dll_storage_attrs
351                && !self.tcx.is_foreign_item(def_id)
352                // Local definitions can never be imported, so we must not apply
353                // the DLLImport annotation.
354                && !dso_local
355                // Linker plugin ThinLTO doesn't create the self-dllimport Rust uses for rlibs
356                // as the code generation happens out of process. Instead we assume static linkage
357                // and disallow dynamic linking when linker plugin based LTO is enabled.
358                // Regular in-process ThinLTO doesn't need this workaround.
359                && !self.tcx.sess.opts.cg.linker_plugin_lto.enabled();
360
361            // If this assertion triggers, there's something wrong with commandline
362            // argument validation.
363            assert!(
364                !(self.tcx.sess.opts.cg.linker_plugin_lto.enabled()
365                    && self.tcx.sess.target.is_like_windows
366                    && self.tcx.sess.opts.cg.prefer_dynamic)
367            );
368
369            if needs_dll_storage_attr {
370                // This item is external but not foreign, i.e., it originates from an external Rust
371                // crate. Since we don't know whether this crate will be linked dynamically or
372                // statically in the final application, we always mark such symbols as 'dllimport'.
373                // If final linkage happens to be static, we rely on compiler-emitted __imp_ stubs
374                // to make things work.
375                //
376                // However, in some scenarios we defer emission of statics to downstream
377                // crates, so there are cases where a static with an upstream DefId
378                // is actually present in the current crate. We can find out via the
379                // is_codegened_item query.
380                if !self.tcx.is_codegened_item(def_id) {
381                    llvm::set_dllimport_storage_class(g);
382                }
383            }
384        }
385
386        if self.use_dll_storage_attrs
387            && let Some(library) = self.tcx.native_library(def_id)
388            && library.kind.is_dllimport()
389        {
390            // For foreign (native) libs we know the exact storage type to use.
391            llvm::set_dllimport_storage_class(g);
392        }
393
394        self.instances.borrow_mut().insert(instance, g);
395        g
396    }

398    fn codegen_static_item(&mut self, def_id: DefId) {
399        assert!(
400            llvm::LLVMGetInitializer(
401                self.instances.borrow().get(&Instance::mono(self.tcx, def_id)).unwrap()
402            )
403            .is_none()
404        );
405        let attrs = self.tcx.codegen_fn_attrs(def_id);
406
407        let Ok((v, alloc)) = codegen_static_initializer(self, def_id) else {
408            // Error has already been reported
409            return;
410        };
411        let alloc = alloc.inner();
412
413        let val_llty = self.val_ty(v);
414
415        let g = self.get_static_inner(def_id, val_llty);
416        let llty = self.get_type_of_global(g);
417
418        let g = if val_llty == llty {
419            g
420        } else {
421            // codegen_static_initializer creates the global value just from the
422            // `Allocation` data by generating one big struct value that is just
423            // all the bytes and pointers after each other. This will almost never
424            // match the type that the static was declared with. Unfortunately
425            // we can't just LLVMConstBitCast our way out of it because that has very
426            // specific rules on what can be cast. So instead of adding a new way to
427            // generate static initializers that match the static's type, we picked
428            // the easier option and retroactively change the type of the static item itself.
429            let name = String::from_utf8(llvm::get_value_name(g))
430                .expect("we declare our statics with a utf8-valid name");
431            llvm::set_value_name(g, b"");
432
433            let linkage = llvm::get_linkage(g);
434            let visibility = llvm::get_visibility(g);
435
436            let new_g = self.declare_global(&name, val_llty);
437
438            llvm::set_linkage(new_g, linkage);
439            llvm::set_visibility(new_g, visibility);
440
441            // The old global has had its name removed but is returned by
442            // get_static since it is in the instance cache. Provide an
443            // alternative lookup that points to the new global so that
444            // global_asm! can compute the correct mangled symbol name
445            // for the global.
446            self.renamed_statics.borrow_mut().insert(def_id, new_g);
447
448            // To avoid breaking any invariants, we leave around the old
449            // global for the moment; we'll replace all references to it
450            // with the new global later. (See base::codegen_backend.)
451            self.statics_to_rauw.borrow_mut().push((g, new_g));
452            new_g
453        };
454        set_global_alignment(self, g, alloc.align);
455        llvm::set_initializer(g, v);
456
457        self.assume_dso_local(g, true);
458
459        // Forward the allocation's mutability (picked by the const interner) to LLVM.
460        if alloc.mutability.is_not() {
461            llvm::set_global_constant(g, true);
462        }
463
464        debuginfo::build_global_var_di_node(self, def_id, g);
465
466        if attrs.flags.contains(CodegenFnAttrFlags::THREAD_LOCAL) {
467            llvm::set_thread_local_mode(g, self.tls_model);
468        }
469
470        // Wasm statics with custom link sections get special treatment as they
471        // go into custom sections of the wasm executable. The exception to this
472        // is the `.init_array` section which are treated specially by the wasm linker.
473        if self.tcx.sess.target.is_like_wasm
474            && attrs
475                .link_section
476                .map(|link_section| !link_section.as_str().starts_with(".init_array"))
477                .unwrap_or(true)
478        {
479            if let Some(section) = attrs.link_section {
480                let section = self.create_metadata(section.as_str().as_bytes());
481                assert!(alloc.provenance().ptrs().is_empty());
482
483                // The `inspect` method is okay here because we checked for provenance, and
484                // because we are doing this access to inspect the final interpreter state (not
485                // as part of the interpreter execution).
486                let bytes = alloc.inspect_with_uninit_and_ptr_outside_interpreter(0..alloc.len());
487                let alloc = self.create_metadata(bytes);
488                let data = [section, alloc];
489                let meta =
490                    unsafe { llvm::LLVMMDNodeInContext2(self.llcx, data.as_ptr(), data.len()) };
491                let val = self.get_metadata_value(meta);
492                unsafe {
493                    llvm::LLVMAddNamedMetadataOperand(
494                        self.llmod,
495                        c"wasm.custom_sections".as_ptr(),
496                        val,
497                    )
498                };
499            }
500        } else {
501            base::set_link_section(g, attrs);
502        }
503
504        base::set_variable_sanitizer_attrs(g, attrs);
505
506        if attrs.flags.contains(CodegenFnAttrFlags::USED_COMPILER) {
507            // `USED` and `USED_LINKER` can't be used together.
508            assert!(!attrs.flags.contains(CodegenFnAttrFlags::USED_LINKER));
509
510            // The semantics of #[used] in Rust only require the symbol to make it into the
511            // object file. It is explicitly allowed for the linker to strip the symbol if it
512            // is dead, which means we are allowed to use `llvm.compiler.used` instead of
513            // `llvm.used` here.
514            //
515            // Additionally, https://reviews.llvm.org/D97448 in LLVM 13 started emitting unique
516            // sections with SHF_GNU_RETAIN flag for llvm.used symbols, which may trigger bugs
517            // in the handling of `.init_array` (the static constructor list) in versions of
518            // the gold linker (prior to the one released with binutils 2.36).
519            //
520            // That said, we only ever emit these when `#[used(compiler)]` is explicitly
521            // requested. This is to avoid similar breakage on other targets, in particular
522            // MachO targets have *their* static constructor lists broken if `llvm.compiler.used`
523            // is emitted rather than `llvm.used`. However, that check happens when assigning
524            // the `CodegenFnAttrFlags` in the `codegen_fn_attrs` query, so we don't need to
525            // take care of it here.
526            self.add_compiler_used_global(g);
527        }
528        if attrs.flags.contains(CodegenFnAttrFlags::USED_LINKER) {
529            // `USED` and `USED_LINKER` can't be used together.
530            assert!(!attrs.flags.contains(CodegenFnAttrFlags::USED_COMPILER));
531
532            self.add_used_global(g);
533        }
534    }

541    /// Add a global value to a list to be stored in the `llvm.compiler.used` variable,
542    /// an array of ptr.
543    pub(crate) fn add_compiler_used_global(&mut self, global: &'ll Value) {
544        self.compiler_used_statics.push(global);
545    }
546}

548impl<'ll> StaticCodegenMethods for CodegenCx<'ll, '_> {
549    /// Get a pointer to a global variable.
550    ///
551    /// The pointer will always be in the default address space. If global variables default to a
552    /// different address space, an addrspacecast is inserted.
553    fn static_addr_of(&self, cv: &'ll Value, align: Align, kind: Option<&str>) -> &'ll Value {
554        let gv = self.static_addr_of_impl(cv, align, kind);
555        // static_addr_of_impl returns the bare global variable, which might not be in the default
556        // address space. Cast to the default address space if necessary.
557        self.const_pointercast(gv, self.type_ptr())
558    }
559
560    fn codegen_static(&mut self, def_id: DefId) {
561        self.codegen_static_item(def_id)
562    }
