1use rustc_ast::attr;

7fn proc_macro_decls_static(tcx: TyCtxt<'_>, (): ()) -> Option<LocalDefId> {
8    let mut decls = None;
9
10    for id in tcx.hir_free_items() {
11        let attrs = tcx.hir_attrs(id.hir_id());
12        if attr::contains_name(attrs, sym::rustc_proc_macro_decls) {
13            decls = Some(id.owner_id.def_id);
14        }
15    }
16
17    decls
18}
19
20pub(crate) fn provide(providers: &mut Providers) {
21    *providers = Providers { proc_macro_decls_static, ..*providers };
