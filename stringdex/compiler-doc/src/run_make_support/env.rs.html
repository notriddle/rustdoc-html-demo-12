3#[track_caller]
4#[must_use]
5pub fn env_var(name: &str) -> String {
6    match std::env::var(name) {
7        Ok(v) => v,
8        Err(err) => panic!("failed to retrieve environment variable {name:?}: {err:?}"),
9    }
10}

15    match std::env::var_os(name) {
16        Some(v) => v,
17        None => panic!("failed to retrieve environment variable {name:?}"),
18    }

21/// Check if staged `rustc`-under-test was built with debug assertions.
22#[track_caller]

24pub fn rustc_debug_assertions_enabled() -> bool {

37/// A wrapper around [`std::env::set_current_dir`] which includes the directory
38/// path in the panic message.
39#[track_caller]
40pub fn set_current_dir<P: AsRef<std::path::Path>>(dir: P) {
41    std::env::set_current_dir(dir.as_ref())
42        .expect(&format!("could not set current directory to \"{}\"", dir.as_ref().display()));
