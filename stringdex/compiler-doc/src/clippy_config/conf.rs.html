102struct ConfError {
103    message: String,

108impl ConfError {
109    fn from_toml(file: &SourceFile, error: &toml::de::Error) -> Self {
110        let span = error.span().unwrap_or(0..file.source_len.0 as usize);
111        Self::spanned(file, error.message(), None, span)
112    }
113
114    fn spanned(
115        file: &SourceFile,
116        message: impl Into<String>,
117        suggestion: Option<Suggestion>,
118        span: Range<usize>,
119    ) -> Self {
120        Self {
121            message: message.into(),
122            suggestion,
123            span: span_from_toml_range(file, span),
124        }
125    }

128// Remove code tags and code behind '# 's, as they are not needed for the lint docs and --explain
129pub fn sanitize_explanation(raw_docs: &str) -> String {
130    // Remove tags and hidden code:
131    let mut explanation = String::with_capacity(128);
132    let mut in_code = false;
133    for line in raw_docs.lines() {
134        let line = line.strip_prefix(' ').unwrap_or(line);
135
136        if let Some(lang) = line.strip_prefix("```") {
137            let tag = lang.split_once(',').map_or(lang, |(left, _)| left);
138            if !in_code && matches!(tag, "" | "rust" | "ignore" | "should_panic" | "no_run" | "compile_fail") {
139                explanation += "```rust\n";
140            } else {
141                explanation += line;
142                explanation.push('\n');
143            }
144            in_code = !in_code;
145        } else if !(in_code && line.starts_with("# ")) {
146            explanation += line;
147            explanation.push('\n');
148        }
149    }
150
151    explanation
152}

174macro_rules! deserialize {
175    ($map:expr, $ty:ty, $errors:expr, $file:expr) => {{
176        let raw_value = $map.next_value::<toml::Spanned<toml::Value>>()?;
177        let value_span = raw_value.span();
178        let value = match <$ty>::deserialize(raw_value.into_inner()) {
179            Err(e) => {
180                $errors.push(ConfError::spanned(
181                    $file,
182                    e.to_string().replace('\n', " ").trim(),
183                    None,
184                    value_span,
185                ));
186                continue;
187            },
188            Ok(value) => value,
189        };
190        (value, value_span)
191    }};
192
193    ($map:expr, $ty:ty, $errors:expr, $file:expr, $replacements_allowed:expr) => {{
194        let array = $map.next_value::<Vec<toml::Spanned<toml::Value>>>()?;
195        let mut disallowed_paths_span = Range {
196            start: usize::MAX,
197            end: usize::MIN,
198        };
199        let mut disallowed_paths = Vec::new();
200        for raw_value in array {
201            let value_span = raw_value.span();
202            let mut disallowed_path = match DisallowedPath::<$replacements_allowed>::deserialize(raw_value.into_inner())
203            {
204                Err(e) => {
205                    $errors.push(ConfError::spanned(
206                        $file,
207                        e.to_string().replace('\n', " ").trim(),
208                        None,
209                        value_span,
210                    ));
211                    continue;
212                },
213                Ok(disallowed_path) => disallowed_path,
214            };
215            disallowed_paths_span = union(&disallowed_paths_span, &value_span);
216            disallowed_path.set_span(span_from_toml_range($file, value_span));
217            disallowed_paths.push(disallowed_path);
218        }
219        (disallowed_paths, disallowed_paths_span)
220    }};
221}

223macro_rules! define_Conf {
224    ($(
225        $(#[doc = $doc:literal])+
226        $(#[conf_deprecated($dep:literal, $new_conf:ident)])?
227        $(#[default_text = $default_text:expr])?
228        $(#[disallowed_paths_allow_replacements = $replacements_allowed:expr])?
229        $(#[lints($($for_lints:ident),* $(,)?)])?
230        $name:ident: $ty:ty = $default:expr,
231    )*) => {
232        /// Clippy lint configuration
233        pub struct Conf {
234            $($(#[cfg_attr(doc, doc = $doc)])+ pub $name: $ty,)*
235        }
236
237        mod defaults {
238            use super::*;
239            $(pub fn $name() -> $ty { $default })*
240        }
241
242        impl Default for Conf {
243            fn default() -> Self {
244                Self { $($name: defaults::$name(),)* }
245            }
246        }
247
248        #[derive(Deserialize)]
249        #[serde(field_identifier, rename_all = "kebab-case")]
250        #[allow(non_camel_case_types)]
251        enum Field { $($name,)* third_party, }
252
253        struct ConfVisitor<'a>(&'a SourceFile);
254
255        impl<'de> Visitor<'de> for ConfVisitor<'_> {
256            type Value = TryConf;
257
258            fn expecting(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
259                formatter.write_str("Conf")
260            }
261
262            fn visit_map<V>(self, mut map: V) -> Result<Self::Value, V::Error> where V: MapAccess<'de> {
263                let mut value_spans = HashMap::new();
264                let mut errors = Vec::new();
265                let mut warnings = Vec::new();
266
267                // Declare a local variable for each field available to a configuration file.
268                $(let mut $name = None;)*
269
270                // could get `Field` here directly, but get `String` first for diagnostics
271                while let Some(name) = map.next_key::<toml::Spanned<String>>()? {
272                    let field = match Field::deserialize(name.get_ref().as_str().into_deserializer()) {
273                        Err(e) => {
274                            let e: FieldError = e;
275                            errors.push(ConfError::spanned(self.0, e.error, e.suggestion, name.span()));
276                            continue;
277                        }
278                        Ok(field) => field
279                    };
280
281                    match field {
282                        $(Field::$name => {
283                            // Is this a deprecated field, i.e., is `$dep` set? If so, push a warning.
284                            $(warnings.push(ConfError::spanned(self.0, format!("deprecated field `{}`. {}", name.get_ref(), $dep), None, name.span()));)?
285                            let (value, value_span) =
286                                deserialize!(map, $ty, errors, self.0 $(, $replacements_allowed)?);
287                            // Was this field set previously?
288                            if $name.is_some() {
289                                errors.push(ConfError::spanned(self.0, format!("duplicate field `{}`", name.get_ref()), None, name.span()));
290                                continue;
291                            }
292                            $name = Some(value);
293                            value_spans.insert(name.get_ref().as_str().to_string(), value_span);
294                            // If this is a deprecated field, was the new field (`$new_conf`) set previously?
295                            // Note that `$new_conf` is one of the defined `$name`s.
296                            $(match $new_conf {
297                                Some(_) => errors.push(ConfError::spanned(self.0, concat!(
298                                    "duplicate field `", stringify!($new_conf),
299                                    "` (provided as `", stringify!($name), "`)"
300                                ), None, name.span())),
301                                None => $new_conf = $name.clone(),
302                            })?
303                        })*
304                        // ignore contents of the third_party key
305                        Field::third_party => drop(map.next_value::<IgnoredAny>())
306                    }
307                }
308                let conf = Conf { $($name: $name.unwrap_or_else(defaults::$name),)* };
309                Ok(TryConf { conf, value_spans, errors, warnings })
310            }
311        }
312
313        pub fn get_configuration_metadata() -> Vec<ClippyConfiguration> {
314            vec![$(
315                ClippyConfiguration {
316                    name: stringify!($name).replace('_', "-"),
317                    default: default_text!(defaults::$name() $(, $default_text)?),
318                    lints: &[$($(stringify!($for_lints)),*)?],
319                    doc: concat!($($doc, '\n',)*),
320                    deprecation_reason: wrap_option!($($dep)?)
321                },
322            )*]
323        }
324    };
325}

327fn union(x: &Range<usize>, y: &Range<usize>) -> Range<usize> {
328    Range {
329        start: cmp::min(x.start, y.start),
330        end: cmp::max(x.end, y.end),
331    }

334fn span_from_toml_range(file: &SourceFile, span: Range<usize>) -> Span {
335    Span::new(
336        file.start_pos + BytePos::from_usize(span.start),
337        file.start_pos + BytePos::from_usize(span.end),
338        SyntaxContext::root(),
339        None,
340    )

883/// Search for the configuration file.
884///
885/// # Errors
886///
887/// Returns any unexpected filesystem error encountered when searching for the config file
888pub fn lookup_conf_file() -> io::Result<(Option<PathBuf>, Vec<String>)> {
889    /// Possible filename to search for.
890    const CONFIG_FILE_NAMES: [&str; 2] = [".clippy.toml", "clippy.toml"];
891
892    // Start looking for a config file in CLIPPY_CONF_DIR, or failing that, CARGO_MANIFEST_DIR.
893    // If neither of those exist, use ".". (Update documentation if this priority changes)
894    let mut current = env::var_os("CLIPPY_CONF_DIR")
895        .or_else(|| env::var_os("CARGO_MANIFEST_DIR"))
896        .map_or_else(|| PathBuf::from("."), PathBuf::from)
897        .canonicalize()?;
898
899    let mut found_config: Option<PathBuf> = None;
900    let mut warnings = vec![];
901
902    loop {
903        for config_file_name in &CONFIG_FILE_NAMES {
904            if let Ok(config_file) = current.join(config_file_name).canonicalize() {
905                match fs::metadata(&config_file) {
906                    Err(e) if e.kind() == io::ErrorKind::NotFound => {},
907                    Err(e) => return Err(e),
908                    Ok(md) if md.is_dir() => {},
909                    Ok(_) => {
910                        // warn if we happen to find two config files #8323
911                        if let Some(ref found_config) = found_config {
912                            warnings.push(format!(
913                                "using config file `{}`, `{}` will be ignored",
914                                found_config.display(),
915                                config_file.display()
916                            ));
917                        } else {
918                            found_config = Some(config_file);
919                        }
920                    },
921                }
922            }
923        }
924
925        if found_config.is_some() {
926            return Ok((found_config, warnings));
927        }
928
929        // If the current directory has no parent, we're done searching.
930        if !current.pop() {
931            return Ok((None, warnings));
932        }
933    }
934}

936fn deserialize(file: &SourceFile) -> TryConf {
937    match toml::de::Deserializer::new(file.src.as_ref().unwrap()).deserialize_map(ConfVisitor(file)) {
938        Ok(mut conf) => {
939            extend_vec_if_indicator_present(&mut conf.conf.disallowed_names, DEFAULT_DISALLOWED_NAMES);
940            extend_vec_if_indicator_present(&mut conf.conf.allowed_prefixes, DEFAULT_ALLOWED_PREFIXES);
941            extend_vec_if_indicator_present(
942                &mut conf.conf.allow_renamed_params_for,
943                DEFAULT_ALLOWED_TRAITS_WITH_RENAMED_PARAMS,
944            );
945
946            // Confirms that the user has not accidentally configured ordering requirements for groups that
947            // aren't configured.
948            if let SourceItemOrderingWithinModuleItemGroupings::Custom(groupings) =
949                &conf.conf.module_items_ordered_within_groupings
950            {
951                for grouping in groupings {
952                    if !conf.conf.module_item_order_groupings.is_grouping(grouping) {
953                        // Since this isn't fixable by rustfix, don't emit a `Suggestion`. This just adds some useful
954                        // info for the user instead.
955
956                        let names = conf.conf.module_item_order_groupings.grouping_names();
957                        let suggestion = suggest_candidate(grouping, names.iter().map(String::as_str))
958                            .map(|s| format!(" perhaps you meant `{s}`?"))
959                            .unwrap_or_default();
960                        let names = names.iter().map(|s| format!("`{s}`")).join(", ");
961                        let message = format!(
962                            "unknown ordering group: `{grouping}` was not specified in `module-items-ordered-within-groupings`,{suggestion} expected one of: {names}"
963                        );
964
965                        let span = conf
966                            .value_spans
967                            .get("module_item_order_groupings")
968                            .cloned()
969                            .unwrap_or_default();
970                        conf.errors.push(ConfError::spanned(file, message, None, span));
971                    }
972                }
973            }
974
975            // TODO: THIS SHOULD BE TESTED, this comment will be gone soon
976            if conf.conf.allowed_idents_below_min_chars.iter().any(|e| e == "..") {
977                conf.conf
978                    .allowed_idents_below_min_chars
979                    .extend(DEFAULT_ALLOWED_IDENTS_BELOW_MIN_CHARS.iter().map(ToString::to_string));
980            }
981            if conf.conf.doc_valid_idents.iter().any(|e| e == "..") {
982                conf.conf
983                    .doc_valid_idents
984                    .extend(DEFAULT_DOC_VALID_IDENTS.iter().map(ToString::to_string));
985            }
986
987            conf
988        },
989        Err(e) => TryConf::from_toml_error(file, &e),
990    }
991}

993fn extend_vec_if_indicator_present(vec: &mut Vec<String>, default: &[&str]) {
994    if vec.contains(&"..".to_string()) {
995        vec.extend(default.iter().map(ToString::to_string));
996    }

999impl Conf {
1000    pub fn read(sess: &Session, path: &io::Result<(Option<PathBuf>, Vec<String>)>) -> &'static Conf {
1001        static CONF: OnceLock<Conf> = OnceLock::new();
1002        CONF.get_or_init(|| Conf::read_inner(sess, path))
1003    }
1004
1005    fn read_inner(sess: &Session, path: &io::Result<(Option<PathBuf>, Vec<String>)>) -> Conf {
1006        match path {
1007            Ok((_, warnings)) => {
1008                for warning in warnings {
1009                    sess.dcx().warn(warning.clone());
1010                }
1011            },
1012            Err(error) => {
1013                sess.dcx()
1014                    .err(format!("error finding Clippy's configuration file: {error}"));
1015            },
1016        }
1017
1018        let TryConf {
1019            mut conf,
1020            value_spans: _,
1021            errors,
1022            warnings,
1023        } = match path {
1024            Ok((Some(path), _)) => match sess.source_map().load_file(path) {
1025                Ok(file) => deserialize(&file),
1026                Err(error) => {
1027                    sess.dcx().err(format!("failed to read `{}`: {error}", path.display()));
1028                    TryConf::default()
1029                },
1030            },
1031            _ => TryConf::default(),
1032        };
1033
1034        conf.msrv.read_cargo(sess);
1035
1036        // all conf errors are non-fatal, we just use the default conf in case of error
1037        for error in errors {
1038            let mut diag = sess.dcx().struct_span_err(
1039                error.span,
1040                format!("error reading Clippy's configuration file: {}", error.message),
1041            );
1042
1043            if let Some(sugg) = error.suggestion {
1044                diag.span_suggestion(error.span, sugg.message, sugg.suggestion, Applicability::MaybeIncorrect);
1045            }
1046
1047            diag.emit();
1048        }
1049
1050        for warning in warnings {
1051            sess.dcx().span_warn(
1052                warning.span,
1053                format!("error reading Clippy's configuration file: {}", warning.message),
1054            );
1055        }
1056
1057        conf
1058    }
1059}

1096        let metadata = get_configuration_metadata();
1097        let deprecated = metadata
1098            .iter()
1099            .filter_map(|conf| {
1100                if conf.deprecation_reason.is_some() {
1101                    Some(conf.name.as_str())
1102                } else {
1103                    None
1104                }
1105            })
1106            .collect::<Vec<_>>();
1107
1108        let mut expected = expected
1109            .iter()
1110            .copied()
1111            .filter(|name| !deprecated.contains(name))
1112            .collect::<Vec<_>>();
1113        expected.sort_unstable();
1114
1115        let (rows, column_widths) = calculate_dimensions(&expected);
1116
1117        let mut msg = format!("unknown field `{field}`, expected one of");
1118        for row in 0..rows {
1119            writeln!(msg).unwrap();
1120            for (column, column_width) in column_widths.iter().copied().enumerate() {
1121                let index = column * rows + row;
1122                let field = expected.get(index).copied().unwrap_or_default();
1123                write!(msg, "{:SEPARATOR_WIDTH$}{field:column_width$}", " ").unwrap();
1124            }
1125        }
1126
1127        let suggestion = suggest_candidate(field, expected).map(|suggestion| Suggestion {
1128            message: "perhaps you meant",
1129            suggestion,
1130        });
1131
1132        Self { error: msg, suggestion }
1133    }
1134}
1135
1136fn calculate_dimensions(fields: &[&str]) -> (usize, Vec<usize>) {
1137    let columns = env::var("CLIPPY_TERMINAL_WIDTH")
1138        .ok()
1139        .and_then(|s| <usize as FromStr>::from_str(&s).ok())
1140        .map_or(1, |terminal_width| {
1141            let max_field_width = fields.iter().map(|field| field.len()).max().unwrap();
1142            cmp::max(1, terminal_width / (SEPARATOR_WIDTH + max_field_width))
1143        });
1144
1145    let rows = fields.len().div_ceil(columns);
1146
1147    let column_widths = (0..columns)
1148        .map(|column| {
1149            if column < columns - 1 {
1150                (0..rows)
1151                    .map(|row| {
1152                        let index = column * rows + row;
1153                        let field = fields.get(index).copied().unwrap_or_default();
1154                        field.len()
1155                    })
1156                    .max()
1157                    .unwrap()
1158            } else {
1159                // Avoid adding extra space to the last column.
1160                0
1161            }
1162        })
1163        .collect::<Vec<_>>();
1164
1165    (rows, column_widths)
1166}
1167
1168/// Given a user-provided value that couldn't be matched to a known option, finds the most likely
1169/// candidate among candidates that the user might have meant.
1170fn suggest_candidate<'a, I>(value: &str, candidates: I) -> Option<&'a str>
1171where
1172    I: IntoIterator<Item = &'a str>,
1173{
1174    candidates
1175        .into_iter()
1176        .filter_map(|expected| {
1177            let dist = edit_distance(value, expected, 4)?;
1178            Some((dist, expected))
1179        })
1180        .min_by_key(|&(dist, _)| dist)
1181        .map(|(_, suggestion)| suggestion)
1182}
