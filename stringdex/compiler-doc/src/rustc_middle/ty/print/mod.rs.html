24/// Interface for outputting user-facing "type-system entities"
25/// (paths, types, lifetimes, constants, etc.) as a side-effect
26/// (e.g. formatting, like `PrettyPrinter` implementors do) or by
27/// constructing some alternative representation (e.g. an AST),
28/// which the associated types allow passing through the methods.
29///
30/// For pretty-printing/formatting in particular, see `PrettyPrinter`.
31//
32// FIXME(eddyb) find a better name; this is more general than "printing".
33pub trait Printer<'tcx>: Sized {
34    fn tcx<'a>(&'a self) -> TyCtxt<'tcx>;
35
36    fn print_def_path(
37        &mut self,
38        def_id: DefId,
39        args: &'tcx [GenericArg<'tcx>],
40    ) -> Result<(), PrintError> {
41        self.default_print_def_path(def_id, args)
42    }
43
44    fn print_impl_path(
45        &mut self,
46        impl_def_id: DefId,
47        args: &'tcx [GenericArg<'tcx>],
48    ) -> Result<(), PrintError> {
49        let tcx = self.tcx();
50        let self_ty = tcx.type_of(impl_def_id);
51        let impl_trait_ref = tcx.impl_trait_ref(impl_def_id);
52        let (self_ty, impl_trait_ref) = if tcx.generics_of(impl_def_id).count() <= args.len() {
53            (
54                self_ty.instantiate(tcx, args),
55                impl_trait_ref.map(|impl_trait_ref| impl_trait_ref.instantiate(tcx, args)),
56            )
57        } else {
58            // We are probably printing a nested item inside of an impl.
59            // Use the identity substitutions for the impl.
60            (
61                self_ty.instantiate_identity(),
62                impl_trait_ref.map(|impl_trait_ref| impl_trait_ref.instantiate_identity()),
63            )
64        };
65
66        self.default_print_impl_path(impl_def_id, self_ty, impl_trait_ref)
67    }
68
69    fn print_region(&mut self, region: ty::Region<'tcx>) -> Result<(), PrintError>;
70
71    fn print_type(&mut self, ty: Ty<'tcx>) -> Result<(), PrintError>;
72
73    fn print_dyn_existential(
74        &mut self,
75        predicates: &'tcx ty::List<ty::PolyExistentialPredicate<'tcx>>,
76    ) -> Result<(), PrintError>;
77
78    fn print_const(&mut self, ct: ty::Const<'tcx>) -> Result<(), PrintError>;
79
80    fn path_crate(&mut self, cnum: CrateNum) -> Result<(), PrintError>;
81
82    fn path_qualified(
83        &mut self,
84        self_ty: Ty<'tcx>,
85        trait_ref: Option<ty::TraitRef<'tcx>>,
86    ) -> Result<(), PrintError>;
87
88    fn path_append_impl(
89        &mut self,
90        print_prefix: impl FnOnce(&mut Self) -> Result<(), PrintError>,
91        self_ty: Ty<'tcx>,
92        trait_ref: Option<ty::TraitRef<'tcx>>,
93    ) -> Result<(), PrintError>;
94
95    fn path_append(
96        &mut self,
97        print_prefix: impl FnOnce(&mut Self) -> Result<(), PrintError>,
98        disambiguated_data: &DisambiguatedDefPathData,
99    ) -> Result<(), PrintError>;
100
101    fn path_generic_args(
102        &mut self,
103        print_prefix: impl FnOnce(&mut Self) -> Result<(), PrintError>,
104        args: &[GenericArg<'tcx>],
105    ) -> Result<(), PrintError>;
106
107    fn should_truncate(&mut self) -> bool {
108        false
109    }
110
111    // Defaults (should not be overridden):
112
113    #[instrument(skip(self), level = "debug")]
114    fn default_print_def_path(
115        &mut self,
116        def_id: DefId,
117        args: &'tcx [GenericArg<'tcx>],
118    ) -> Result<(), PrintError> {
119        let key = self.tcx().def_key(def_id);
120        debug!(?key);
121
122        match key.disambiguated_data.data {
123            DefPathData::CrateRoot => {
124                assert!(key.parent.is_none());
125                self.path_crate(def_id.krate)
126            }
127
128            DefPathData::Impl => self.print_impl_path(def_id, args),
129
130            _ => {
131                let parent_def_id = DefId { index: key.parent.unwrap(), ..def_id };
132
133                let mut parent_args = args;
134                let mut trait_qualify_parent = false;
135                if !args.is_empty() {
136                    let generics = self.tcx().generics_of(def_id);
137                    parent_args = &args[..generics.parent_count.min(args.len())];
138
139                    match key.disambiguated_data.data {
140                        DefPathData::Closure => {
141                            // We need to additionally print the `kind` field of a coroutine if
142                            // it is desugared from a coroutine-closure.
143                            if let Some(hir::CoroutineKind::Desugared(
144                                _,
145                                hir::CoroutineSource::Closure,
146                            )) = self.tcx().coroutine_kind(def_id)
147                                && args.len() > parent_args.len()
148                            {
149                                return self.path_generic_args(
150                                    |p| p.print_def_path(def_id, parent_args),
151                                    &args[..parent_args.len() + 1][..1],
152                                );
153                            } else {
154                                // Closures' own generics are only captures, don't print them.
155                            }
156                        }
157                        DefPathData::SyntheticCoroutineBody => {
158                            // Synthetic coroutine bodies have no distinct generics, since like
159                            // closures they're all just internal state of the coroutine.
160                        }
161                        // This covers both `DefKind::AnonConst` and `DefKind::InlineConst`.
162                        // Anon consts doesn't have their own generics, and inline consts' own
163                        // generics are their inferred types, so don't print them.
164                        DefPathData::AnonConst => {}
165
166                        // If we have any generic arguments to print, we do that
167                        // on top of the same path, but without its own generics.
168                        _ => {
169                            if !generics.is_own_empty() && args.len() >= generics.count() {
170                                let args = generics.own_args_no_defaults(self.tcx(), args);
171                                return self.path_generic_args(
172                                    |p| p.print_def_path(def_id, parent_args),
173                                    args,
174                                );
175                            }
176                        }
177                    }
178
179                    // FIXME(eddyb) try to move this into the parent's printing
180                    // logic, instead of doing it when printing the child.
181                    trait_qualify_parent = generics.has_self
182                        && generics.parent == Some(parent_def_id)
183                        && parent_args.len() == generics.parent_count
184                        && self.tcx().generics_of(parent_def_id).parent_count == 0;
185                }
186
187                self.path_append(
188                    |p: &mut Self| {
189                        if trait_qualify_parent {
190                            let trait_ref = ty::TraitRef::new(
191                                p.tcx(),
192                                parent_def_id,
193                                parent_args.iter().copied(),
194                            );
195                            p.path_qualified(trait_ref.self_ty(), Some(trait_ref))
196                        } else {
197                            p.print_def_path(parent_def_id, parent_args)
198                        }
199                    },
200                    &key.disambiguated_data,
201                )
202            }
203        }
204    }
205
206    fn default_print_impl_path(
207        &mut self,
208        impl_def_id: DefId,
209        self_ty: Ty<'tcx>,
210        impl_trait_ref: Option<ty::TraitRef<'tcx>>,
211    ) -> Result<(), PrintError> {
212        debug!(
213            "default_print_impl_path: impl_def_id={:?}, self_ty={}, impl_trait_ref={:?}",
214            impl_def_id, self_ty, impl_trait_ref
215        );
216
217        let key = self.tcx().def_key(impl_def_id);
218        let parent_def_id = DefId { index: key.parent.unwrap(), ..impl_def_id };
219
220        // Decide whether to print the parent path for the impl.
221        // Logically, since impls are global, it's never needed, but
222        // users may find it useful. Currently, we omit the parent if
223        // the impl is either in the same module as the self-type or
224        // as the trait.
225        let in_self_mod = match characteristic_def_id_of_type(self_ty) {
226            None => false,
227            Some(ty_def_id) => self.tcx().parent(ty_def_id) == parent_def_id,
228        };
229        let in_trait_mod = match impl_trait_ref {
230            None => false,
231            Some(trait_ref) => self.tcx().parent(trait_ref.def_id) == parent_def_id,
232        };
233
234        if !in_self_mod && !in_trait_mod {
235            // If the impl is not co-located with either self-type or
236            // trait-type, then fallback to a format that identifies
237            // the module more clearly.
238            self.path_append_impl(|p| p.print_def_path(parent_def_id, &[]), self_ty, impl_trait_ref)
239        } else {
240            // Otherwise, try to give a good form that would be valid language
241            // syntax. Preferably using associated item notation.
242            self.path_qualified(self_ty, impl_trait_ref)
243        }
244    }
245}

247/// As a heuristic, when we see an impl, if we see that the
248/// 'self type' is a type defined in the same module as the impl,
249/// we can omit including the path to the impl itself. This
250/// function tries to find a "characteristic `DefId`" for a
251/// type. It's just a heuristic so it makes some questionable
252/// decisions and we may want to adjust it later.
253///
254/// Visited set is needed to avoid full iteration over
255/// deeply nested tuples that have no DefId.
256fn characteristic_def_id_of_type_cached<'a>(
257    ty: Ty<'a>,
258    visited: &mut SsoHashSet<Ty<'a>>,
259) -> Option<DefId> {
260    match *ty.kind() {
261        ty::Adt(adt_def, _) => Some(adt_def.did()),
262
263        ty::Dynamic(data, ..) => data.principal_def_id(),
264
265        ty::Pat(subty, _) | ty::Array(subty, _) | ty::Slice(subty) => {
266            characteristic_def_id_of_type_cached(subty, visited)
267        }
268
269        ty::RawPtr(ty, _) => characteristic_def_id_of_type_cached(ty, visited),
270
271        ty::Ref(_, ty, _) => characteristic_def_id_of_type_cached(ty, visited),
272
273        ty::Tuple(tys) => tys.iter().find_map(|ty| {
274            if visited.insert(ty) {
275                return characteristic_def_id_of_type_cached(ty, visited);
276            }
277            return None;
278        }),
279
280        ty::FnDef(def_id, _)
281        | ty::Closure(def_id, _)
282        | ty::CoroutineClosure(def_id, _)
283        | ty::Coroutine(def_id, _)
284        | ty::CoroutineWitness(def_id, _)
285        | ty::Foreign(def_id) => Some(def_id),
286
287        ty::Bool
288        | ty::Char
289        | ty::Int(_)
290        | ty::Uint(_)
291        | ty::Str
292        | ty::FnPtr(..)
293        | ty::UnsafeBinder(_)
294        | ty::Alias(..)
295        | ty::Placeholder(..)
296        | ty::Param(_)
297        | ty::Infer(_)
298        | ty::Bound(..)
299        | ty::Error(_)
300        | ty::Never
301        | ty::Float(_) => None,
302    }
303}
304pub fn characteristic_def_id_of_type(ty: Ty<'_>) -> Option<DefId> {
305    characteristic_def_id_of_type_cached(ty, &mut SsoHashSet::new())
306}

332// This is only used by query descriptions
333pub fn describe_as_module(def_id: impl Into<LocalDefId>, tcx: TyCtxt<'_>) -> String {
334    let def_id = def_id.into();
335    if def_id.is_top_level_module() {
336        "top-level module".to_string()
337    } else {
338        format!("module `{}`", tcx.def_path_str(def_id))
339    }
340}

342impl<T> rustc_type_ir::ir_print::IrPrint<T> for TyCtxt<'_>
343where
344    T: Copy + for<'a, 'tcx> Lift<TyCtxt<'tcx>, Lifted: Print<'tcx, FmtPrinter<'a, 'tcx>>>,
345{
346    fn print(t: &T, fmt: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
347        ty::tls::with(|tcx| {
348            let mut p = FmtPrinter::new(tcx, Namespace::TypeNS);
349            tcx.lift(*t).expect("could not lift for printing").print(&mut p)?;
350            fmt.write_str(&p.into_buffer())?;
351            Ok(())
352        })
353    }
354
355    fn print_debug(t: &T, fmt: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
356        with_no_trimmed_paths!(Self::print(t, fmt))
357    }
358}

360/// Format instance name that is already known to be too long for rustc.
361/// Show only the first 2 types if it is longer than 32 characters to avoid blasting
362/// the user's terminal with thousands of lines of type-name.
363///
364/// If the type name is longer than before+after, it will be written to a file.
365pub fn shrunk_instance_name<'tcx>(
366    tcx: TyCtxt<'tcx>,
367    instance: ty::Instance<'tcx>,
368) -> (String, Option<PathBuf>) {
369    let s = instance.to_string();
370
371    // Only use the shrunk version if it's really shorter.
372    // This also avoids the case where before and after slices overlap.
373    if s.chars().nth(33).is_some() {
374        let shrunk = format!("{}", ShortInstance(instance, 4));
375        if shrunk == s {
376            return (s, None);
377        }
378
379        let path = tcx.output_filenames(()).temp_path_for_diagnostic("long-type.txt");
380        let written_to_path = std::fs::write(&path, s).ok().map(|_| path);
381
382        (shrunk, written_to_path)
383    } else {
384        (s, None)
385    }
