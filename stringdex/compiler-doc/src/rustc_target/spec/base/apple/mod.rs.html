18pub(crate) enum Arch {
19    Armv7k,
20    Armv7s,
21    Arm64,
22    Arm64e,
23    Arm64_32,
24    I386,
25    I686,
26    X86_64,
27    X86_64h,
28}
29
30impl Arch {
31    fn target_name(self) -> &'static str {
32        match self {
33            Armv7k => "armv7k",
34            Armv7s => "armv7s",
35            Arm64 => "arm64",
36            Arm64e => "arm64e",
37            Arm64_32 => "arm64_32",
38            I386 => "i386",
39            I686 => "i686",
40            X86_64 => "x86_64",
41            X86_64h => "x86_64h",
42        }
43    }
44
45    pub(crate) fn target_arch(self) -> Cow<'static, str> {
46        Cow::Borrowed(match self {
47            Armv7k | Armv7s => "arm",
48            Arm64 | Arm64e | Arm64_32 => "aarch64",
49            I386 | I686 => "x86",
50            X86_64 | X86_64h => "x86_64",
51        })
52    }
53
54    fn target_cpu(self, env: TargetEnv) -> &'static str {
55        match self {
56            Armv7k => "cortex-a8",
57            Armv7s => "swift", // iOS 10 is only supported on iPhone 5 or higher.
58            Arm64 => match env {
59                TargetEnv::Normal => "apple-a7",
60                TargetEnv::Simulator => "apple-a12",
61                TargetEnv::MacCatalyst => "apple-a12",
62            },
63            Arm64e => "apple-a12",
64            Arm64_32 => "apple-s4",
65            // Only macOS 10.12+ is supported, which means
66            // all x86_64/x86 CPUs must be running at least penryn
67            // https://github.com/llvm/llvm-project/blob/01f924d0e37a5deae51df0d77e10a15b63aa0c0f/clang/lib/Driver/ToolChains/Arch/X86.cpp#L79-L82
68            I386 | I686 => "penryn",
69            X86_64 => "penryn",
70            // Note: `core-avx2` is slightly more advanced than `x86_64h`, see
71            // comments (and disabled features) in `x86_64h_apple_darwin` for
72            // details. It is a higher baseline then `penryn` however.
73            X86_64h => "core-avx2",
74        }
75    }
76
77    fn stack_probes(self) -> StackProbeType {
78        match self {
79            Armv7k | Armv7s => StackProbeType::None,
80            Arm64 | Arm64e | Arm64_32 | I386 | I686 | X86_64 | X86_64h => StackProbeType::Inline,
81        }
82    }
83}

102/// Get the base target options, unversioned LLVM target and `target_arch` from the three
103/// things that uniquely identify Rust's Apple targets: The OS, the architecture, and the ABI.
104pub(crate) fn base(
105    os: &'static str,
106    arch: Arch,
107    env: TargetEnv,
108) -> (TargetOptions, StaticCow<str>, StaticCow<str>) {
109    let mut opts = TargetOptions {
110        llvm_floatabi: Some(FloatAbi::Hard),
111        os: os.into(),
112        env: env.target_env().into(),
113        // NOTE: We originally set `cfg(target_abi = "macabi")` / `cfg(target_abi = "sim")`,
114        // before it was discovered that those are actually environments:
115        // https://github.com/rust-lang/rust/issues/133331
116        //
117        // But let's continue setting them for backwards compatibility.
118        // FIXME(madsmtm): Warn about using these in the future.
119        abi: env.target_env().into(),
120        cpu: arch.target_cpu(env).into(),
121        link_env_remove: link_env_remove(os),
122        vendor: "apple".into(),
123        linker_flavor: LinkerFlavor::Darwin(Cc::Yes, Lld::No),
124        // macOS has -dead_strip, which doesn't rely on function_sections
125        function_sections: false,
126        dynamic_linking: true,
127        families: cvs!["unix"],
128        is_like_darwin: true,
129        binary_format: BinaryFormat::MachO,
130        // LLVM notes that macOS 10.11+ and iOS 9+ default
131        // to v4, so we do the same.
132        // https://github.com/llvm/llvm-project/blob/378778a0d10c2f8d5df8ceff81f95b6002984a4b/clang/lib/Driver/ToolChains/Darwin.cpp#L1203
133        default_dwarf_version: 4,
134        frame_pointer: match arch {
135            // clang ignores `-fomit-frame-pointer` for Armv7, it only accepts `-momit-leaf-frame-pointer`
136            Armv7k | Armv7s => FramePointer::Always,
137            // clang supports omitting frame pointers for the rest, but... don't?
138            Arm64 | Arm64e | Arm64_32 => FramePointer::NonLeaf,
139            I386 | I686 | X86_64 | X86_64h => FramePointer::Always,
140        },
141        has_rpath: true,
142        dll_suffix: ".dylib".into(),
143        archive_format: "darwin".into(),
144        // Thread locals became available with iOS 8 and macOS 10.7,
145        // and both are far below our minimum.
146        has_thread_local: true,
147        abi_return_struct_as_int: true,
148        emit_debug_gdb_scripts: false,
149        eh_frame_header: false,
150        stack_probes: arch.stack_probes(),
151
152        debuginfo_kind: DebuginfoKind::DwarfDsym,
153        // The historical default for macOS targets is to run `dsymutil` which
154        // generates a packed version of debuginfo split from the main file.
155        split_debuginfo: SplitDebuginfo::Packed,
156        supported_split_debuginfo: Cow::Borrowed(&[
157            SplitDebuginfo::Packed,
158            SplitDebuginfo::Unpacked,
159            SplitDebuginfo::Off,
160        ]),
161
162        // This environment variable is pretty magical but is intended for
163        // producing deterministic builds. This was first discovered to be used
164        // by the `ar` tool as a way to control whether or not mtime entries in
165        // the archive headers were set to zero or not. It appears that
166        // eventually the linker got updated to do the same thing and now reads
167        // this environment variable too in recent versions.
168        //
169        // For some more info see the commentary on #47086
170        link_env: Cow::Borrowed(&[(Cow::Borrowed("ZERO_AR_DATE"), Cow::Borrowed("1"))]),
171
172        ..Default::default()
173    };
174    if matches!(arch, Arch::I386 | Arch::I686) {
175        // All Apple x86-32 targets have SSE2.
176        opts.rustc_abi = Some(RustcAbi::X86Sse2);
177    }
178    (opts, unversioned_llvm_target(os, arch, env), arch.target_arch())
179}

181/// Generate part of the LLVM target triple.
182///
183/// See `rustc_codegen_ssa::back::versioned_llvm_target` for the full triple passed to LLVM and
184/// Clang.
185fn unversioned_llvm_target(os: &str, arch: Arch, env: TargetEnv) -> StaticCow<str> {
186    let arch = arch.target_name();
187    // Convert to the "canonical" OS name used by LLVM:
188    // https://github.com/llvm/llvm-project/blob/llvmorg-18.1.8/llvm/lib/TargetParser/Triple.cpp#L236-L282
189    let os = match os {
190        "macos" => "macosx",
191        "ios" => "ios",
192        "watchos" => "watchos",
193        "tvos" => "tvos",
194        "visionos" => "xros",
195        _ => unreachable!("tried to get LLVM target OS for non-Apple platform"),
196    };
197    let environment = match env {
198        TargetEnv::Normal => "",
199        TargetEnv::MacCatalyst => "-macabi",
200        TargetEnv::Simulator => "-simulator",
201    };
202    format!("{arch}-apple-{os}{environment}").into()
203}

205fn link_env_remove(os: &'static str) -> StaticCow<[StaticCow<str>]> {
206    // Apple platforms only officially support macOS as a host for any compilation.
207    //
208    // If building for macOS, we go ahead and remove any erroneous environment state
209    // that's only applicable to cross-OS compilation. Always leave anything for the
210    // host OS alone though.
211    if os == "macos" {
212        let mut env_remove = Vec::with_capacity(2);
213        // Remove the `SDKROOT` environment variable if it's clearly set for the wrong platform, which
214        // may occur when we're linking a custom build script while targeting iOS for example.
215        if let Ok(sdkroot) = env::var("SDKROOT") {
216            if sdkroot.contains("iPhoneOS.platform")
217                || sdkroot.contains("iPhoneSimulator.platform")
218                || sdkroot.contains("AppleTVOS.platform")
219                || sdkroot.contains("AppleTVSimulator.platform")
220                || sdkroot.contains("WatchOS.platform")
221                || sdkroot.contains("WatchSimulator.platform")
222                || sdkroot.contains("XROS.platform")
223                || sdkroot.contains("XRSimulator.platform")
224            {
225                env_remove.push("SDKROOT".into())
226            }
227        }
228        // Additionally, `IPHONEOS_DEPLOYMENT_TARGET` must not be set when using the Xcode linker at
229        // "/Applications/Xcode.app/Contents/Developer/Toolchains/XcodeDefault.xctoolchain/usr/bin/ld",
230        // although this is apparently ignored when using the linker at "/usr/bin/ld".
231        env_remove.push("IPHONEOS_DEPLOYMENT_TARGET".into());
232        env_remove.push("TVOS_DEPLOYMENT_TARGET".into());
233        env_remove.push("XROS_DEPLOYMENT_TARGET".into());
234        env_remove.into()
235    } else {
236        // Otherwise if cross-compiling for a different OS/SDK (including Mac Catalyst), remove any part
237        // of the linking environment that's wrong and reversed.
238        cvs!["MACOSX_DEPLOYMENT_TARGET"]
239    }
240}

242/// Deployment target or SDK version.
243///
244/// The size of the numbers in here are limited by Mach-O's `LC_BUILD_VERSION`.
245#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
246pub struct OSVersion {
247    pub major: u16,
248    pub minor: u8,
249    pub patch: u8,
250}
251
252impl FromStr for OSVersion {
253    type Err = ParseIntError;
254
255    /// Parse an OS version triple (SDK version or deployment target).
256    fn from_str(version: &str) -> Result<Self, ParseIntError> {
257        if let Some((major, minor)) = version.split_once('.') {
258            let major = major.parse()?;
259            if let Some((minor, patch)) = minor.split_once('.') {
260                Ok(Self { major, minor: minor.parse()?, patch: patch.parse()? })
261            } else {
262                Ok(Self { major, minor: minor.parse()?, patch: 0 })
263            }
264        } else {
265            Ok(Self { major: version.parse()?, minor: 0, patch: 0 })
266        }
267    }
268}
269
270impl OSVersion {
271    pub fn new(major: u16, minor: u8, patch: u8) -> Self {
272        Self { major, minor, patch }
273    }
274
275    pub fn fmt_pretty(self) -> impl Display {
276        let Self { major, minor, patch } = self;
277        from_fn(move |f| {
278            write!(f, "{major}.{minor}")?;
279            if patch != 0 {
280                write!(f, ".{patch}")?;
281            }
282            Ok(())
283        })
284    }
285
286    pub fn fmt_full(self) -> impl Display {
287        let Self { major, minor, patch } = self;
288        from_fn(move |f| write!(f, "{major}.{minor}.{patch}"))
289    }
290
291    /// Minimum operating system versions currently supported by `rustc`.
292    pub fn os_minimum_deployment_target(os: &str) -> Self {
293        // When bumping a version in here, remember to update the platform-support docs too.
294        //
295        // NOTE: The defaults may change in future `rustc` versions, so if you are looking for the
296        // default deployment target, prefer:
297        // ```
298        // $ rustc --print deployment-target
299        // ```
300        let (major, minor, patch) = match os {
301            "macos" => (10, 12, 0),
302            "ios" => (10, 0, 0),
303            "tvos" => (10, 0, 0),
304            "watchos" => (5, 0, 0),
305            "visionos" => (1, 0, 0),
306            _ => unreachable!("tried to get deployment target for non-Apple platform"),
307        };
308        Self { major, minor, patch }
309    }
310
311    /// The deployment target for the given target.
312    ///
313    /// This is similar to `os_minimum_deployment_target`, except that on certain targets it makes sense
314    /// to raise the minimum OS version.
315    ///
316    /// This matches what LLVM does, see in part:
317    /// <https://github.com/llvm/llvm-project/blob/llvmorg-18.1.8/llvm/lib/TargetParser/Triple.cpp#L1900-L1932>
318    pub fn minimum_deployment_target(target: &Target) -> Self {
319        let (major, minor, patch) = match (&*target.os, &*target.arch, &*target.env) {
320            ("macos", "aarch64", _) => (11, 0, 0),
321            ("ios", "aarch64", "macabi") => (14, 0, 0),
322            ("ios", "aarch64", "sim") => (14, 0, 0),
323            ("ios", _, _) if target.llvm_target.starts_with("arm64e") => (14, 0, 0),
324            // Mac Catalyst defaults to 13.1 in Clang.
325            ("ios", _, "macabi") => (13, 1, 0),
326            ("tvos", "aarch64", "sim") => (14, 0, 0),
327            ("watchos", "aarch64", "sim") => (7, 0, 0),
328            (os, _, _) => return Self::os_minimum_deployment_target(os),
329        };
330        Self { major, minor, patch }
331    }
332}
