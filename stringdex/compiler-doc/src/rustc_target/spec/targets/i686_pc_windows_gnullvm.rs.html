1use crate::spec::{Cc, FramePointer, LinkerFlavor, Lld, RustcAbi, Target, TargetMetadata, base};
2
3pub(crate) fn target() -> Target {
4    let mut base = base::windows_gnullvm::opts();
5    base.rustc_abi = Some(RustcAbi::X86Sse2);
6    base.cpu = "pentium4".into();
7    base.max_atomic_width = Some(64);
8    base.frame_pointer = FramePointer::Always; // Required for backtraces
9    base.linker = Some("i686-w64-mingw32-clang".into());
10
11    // Mark all dynamic libraries and executables as compatible with the larger 4GiB address
12    // space available to x86 Windows binaries on x86_64.
13    base.add_pre_link_args(
14        LinkerFlavor::Gnu(Cc::No, Lld::No),
15        &["-m", "i386pe", "--large-address-aware"],
16    );
17
18    Target {
19        llvm_target: "i686-pc-windows-gnu".into(),
20        metadata: TargetMetadata {
21            description: Some("32-bit x86 MinGW (Windows 10+), LLVM ABI".into()),
22            tier: Some(2),
23            host_tools: Some(false),
24            std: Some(true),
25        },
26        pointer_width: 32,
27        data_layout: "e-m:x-p:32:32-p270:32:32-p271:32:32-p272:64:64-\
28            i64:64-i128:128-f80:32-n8:16:32-a:0:32-S32"
29            .into(),
30        arch: "x86".into(),
31        options: base,
32    }
33}
