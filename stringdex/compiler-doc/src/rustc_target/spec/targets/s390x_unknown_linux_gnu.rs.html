1use rustc_abi::{Align, Endian};
2
3use crate::spec::{SanitizerSet, StackProbeType, Target, TargetMetadata, base};
4
5pub(crate) fn target() -> Target {
6    let mut base = base::linux_gnu::opts();
7    base.endian = Endian::Big;
8    // z10 is the oldest CPU supported by LLVM
9    base.cpu = "z10".into();
10    base.max_atomic_width = Some(128);
11    base.min_global_align = Some(Align::from_bits(16).unwrap());
12    base.stack_probes = StackProbeType::Inline;
13    base.supported_sanitizers =
14        SanitizerSet::ADDRESS | SanitizerSet::LEAK | SanitizerSet::MEMORY | SanitizerSet::THREAD;
15
16    Target {
17        llvm_target: "s390x-unknown-linux-gnu".into(),
18        metadata: TargetMetadata {
19            description: Some("S390x Linux (kernel 3.2, glibc 2.17)".into()),
20            tier: Some(2),
21            host_tools: Some(true),
22            std: Some(true),
23        },
24        pointer_width: 64,
25        data_layout: "E-m:e-i1:8:16-i8:8:16-i64:64-f128:64-v128:64-a:8:16-n32:64".into(),
26        arch: "s390x".into(),
27        options: base,
28    }
29}
