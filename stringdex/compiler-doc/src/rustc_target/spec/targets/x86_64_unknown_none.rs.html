1// Generic x86-64 target for bare-metal code - Floating point disabled
2//
3// Can be used in conjunction with the `target-feature` and
4// `target-cpu` compiler flags to opt-in more hardware-specific
5// features.
6
7use crate::spec::{
8    Cc, CodeModel, LinkerFlavor, Lld, PanicStrategy, RelroLevel, RustcAbi, SanitizerSet,
9    StackProbeType, Target, TargetMetadata, TargetOptions,
10};
11
12pub(crate) fn target() -> Target {
13    let opts = TargetOptions {
14        cpu: "x86-64".into(),
15        plt_by_default: false,
16        max_atomic_width: Some(64),
17        stack_probes: StackProbeType::Inline,
18        position_independent_executables: true,
19        static_position_independent_executables: true,
20        relro_level: RelroLevel::Full,
21        linker_flavor: LinkerFlavor::Gnu(Cc::No, Lld::Yes),
22        linker: Some("rust-lld".into()),
23        rustc_abi: Some(RustcAbi::X86Softfloat),
24        features: "-mmx,-sse,-sse2,-sse3,-ssse3,-sse4.1,-sse4.2,-avx,-avx2,+soft-float".into(),
25        supported_sanitizers: SanitizerSet::KCFI | SanitizerSet::KERNELADDRESS,
26        disable_redzone: true,
27        panic_strategy: PanicStrategy::Abort,
28        code_model: Some(CodeModel::Kernel),
29        ..Default::default()
30    };
31    Target {
32        llvm_target: "x86_64-unknown-none-elf".into(),
33        metadata: TargetMetadata {
34            description: Some("Freestanding/bare-metal x86_64 softfloat".into()),
35            tier: Some(2),
36            host_tools: Some(false),
37            std: Some(false),
38        },
39        pointer_width: 64,
40        data_layout:
41            "e-m:e-p270:32:32-p271:32:32-p272:64:64-i64:64-i128:128-f80:128-n8:16:32:64-S128".into(),
42        arch: "x86_64".into(),
43        options: opts,
44    }
45}
