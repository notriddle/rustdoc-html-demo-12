1use crate::callconv::{ArgAbi, FnAbi};
2use crate::spec::HasTargetSpec;
3
4fn classify_ret<Ty>(ret: &mut ArgAbi<'_, Ty>) {
5    if ret.layout.is_aggregate() {
6        ret.make_indirect();
7    } else {
8        ret.extend_integer_width_to(32);
9    }
10}
11
12fn classify_arg<Ty>(cx: &impl HasTargetSpec, arg: &mut ArgAbi<'_, Ty>) {
13    if arg.is_ignore() {
14        // powerpc-unknown-linux-{gnu,musl,uclibc} doesn't ignore ZSTs.
15        if cx.target_spec().os == "linux"
16            && matches!(&*cx.target_spec().env, "gnu" | "musl" | "uclibc")
17            && arg.layout.is_zst()
18        {
19            arg.make_indirect_from_ignore();
20        }
21        return;
22    }
23    if arg.layout.is_aggregate() {
24        arg.make_indirect();
25    } else {
26        arg.extend_integer_width_to(32);
27    }
28}
29
30pub(crate) fn compute_abi_info<Ty>(cx: &impl HasTargetSpec, fn_abi: &mut FnAbi<'_, Ty>) {
31    if !fn_abi.ret.is_ignore() {
32        classify_ret(&mut fn_abi.ret);
33    }
34
35    for arg in fn_abi.args.iter_mut() {
36        classify_arg(cx, arg);
37    }
38}
