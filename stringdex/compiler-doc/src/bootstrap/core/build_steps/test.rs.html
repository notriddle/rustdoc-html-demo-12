45/// Runs `cargo test` on various internal tools used by bootstrap.
46#[derive(Debug, Clone, PartialEq, Eq, Hash)]
47pub struct CrateBootstrap {
48    path: PathBuf,
49    host: TargetSelection,
50}
51
52impl Step for CrateBootstrap {
53    type Output = ();
54    const ONLY_HOSTS: bool = true;
55    const DEFAULT: bool = true;
56
57    fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_> {
58        // This step is responsible for several different tool paths.
59        //
60        // By default, it will test all of them, but requesting specific tools on the command-line
61        // (e.g. `./x test src/tools/coverage-dump`) will test only the specified tools.
62        run.path("src/tools/jsondoclint")
63            .path("src/tools/replace-version-placeholder")
64            .path("src/tools/coverage-dump")
65            // We want `./x test tidy` to _run_ the tidy tool, not its tests.
66            // So we need a separate alias to test the tidy tool itself.
67            .alias("tidyselftest")
68    }
69
70    fn make_run(run: RunConfig<'_>) {
71        // Create and ensure a separate instance of this step for each path
72        // that was selected on the command-line (or selected by default).
73        for path in run.paths {
74            let path = path.assert_single_path().path.clone();
75            run.builder.ensure(CrateBootstrap { host: run.target, path });
76        }
77    }
78
79    fn run(self, builder: &Builder<'_>) {
80        let bootstrap_host = builder.config.host_target;
81        let compiler = builder.compiler(0, bootstrap_host);
82        let mut path = self.path.to_str().unwrap();
83
84        // Map alias `tidyselftest` back to the actual crate path of tidy.
85        if path == "tidyselftest" {
86            path = "src/tools/tidy";
87        }
88
89        let cargo = tool::prepare_tool_cargo(
90            builder,
91            compiler,
92            Mode::ToolBootstrap,
93            bootstrap_host,
94            Kind::Test,
95            path,
96            SourceType::InTree,
97            &[],
98        );
99
100        let crate_name = path.rsplit_once('/').unwrap().1;
101        run_cargo_test(cargo, &[], &[], crate_name, bootstrap_host, builder);
102    }
103}

107    host: TargetSelection,
108}
109
110impl Step for Linkcheck {
111    type Output = ();
112    const ONLY_HOSTS: bool = true;
113    const DEFAULT: bool = true;
114
115    /// Runs the `linkchecker` tool as compiled in `stage` by the `host` compiler.
116    ///
117    /// This tool in `src/tools` will verify the validity of all our links in the
118    /// documentation to ensure we don't have a bunch of dead ones.
119    fn run(self, builder: &Builder<'_>) {
120        let host = self.host;
121        let hosts = &builder.hosts;
122        let targets = &builder.targets;
123
124        // if we have different hosts and targets, some things may be built for
125        // the host (e.g. rustc) and others for the target (e.g. std). The
126        // documentation built for each will contain broken links to
127        // docs built for the other platform (e.g. rustc linking to cargo)
128        if (hosts != targets) && !hosts.is_empty() && !targets.is_empty() {
129            panic!(
130                "Linkcheck currently does not support builds with different hosts and targets.
131You can skip linkcheck with --skip src/tools/linkchecker"
132            );
133        }
134
135        builder.info(&format!("Linkcheck ({host})"));
136
137        // Test the linkchecker itself.
138        let bootstrap_host = builder.config.host_target;
139        let compiler = builder.compiler(0, bootstrap_host);
140
141        let cargo = tool::prepare_tool_cargo(
142            builder,
143            compiler,
144            Mode::ToolBootstrap,
145            bootstrap_host,
146            Kind::Test,
147            "src/tools/linkchecker",
148            SourceType::InTree,
149            &[],
150        );
151        run_cargo_test(cargo, &[], &[], "linkchecker self tests", bootstrap_host, builder);
152
153        if builder.doc_tests == DocTests::No {
154            return;
155        }
156
157        // Build all the default documentation.
158        builder.default_doc(&[]);
159
160        // Build the linkchecker before calling `msg`, since GHA doesn't support nested groups.
161        let linkchecker = builder.tool_cmd(Tool::Linkchecker);
162
163        // Run the linkchecker.
164        let _guard =
165            builder.msg(Kind::Test, compiler.stage, "Linkcheck", bootstrap_host, bootstrap_host);
166        let _time = helpers::timeit(builder);
167        linkchecker.delay_failure().arg(builder.out.join(host).join("doc")).run(builder);
168    }
169
170    fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_> {
171        let builder = run.builder;
172        let run = run.path("src/tools/linkchecker");
173        run.default_condition(builder.config.docs)
174    }
175
176    fn make_run(run: RunConfig<'_>) {
177        run.builder.ensure(Linkcheck { host: run.target });
178    }
179}

229/// Builds cargo and then runs the `src/tools/cargotest` tool, which checks out
230/// some representative crate repositories and runs `cargo test` on them, in
231/// order to test cargo.
232#[derive(Debug, Clone, PartialEq, Eq, Hash)]
233pub struct Cargotest {
234    build_compiler: Compiler,
235    host: TargetSelection,
236}
237
238impl Step for Cargotest {
239    type Output = ();
240    const ONLY_HOSTS: bool = true;
241
242    fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_> {
243        run.path("src/tools/cargotest")
244    }
245
246    fn make_run(run: RunConfig<'_>) {
247        if run.builder.top_stage == 0 {
248            eprintln!(
249                "ERROR: running cargotest with stage 0 is currently unsupported. Use at least stage 1."
250            );
251            exit!(1);
252        }
253        // We want to build cargo stage N (where N == top_stage), and rustc stage N,
254        // and test both of these together.
255        // So we need to get a build compiler stage N-1 to build the stage N components.
256        run.builder.ensure(Cargotest {
257            build_compiler: run.builder.compiler(run.builder.top_stage - 1, run.target),
258            host: run.target,
259        });
260    }
261
262    /// Runs the `cargotest` tool as compiled in `stage` by the `host` compiler.
263    ///
264    /// This tool in `src/tools` will check out a few Rust projects and run `cargo
265    /// test` to ensure that we don't regress the test suites there.
266    fn run(self, builder: &Builder<'_>) {
267        // cargotest's staging has several pieces:
268        // consider ./x test cargotest --stage=2.
269        //
270        // The test goal is to exercise a (stage 2 cargo, stage 2 rustc) pair through a stage 2
271        // cargotest tool.
272        // To produce the stage 2 cargo and cargotest, we need to do so with the stage 1 rustc and std.
273        // Importantly, the stage 2 rustc being tested (`tested_compiler`) via stage 2 cargotest is
274        // the rustc built by an earlier stage 1 rustc (the build_compiler). These are two different
275        // compilers!
276        let cargo =
277            builder.ensure(tool::Cargo::from_build_compiler(self.build_compiler, self.host));
278        let tested_compiler = builder.compiler(self.build_compiler.stage + 1, self.host);
279        builder.std(tested_compiler, self.host);
280
281        // Note that this is a short, cryptic, and not scoped directory name. This
282        // is currently to minimize the length of path on Windows where we otherwise
283        // quickly run into path name limit constraints.
284        let out_dir = builder.out.join("ct");
285        t!(fs::create_dir_all(&out_dir));
286
287        let _time = helpers::timeit(builder);
288        let mut cmd = builder.tool_cmd(Tool::CargoTest);
289        cmd.arg(&cargo.tool_path)
290            .arg(&out_dir)
291            .args(builder.config.test_args())
292            .env("RUSTC", builder.rustc(tested_compiler))
293            .env("RUSTDOC", builder.rustdoc_for_compiler(tested_compiler));
294        add_rustdoc_cargo_linker_args(&mut cmd, builder, tested_compiler.host, LldThreads::No);
295        cmd.delay_failure().run(builder);
296    }
297
298    fn metadata(&self) -> Option<StepMetadata> {
299        Some(StepMetadata::test("cargotest", self.host).stage(self.build_compiler.stage + 1))
300    }
301}

303/// Runs `cargo test` for cargo itself.
304/// We label these tests as "cargo self-tests".
305#[derive(Debug, Clone, PartialEq, Eq, Hash)]
306pub struct Cargo {
307    build_compiler: Compiler,
308    host: TargetSelection,
309}
310
311impl Cargo {
312    const CRATE_PATH: &str = "src/tools/cargo";
313}
314
315impl Step for Cargo {
316    type Output = ();
317    const ONLY_HOSTS: bool = true;
318
319    fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_> {
320        run.path(Self::CRATE_PATH)
321    }
322
323    fn make_run(run: RunConfig<'_>) {
324        run.builder.ensure(Cargo {
325            build_compiler: get_tool_target_compiler(
326                run.builder,
327                ToolTargetBuildMode::Build(run.target),
328            ),
329            host: run.target,
330        });
331    }
332
333    /// Runs `cargo test` for `cargo` packaged with Rust.
334    fn run(self, builder: &Builder<'_>) {
335        // When we do a "stage 1 cargo self-test", it means that we test the stage 1 rustc
336        // using stage 1 cargo. So we actually build cargo using the stage 0 compiler, and then
337        // run its tests against the stage 1 compiler (called `tested_compiler` below).
338        builder.ensure(tool::Cargo::from_build_compiler(self.build_compiler, self.host));
339
340        let tested_compiler = builder.compiler(self.build_compiler.stage + 1, self.host);
341        builder.std(tested_compiler, self.host);
342        // We also need to build rustdoc for cargo tests
343        // It will be located in the bindir of `tested_compiler`, so we don't need to explicitly
344        // pass its path to Cargo.
345        builder.rustdoc_for_compiler(tested_compiler);
346
347        let cargo = tool::prepare_tool_cargo(
348            builder,
349            self.build_compiler,
350            Mode::ToolTarget,
351            self.host,
352            Kind::Test,
353            Self::CRATE_PATH,
354            SourceType::Submodule,
355            &[],
356        );
357
358        // NOTE: can't use `run_cargo_test` because we need to overwrite `PATH`
359        let mut cargo = prepare_cargo_test(cargo, &[], &[], self.host, builder);
360
361        // Don't run cross-compile tests, we may not have cross-compiled libstd libs
362        // available.
363        cargo.env("CFG_DISABLE_CROSS_TESTS", "1");
364        // Forcibly disable tests using nightly features since any changes to
365        // those features won't be able to land.
366        cargo.env("CARGO_TEST_DISABLE_NIGHTLY", "1");
367
368        // Configure PATH to find the right rustc. NB. we have to use PATH
369        // and not RUSTC because the Cargo test suite has tests that will
370        // fail if rustc is not spelled `rustc`.
371        cargo.env("PATH", bin_path_for_cargo(builder, tested_compiler));
372
373        // The `cargo` command configured above has dylib dir path set to the `build_compiler`'s
374        // libdir. That causes issues in cargo test, because the programs that cargo compiles are
375        // incorrectly picking that libdir, even though they should be picking the
376        // `tested_compiler`'s libdir. We thus have to override the precedence here.
377        let mut existing_dylib_paths = cargo
378            .get_envs()
379            .find(|(k, _)| *k == OsStr::new(dylib_path_var()))
380            .and_then(|(_, v)| v)
381            .map(|value| split_paths(value).collect::<Vec<PathBuf>>())
382            .unwrap_or_default();
383        existing_dylib_paths.insert(0, builder.rustc_libdir(tested_compiler));
384        add_dylib_path(existing_dylib_paths, &mut cargo);
385
386        // Cargo's test suite uses `CARGO_RUSTC_CURRENT_DIR` to determine the path that `file!` is
387        // relative to. Cargo no longer sets this env var, so we have to do that. This has to be the
388        // same value as `-Zroot-dir`.
389        cargo.env("CARGO_RUSTC_CURRENT_DIR", builder.src.display().to_string());
390
391        #[cfg(feature = "build-metrics")]
392        builder.metrics.begin_test_suite(
393            build_helper::metrics::TestSuiteMetadata::CargoPackage {
394                crates: vec!["cargo".into()],
395                target: self.host.triple.to_string(),
396                host: self.host.triple.to_string(),
397                stage: self.build_compiler.stage + 1,
398            },
399            builder,
400        );
401
402        let _time = helpers::timeit(builder);
403        add_flags_and_try_run_tests(builder, &mut cargo);
404    }
405}

518    target: TargetSelection,

521impl Miri {
522    /// Run `cargo miri setup` for the given target, return where the Miri sysroot was put.
523    pub fn build_miri_sysroot(
524        builder: &Builder<'_>,
525        compiler: Compiler,
526        target: TargetSelection,
527    ) -> PathBuf {
528        let miri_sysroot = builder.out.join(compiler.host).join("miri-sysroot");
529        let mut cargo = builder::Cargo::new(
530            builder,
531            compiler,
532            Mode::Std,
533            SourceType::Submodule,
534            target,
535            Kind::MiriSetup,
536        );
537
538        // Tell `cargo miri setup` where to find the sources.
539        cargo.env("MIRI_LIB_SRC", builder.src.join("library"));
540        // Tell it where to put the sysroot.
541        cargo.env("MIRI_SYSROOT", &miri_sysroot);
542
543        let mut cargo = BootstrapCommand::from(cargo);
544        let _guard =
545            builder.msg(Kind::Build, compiler.stage, "miri sysroot", compiler.host, target);
546        cargo.run(builder);
547
548        // # Determine where Miri put its sysroot.
549        // To this end, we run `cargo miri setup --print-sysroot` and capture the output.
550        // (We do this separately from the above so that when the setup actually
551        // happens we get some output.)
552        // We re-use the `cargo` from above.
553        cargo.arg("--print-sysroot");
554
555        builder.verbose(|| println!("running: {cargo:?}"));
556        let stdout = cargo.run_capture_stdout(builder).stdout();
557        // Output is "<sysroot>\n".
558        let sysroot = stdout.trim_end();
559        builder.verbose(|| println!("`cargo miri setup --print-sysroot` said: {sysroot:?}"));
560        PathBuf::from(sysroot)
561    }
562}

564impl Step for Miri {
565    type Output = ();
566    const ONLY_HOSTS: bool = false;
567
568    fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_> {
569        run.path("src/tools/miri")
570    }
571
572    fn make_run(run: RunConfig<'_>) {
573        run.builder.ensure(Miri { target: run.target });
574    }
575
576    /// Runs `cargo test` for miri.
577    fn run(self, builder: &Builder<'_>) {
578        let host = builder.build.host_target;
579        let target = self.target;
580        let stage = builder.top_stage;
581        if stage == 0 {
582            eprintln!("miri cannot be tested at stage 0");
583            std::process::exit(1);
584        }
585
586        // This compiler runs on the host, we'll just use it for the target.
587        let compilers = RustcPrivateCompilers::new(builder, stage, host);
588
589        // Build our tools.
590        let miri = builder.ensure(tool::Miri::from_compilers(compilers));
591        // the ui tests also assume cargo-miri has been built
592        builder.ensure(tool::CargoMiri::from_compilers(compilers));
593
594        let target_compiler = compilers.target_compiler();
595
596        // We also need sysroots, for Miri and for the host (the latter for build scripts).
597        // This is for the tests so everything is done with the target compiler.
598        let miri_sysroot = Miri::build_miri_sysroot(builder, target_compiler, target);
599        builder.std(target_compiler, host);
600        let host_sysroot = builder.sysroot(target_compiler);
601
602        // Miri has its own "target dir" for ui test dependencies. Make sure it gets cleared when
603        // the sysroot gets rebuilt, to avoid "found possibly newer version of crate `std`" errors.
604        if !builder.config.dry_run() {
605            // This has to match `CARGO_TARGET_TMPDIR` in Miri's `ui.rs`.
606            // This means we need `host` here as that's the target `ui.rs` is built for.
607            let ui_test_dep_dir = builder
608                .stage_out(miri.build_compiler, Mode::ToolStd)
609                .join(host)
610                .join("tmp")
611                .join("miri_ui");
612            // The mtime of `miri_sysroot` changes when the sysroot gets rebuilt (also see
613            // <https://github.com/RalfJung/rustc-build-sysroot/commit/10ebcf60b80fe2c3dc765af0ff19fdc0da4b7466>).
614            // We can hence use that directly as a signal to clear the ui test dir.
615            build_stamp::clear_if_dirty(builder, &ui_test_dep_dir, &miri_sysroot);
616        }
617
618        // Run `cargo test`.
619        // This is with the Miri crate, so it uses the host compiler.
620        let mut cargo = tool::prepare_tool_cargo(
621            builder,
622            miri.build_compiler,
623            Mode::ToolRustc,
624            host,
625            Kind::Test,
626            "src/tools/miri",
627            SourceType::InTree,
628            &[],
629        );
630
631        cargo.add_rustc_lib_path(builder);
632
633        // We can NOT use `run_cargo_test` since Miri's integration tests do not use the usual test
634        // harness and therefore do not understand the flags added by `add_flags_and_try_run_test`.
635        let mut cargo = prepare_cargo_test(cargo, &[], &[], host, builder);
636
637        // miri tests need to know about the stage sysroot
638        cargo.env("MIRI_SYSROOT", &miri_sysroot);
639        cargo.env("MIRI_HOST_SYSROOT", &host_sysroot);
640        cargo.env("MIRI", &miri.tool_path);
641
642        // Set the target.
643        cargo.env("MIRI_TEST_TARGET", target.rustc_target_arg());
644
645        {
646            let _guard = builder.msg_rustc_tool(Kind::Test, stage, "miri", host, target);
647            let _time = helpers::timeit(builder);
648            cargo.run(builder);
649        }
650
651        // Run it again for mir-opt-level 4 to catch some miscompilations.
652        if builder.config.test_args().is_empty() {
653            cargo.env("MIRIFLAGS", "-O -Zmir-opt-level=4 -Cdebug-assertions=yes");
654            // Optimizations can change backtraces
655            cargo.env("MIRI_SKIP_UI_CHECKS", "1");
656            // `MIRI_SKIP_UI_CHECKS` and `RUSTC_BLESS` are incompatible
657            cargo.env_remove("RUSTC_BLESS");
658            // Optimizations can change error locations and remove UB so don't run `fail` tests.
659            cargo.args(["tests/pass", "tests/panic"]);
660
661            {
662                let _guard = builder.msg_rustc_tool(
663                    Kind::Test,
664                    stage,
665                    "miri (mir-opt-level 4)",
666                    host,
667                    target,
668                );
669                let _time = helpers::timeit(builder);
670                cargo.run(builder);
671            }
672        }
673    }

809    compilers: RustcPrivateCompilers,
810}
811
812impl Step for Clippy {
813    type Output = ();
814    const ONLY_HOSTS: bool = true;
815    const DEFAULT: bool = false;
816
817    fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_> {
818        run.suite_path("src/tools/clippy/tests").path("src/tools/clippy")
819    }
820
821    fn make_run(run: RunConfig<'_>) {
822        run.builder.ensure(Clippy {
823            compilers: RustcPrivateCompilers::new(
824                run.builder,
825                run.builder.top_stage,
826                run.builder.host_target,
827            ),
828        });
829    }
830
831    /// Runs `cargo test` for clippy.
832    fn run(self, builder: &Builder<'_>) {
833        let host = self.compilers.target();
834
835        // We need to carefully distinguish the compiler that builds clippy, and the compiler
836        // that is linked into the clippy being tested. `target_compiler` is the latter,
837        // and it must also be used by clippy's test runner to build tests and their dependencies.
838        let compilers = self.compilers;
839        let target_compiler = compilers.target_compiler();
840
841        let tool_result = builder.ensure(tool::Clippy::from_compilers(compilers));
842        let build_compiler = tool_result.build_compiler;
843        let mut cargo = tool::prepare_tool_cargo(
844            builder,
845            build_compiler,
846            Mode::ToolRustc,
847            host,
848            Kind::Test,
849            "src/tools/clippy",
850            SourceType::InTree,
851            &[],
852        );
853
854        cargo.env("RUSTC_TEST_SUITE", builder.rustc(build_compiler));
855        cargo.env("RUSTC_LIB_PATH", builder.rustc_libdir(build_compiler));
856        let host_libs =
857            builder.stage_out(build_compiler, Mode::ToolRustc).join(builder.cargo_dir());
858        cargo.env("HOST_LIBS", host_libs);
859
860        // Build the standard library that the tests can use.
861        builder.std(target_compiler, host);
862        cargo.env("TEST_SYSROOT", builder.sysroot(target_compiler));
863        cargo.env("TEST_RUSTC", builder.rustc(target_compiler));
864        cargo.env("TEST_RUSTC_LIB", builder.rustc_libdir(target_compiler));
865
866        // Collect paths of tests to run
867        'partially_test: {
868            let paths = &builder.config.paths[..];
869            let mut test_names = Vec::new();
870            for path in paths {
871                if let Some(path) =
872                    helpers::is_valid_test_suite_arg(path, "src/tools/clippy/tests", builder)
873                {
874                    test_names.push(path);
875                } else if path.ends_with("src/tools/clippy") {
876                    // When src/tools/clippy is called directly, all tests should be run.
877                    break 'partially_test;
878                }
879            }
880            cargo.env("TESTNAME", test_names.join(","));
881        }
882
883        cargo.add_rustc_lib_path(builder);
884        let cargo = prepare_cargo_test(cargo, &[], &[], host, builder);
885
886        let _guard = builder.msg_rustc_tool(Kind::Test, build_compiler.stage, "clippy", host, host);
887
888        // Clippy reports errors if it blessed the outputs
889        if cargo.allow_failure().run(builder) {
890            // The tests succeeded; nothing to do.
891            return;
892        }
893
894        if !builder.config.cmd.bless() {
895            crate::exit!(1);
896        }
897    }
898}

1168    ///
1169    /// This tool in `src/tools` checks up on various bits and pieces of style and
1170    /// otherwise just implements a few lint-like checks that are specific to the
1171    /// compiler itself.
1172    ///
1173    /// Once tidy passes, this step also runs `fmt --check` if tests are being run
1174    /// for the `dev` or `nightly` channels.
1175    fn run(self, builder: &Builder<'_>) {

1181        let jobs = builder.config.jobs.unwrap_or_else(|| {
1182            8 * std::thread::available_parallelism().map_or(1, std::num::NonZeroUsize::get) as u32
1183        });

1190        }

1192            cmd.arg("--verbose");
1193        }
1194        if builder.config.cmd.bless() {
1195            cmd.arg("--bless");
1196        }
1197        if let Some(s) =
1198            builder.config.cmd.extra_checks().or(builder.config.tidy_extra_checks.as_deref())
1199        {
1200            cmd.arg(format!("--extra-checks={s}"));
1201        }
1202        let mut args = std::env::args_os();

1204            cmd.arg("--");

1209            if !builder.config.json_output {
1210                builder.info("fmt check");

1212                    let inferred_rustfmt_dir = builder.initial_sysroot.join("bin");
1213                    eprintln!(
1214                        "\
1215ERROR: no `rustfmt` binary found in {PATH}
1216INFO: `rust.channel` is currently set to \"{CHAN}\"
1217HELP: if you are testing a beta branch, set `rust.channel` to \"beta\" in the `bootstrap.toml` file
1218HELP: to skip test's attempt to check tidiness, pass `--skip src/tools/tidy` to `x.py test`",
1219                        PATH = inferred_rustfmt_dir.display(),

1275macro_rules! test {
1276    (
1277        $( #[$attr:meta] )* // allow docstrings and attributes
1278        $name:ident {
1279            path: $path:expr,
1280            mode: $mode:expr,
1281            suite: $suite:expr,
1282            default: $default:expr
1283            $( , only_hosts: $only_hosts:expr )? // default: false
1284            $( , compare_mode: $compare_mode:expr )? // default: None
1285            $( , )? // optional trailing comma
1286        }
1287    ) => {
1288        $( #[$attr] )*
1289        #[derive(Debug, Clone, PartialEq, Eq, Hash)]
1290        pub struct $name {
1291            pub compiler: Compiler,
1292            pub target: TargetSelection,
1293        }
1294
1295        impl Step for $name {
1296            type Output = ();
1297            const DEFAULT: bool = $default;
1298            const ONLY_HOSTS: bool = (const {
1299                #[allow(unused_assignments, unused_mut)]
1300                let mut value = false;
1301                $( value = $only_hosts; )?
1302                value
1303            });
1304
1305            fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_> {
1306                run.suite_path($path)
1307            }
1308
1309            fn make_run(run: RunConfig<'_>) {
1310                let compiler = run.builder.compiler(run.builder.top_stage, run.build_triple());
1311
1312                run.builder.ensure($name { compiler, target: run.target });
1313            }
1314
1315            fn run(self, builder: &Builder<'_>) {
1316                builder.ensure(Compiletest {
1317                    compiler: self.compiler,
1318                    target: self.target,
1319                    mode: $mode,
1320                    suite: $suite,
1321                    path: $path,
1322                    compare_mode: (const {
1323                        #[allow(unused_assignments, unused_mut)]
1324                        let mut value = None;
1325                        $( value = $compare_mode; )?
1326                        value
1327                    }),
1328                })
1329            }
1330
1331            fn metadata(&self) -> Option<StepMetadata> {
1332                Some(
1333                    StepMetadata::test(stringify!($name), self.target)
1334                )
1335            }
1336        }
1337    };

1497/// Runs the coverage test suite at `tests/coverage` in some or all of the
1498/// coverage test modes.
1499#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
1500pub struct Coverage {
1501    pub compiler: Compiler,
1502    pub target: TargetSelection,
1503    pub mode: &'static str,
1504}
1505
1506impl Coverage {
1507    const PATH: &'static str = "tests/coverage";
1508    const SUITE: &'static str = "coverage";
1509    const ALL_MODES: &[&str] = &["coverage-map", "coverage-run"];
1510}
1511
1512impl Step for Coverage {
1513    type Output = ();
1514    const DEFAULT: bool = true;
1515    /// Compiletest will automatically skip the "coverage-run" tests if necessary.
1516    const ONLY_HOSTS: bool = false;
1517
1518    fn should_run(mut run: ShouldRun<'_>) -> ShouldRun<'_> {
1519        // Support various invocation styles, including:
1520        // - `./x test coverage`
1521        // - `./x test tests/coverage/trivial.rs`
1522        // - `./x test coverage-map`
1523        // - `./x test coverage-run -- tests/coverage/trivial.rs`
1524        run = run.suite_path(Self::PATH);
1525        for mode in Self::ALL_MODES {
1526            run = run.alias(mode);
1527        }
1528        run
1529    }
1530
1531    fn make_run(run: RunConfig<'_>) {
1532        let compiler = run.builder.compiler(run.builder.top_stage, run.build_triple());
1533        let target = run.target;
1534
1535        // List of (coverage) test modes that the coverage test suite will be
1536        // run in. It's OK for this to contain duplicates, because the call to
1537        // `Builder::ensure` below will take care of deduplication.
1538        let mut modes = vec![];
1539
1540        // From the pathsets that were selected on the command-line (or by default),
1541        // determine which modes to run in.
1542        for path in &run.paths {
1543            match path {
1544                PathSet::Set(_) => {
1545                    for mode in Self::ALL_MODES {
1546                        if path.assert_single_path().path == Path::new(mode) {
1547                            modes.push(mode);
1548                            break;
1549                        }
1550                    }
1551                }
1552                PathSet::Suite(_) => {
1553                    modes.extend(Self::ALL_MODES);
1554                    break;
1555                }
1556            }
1557        }
1558
1559        // Skip any modes that were explicitly skipped/excluded on the command-line.
1560        // FIXME(Zalathar): Integrate this into central skip handling somehow?
1561        modes.retain(|mode| !run.builder.config.skip.iter().any(|skip| skip == Path::new(mode)));
1562
1563        // FIXME(Zalathar): Make these commands skip all coverage tests, as expected:
1564        // - `./x test --skip=tests`
1565        // - `./x test --skip=tests/coverage`
1566        // - `./x test --skip=coverage`
1567        // Skip handling currently doesn't have a way to know that skipping the coverage
1568        // suite should also skip the `coverage-map` and `coverage-run` aliases.
1569
1570        for mode in modes {
1571            run.builder.ensure(Coverage { compiler, target, mode });
1572        }
1573    }
1574
1575    fn run(self, builder: &Builder<'_>) {
1576        let Self { compiler, target, mode } = self;
1577        // Like other compiletest suite test steps, delegate to an internal
1578        // compiletest task to actually run the tests.
1579        builder.ensure(Compiletest {
1580            compiler,
1581            target,
1582            mode,
1583            suite: Self::SUITE,
1584            path: Self::PATH,
1585            compare_mode: None,
1586        });
1587    }
1588}

1657#[derive(Debug, Clone, PartialEq, Eq, Hash)]
1658struct Compiletest {
1659    compiler: Compiler,
1660    target: TargetSelection,
1661    mode: &'static str,
1662    suite: &'static str,
1663    path: &'static str,
1664    compare_mode: Option<&'static str>,
1665}
1666
1667impl Step for Compiletest {
1668    type Output = ();
1669
1670    fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_> {
1671        run.never()
1672    }
1673
1674    /// Executes the `compiletest` tool to run a suite of tests.
1675    ///
1676    /// Compiles all tests with `compiler` for `target` with the specified
1677    /// compiletest `mode` and `suite` arguments. For example `mode` can be
1678    /// "run-pass" or `suite` can be something like `debuginfo`.
1679    fn run(self, builder: &Builder<'_>) {
1680        if builder.doc_tests == DocTests::Only {
1681            return;
1682        }
1683
1684        if builder.top_stage == 0 && !builder.config.compiletest_allow_stage0 {
1685            eprintln!("\
1686ERROR: `--stage 0` runs compiletest on the stage0 (precompiled) compiler, not your local changes, and will almost always cause tests to fail
1687HELP: to test the compiler or standard library, omit the stage or explicitly use `--stage 1` instead
1688NOTE: if you're sure you want to do this, please open an issue as to why. In the meantime, you can override this with `--set build.compiletest-allow-stage0=true`."
1689            );
1690            crate::exit!(1);
1691        }
1692
1693        let mut compiler = self.compiler;
1694        let target = self.target;
1695        let mode = self.mode;
1696        let suite = self.suite;
1697
1698        // Path for test suite
1699        let suite_path = self.path;
1700
1701        // Skip codegen tests if they aren't enabled in configuration.
1702        if !builder.config.codegen_tests && mode == "codegen" {
1703            return;
1704        }
1705
1706        // Support stage 1 ui-fulldeps. This is somewhat complicated: ui-fulldeps tests for the most
1707        // part test the *API* of the compiler, not how it compiles a given file. As a result, we
1708        // can run them against the stage 1 sources as long as we build them with the stage 0
1709        // bootstrap compiler.
1710        // NOTE: Only stage 1 is special cased because we need the rustc_private artifacts to match the
1711        // running compiler in stage 2 when plugins run.
1712        let query_compiler;
1713        let (stage, stage_id) = if suite == "ui-fulldeps" && compiler.stage == 1 {
1714            // Even when using the stage 0 compiler, we also need to provide the stage 1 compiler
1715            // so that compiletest can query it for target information.
1716            query_compiler = Some(compiler);
1717            // At stage 0 (stage - 1) we are using the stage0 compiler. Using `self.target` can lead
1718            // finding an incorrect compiler path on cross-targets, as the stage 0 is always equal to
1719            // `build.build` in the configuration.
1720            let build = builder.build.host_target;
1721            compiler = builder.compiler(compiler.stage - 1, build);
1722            let test_stage = compiler.stage + 1;
1723            (test_stage, format!("stage{test_stage}-{build}"))
1724        } else {
1725            query_compiler = None;
1726            let stage = compiler.stage;
1727            (stage, format!("stage{stage}-{target}"))
1728        };
1729
1730        if suite.ends_with("fulldeps") {
1731            builder.ensure(compile::Rustc::new(compiler, target));
1732        }
1733
1734        if suite == "debuginfo" {
1735            builder.ensure(dist::DebuggerScripts {
1736                sysroot: builder.sysroot(compiler).to_path_buf(),
1737                host: target,
1738            });
1739        }
1740        if suite == "run-make" {
1741            builder.tool_exe(Tool::RunMakeSupport);
1742        }
1743
1744        // ensure that `libproc_macro` is available on the host.
1745        if suite == "mir-opt" {
1746            builder.ensure(compile::Std::new(compiler, compiler.host).is_for_mir_opt_tests(true));
1747        } else {
1748            builder.std(compiler, compiler.host);
1749        }
1750
1751        let mut cmd = builder.tool_cmd(Tool::Compiletest);
1752
1753        if suite == "mir-opt" {
1754            builder.ensure(compile::Std::new(compiler, target).is_for_mir_opt_tests(true));
1755        } else {
1756            builder.std(compiler, target);
1757        }
1758
1759        builder.ensure(RemoteCopyLibs { compiler, target });
1760
1761        // compiletest currently has... a lot of arguments, so let's just pass all
1762        // of them!
1763
1764        cmd.arg("--stage").arg(stage.to_string());
1765        cmd.arg("--stage-id").arg(stage_id);
1766
1767        cmd.arg("--compile-lib-path").arg(builder.rustc_libdir(compiler));
1768        cmd.arg("--run-lib-path").arg(builder.sysroot_target_libdir(compiler, target));
1769        cmd.arg("--rustc-path").arg(builder.rustc(compiler));
1770        if let Some(query_compiler) = query_compiler {
1771            cmd.arg("--query-rustc-path").arg(builder.rustc(query_compiler));
1772        }
1773
1774        // Minicore auxiliary lib for `no_core` tests that need `core` stubs in cross-compilation
1775        // scenarios.
1776        cmd.arg("--minicore-path")
1777            .arg(builder.src.join("tests").join("auxiliary").join("minicore.rs"));
1778
1779        let is_rustdoc = suite == "rustdoc-ui" || suite == "rustdoc-js";
1780
1781        if mode == "run-make" {
1782            let cargo_path = if builder.top_stage == 0 {
1783                // If we're using `--stage 0`, we should provide the bootstrap cargo.
1784                builder.initial_cargo.clone()
1785            } else {
1786                builder.ensure(tool::Cargo::from_build_compiler(compiler, compiler.host)).tool_path
1787            };
1788
1789            cmd.arg("--cargo-path").arg(cargo_path);
1790
1791            // We need to pass the compiler that was used to compile run-make-support,
1792            // because we have to use the same compiler to compile rmake.rs recipes.
1793            let stage0_rustc_path = builder.compiler(0, compiler.host);
1794            cmd.arg("--stage0-rustc-path").arg(builder.rustc(stage0_rustc_path));
1795        }
1796
1797        // Avoid depending on rustdoc when we don't need it.
1798        if mode == "rustdoc"
1799            || mode == "run-make"
1800            || (mode == "ui" && is_rustdoc)
1801            || mode == "rustdoc-js"
1802            || mode == "rustdoc-json"
1803            || suite == "coverage-run-rustdoc"
1804        {
1805            cmd.arg("--rustdoc-path").arg(builder.rustdoc_for_compiler(compiler));
1806        }
1807
1808        if mode == "rustdoc-json" {
1809            // Use the stage0 compiler for jsondocck
1810            let json_compiler = compiler.with_stage(0);
1811            cmd.arg("--jsondocck-path")
1812                .arg(builder.ensure(tool::JsonDocCk { compiler: json_compiler, target }).tool_path);
1813            cmd.arg("--jsondoclint-path").arg(
1814                builder.ensure(tool::JsonDocLint { compiler: json_compiler, target }).tool_path,
1815            );
1816        }
1817
1818        if matches!(mode, "coverage-map" | "coverage-run") {
1819            let coverage_dump = builder.tool_exe(Tool::CoverageDump);
1820            cmd.arg("--coverage-dump-path").arg(coverage_dump);
1821        }
1822
1823        cmd.arg("--src-root").arg(&builder.src);
1824        cmd.arg("--src-test-suite-root").arg(builder.src.join("tests").join(suite));
1825
1826        // N.B. it's important to distinguish between the *root* build directory, the *host* build
1827        // directory immediately under the root build directory, and the test-suite-specific build
1828        // directory.
1829        cmd.arg("--build-root").arg(&builder.out);
1830        cmd.arg("--build-test-suite-root").arg(testdir(builder, compiler.host).join(suite));
1831
1832        // When top stage is 0, that means that we're testing an externally provided compiler.
1833        // In that case we need to use its specific sysroot for tests to pass.
1834        let sysroot = if builder.top_stage == 0 {
1835            builder.initial_sysroot.clone()
1836        } else {
1837            builder.sysroot(compiler)
1838        };
1839
1840        cmd.arg("--sysroot-base").arg(sysroot);
1841
1842        cmd.arg("--suite").arg(suite);
1843        cmd.arg("--mode").arg(mode);
1844        cmd.arg("--target").arg(target.rustc_target_arg());
1845        cmd.arg("--host").arg(&*compiler.host.triple);
1846        cmd.arg("--llvm-filecheck").arg(builder.llvm_filecheck(builder.config.host_target));
1847
1848        if let Some(codegen_backend) = builder.config.default_codegen_backend(compiler.host) {
1849            // Tells compiletest which codegen backend is used by default by the compiler.
1850            // It is used to e.g. ignore tests that don't support that codegen backend.
1851            cmd.arg("--codegen-backend").arg(codegen_backend.name());
1852        }
1853
1854        if builder.build.config.llvm_enzyme {
1855            cmd.arg("--has-enzyme");
1856        }

2314    fn run_ext_doc(self, builder: &Builder<'_>) {
2315        let compiler = self.compiler;
2316
2317        builder.std(compiler, compiler.host);
2318
2319        // mdbook just executes a binary named "rustdoc", so we need to update
2320        // PATH so that it points to our rustdoc.
2321        let mut rustdoc_path = builder.rustdoc_for_compiler(compiler);
2322        rustdoc_path.pop();
2323        let old_path = env::var_os("PATH").unwrap_or_default();
2324        let new_path = env::join_paths(iter::once(rustdoc_path).chain(env::split_paths(&old_path)))
2325            .expect("could not add rustdoc to PATH");
2326

2329        // Books often have feature-gated example text.
2330        rustbook_cmd.env("RUSTC_BOOTSTRAP", "1");

2333        // Books may also need to build dependencies. For example, `TheBook` has
2334        // code samples which use the `trpl` crate. For the `rustdoc` invocation
2335        // to find them them successfully, they need to be built first and their
2336        // paths used to generate the
2337        let libs = if !self.dependencies.is_empty() {

2340                let mode = Mode::ToolRustc;

2344                    compiler,

2349                    SourceType::Submodule,

2351                );

2358                    .into_iter()
2359                    .filter_map(|p| p.parent().map(ToOwned::to_owned))
2360                    .fold(HashSet::new(), |mut set, dir| {
2361                        set.insert(dir);
2362                        set
2363                    });

2366            }
2367            lib_paths
2368        } else {
2369            vec![]
2370        };

2374                .into_iter()

2376                .collect::<Vec<OsString>>()
2377                .join(OsStr::new(","));

2383            Kind::Test,

2388        );

2390        let toolstate = if rustbook_cmd.delay_failure().run(builder) {
2391            ToolState::TestPass
2392        } else {

2394        };

2396    }

2399    fn run_local_doc(self, builder: &Builder<'_>) {
2400        let compiler = self.compiler;
2401        let host = self.compiler.host;
2402
2403        builder.std(compiler, host);
2404

2406            builder.msg(Kind::Test, compiler.stage, format!("book {}", self.name), host, host);
2407

2409        // tests for all files that end in `*.md`
2410        let mut stack = vec![builder.src.join(self.path)];
2411        let _time = helpers::timeit(builder);
2412        let mut files = Vec::new();

2414            if p.is_dir() {
2415                stack.extend(t!(p.read_dir()).map(|p| t!(p).path()));

2424        }
2425
2426        files.sort();

2428        for file in files {
2429            markdown_test(builder, compiler, &file);
2430        }
2431    }

2670/// Given a `cargo test` subcommand, pass it the appropriate test flags given a `builder`.
2671fn prepare_cargo_test(
2672    cargo: builder::Cargo,
2673    libtest_args: &[&str],
2674    crates: &[String],
2675    target: TargetSelection,
2676    builder: &Builder<'_>,
2677) -> BootstrapCommand {
2678    let compiler = cargo.compiler();
2679    let mut cargo: BootstrapCommand = cargo.into();
2680
2681    // Propagate `--bless` if it has not already been set/unset
2682    // Any tools that want to use this should bless if `RUSTC_BLESS` is set to
2683    // anything other than `0`.
2684    if builder.config.cmd.bless() && !cargo.get_envs().any(|v| v.0 == "RUSTC_BLESS") {
2685        cargo.env("RUSTC_BLESS", "Gesundheit");
2686    }
2687
2688    // Pass in some standard flags then iterate over the graph we've discovered
2689    // in `cargo metadata` with the maps above and figure out what `-p`
2690    // arguments need to get passed.
2691    if builder.kind == Kind::Test && !builder.fail_fast {
2692        cargo.arg("--no-fail-fast");
2693    }
2694
2695    if builder.config.json_output {
2696        cargo.arg("--message-format=json");
2697    }
2698
2699    match builder.doc_tests {
2700        DocTests::Only => {
2701            cargo.arg("--doc");
2702        }
2703        DocTests::No => {
2704            cargo.args(["--bins", "--examples", "--tests", "--benches"]);
2705        }
2706        DocTests::Yes => {}
2707    }
2708
2709    for krate in crates {
2710        cargo.arg("-p").arg(krate);
2711    }
2712
2713    cargo.arg("--").args(builder.config.test_args()).args(libtest_args);
2714    if !builder.config.verbose_tests {
2715        cargo.arg("--quiet");
2716    }
2717
2718    // The tests are going to run with the *target* libraries, so we need to
2719    // ensure that those libraries show up in the LD_LIBRARY_PATH equivalent.
2720    //
2721    // Note that to run the compiler we need to run with the *host* libraries,
2722    // but our wrapper scripts arrange for that to be the case anyway.
2723    //
2724    // We skip everything on Miri as then this overwrites the libdir set up
2725    // by `Cargo::new` and that actually makes things go wrong.
2726    if builder.kind != Kind::Miri {
2727        let mut dylib_paths = builder.rustc_lib_paths(compiler);
2728        dylib_paths.push(builder.sysroot_target_libdir(compiler, target));
2729        helpers::add_dylib_path(dylib_paths, &mut cargo);
2730    }
2731
2732    if builder.remote_tested(target) {
2733        cargo.env(
2734            format!("CARGO_TARGET_{}_RUNNER", envify(&target.triple)),
2735            format!("{} run 0", builder.tool_exe(Tool::RemoteTestClient).display()),
2736        );
2737    } else if let Some(tool) = builder.runner(target) {
2738        cargo.env(format!("CARGO_TARGET_{}_RUNNER", envify(&target.triple)), tool);
2739    }
2740
2741    cargo
2742}

2781    /// by a `Cargo.toml` (single manifest)

2783    /// This is what runs tests for crates like the standard library, compiler, etc.
2784    /// It essentially is the driver for running `cargo test`.

2786    /// Currently this runs all tests for a DAG by passing a bunch of `-p foo`
2787    /// arguments, and those arguments are discovered from `cargo metadata`.
2788    fn run(self, builder: &Builder<'_>) {
2789        let compiler = self.compiler;
2790        let target = self.target;
2791        let mode = self.mode;
2792
2793        // Prepare sysroot
2794        // See [field@compile::Std::force_recompile].
2795        builder.ensure(Std::new(compiler, compiler.host).force_recompile(true));
2796
2797        // If we're not doing a full bootstrap but we're testing a stage2
2798        // version of libstd, then what we're actually testing is the libstd
2799        // produced in stage1. Reflect that here by updating the compiler that
2800        // we're working with automatically.
2801        let compiler = builder.compiler_for(compiler.stage, compiler.host, target);

2803        let mut cargo = if builder.kind == Kind::Miri {
2804            if builder.top_stage == 0 {
2805                eprintln!("ERROR: `x.py miri` requires stage 1 or higher");
2806                std::process::exit(1);
2807            }
2808
2809            // Build `cargo miri test` command
2810            // (Implicitly prepares target sysroot)
2811            let mut cargo = builder::Cargo::new(
2812                builder,
2813                compiler,
2814                mode,
2815                SourceType::InTree,
2816                target,
2817                Kind::MiriTest,
2818            );
2819            // This hack helps bootstrap run standard library tests in Miri. The issue is as
2820            // follows: when running `cargo miri test` on libcore, cargo builds a local copy of core
2821            // and makes it a dependency of the integration test crate. This copy duplicates all the
2822            // lang items, so the build fails. (Regular testing avoids this because the sysroot is a
2823            // literal copy of what `cargo build` produces, but since Miri builds its own sysroot
2824            // this does not work for us.) So we need to make it so that the locally built libcore
2825            // contains all the items from `core`, but does not re-define them -- we want to replace
2826            // the entire crate but a re-export of the sysroot crate. We do this by swapping out the
2827            // source file: if `MIRI_REPLACE_LIBRS_IF_NOT_TEST` is set and we are building a
2828            // `lib.rs` file, and a `lib.miri.rs` file exists in the same folder, we build that
2829            // instead. But crucially we only do that for the library, not the test builds.
2830            cargo.env("MIRI_REPLACE_LIBRS_IF_NOT_TEST", "1");
2831            // std needs to be built with `-Zforce-unstable-if-unmarked`. For some reason the builder
2832            // does not set this directly, but relies on the rustc wrapper to set it, and we are not using
2833            // the wrapper -- hence we have to set it ourselves.
2834            cargo.rustflag("-Zforce-unstable-if-unmarked");

2836        } else {
2837            // Also prepare a sysroot for the target.
2838            if !builder.config.is_host_target(target) {
2839                builder.ensure(compile::Std::new(compiler, target).force_recompile(true));
2840                builder.ensure(RemoteCopyLibs { compiler, target });
2841            }
2842
2843            // Build `cargo test` command
2844            builder::Cargo::new(builder, compiler, mode, SourceType::InTree, target, builder.kind)
2845        };
2846
2847        match mode {
2848            Mode::Std => {
2849                if builder.kind == Kind::Miri {
2850                    // We can't use `std_cargo` as that uses `optimized-compiler-builtins` which
2851                    // needs host tools for the given target. This is similar to what `compile::Std`
2852                    // does when `is_for_mir_opt_tests` is true. There's probably a chance for
2853                    // de-duplication here... `std_cargo` should support a mode that avoids needing
2854                    // host tools.
2855                    cargo
2856                        .arg("--manifest-path")
2857                        .arg(builder.src.join("library/sysroot/Cargo.toml"));
2858                } else {
2859                    compile::std_cargo(builder, target, &mut cargo);
2860                }
2861            }
2862            Mode::Rustc => {
2863                compile::rustc_cargo(builder, &mut cargo, target, &compiler, &self.crates);
2864            }
2865            _ => panic!("can only test libraries"),
2866        };
2867
2868        let mut crates = self.crates.clone();
2869        // The core and alloc crates can't directly be tested. We
2870        // could silently ignore them, but adding their own test
2871        // crates is less confusing for users. We still keep core and
2872        // alloc themself for doctests
2873        if crates.iter().any(|crate_| crate_ == "core") {
2874            crates.push("coretests".to_owned());
2875        }
2876        if crates.iter().any(|crate_| crate_ == "alloc") {
2877            crates.push("alloctests".to_owned());
2878        }
2879
2880        run_cargo_test(cargo, &[], &crates, &*crate_description(&self.crates), target, builder);
2881    }
2882}
