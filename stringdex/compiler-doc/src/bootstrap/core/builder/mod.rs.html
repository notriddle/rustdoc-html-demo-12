92pub trait Step: 'static + Clone + Debug + PartialEq + Eq + Hash {
93    /// Result type of `Step::run`.
94    type Output: Clone;
95
96    /// Whether this step is run by default as part of its respective phase, as defined by the `describe`
97    /// macro in [`Builder::get_step_descriptions`].
98    ///
99    /// Note: Even if set to `true`, it can still be overridden with [`ShouldRun::default_condition`]
100    /// by `Step::should_run`.
101    const DEFAULT: bool = false;
102
103    /// If true, then this rule should be skipped if --target was specified, but --host was not
104    const ONLY_HOSTS: bool = false;
105
106    /// Primary function to implement `Step` logic.
107    ///
108    /// This function can be triggered in two ways:
109    /// 1. Directly from [`Builder::execute_cli`].
110    /// 2. Indirectly by being called from other `Step`s using [`Builder::ensure`].
111    ///
112    /// When called with [`Builder::execute_cli`] (as done by `Build::build`), this function is executed twice:
113    /// - First in "dry-run" mode to validate certain things (like cyclic Step invocations,
114    ///   directory creation, etc) super quickly.
115    /// - Then it's called again to run the actual, very expensive process.
116    ///
117    /// When triggered indirectly from other `Step`s, it may still run twice (as dry-run and real mode)
118    /// depending on the `Step::run` implementation of the caller.
119    fn run(self, builder: &Builder<'_>) -> Self::Output;
120
121    /// Determines if this `Step` should be run when given specific paths (e.g., `x build $path`).
122    fn should_run(run: ShouldRun<'_>) -> ShouldRun<'_>;
123
124    /// Called directly by the bootstrap `Step` handler when not triggered indirectly by other `Step`s using [`Builder::ensure`].
125    /// For example, `./x.py test bootstrap` runs this for `test::Bootstrap`. Similarly, `./x.py test` runs it for every step
126    /// that is listed by the `describe` macro in [`Builder::get_step_descriptions`].
127    fn make_run(_run: RunConfig<'_>) {
128        // It is reasonable to not have an implementation of make_run for rules
129        // who do not want to get called from the root context. This means that
130        // they are likely dependencies (e.g., sysroot creation) or similar, and
131        // as such calling them from ./x.py isn't logical.
132        unimplemented!()
133    }
134
135    /// Returns metadata of the step, for tests
136    fn metadata(&self) -> Option<StepMetadata> {
137        None
138    }
139}

224    pub fn cargo_crates_in_set(&self) -> Vec<String> {
225        let mut crates = Vec::new();
226        for krate in &self.paths {

230                .builder

240    /// Given an `alias` selected by the `Step` and the paths passed on the command line,
241    /// return a list of the crates that should be built.
242    ///
243    /// Normally, people will pass *just* `library` if they pass it.
244    /// But it's possible (although strange) to pass something like `library std core`.
245    /// Build all crates anyway, as if they hadn't passed the other args.
246    pub fn make_run_crates(&self, alias: Alias) -> Vec<String> {
247        let has_alias =
248            self.paths.iter().any(|set| set.assert_single_path().path.ends_with(alias.as_str()));
249        if !has_alias {
250            return self.cargo_crates_in_set();
251        }
252
253        let crates = match alias {
254            Alias::Library => self.builder.in_tree_crates("sysroot", Some(self.target)),
255            Alias::Compiler => self.builder.in_tree_crates("rustc-main", Some(self.target)),
256        };
257
258        crates.into_iter().map(|krate| krate.name.to_string()).collect()
259    }

319/// Collection of paths used to match a task rule.
320#[derive(Debug, Clone, PartialOrd, Ord, PartialEq, Eq)]
321pub enum PathSet {
322    /// A collection of individual paths or aliases.
323    ///
324    /// These are generally matched as a path suffix. For example, a
325    /// command-line value of `std` will match if `library/std` is in the
326    /// set.
327    ///
328    /// NOTE: the paths within a set should always be aliases of one another.
329    /// For example, `src/librustdoc` and `src/tools/rustdoc` should be in the same set,
330    /// but `library/core` and `library/std` generally should not, unless there's no way (for that Step)
331    /// to build them separately.
332    Set(BTreeSet<TaskPath>),
333    /// A "suite" of paths.
334    ///
335    /// These can match as a path suffix (like `Set`), or as a prefix. For
336    /// example, a command-line value of `tests/ui/abi/variadic-ffi.rs`
337    /// will match `tests/ui`. A command-line value of `ui` would also
338    /// match `tests/ui`.
339    Suite(TaskPath),
340}
341
342impl PathSet {
343    fn empty() -> PathSet {
344        PathSet::Set(BTreeSet::new())
345    }
346
347    fn one<P: Into<PathBuf>>(path: P, kind: Kind) -> PathSet {
348        let mut set = BTreeSet::new();
349        set.insert(TaskPath { path: path.into(), kind: Some(kind) });
350        PathSet::Set(set)
351    }
352
353    fn has(&self, needle: &Path, module: Kind) -> bool {
354        match self {
355            PathSet::Set(set) => set.iter().any(|p| Self::check(p, needle, module)),
356            PathSet::Suite(suite) => Self::check(suite, needle, module),
357        }
358    }
359
360    // internal use only
361    fn check(p: &TaskPath, needle: &Path, module: Kind) -> bool {
362        let check_path = || {
363            // This order is important for retro-compatibility, as `starts_with` was introduced later.
364            p.path.ends_with(needle) || p.path.starts_with(needle)
365        };
366        if let Some(p_kind) = &p.kind { check_path() && *p_kind == module } else { check_path() }
367    }
368
369    /// Return all `TaskPath`s in `Self` that contain any of the `needles`, removing the
370    /// matched needles.
371    ///
372    /// This is used for `StepDescription::krate`, which passes all matching crates at once to
373    /// `Step::make_run`, rather than calling it many times with a single crate.
374    /// See `tests.rs` for examples.
375    fn intersection_removing_matches(&self, needles: &mut [CLIStepPath], module: Kind) -> PathSet {
376        let mut check = |p| {
377            let mut result = false;
378            for n in needles.iter_mut() {
379                let matched = Self::check(p, &n.path, module);
380                if matched {
381                    n.will_be_executed = true;
382                    result = true;
383                }
384            }
385            result
386        };
387        match self {
388            PathSet::Set(set) => PathSet::Set(set.iter().filter(|&p| check(p)).cloned().collect()),
389            PathSet::Suite(suite) => {
390                if check(suite) {
391                    self.clone()
392                } else {
393                    PathSet::empty()
394                }
395            }
396        }
397    }
398
399    /// A convenience wrapper for Steps which know they have no aliases and all their sets contain only a single path.
400    ///
401    /// This can be used with [`ShouldRun::crate_or_deps`], [`ShouldRun::path`], or [`ShouldRun::alias`].
402    #[track_caller]
403    pub fn assert_single_path(&self) -> &TaskPath {
404        match self {
405            PathSet::Set(set) => {
406                assert_eq!(set.len(), 1, "called assert_single_path on multiple paths");
407                set.iter().next().unwrap()
408            }
409            PathSet::Suite(_) => unreachable!("called assert_single_path on a Suite path"),
410        }
411    }
412}

447fn remap_paths(paths: &mut Vec<PathBuf>) {
448    let mut remove = vec![];
449    let mut add = vec![];
450    for (i, path) in paths.iter().enumerate().filter_map(|(i, path)| path.to_str().map(|s| (i, s)))
451    {
452        for &(search, replace) in PATH_REMAP {
453            // Remove leading and trailing slashes so `tests/` and `tests` are equivalent
454            if path.trim_matches(std::path::is_separator) == search {
455                remove.push(i);
456                add.extend(replace.iter().map(PathBuf::from));
457                break;
458            }
459        }
460    }
461    remove.sort();
462    remove.dedup();
463    for idx in remove.into_iter().rev() {
464        paths.remove(idx);
465    }
466    paths.append(&mut add);
467}

495impl StepDescription {
496    fn from<S: Step>(kind: Kind) -> StepDescription {
497        StepDescription {
498            default: S::DEFAULT,
499            only_hosts: S::ONLY_HOSTS,
500            should_run: S::should_run,
501            make_run: S::make_run,
502            name: std::any::type_name::<S>(),
503            kind,
504        }
505    }
506
507    fn maybe_run(&self, builder: &Builder<'_>, mut pathsets: Vec<PathSet>) {
508        pathsets.retain(|set| !self.is_excluded(builder, set));
509
510        if pathsets.is_empty() {
511            return;
512        }
513
514        // Determine the targets participating in this rule.
515        let targets = if self.only_hosts { &builder.hosts } else { &builder.targets };
516
517        for target in targets {
518            let run = RunConfig { builder, paths: pathsets.clone(), target: *target };
519            (self.make_run)(run);
520        }
521    }
522
523    fn is_excluded(&self, builder: &Builder<'_>, pathset: &PathSet) -> bool {
524        if builder.config.skip.iter().any(|e| pathset.has(e, builder.kind)) {
525            if !matches!(builder.config.get_dry_run(), DryRun::SelfCheck) {
526                println!("Skipping {pathset:?} because it is excluded");
527            }
528            return true;
529        }
530
531        if !builder.config.skip.is_empty()
532            && !matches!(builder.config.get_dry_run(), DryRun::SelfCheck)
533        {
534            builder.verbose(|| {
535                println!(
536                    "{:?} not skipped for {:?} -- not in {:?}",
537                    pathset, self.name, builder.config.skip
538                )
539            });
540        }
541        false
542    }
543
544    fn run(v: &[StepDescription], builder: &Builder<'_>, paths: &[PathBuf]) {
545        let should_runs = v
546            .iter()
547            .map(|desc| (desc.should_run)(ShouldRun::new(builder, desc.kind)))
548            .collect::<Vec<_>>();
549
550        if builder.download_rustc() && (builder.kind == Kind::Dist || builder.kind == Kind::Install)
551        {
552            eprintln!(
553                "ERROR: '{}' subcommand is incompatible with `rust.download-rustc`.",
554                builder.kind.as_str()
555            );
556            crate::exit!(1);
557        }
558
559        // sanity checks on rules
560        for (desc, should_run) in v.iter().zip(&should_runs) {
561            assert!(
562                !should_run.paths.is_empty(),
563                "{:?} should have at least one pathset",
564                desc.name
565            );
566        }
567
568        if paths.is_empty() || builder.config.include_default_paths {
569            for (desc, should_run) in v.iter().zip(&should_runs) {
570                if desc.default && should_run.is_really_default() {
571                    desc.maybe_run(builder, should_run.paths.iter().cloned().collect());
572                }
573            }
574        }
575
576        // Attempt to resolve paths to be relative to the builder source directory.
577        let mut paths: Vec<PathBuf> = paths
578            .iter()
579            .map(|p| {
580                // If the path does not exist, it may represent the name of a Step, such as `tidy` in `x test tidy`
581                if !p.exists() {
582                    return p.clone();
583                }
584
585                // Make the path absolute, strip the prefix, and convert to a PathBuf.
586                match std::path::absolute(p) {
587                    Ok(p) => p.strip_prefix(&builder.src).unwrap_or(&p).to_path_buf(),
588                    Err(e) => {
589                        eprintln!("ERROR: {e:?}");
590                        panic!("Due to the above error, failed to resolve path: {p:?}");
591                    }
592                }
593            })
594            .collect();
595
596        remap_paths(&mut paths);
597
598        // Handle all test suite paths.
599        // (This is separate from the loop below to avoid having to handle multiple paths in `is_suite_path` somehow.)
600        paths.retain(|path| {
601            for (desc, should_run) in v.iter().zip(&should_runs) {
602                if let Some(suite) = should_run.is_suite_path(path) {
603                    desc.maybe_run(builder, vec![suite.clone()]);
604                    return false;
605                }
606            }
607            true
608        });
609
610        if paths.is_empty() {
611            return;
612        }
613
614        let mut paths: Vec<CLIStepPath> = paths.into_iter().map(|p| p.into()).collect();
615        let mut path_lookup: Vec<(CLIStepPath, bool)> =
616            paths.clone().into_iter().map(|p| (p, false)).collect();
617
618        // List of `(usize, &StepDescription, Vec<PathSet>)` where `usize` is the closest index of a path
619        // compared to the given CLI paths. So we can respect to the CLI order by using this value to sort
620        // the steps.
621        let mut steps_to_run = vec![];
622
623        for (desc, should_run) in v.iter().zip(&should_runs) {
624            let pathsets = should_run.pathset_for_paths_removing_matches(&mut paths, desc.kind);
625
626            // This value is used for sorting the step execution order.
627            // By default, `usize::MAX` is used as the index for steps to assign them the lowest priority.
628            //
629            // If we resolve the step's path from the given CLI input, this value will be updated with
630            // the step's actual index.
631            let mut closest_index = usize::MAX;
632
633            // Find the closest index from the original list of paths given by the CLI input.
634            for (index, (path, is_used)) in path_lookup.iter_mut().enumerate() {
635                if !*is_used && !paths.contains(path) {
636                    closest_index = index;
637                    *is_used = true;
638                    break;
639                }
640            }
641
642            steps_to_run.push((closest_index, desc, pathsets));
643        }
644
645        // Sort the steps before running them to respect the CLI order.
646        steps_to_run.sort_by_key(|(index, _, _)| *index);
647
648        // Handle all PathSets.
649        for (_index, desc, pathsets) in steps_to_run {
650            if !pathsets.is_empty() {
651                desc.maybe_run(builder, pathsets);
652            }
653        }
654
655        paths.retain(|p| !p.will_be_executed);
656
657        if !paths.is_empty() {
658            eprintln!("ERROR: no `{}` rules matched {:?}", builder.kind.as_str(), paths);
659            eprintln!(
660                "HELP: run `x.py {} --help --verbose` to show a list of available paths",

671enum ReallyDefault<'a> {
672    Bool(bool),
673    Lazy(LazyLock<bool, Box<dyn Fn() -> bool + 'a>>),
674}
675
676pub struct ShouldRun<'a> {
677    pub builder: &'a Builder<'a>,
678    kind: Kind,
679
680    // use a BTreeSet to maintain sort order
681    paths: BTreeSet<PathSet>,
682
683    // If this is a default rule, this is an additional constraint placed on
684    // its run. Generally something like compiler docs being enabled.
685    is_really_default: ReallyDefault<'a>,
686}
687
688impl<'a> ShouldRun<'a> {
689    fn new(builder: &'a Builder<'_>, kind: Kind) -> ShouldRun<'a> {
690        ShouldRun {
691            builder,
692            kind,
693            paths: BTreeSet::new(),
694            is_really_default: ReallyDefault::Bool(true), // by default no additional conditions
695        }
696    }
697
698    pub fn default_condition(mut self, cond: bool) -> Self {
699        self.is_really_default = ReallyDefault::Bool(cond);
700        self
701    }
702
703    pub fn lazy_default_condition(mut self, lazy_cond: Box<dyn Fn() -> bool + 'a>) -> Self {
704        self.is_really_default = ReallyDefault::Lazy(LazyLock::new(lazy_cond));
705        self
706    }
707
708    pub fn is_really_default(&self) -> bool {
709        match &self.is_really_default {
710            ReallyDefault::Bool(val) => *val,
711            ReallyDefault::Lazy(lazy) => *lazy.deref(),
712        }
713    }
714
715    /// Indicates it should run if the command-line selects the given crate or
716    /// any of its (local) dependencies.
717    ///
718    /// `make_run` will be called a single time with all matching command-line paths.
719    pub fn crate_or_deps(self, name: &str) -> Self {
720        let crates = self.builder.in_tree_crates(name, None);
721        self.crates(crates)
722    }
723
724    /// Indicates it should run if the command-line selects any of the given crates.
725    ///
726    /// `make_run` will be called a single time with all matching command-line paths.
727    ///
728    /// Prefer [`ShouldRun::crate_or_deps`] to this function where possible.
729    pub(crate) fn crates(mut self, crates: Vec<&Crate>) -> Self {
730        for krate in crates {
731            let path = krate.local_path(self.builder);
732            self.paths.insert(PathSet::one(path, self.kind));
733        }
734        self
735    }
736
737    // single alias, which does not correspond to any on-disk path
738    pub fn alias(mut self, alias: &str) -> Self {
739        // exceptional case for `Kind::Setup` because its `library`
740        // and `compiler` options would otherwise naively match with
741        // `compiler` and `library` folders respectively.
742        assert!(
743            self.kind == Kind::Setup || !self.builder.src.join(alias).exists(),
744            "use `builder.path()` for real paths: {alias}"
745        );
746        self.paths.insert(PathSet::Set(
747            std::iter::once(TaskPath { path: alias.into(), kind: Some(self.kind) }).collect(),
748        ));
749        self
750    }
751
752    /// single, non-aliased path
753    ///
754    /// Must be an on-disk path; use `alias` for names that do not correspond to on-disk paths.
755    pub fn path(self, path: &str) -> Self {
756        self.paths(&[path])
757    }
758
759    /// Multiple aliases for the same job.
760    ///
761    /// This differs from [`path`] in that multiple calls to path will end up calling `make_run`
762    /// multiple times, whereas a single call to `paths` will only ever generate a single call to
763    /// `make_run`.
764    ///
765    /// This is analogous to `all_krates`, although `all_krates` is gone now. Prefer [`path`] where possible.
766    ///
767    /// [`path`]: ShouldRun::path
768    pub fn paths(mut self, paths: &[&str]) -> Self {
769        let submodules_paths = self.builder.submodule_paths();
770
771        self.paths.insert(PathSet::Set(
772            paths
773                .iter()
774                .map(|p| {
775                    // assert only if `p` isn't submodule
776                    if !submodules_paths.iter().any(|sm_p| p.contains(sm_p)) {
777                        assert!(
778                            self.builder.src.join(p).exists(),
779                            "`should_run.paths` should correspond to real on-disk paths - use `alias` if there is no relevant path: {p}"
780                        );
781                    }
782
783                    TaskPath { path: p.into(), kind: Some(self.kind) }
784                })
785                .collect(),
786        ));
787        self
788    }
789
790    /// Handles individual files (not directories) within a test suite.
791    fn is_suite_path(&self, requested_path: &Path) -> Option<&PathSet> {
792        self.paths.iter().find(|pathset| match pathset {
793            PathSet::Suite(suite) => requested_path.starts_with(&suite.path),
794            PathSet::Set(_) => false,
795        })
796    }
797
798    pub fn suite_path(mut self, suite: &str) -> Self {
799        self.paths.insert(PathSet::Suite(TaskPath { path: suite.into(), kind: Some(self.kind) }));
800        self
801    }
802
803    // allows being more explicit about why should_run in Step returns the value passed to it
804    pub fn never(mut self) -> ShouldRun<'a> {
805        self.paths.insert(PathSet::empty());
806        self
807    }
808
809    /// Given a set of requested paths, return the subset which match the Step for this `ShouldRun`,
810    /// removing the matches from `paths`.
811    ///
812    /// NOTE: this returns multiple PathSets to allow for the possibility of multiple units of work
813    /// within the same step. For example, `test::Crate` allows testing multiple crates in the same
814    /// cargo invocation, which are put into separate sets because they aren't aliases.
815    ///
816    /// The reason we return PathSet instead of PathBuf is to allow for aliases that mean the same thing
817    /// (for now, just `all_krates` and `paths`, but we may want to add an `aliases` function in the future?)
818    fn pathset_for_paths_removing_matches(
819        &self,
820        paths: &mut [CLIStepPath],
821        kind: Kind,
822    ) -> Vec<PathSet> {
823        let mut sets = vec![];
824        for pathset in &self.paths {
825            let subset = pathset.intersection_removing_matches(paths, kind);
826            if subset != PathSet::empty() {
827                sets.push(subset);
828            }
829        }
830        sets
831    }
832}

1219    pub fn get_help(build: &Build, kind: Kind) -> Option<String> {
1220        let step_descriptions = Builder::get_step_descriptions(kind);
1221        if step_descriptions.is_empty() {
1222            return None;
1223        }
1224
1225        let builder = Self::new_internal(build, kind, vec![]);
1226        let builder = &builder;
1227        // The "build" kind here is just a placeholder, it will be replaced with something else in
1228        // the following statement.
1229        let mut should_run = ShouldRun::new(builder, Kind::Build);
1230        for desc in step_descriptions {
1231            should_run.kind = desc.kind;
1232            should_run = (desc.should_run)(should_run);
1233        }
1234        let mut help = String::from("Available paths:\n");
1235        let mut add_path = |path: &Path| {
1236            t!(write!(help, "    ./x.py {} {}\n", kind.as_str(), path.display()));
1237        };
1238        for pathset in should_run.paths {
1239            match pathset {
1240                PathSet::Set(set) => {
1241                    for path in set {
1242                        add_path(&path.path);
1243                    }
1244                }
1245                PathSet::Suite(path) => {
1246                    add_path(&path.path.join("..."));
1247                }
1248            }
1249        }
1250        Some(help)
1251    }

1396    /// Obtain a standard library for the given target that will be built by the passed compiler.
1397    /// The standard library will be linked to the sysroot of the passed compiler.
1398    ///
1399    /// Prefer using this method rather than manually invoking `Std::new`.
1400    #[cfg_attr(
1401        feature = "tracing",
1402        instrument(
1403            level = "trace",
1404            name = "Builder::std",
1405            target = "STD",
1406            skip_all,
1407            fields(
1408                compiler = ?compiler,
1409                target = ?target,
1410            ),
1411        ),
1412    )]
1413    pub fn std(&self, compiler: Compiler, target: TargetSelection) {
1414        // FIXME: make the `Std` step return some type-level "proof" that std was indeed built,
1415        // and then require passing that to all Cargo invocations that we do.
1416
1417        // The "stage 0" std is always precompiled and comes with the stage0 compiler, so we have
1418        // special logic for it, to avoid creating needless and confusing Std steps that don't
1419        // actually build anything.
1420        if compiler.stage == 0 {
1421            if target != compiler.host {
1422                panic!(
1423                    r"It is not possible to build the standard library for `{target}` using the stage0 compiler.
1424You have to build a stage1 compiler for `{}` first, and then use it to build a standard library for `{target}`.
1425",
1426                    compiler.host
1427                )
1428            }
1429
1430            // We still need to link the prebuilt standard library into the ephemeral stage0 sysroot
1431            self.ensure(StdLink::from_std(Std::new(compiler, target), compiler));
1432        } else {
1433            // This step both compiles the std and links it into the compiler's sysroot.
1434            // Yes, it's quite magical and side-effecty.. would be nice to refactor later.
1435            self.ensure(Std::new(compiler, target));
1436        }
1437    }

1584        assert!(run_compiler.stage > 0, "miri can not be invoked at stage 0");
1585

1587            RustcPrivateCompilers::new(self, run_compiler.stage, self.build.host_target);

1605        add_dylib_path(self.rustc_lib_paths(run_compiler), &mut cmd);

1657    /// Ensure that a given step is built, returning its output. This will
1658    /// cache the step, so it is safe (and good!) to call this as often as
1659    /// needed to ensure that all dependencies are built.
1660    pub fn ensure<S: Step>(&'a self, step: S) -> S::Output {
1661        {
1662            let mut stack = self.stack.borrow_mut();
1663            for stack_step in stack.iter() {
1664                // should skip
1665                if stack_step.downcast_ref::<S>().is_none_or(|stack_step| *stack_step != step) {
1666                    continue;
1667                }
1668                let mut out = String::new();
1669                out += &format!("\n\nCycle in build detected when adding {step:?}\n");
1670                for el in stack.iter().rev() {
1671                    out += &format!("\t{el:?}\n");
1672                }
1673                panic!("{}", out);
1674            }
1675            if let Some(out) = self.cache.get(&step) {
1676                self.verbose_than(1, || println!("{}c {:?}", "  ".repeat(stack.len()), step));
1677
1678                #[cfg(feature = "tracing")]
1679                {
1680                    if let Some(parent) = stack.last() {
1681                        let mut graph = self.build.step_graph.borrow_mut();
1682                        graph.register_cached_step(&step, parent, self.config.dry_run());
1683                    }
1684                }
1685                return out;
1686            }
1687            self.verbose_than(1, || println!("{}> {:?}", "  ".repeat(stack.len()), step));
1688
1689            #[cfg(feature = "tracing")]
1690            {
1691                let parent = stack.last();
1692                let mut graph = self.build.step_graph.borrow_mut();
1693                graph.register_step_execution(&step, parent, self.config.dry_run());
1694            }
1695
1696            stack.push(Box::new(step.clone()));
1697        }
1698
1699        #[cfg(feature = "build-metrics")]
1700        self.metrics.enter_step(&step, self);
1701
1702        let (out, dur) = {
1703            let start = Instant::now();
1704            let zero = Duration::new(0, 0);
1705            let parent = self.time_spent_on_dependencies.replace(zero);
1706            let out = step.clone().run(self);
1707            let dur = start.elapsed();
1708            let deps = self.time_spent_on_dependencies.replace(parent + dur);
1709            (out, dur.saturating_sub(deps))
1710        };
1711
1712        if self.config.print_step_timings && !self.config.dry_run() {
1713            let step_string = format!("{step:?}");
1714            let brace_index = step_string.find('{').unwrap_or(0);
1715            let type_string = type_name::<S>();
1716            println!(
1717                "[TIMING] {} {} -- {}.{:03}",
1718                &type_string.strip_prefix("bootstrap::").unwrap_or(type_string),
1719                &step_string[brace_index..],
1720                dur.as_secs(),
1721                dur.subsec_millis()
1722            );
1723        }
1724
1725        #[cfg(feature = "build-metrics")]
1726        self.metrics.exit_step(self);
1727
1728        {
1729            let mut stack = self.stack.borrow_mut();
1730            let cur_step = stack.pop().expect("step stack empty");
1731            assert_eq!(cur_step.downcast_ref(), Some(&step));
1732        }
1733        self.verbose_than(1, || println!("{}< {:?}", "  ".repeat(self.stack.borrow().len()), step));
1734        self.cache.put(step, out.clone());
1735        out
1736    }

1738    /// Ensure that a given step is built *only if it's supposed to be built by default*, returning
1739    /// its output. This will cache the step, so it's safe (and good!) to call this as often as
1740    /// needed to ensure that all dependencies are build.
1741    pub(crate) fn ensure_if_default<T, S: Step<Output = T>>(
1742        &'a self,
1743        step: S,
1744        kind: Kind,
1745    ) -> Option<S::Output> {
1746        let desc = StepDescription::from::<S>(kind);
1747        let should_run = (desc.should_run)(ShouldRun::new(self, desc.kind));
1748
1749        // Avoid running steps contained in --skip
1750        for pathset in &should_run.paths {
1751            if desc.is_excluded(self, pathset) {
1752                return None;
1753            }
1754        }
1755
1756        // Only execute if it's supposed to run as default
1757        if desc.default && should_run.is_really_default() { Some(self.ensure(step)) } else { None }
1758    }
