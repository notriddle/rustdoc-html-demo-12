60fn target_from_impl_item<'tcx>(tcx: TyCtxt<'tcx>, impl_item: &hir::ImplItem<'_>) -> Target {
61    match impl_item.kind {
62        hir::ImplItemKind::Const(..) => Target::AssocConst,
63        hir::ImplItemKind::Fn(..) => {
64            let parent_def_id = tcx.hir_get_parent_item(impl_item.hir_id()).def_id;
65            let containing_item = tcx.hir_expect_item(parent_def_id);
66            let containing_impl_is_for_trait = match &containing_item.kind {
67                hir::ItemKind::Impl(impl_) => impl_.of_trait.is_some(),
68                _ => bug!("parent of an ImplItem must be an Impl"),
69            };
70            if containing_impl_is_for_trait {
71                Target::Method(MethodKind::Trait { body: true })
72            } else {
73                Target::Method(MethodKind::Inherent)
74            }
75        }
76        hir::ImplItemKind::Type(..) => Target::AssocTy,
77    }
78}

116    /// Checks any attribute.
117    fn check_attributes(
118        &self,
119        hir_id: HirId,
120        span: Span,
121        target: Target,
122        item: Option<ItemLike<'_>>,
123    ) {
124        let mut doc_aliases = FxHashMap::default();
125        let mut specified_inline = None;
126        let mut seen = FxHashMap::default();
127        let attrs = self.tcx.hir_attrs(hir_id);
128        for attr in attrs {
129            let mut style = None;
130            match attr {
131                Attribute::Parsed(AttributeKind::ProcMacro(_)) => {
132                    self.check_proc_macro(hir_id, target, ProcMacroKind::FunctionLike)
133                }
134                Attribute::Parsed(AttributeKind::ProcMacroAttribute(_)) => {
135                    self.check_proc_macro(hir_id, target, ProcMacroKind::Attribute);
136                }
137                Attribute::Parsed(AttributeKind::ProcMacroDerive { span: attr_span, .. }) => {
138                    self.check_generic_attr(
139                        hir_id,
140                        sym::proc_macro_derive,
141                        *attr_span,
142                        target,
143                        Target::Fn,
144                    );
145                    self.check_proc_macro(hir_id, target, ProcMacroKind::Derive)
146                }
147                Attribute::Parsed(
148                    AttributeKind::SkipDuringMethodDispatch { span: attr_span, .. }
149                    | AttributeKind::Coinductive(attr_span)
150                    | AttributeKind::ConstTrait(attr_span)
151                    | AttributeKind::DenyExplicitImpl(attr_span)
152                    | AttributeKind::DoNotImplementViaObject(attr_span),
153                ) => {
154                    self.check_must_be_applied_to_trait(*attr_span, span, target);
155                }
156                &Attribute::Parsed(
157                    AttributeKind::SpecializationTrait(attr_span)
158                    | AttributeKind::UnsafeSpecializationMarker(attr_span)
159                    | AttributeKind::ParenSugar(attr_span),
160                ) => {
161                    // FIXME: more validation is needed
162                    self.check_must_be_applied_to_trait(attr_span, span, target);
163                }
164                &Attribute::Parsed(AttributeKind::TypeConst(attr_span)) => {
165                    self.check_type_const(hir_id, attr_span, target)
166                }
167                &Attribute::Parsed(AttributeKind::Marker(attr_span)) => {
168                    self.check_marker(hir_id, attr_span, span, target)
169                }
170                Attribute::Parsed(AttributeKind::Fundamental | AttributeKind::CoherenceIsCore) => {
171                    // FIXME: add validation
172                }
173                &Attribute::Parsed(AttributeKind::AllowIncoherentImpl(attr_span)) => {
174                    self.check_allow_incoherent_impl(attr_span, span, target)
175                }
176                Attribute::Parsed(AttributeKind::Confusables { first_span, .. }) => {
177                    self.check_confusables(*first_span, target);
178                }
179                Attribute::Parsed(AttributeKind::AutomaticallyDerived(attr_span)) => self
180                    .check_generic_attr(
181                        hir_id,
182                        sym::automatically_derived,
183                        *attr_span,
184                        target,
185                        Target::Impl { of_trait: true },
186                    ),
187                Attribute::Parsed(
188                    AttributeKind::Stability {
189                        span: attr_span,
190                        stability: Stability { level, feature },
191                    }
192                    | AttributeKind::ConstStability {
193                        span: attr_span,
194                        stability: PartialConstStability { level, feature, .. },
195                    },
196                ) => self.check_stability(*attr_span, span, level, *feature, target),
197                Attribute::Parsed(AttributeKind::Inline(InlineAttr::Force { .. }, ..)) => {} // handled separately below
198                Attribute::Parsed(AttributeKind::Inline(kind, attr_span)) => {
199                    self.check_inline(hir_id, *attr_span, span, kind, target)
200                }
201                Attribute::Parsed(AttributeKind::Optimize(_, attr_span)) => {
202                    self.check_optimize(hir_id, *attr_span, span, target)
203                }
204                Attribute::Parsed(AttributeKind::LoopMatch(attr_span)) => {
205                    self.check_loop_match(hir_id, *attr_span, target)
206                }
207                Attribute::Parsed(AttributeKind::ConstContinue(attr_span)) => {
208                    self.check_const_continue(hir_id, *attr_span, target)
209                }
210                Attribute::Parsed(AttributeKind::AllowInternalUnstable(_, first_span)) => {
211                    self.check_allow_internal_unstable(hir_id, *first_span, span, target, attrs)
212                }
213                Attribute::Parsed(AttributeKind::AllowConstFnUnstable(_, first_span)) => {
214                    self.check_rustc_allow_const_fn_unstable(hir_id, *first_span, span, target)
215                }
216                Attribute::Parsed(AttributeKind::Deprecation { .. }) => {
217                    self.check_deprecated(hir_id, attr, span, target)
218                }
219                Attribute::Parsed(AttributeKind::TargetFeature(_, attr_span)) => {
220                    self.check_target_feature(hir_id, *attr_span, span, target, attrs)
221                }
222                Attribute::Parsed(AttributeKind::DocComment { .. }) => { /* `#[doc]` is actually a lot more than just doc comments, so is checked below*/
223                }
224                Attribute::Parsed(AttributeKind::Repr { .. }) => { /* handled below this loop and elsewhere */
225                }
226                Attribute::Parsed(AttributeKind::RustcObjectLifetimeDefault) => {
227                    self.check_object_lifetime_default(hir_id);
228                }
229                &Attribute::Parsed(AttributeKind::PubTransparent(attr_span)) => {
230                    self.check_rustc_pub_transparent(attr_span, span, attrs)
231                }
232                Attribute::Parsed(AttributeKind::Cold(attr_span)) => {
233                    self.check_cold(hir_id, *attr_span, span, target)
234                }
235                Attribute::Parsed(AttributeKind::ExportName { span: attr_span, .. }) => {
236                    self.check_export_name(hir_id, *attr_span, span, target)
237                }
238                Attribute::Parsed(AttributeKind::Align { align, span: attr_span }) => {
239                    self.check_align(span, hir_id, target, *align, *attr_span)
240                }
241                Attribute::Parsed(AttributeKind::LinkSection { span: attr_span, .. }) => {
242                    self.check_link_section(hir_id, *attr_span, span, target)
243                }
244                Attribute::Parsed(AttributeKind::MacroUse { span, .. }) => {
245                    self.check_macro_use(hir_id, sym::macro_use, *span, target)
246                }
247                Attribute::Parsed(AttributeKind::MacroEscape(span)) => {
248                    self.check_macro_use(hir_id, sym::macro_escape, *span, target)
249                }
250                Attribute::Parsed(AttributeKind::Naked(attr_span)) => {
251                    self.check_naked(hir_id, *attr_span, span, target)
252                }
253                Attribute::Parsed(AttributeKind::NoImplicitPrelude(attr_span)) => self
254                    .check_generic_attr(
255                        hir_id,
256                        sym::no_implicit_prelude,
257                        *attr_span,
258                        target,
259                        Target::Mod,
260                    ),
261                Attribute::Parsed(AttributeKind::Path(_, attr_span)) => {
262                    self.check_generic_attr(hir_id, sym::path, *attr_span, target, Target::Mod)
263                }
264                Attribute::Parsed(AttributeKind::TrackCaller(attr_span)) => {
265                    self.check_track_caller(hir_id, *attr_span, attrs, span, target)
266                }
267                Attribute::Parsed(AttributeKind::NonExhaustive(attr_span)) => {
268                    self.check_non_exhaustive(hir_id, *attr_span, span, target, item)
269                }
270                Attribute::Parsed(
271                    AttributeKind::RustcLayoutScalarValidRangeStart(_num, attr_span)
272                    | AttributeKind::RustcLayoutScalarValidRangeEnd(_num, attr_span),
273                ) => self.check_rustc_layout_scalar_valid_range(*attr_span, span, target),
274                Attribute::Parsed(AttributeKind::ExportStable) => {
275                    // handled in `check_export`
276                }
277                &Attribute::Parsed(AttributeKind::FfiConst(attr_span)) => {
278                    self.check_ffi_const(attr_span, target)
279                }
280                &Attribute::Parsed(AttributeKind::FfiPure(attr_span)) => {
281                    self.check_ffi_pure(attr_span, attrs, target)
282                }
283                Attribute::Parsed(AttributeKind::UnstableFeatureBound(syms)) => {
284                    self.check_unstable_feature_bound(syms.first().unwrap().1, span, target)
285                }
286                Attribute::Parsed(
287                    AttributeKind::BodyStability { .. }
288                    | AttributeKind::ConstStabilityIndirect
289                    | AttributeKind::MacroTransparency(_)
290                    | AttributeKind::Pointee(..)
291                    | AttributeKind::Dummy
292                    | AttributeKind::RustcBuiltinMacro { .. },
293                ) => { /* do nothing  */ }
294                Attribute::Parsed(AttributeKind::AsPtr(attr_span)) => {
295                    self.check_applied_to_fn_or_method(hir_id, *attr_span, span, target)
296                }
297                Attribute::Parsed(AttributeKind::LinkName { span: attr_span, name }) => {
298                    self.check_link_name(hir_id, *attr_span, *name, span, target)
299                }
300                Attribute::Parsed(AttributeKind::LinkOrdinal { span: attr_span, .. }) => {
301                    self.check_link_ordinal(*attr_span, span, target)
302                }
303                Attribute::Parsed(AttributeKind::MayDangle(attr_span)) => {
304                    self.check_may_dangle(hir_id, *attr_span)
305                }
306                Attribute::Parsed(AttributeKind::Ignore { span, .. }) => {
307                    self.check_generic_attr(hir_id, sym::ignore, *span, target, Target::Fn)
308                }
309                Attribute::Parsed(AttributeKind::MustUse { span, .. }) => {
310                    self.check_must_use(hir_id, *span, target)
311                }
312                Attribute::Parsed(AttributeKind::NoMangle(attr_span)) => {
313                    self.check_no_mangle(hir_id, *attr_span, span, target)
314                }
315                Attribute::Parsed(AttributeKind::Used { span: attr_span, .. }) => {
316                    self.check_used(*attr_span, target, span);
317                }
318                Attribute::Parsed(AttributeKind::ShouldPanic { span: attr_span, .. }) => self
319                    .check_generic_attr(hir_id, sym::should_panic, *attr_span, target, Target::Fn),
320                &Attribute::Parsed(AttributeKind::PassByValue(attr_span)) => {
321                    self.check_pass_by_value(attr_span, span, target)
322                }
323                &Attribute::Parsed(AttributeKind::StdInternalSymbol(attr_span)) => {
324                    self.check_rustc_std_internal_symbol(attr_span, span, target)
325                }
326                &Attribute::Parsed(AttributeKind::Coverage(attr_span, _)) => {
327                    self.check_coverage(attr_span, span, target)
328                }
329                &Attribute::Parsed(AttributeKind::Coroutine(attr_span)) => {
330                    self.check_coroutine(attr_span, target)
331                }
332                Attribute::Unparsed(attr_item) => {
333                    style = Some(attr_item.style);
334                    match attr.path().as_slice() {
335                        [sym::diagnostic, sym::do_not_recommend, ..] => {
336                            self.check_do_not_recommend(attr.span(), hir_id, target, attr, item)
337                        }
338                        [sym::diagnostic, sym::on_unimplemented, ..] => {
339                            self.check_diagnostic_on_unimplemented(attr.span(), hir_id, target)
340                        }
341                        [sym::no_sanitize, ..] => {
342                            self.check_no_sanitize(attr, span, target)
343                        }
344                        [sym::thread_local, ..] => self.check_thread_local(attr, span, target),
345                        [sym::doc, ..] => self.check_doc_attrs(
346                            attr,
347                            attr_item.style,
348                            hir_id,
349                            target,
350                            &mut specified_inline,
351                            &mut doc_aliases,
352                        ),
353                        [sym::no_link, ..] => self.check_no_link(hir_id, attr, span, target),
354                        [sym::debugger_visualizer, ..] => self.check_debugger_visualizer(attr, target),
355                        [sym::rustc_no_implicit_autorefs, ..] => {
356                            self.check_applied_to_fn_or_method(hir_id, attr.span(), span, target)
357                        }
358                        [sym::rustc_never_returns_null_ptr, ..] => {
359                            self.check_applied_to_fn_or_method(hir_id, attr.span(), span, target)
360                        }
361                        [sym::rustc_legacy_const_generics, ..] => {
362                            self.check_rustc_legacy_const_generics(hir_id, attr, span, target, item)
363                        }
364                        [sym::rustc_lint_query_instability, ..] => {
365                            self.check_applied_to_fn_or_method(hir_id, attr.span(), span, target)
366                        }
367                        [sym::rustc_lint_untracked_query_information, ..] => {
368                            self.check_applied_to_fn_or_method(hir_id, attr.span(), span, target)
369                        }
370                        [sym::rustc_lint_diagnostics, ..] => {
371                            self.check_applied_to_fn_or_method(hir_id, attr.span(), span, target)
372                        }
373                        [sym::rustc_lint_opt_ty, ..] => self.check_rustc_lint_opt_ty(attr, span, target),
374                        [sym::rustc_lint_opt_deny_field_access, ..] => {
375                            self.check_rustc_lint_opt_deny_field_access(attr, span, target)
376                        }
377                        [sym::rustc_clean, ..]
378                        | [sym::rustc_dirty, ..]
379                        | [sym::rustc_if_this_changed, ..]
380                        | [sym::rustc_then_this_would_need, ..] => self.check_rustc_dirty_clean(attr),
381                        [sym::rustc_must_implement_one_of, ..] => self.check_must_be_applied_to_trait(attr.span(), span, target),
382                        [sym::collapse_debuginfo, ..] => self.check_collapse_debuginfo(attr, span, target),
383                        [sym::must_not_suspend, ..] => self.check_must_not_suspend(attr, span, target),
384                        [sym::rustc_has_incoherent_inherent_impls, ..] => {
385                            self.check_has_incoherent_inherent_impls(attr, span, target)
386                        }
387                        [sym::ffi_pure, ..] => self.check_ffi_pure(attr.span(), attrs, target),
388                        [sym::ffi_const, ..] => self.check_ffi_const(attr.span(), target),
389                        [sym::link, ..] => self.check_link(hir_id, attr, span, target),
390                        [sym::path, ..] => self.check_generic_attr_unparsed(hir_id, attr, target, Target::Mod),
391                        [sym::macro_export, ..] => self.check_macro_export(hir_id, attr, target),
392                        [sym::autodiff_forward, ..] | [sym::autodiff_reverse, ..] => {
393                            self.check_autodiff(hir_id, attr, span, target)
394                        }
395                        [sym::linkage, ..] => self.check_linkage(attr, span, target),
396                        [
397                            // ok
398                            sym::allow
399                            | sym::expect
400                            | sym::warn
401                            | sym::deny
402                            | sym::forbid
403                            | sym::cfg
404                            | sym::cfg_attr
405                            | sym::cfg_trace
406                            | sym::cfg_attr_trace
407                            // need to be fixed
408                            | sym::cfi_encoding // FIXME(cfi_encoding)
409                            | sym::instruction_set // broken on stable!!!
410                            | sym::windows_subsystem // broken on stable!!!
411                            | sym::patchable_function_entry // FIXME(patchable_function_entry)
412                            | sym::deprecated_safe // FIXME(deprecated_safe)
413                            // internal
414                            | sym::prelude_import
415                            | sym::panic_handler
416                            | sym::allow_internal_unsafe
417                            | sym::lang
418                            | sym::needs_allocator
419                            | sym::default_lib_allocator
420                            | sym::custom_mir,
421                            ..
422                        ] => {}
423                        [name, rest@..] => {
424                            match BUILTIN_ATTRIBUTE_MAP.get(name) {
425                                // checked below
426                                Some(BuiltinAttribute { type_: AttributeType::CrateLevel, .. }) => {}
427                                Some(_) => {
428                                    if rest.len() > 0 && AttributeParser::<Late>::is_parsed_attribute(slice::from_ref(name)) {
429                                        // Check if we tried to use a builtin attribute as an attribute namespace, like `#[must_use::skip]`.
430                                        // This check is here to solve https://github.com/rust-lang/rust/issues/137590
431                                        // An error is already produced for this case elsewhere
432                                        continue
433                                    }
434
435                                    // FIXME: differentiate between unstable and internal attributes just
436                                    // like we do with features instead of just accepting `rustc_`
437                                    // attributes by name. That should allow trimming the above list, too.
438                                    if !name.as_str().starts_with("rustc_") {
439                                        span_bug!(
440                                            attr.span(),
441                                            "builtin attribute {name:?} not handled by `CheckAttrVisitor`"
442                                        )
443                                    }
444                                }
445                                None => (),
446                            }
447                        }
448                        [] => unreachable!(),
449                    }
450                }
451            }
452
453            let builtin = attr.ident().and_then(|ident| BUILTIN_ATTRIBUTE_MAP.get(&ident.name));
454
455            if hir_id != CRATE_HIR_ID {
456                if let Some(BuiltinAttribute { type_: AttributeType::CrateLevel, .. }) =
457                    attr.ident().and_then(|ident| BUILTIN_ATTRIBUTE_MAP.get(&ident.name))
458                {
459                    match style {
460                        Some(ast::AttrStyle::Outer) => self.tcx.emit_node_span_lint(
461                            UNUSED_ATTRIBUTES,
462                            hir_id,
463                            attr.span(),
464                            errors::OuterCrateLevelAttr,
465                        ),
466                        Some(ast::AttrStyle::Inner) | None => self.tcx.emit_node_span_lint(
467                            UNUSED_ATTRIBUTES,
468                            hir_id,
469                            attr.span(),
470                            errors::InnerCrateLevelAttr,
471                        ),
472                    }
473                }
474            }
475
476            if let Some(BuiltinAttribute { duplicates, .. }) = builtin {
477                check_duplicates(self.tcx, attr, hir_id, *duplicates, &mut seen);
478            }
479
480            self.check_unused_attribute(hir_id, attr, style)
481        }
482
483        self.check_repr(attrs, span, target, item, hir_id);
484        self.check_rustc_force_inline(hir_id, attrs, span, target);
485        self.check_mix_no_mangle_export(hir_id, attrs);
486    }

506    /// Checks if `#[diagnostic::do_not_recommend]` is applied on a trait impl and that it has no
507    /// arguments.
508    fn check_do_not_recommend(
509        &self,
510        attr_span: Span,
511        hir_id: HirId,
512        target: Target,
513        attr: &Attribute,
514        item: Option<ItemLike<'_>>,
515    ) {
516        if !matches!(target, Target::Impl { .. })
517            || matches!(
518                item,
519                Some(ItemLike::Item(hir::Item {  kind: hir::ItemKind::Impl(_impl),.. }))
520                    if _impl.of_trait.is_none()
521            )
522        {
523            self.tcx.emit_node_span_lint(
524                MISPLACED_DIAGNOSTIC_ATTRIBUTES,
525                hir_id,
526                attr_span,
527                errors::IncorrectDoNotRecommendLocation,
528            );
529        }
530        if !attr.is_word() {
531            self.tcx.emit_node_span_lint(
532                MALFORMED_DIAGNOSTIC_ATTRIBUTES,
533                hir_id,
534                attr_span,
535                errors::DoNotRecommendDoesNotExpectArgs,
536            );
537        }

975    fn check_doc_alias_value(
976        &self,
977        meta: &MetaItemInner,
978        doc_alias: Symbol,
979        hir_id: HirId,
980        target: Target,
981        is_list: bool,
982        aliases: &mut FxHashMap<String, Span>,
983    ) {
984        let tcx = self.tcx;
985        let span = meta.name_value_literal_span().unwrap_or_else(|| meta.span());
986        let attr_str =
987            &format!("`#[doc(alias{})]`", if is_list { "(\"...\")" } else { " = \"...\"" });
988        if doc_alias == sym::empty {
989            tcx.dcx().emit_err(errors::DocAliasEmpty { span, attr_str });
990            return;
991        }
992
993        let doc_alias_str = doc_alias.as_str();
994        if let Some(c) = doc_alias_str
995            .chars()
996            .find(|&c| c == '"' || c == '\'' || (c.is_whitespace() && c != ' '))
997        {
998            tcx.dcx().emit_err(errors::DocAliasBadChar { span, attr_str, char_: c });
999            return;
1000        }
1001        if doc_alias_str.starts_with(' ') || doc_alias_str.ends_with(' ') {
1002            tcx.dcx().emit_err(errors::DocAliasStartEnd { span, attr_str });
1003            return;
1004        }
1005
1006        let span = meta.span();
1007        if let Some(location) = match target {
1008            Target::AssocTy => {
1009                if let DefKind::Impl { .. } =
1010                    self.tcx.def_kind(self.tcx.local_parent(hir_id.owner.def_id))
1011                {
1012                    Some("type alias in implementation block")
1013                } else {
1014                    None
1015                }
1016            }
1017            Target::AssocConst => {
1018                let parent_def_id = self.tcx.hir_get_parent_item(hir_id).def_id;
1019                let containing_item = self.tcx.hir_expect_item(parent_def_id);
1020                // We can't link to trait impl's consts.
1021                let err = "associated constant in trait implementation block";
1022                match containing_item.kind {
1023                    ItemKind::Impl(hir::Impl { of_trait: Some(_), .. }) => Some(err),
1024                    _ => None,
1025                }
1026            }
1027            // we check the validity of params elsewhere
1028            Target::Param => return,
1029            Target::Expression
1030            | Target::Statement
1031            | Target::Arm
1032            | Target::ForeignMod
1033            | Target::Closure
1034            | Target::Impl { .. }
1035            | Target::WherePredicate => Some(target.name()),
1036            Target::ExternCrate
1037            | Target::Use
1038            | Target::Static
1039            | Target::Const
1040            | Target::Fn
1041            | Target::Mod
1042            | Target::GlobalAsm
1043            | Target::TyAlias
1044            | Target::Enum
1045            | Target::Variant
1046            | Target::Struct
1047            | Target::Field
1048            | Target::Union
1049            | Target::Trait
1050            | Target::TraitAlias
1051            | Target::Method(..)
1052            | Target::ForeignFn
1053            | Target::ForeignStatic
1054            | Target::ForeignTy
1055            | Target::GenericParam { .. }
1056            | Target::MacroDef
1057            | Target::PatField
1058            | Target::ExprField => None,
1059        } {
1060            tcx.dcx().emit_err(errors::DocAliasBadLocation { span, attr_str, location });
1061            return;
1062        }
1063        if self.tcx.hir_opt_name(hir_id) == Some(doc_alias) {
1064            tcx.dcx().emit_err(errors::DocAliasNotAnAlias { span, attr_str });
1065            return;
1066        }
1067        if let Err(entry) = aliases.try_insert(doc_alias_str.to_owned(), span) {
1068            self.tcx.emit_node_span_lint(
1069                UNUSED_ATTRIBUTES,
1070                hir_id,
1071                span,
1072                errors::DocAliasDuplicated { first_defn: *entry.entry.get() },
1073            );
1074        }
1075    }

1781        &self,

1784        span: Span,
1785        target: Target,

1796        }

1804            kind: ItemKind::Fn { sig: FnSig { decl, .. }, generics, .. },
1805            ..
1806        })) = item
1807        else {
1808            bug!("should be a function item");
1809        };

1812            match param.kind {
1813                hir::GenericParamKind::Const { .. } => {}

1826                attr_span: attr.span(),
1827                generics_span: generics.span,
1828            });
1829            return;

1839                        span,
1840                        arg_count: arg_count as usize,

1842                    return;
1843                }
1844            } else {
1845                invalid_args.push(meta.span());
1846            }
1847        }

1850            self.dcx().emit_err(errors::RustcLegacyConstGenericsIndexNegative { invalid_args });

2013    /// Checks if the `#[repr]` attributes on `item` are valid.
2014    fn check_repr(
2015        &self,
2016        attrs: &[Attribute],
2017        span: Span,
2018        target: Target,
2019        item: Option<ItemLike<'_>>,
2020        hir_id: HirId,
2021    ) {
2022        // Extract the names of all repr hints, e.g., [foo, bar, align] for:
2023        // ```
2024        // #[repr(foo)]
2025        // #[repr(bar, align(8))]
2026        // ```
2027        let (reprs, first_attr_span) = find_attr!(attrs, AttributeKind::Repr { reprs, first_span } => (reprs.as_slice(), Some(*first_span))).unwrap_or((&[], None));
2028
2029        let mut int_reprs = 0;
2030        let mut is_explicit_rust = false;
2031        let mut is_c = false;
2032        let mut is_simd = false;
2033        let mut is_transparent = false;
2034
2035        for (repr, repr_span) in reprs {
2036            match repr {
2037                ReprAttr::ReprRust => {
2038                    is_explicit_rust = true;
2039                    match target {
2040                        Target::Struct | Target::Union | Target::Enum => continue,
2041                        _ => {
2042                            self.dcx().emit_err(errors::AttrApplication::StructEnumUnion {
2043                                hint_span: *repr_span,
2044                                span,
2045                            });
2046                        }
2047                    }
2048                }
2049                ReprAttr::ReprC => {
2050                    is_c = true;
2051                    match target {
2052                        Target::Struct | Target::Union | Target::Enum => continue,
2053                        _ => {
2054                            self.dcx().emit_err(errors::AttrApplication::StructEnumUnion {
2055                                hint_span: *repr_span,
2056                                span,
2057                            });
2058                        }
2059                    }
2060                }
2061                ReprAttr::ReprAlign(align) => {
2062                    match target {
2063                        Target::Struct | Target::Union | Target::Enum => {}
2064                        Target::Fn | Target::Method(_) => {
2065                            self.dcx().emit_err(errors::ReprAlignShouldBeAlign {
2066                                span: *repr_span,
2067                                item: target.name(),
2068                            });
2069                        }
2070                        _ => {
2071                            self.dcx().emit_err(errors::AttrApplication::StructEnumUnion {
2072                                hint_span: *repr_span,
2073                                span,
2074                            });
2075                        }
2076                    }
2077
2078                    self.check_align_value(*align, *repr_span);
2079                }
2080                ReprAttr::ReprPacked(_) => {
2081                    if target != Target::Struct && target != Target::Union {
2082                        self.dcx().emit_err(errors::AttrApplication::StructUnion {
2083                            hint_span: *repr_span,
2084                            span,
2085                        });
2086                    } else {
2087                        continue;
2088                    }
2089                }
2090                ReprAttr::ReprSimd => {
2091                    is_simd = true;
2092                    if target != Target::Struct {
2093                        self.dcx().emit_err(errors::AttrApplication::Struct {
2094                            hint_span: *repr_span,
2095                            span,
2096                        });
2097                    } else {
2098                        continue;
2099                    }
2100                }
2101                ReprAttr::ReprTransparent => {
2102                    is_transparent = true;
2103                    match target {
2104                        Target::Struct | Target::Union | Target::Enum => continue,
2105                        _ => {
2106                            self.dcx().emit_err(errors::AttrApplication::StructEnumUnion {
2107                                hint_span: *repr_span,
2108                                span,
2109                            });
2110                        }
2111                    }
2112                }
2113                ReprAttr::ReprInt(_) => {
2114                    int_reprs += 1;
2115                    if target != Target::Enum {
2116                        self.dcx().emit_err(errors::AttrApplication::Enum {
2117                            hint_span: *repr_span,
2118                            span,
2119                        });
2120                    } else {
2121                        continue;
2122                    }
2123                }
2124            };
2125        }
2126
2127        // catch `repr()` with no arguments, applied to an item (i.e. not `#![repr()]`)
2128        if let Some(first_attr_span) = first_attr_span
2129            && reprs.is_empty()
2130            && item.is_some()
2131        {
2132            match target {
2133                Target::Struct | Target::Union | Target::Enum => {}
2134                Target::Fn | Target::Method(_) => {
2135                    self.dcx().emit_err(errors::ReprAlignShouldBeAlign {
2136                        span: first_attr_span,
2137                        item: target.name(),
2138                    });
2139                }
2140                _ => {
2141                    self.dcx().emit_err(errors::AttrApplication::StructEnumUnion {
2142                        hint_span: first_attr_span,
2143                        span,
2144                    });
2145                }
2146            }
2147            return;
2148        }
2149
2150        // Just point at all repr hints if there are any incompatibilities.
2151        // This is not ideal, but tracking precisely which ones are at fault is a huge hassle.
2152        let hint_spans = reprs.iter().map(|(_, span)| *span);
2153
2154        // Error on repr(transparent, <anything else>).
2155        if is_transparent && reprs.len() > 1 {
2156            let hint_spans = hint_spans.clone().collect();
2157            self.dcx().emit_err(errors::TransparentIncompatible {
2158                hint_spans,
2159                target: target.to_string(),
2160            });
2161        }
2162        if is_explicit_rust && (int_reprs > 0 || is_c || is_simd) {
2163            let hint_spans = hint_spans.clone().collect();
2164            self.dcx().emit_err(errors::ReprConflicting { hint_spans });
2165        }
2166        // Warn on repr(u8, u16), repr(C, simd), and c-like-enum-repr(C, u8)
2167        if (int_reprs > 1)
2168            || (is_simd && is_c)
2169            || (int_reprs == 1
2170                && is_c
2171                && item.is_some_and(|item| {
2172                    if let ItemLike::Item(item) = item { is_c_like_enum(item) } else { false }
2173                }))
2174        {
2175            self.tcx.emit_node_span_lint(
2176                CONFLICTING_REPR_HINTS,
2177                hir_id,
2178                hint_spans.collect::<Vec<Span>>(),
2179                errors::ReprConflictingLint,
2180            );
2181        }
2182    }

2545    /// A best effort attempt to create an error for a mismatching proc macro signature.
2546    ///
2547    /// If this best effort goes wrong, it will just emit a worse error later (see #102923)
2548    fn check_proc_macro(&self, hir_id: HirId, target: Target, kind: ProcMacroKind) {
2549        if target != Target::Fn {
2550            return;
2551        }
2552
2553        let tcx = self.tcx;
2554        let Some(token_stream_def_id) = tcx.get_diagnostic_item(sym::TokenStream) else {
2555            return;
2556        };
2557        let Some(token_stream) = tcx.type_of(token_stream_def_id).no_bound_vars() else {
2558            return;
2559        };
2560
2561        let def_id = hir_id.expect_owner().def_id;
2562        let param_env = ty::ParamEnv::empty();
2563
2564        let infcx = tcx.infer_ctxt().build(TypingMode::non_body_analysis());
2565        let ocx = ObligationCtxt::new_with_diagnostics(&infcx);
2566
2567        let span = tcx.def_span(def_id);
2568        let fresh_args = infcx.fresh_args_for_item(span, def_id.to_def_id());
2569        let sig = tcx.liberate_late_bound_regions(
2570            def_id.to_def_id(),
2571            tcx.fn_sig(def_id).instantiate(tcx, fresh_args),
2572        );
2573
2574        let mut cause = ObligationCause::misc(span, def_id);
2575        let sig = ocx.normalize(&cause, param_env, sig);
2576
2577        // proc macro is not WF.
2578        let errors = ocx.select_where_possible();
2579        if !errors.is_empty() {
2580            return;
2581        }
2582
2583        let expected_sig = tcx.mk_fn_sig(
2584            std::iter::repeat(token_stream).take(match kind {
2585                ProcMacroKind::Attribute => 2,
2586                ProcMacroKind::Derive | ProcMacroKind::FunctionLike => 1,
2587            }),
2588            token_stream,
2589            false,
2590            Safety::Safe,
2591            ExternAbi::Rust,
2592        );
2593
2594        if let Err(terr) = ocx.eq(&cause, param_env, expected_sig, sig) {
2595            let mut diag = tcx.dcx().create_err(errors::ProcMacroBadSig { span, kind });
2596
2597            let hir_sig = tcx.hir_fn_sig_by_hir_id(hir_id);
2598            if let Some(hir_sig) = hir_sig {
2599                #[allow(rustc::diagnostic_outside_of_impl)] // FIXME
2600                match terr {
2601                    TypeError::ArgumentMutability(idx) | TypeError::ArgumentSorts(_, idx) => {
2602                        if let Some(ty) = hir_sig.decl.inputs.get(idx) {
2603                            diag.span(ty.span);
2604                            cause.span = ty.span;
2605                        } else if idx == hir_sig.decl.inputs.len() {
2606                            let span = hir_sig.decl.output.span();
2607                            diag.span(span);
2608                            cause.span = span;
2609                        }
2610                    }
2611                    TypeError::ArgCount => {
2612                        if let Some(ty) = hir_sig.decl.inputs.get(expected_sig.inputs().len()) {
2613                            diag.span(ty.span);
2614                            cause.span = ty.span;
2615                        }
2616                    }
2617                    TypeError::SafetyMismatch(_) => {
2618                        // FIXME: Would be nice if we had a span here..
2619                    }
2620                    TypeError::AbiMismatch(_) => {
2621                        // FIXME: Would be nice if we had a span here..
2622                    }
2623                    TypeError::VariadicMismatch(_) => {
2624                        // FIXME: Would be nice if we had a span here..
2625                    }
2626                    _ => {}
2627                }
2628            }
2629
2630            infcx.err_ctxt().note_type_err(
2631                &mut diag,
2632                &cause,
2633                None,
2634                Some(param_env.and(ValuePairs::PolySigs(ExpectedFound {
2635                    expected: ty::Binder::dummy(expected_sig),
2636                    found: ty::Binder::dummy(sig),
2637                }))),
2638                terr,
2639                false,
2640                None,
2641            );
2642            diag.emit();
2643            self.abort.set(true);
2644        }
2645
2646        let errors = ocx.select_all_or_error();
2647        if !errors.is_empty() {
2648            infcx.err_ctxt().report_fulfillment_errors(errors);
2649            self.abort.set(true);
2650        }
2651    }

3040fn check_duplicates(
3041    tcx: TyCtxt<'_>,
3042    attr: &Attribute,
3043    hir_id: HirId,
3044    duplicates: AttributeDuplicates,
3045    seen: &mut FxHashMap<Symbol, Span>,
3046) {
3047    use AttributeDuplicates::*;
3048    if matches!(duplicates, WarnFollowingWordOnly) && !attr.is_word() {
3049        return;
3050    }
3051    let attr_name = attr.name().unwrap();
3052    match duplicates {
3053        DuplicatesOk => {}
3054        WarnFollowing | FutureWarnFollowing | WarnFollowingWordOnly | FutureWarnPreceding => {
3055            match seen.entry(attr_name) {
3056                Entry::Occupied(mut entry) => {
3057                    let (this, other) = if matches!(duplicates, FutureWarnPreceding) {
3058                        let to_remove = entry.insert(attr.span());
3059                        (to_remove, attr.span())
3060                    } else {
3061                        (attr.span(), *entry.get())
3062                    };
3063                    tcx.emit_node_span_lint(
3064                        UNUSED_ATTRIBUTES,
3065                        hir_id,
3066                        this,
3067                        errors::UnusedDuplicate {
3068                            this,
3069                            other,
3070                            warning: matches!(
3071                                duplicates,
3072                                FutureWarnFollowing | FutureWarnPreceding
3073                            ),
3074                        },
3075                    );
3076                }
3077                Entry::Vacant(entry) => {
3078                    entry.insert(attr.span());
3079                }
3080            }
3081        }
3082        ErrorFollowing | ErrorPreceding => match seen.entry(attr_name) {
3083            Entry::Occupied(mut entry) => {
3084                let (this, other) = if matches!(duplicates, ErrorPreceding) {
3085                    let to_remove = entry.insert(attr.span());
3086                    (to_remove, attr.span())
3087                } else {
3088                    (attr.span(), *entry.get())
3089                };
3090                tcx.dcx().emit_err(errors::UnusedMultiple { this, other, name: attr_name });
3091            }
3092            Entry::Vacant(entry) => {
3093                entry.insert(attr.span());
3094            }
3095        },
3096    }
3097}
