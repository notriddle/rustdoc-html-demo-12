35    /// Parses a `mod <foo> { ... }` or `mod <foo>;` item.
36    fn parse_item_mod(&mut self, attrs: &mut AttrVec) -> PResult<'a, ItemKind> {
37        let safety = self.parse_safety(Case::Sensitive);
38        self.expect_keyword(exp!(Mod))?;
39        let ident = self.parse_ident()?;
40        let mod_kind = if self.eat(exp!(Semi)) {
41            ModKind::Unloaded
42        } else {
43            self.expect(exp!(OpenBrace))?;
44            let (inner_attrs, items, inner_span) = self.parse_mod(exp!(CloseBrace))?;
45            attrs.extend(inner_attrs);
46            ModKind::Loaded(items, Inline::Yes, inner_span, Ok(()))
47        };
48        Ok(ItemKind::Mod(safety, ident, mod_kind))
49    }

51    /// Parses the contents of a module (inner attributes followed by module items).
52    /// We exit once we hit `term` which can be either
53    /// - EOF (for files)
54    /// - `}` for mod items
55    pub fn parse_mod(
56        &mut self,
57        term: ExpTokenPair<'_>,
58    ) -> PResult<'a, (AttrVec, ThinVec<Box<Item>>, ModSpans)> {
59        let lo = self.token.span;
60        let attrs = self.parse_inner_attributes()?;
61
62        let post_attr_lo = self.token.span;
63        let mut items: ThinVec<Box<_>> = ThinVec::new();
64
65        // There shouldn't be any stray semicolons before or after items.
66        // `parse_item` consumes the appropriate semicolons so any leftover is an error.
67        loop {
68            while self.maybe_consume_incorrect_semicolon(items.last().map(|x| &**x)) {} // Eat all bad semicolons
69            let Some(item) = self.parse_item(ForceCollect::No)? else {
70                break;
71            };
72            items.push(item);
73        }
74
75        if !self.eat(term) {
76            let token_str = super::token_descr(&self.token);
77            if !self.maybe_consume_incorrect_semicolon(items.last().map(|x| &**x)) {
78                let is_let = self.token.is_keyword(kw::Let);
79                let is_let_mut = is_let && self.look_ahead(1, |t| t.is_keyword(kw::Mut));
80                let let_has_ident = is_let && !is_let_mut && self.is_kw_followed_by_ident(kw::Let);
81
82                let msg = format!("expected item, found {token_str}");
83                let mut err = self.dcx().struct_span_err(self.token.span, msg);
84
85                let label = if is_let {
86                    "`let` cannot be used for global variables"
87                } else {
88                    "expected item"
89                };
90                err.span_label(self.token.span, label);
91
92                if is_let {
93                    if is_let_mut {
94                        err.help("consider using `static` and a `Mutex` instead of `let mut`");
95                    } else if let_has_ident {
96                        err.span_suggestion_short(
97                            self.token.span,
98                            "consider using `static` or `const` instead of `let`",
99                            "static",
100                            Applicability::MaybeIncorrect,
101                        );
102                    } else {
103                        err.help("consider using `static` or `const` instead of `let`");
104                    }
105                }
106                err.note("for a full list of items that can appear in modules, see <https://doc.rust-lang.org/reference/items.html>");
107                return Err(err);
108            }
109        }
110
111        let inject_use_span = post_attr_lo.data().with_hi(post_attr_lo.lo());
112        let mod_spans = ModSpans { inner_span: lo.to(self.prev_token.span), inject_use_span };
113        Ok((attrs, items, mod_spans))
114    }

123    fn parse_item_(
124        &mut self,
125        fn_parse_mode: FnParseMode,
126        force_collect: ForceCollect,
127    ) -> PResult<'a, Option<Item>> {
128        self.recover_vcs_conflict_marker();
129        let attrs = self.parse_outer_attributes()?;
130        self.recover_vcs_conflict_marker();
131        self.parse_item_common(attrs, true, false, fn_parse_mode, force_collect)

134    pub(super) fn parse_item_common(
135        &mut self,
136        attrs: AttrWrapper,
137        mac_allowed: bool,
138        attrs_allowed: bool,
139        fn_parse_mode: FnParseMode,
140        force_collect: ForceCollect,
141    ) -> PResult<'a, Option<Item>> {
142        if let Some(item) =
143            self.eat_metavar_seq(MetaVarKind::Item, |this| this.parse_item(ForceCollect::Yes))
144        {
145            let mut item = item.expect("an actual item");
146            attrs.prepend_to_nt_inner(&mut item.attrs);
147            return Ok(Some(*item));
148        }
149
150        self.collect_tokens(None, attrs, force_collect, |this, mut attrs| {
151            let lo = this.token.span;
152            let vis = this.parse_visibility(FollowedByType::No)?;
153            let mut def = this.parse_defaultness();
154            let kind = this.parse_item_kind(
155                &mut attrs,
156                mac_allowed,
157                lo,
158                &vis,
159                &mut def,
160                fn_parse_mode,
161                Case::Sensitive,
162            )?;
163            if let Some(kind) = kind {
164                this.error_on_unconsumed_default(def, &kind);
165                let span = lo.to(this.prev_token.span);
166                let id = DUMMY_NODE_ID;
167                let item = Item { attrs, id, kind, vis, span, tokens: None };
168                return Ok((Some(item), Trailing::No, UsePreAttrPos::No));
169            }
170
171            // At this point, we have failed to parse an item.
172            if !matches!(vis.kind, VisibilityKind::Inherited) {
173                this.dcx().emit_err(errors::VisibilityNotFollowedByItem { span: vis.span, vis });
174            }
175
176            if let Defaultness::Default(span) = def {
177                this.dcx().emit_err(errors::DefaultNotFollowedByItem { span });
178            }
179
180            if !attrs_allowed {
181                this.recover_attrs_no_item(&attrs)?;
182            }
183            Ok((None, Trailing::No, UsePreAttrPos::No))
184        })
185    }

198    /// Parses one of the items allowed by the flags.
199    fn parse_item_kind(
200        &mut self,
201        attrs: &mut AttrVec,
202        macros_allowed: bool,
203        lo: Span,
204        vis: &Visibility,
205        def: &mut Defaultness,
206        fn_parse_mode: FnParseMode,
207        case: Case,
208    ) -> PResult<'a, Option<ItemKind>> {
209        let check_pub = def == &Defaultness::Final;
210        let mut def_ = || mem::replace(def, Defaultness::Final);
211
212        let info = if !self.is_use_closure() && self.eat_keyword_case(exp!(Use), case) {
213            self.parse_use_item()?
214        } else if self.check_fn_front_matter(check_pub, case) {
215            // FUNCTION ITEM
216            let (ident, sig, generics, contract, body) =
217                self.parse_fn(attrs, fn_parse_mode, lo, vis, case)?;
218            ItemKind::Fn(Box::new(Fn {
219                defaultness: def_(),
220                ident,
221                sig,
222                generics,
223                contract,
224                body,
225                define_opaque: None,
226            }))
227        } else if self.eat_keyword(exp!(Extern)) {
228            if self.eat_keyword(exp!(Crate)) {
229                // EXTERN CRATE
230                self.parse_item_extern_crate()?
231            } else {
232                // EXTERN BLOCK
233                self.parse_item_foreign_mod(attrs, Safety::Default)?
234            }
235        } else if self.is_unsafe_foreign_mod() {
236            // EXTERN BLOCK
237            let safety = self.parse_safety(Case::Sensitive);
238            self.expect_keyword(exp!(Extern))?;
239            self.parse_item_foreign_mod(attrs, safety)?
240        } else if self.is_static_global() {
241            let safety = self.parse_safety(Case::Sensitive);
242            // STATIC ITEM
243            self.bump(); // `static`
244            let mutability = self.parse_mutability();
245            self.parse_static_item(safety, mutability)?
246        } else if self.check_keyword(exp!(Trait)) || self.check_trait_front_matter() {
247            // TRAIT ITEM
248            self.parse_item_trait(attrs, lo)?
249        } else if let Const::Yes(const_span) = self.parse_constness(Case::Sensitive) {
250            // CONST ITEM
251            if self.token.is_keyword(kw::Impl) {
252                // recover from `const impl`, suggest `impl const`
253                self.recover_const_impl(const_span, attrs, def_())?
254            } else {
255                self.recover_const_mut(const_span);
256                self.recover_missing_kw_before_item()?;
257                let (ident, generics, ty, expr) = self.parse_const_item()?;
258                ItemKind::Const(Box::new(ConstItem {
259                    defaultness: def_(),
260                    ident,
261                    generics,
262                    ty,
263                    expr,
264                    define_opaque: None,
265                }))
266            }
267        } else if self.check_keyword(exp!(Impl))
268            || self.check_keyword(exp!(Unsafe)) && self.is_keyword_ahead(1, &[kw::Impl])
269        {
270            // IMPL ITEM
271            self.parse_item_impl(attrs, def_())?
272        } else if self.is_reuse_path_item() {
273            self.parse_item_delegation()?
274        } else if self.check_keyword(exp!(Mod))
275            || self.check_keyword(exp!(Unsafe)) && self.is_keyword_ahead(1, &[kw::Mod])
276        {
277            // MODULE ITEM
278            self.parse_item_mod(attrs)?
279        } else if self.eat_keyword(exp!(Type)) {
280            // TYPE ITEM
281            self.parse_type_alias(def_())?
282        } else if self.eat_keyword(exp!(Enum)) {
283            // ENUM ITEM
284            self.parse_item_enum()?
285        } else if self.eat_keyword(exp!(Struct)) {
286            // STRUCT ITEM
287            self.parse_item_struct()?
288        } else if self.is_kw_followed_by_ident(kw::Union) {
289            // UNION ITEM
290            self.bump(); // `union`
291            self.parse_item_union()?
292        } else if self.is_builtin() {
293            // BUILTIN# ITEM
294            return self.parse_item_builtin();
295        } else if self.eat_keyword(exp!(Macro)) {
296            // MACROS 2.0 ITEM
297            self.parse_item_decl_macro(lo)?
298        } else if let IsMacroRulesItem::Yes { has_bang } = self.is_macro_rules_item() {
299            // MACRO_RULES ITEM
300            self.parse_item_macro_rules(vis, has_bang)?
301        } else if self.isnt_macro_invocation()
302            && (self.token.is_ident_named(sym::import)
303                || self.token.is_ident_named(sym::using)
304                || self.token.is_ident_named(sym::include)
305                || self.token.is_ident_named(sym::require))
306        {
307            return self.recover_import_as_use();
308        } else if self.isnt_macro_invocation() && vis.kind.is_pub() {
309            self.recover_missing_kw_before_item()?;
310            return Ok(None);
311        } else if self.isnt_macro_invocation() && case == Case::Sensitive {
312            _ = def_;
313
314            // Recover wrong cased keywords
315            return self.parse_item_kind(
316                attrs,
317                macros_allowed,
318                lo,
319                vis,
320                def,
321                fn_parse_mode,
322                Case::Insensitive,
323            );
324        } else if macros_allowed && self.check_path() {
325            if self.isnt_macro_invocation() {
326                self.recover_missing_kw_before_item()?;
327            }
328            // MACRO INVOCATION ITEM
329            ItemKind::MacCall(Box::new(self.parse_item_macro(vis)?))
330        } else {
331            return Ok(None);
332        };
333        Ok(Some(info))
334    }

336    fn recover_import_as_use(&mut self) -> PResult<'a, Option<ItemKind>> {
337        let span = self.token.span;
338        let token_name = super::token_descr(&self.token);
339        let snapshot = self.create_snapshot_for_diagnostic();
340        self.bump();
341        match self.parse_use_item() {
342            Ok(u) => {
343                self.dcx().emit_err(errors::RecoverImportAsUse { span, token_name });
344                Ok(Some(u))
345            }
346            Err(e) => {
347                e.cancel();
348                self.restore_snapshot(snapshot);
349                Ok(None)
350            }
351        }
352    }

355        let tree = self.parse_use_tree()?;

367        }
368        Ok(ItemKind::Use(tree))

391    /// Recover on encountering a struct, enum, or method definition where the user
392    /// forgot to add the `struct`, `enum`, or `fn` keyword
393    fn recover_missing_kw_before_item(&mut self) -> PResult<'a, ()> {
394        let is_pub = self.prev_token.is_keyword(kw::Pub);
395        let is_const = self.prev_token.is_keyword(kw::Const);
396        let ident_span = self.token.span;
397        let span = if is_pub { self.prev_token.span.to(ident_span) } else { ident_span };
398        let insert_span = ident_span.shrink_to_lo();
399
400        let ident = if self.token.is_ident()
401            && (!is_const || self.look_ahead(1, |t| *t == token::OpenParen))
402            && self.look_ahead(1, |t| {
403                matches!(t.kind, token::Lt | token::OpenBrace | token::OpenParen)
404            }) {
405            self.parse_ident().unwrap()
406        } else {
407            return Ok(());
408        };
409
410        let mut found_generics = false;
411        if self.check(exp!(Lt)) {
412            found_generics = true;
413            self.eat_to_tokens(&[exp!(Gt)]);
414            self.bump(); // `>`
415        }
416
417        let err = if self.check(exp!(OpenBrace)) {
418            // possible struct or enum definition where `struct` or `enum` was forgotten
419            if self.look_ahead(1, |t| *t == token::CloseBrace) {
420                // `S {}` could be unit enum or struct
421                Some(errors::MissingKeywordForItemDefinition::EnumOrStruct { span })
422            } else if self.look_ahead(2, |t| *t == token::Colon)
423                || self.look_ahead(3, |t| *t == token::Colon)
424            {
425                // `S { f:` or `S { pub f:`
426                Some(errors::MissingKeywordForItemDefinition::Struct { span, insert_span, ident })
427            } else {
428                Some(errors::MissingKeywordForItemDefinition::Enum { span, insert_span, ident })
429            }
430        } else if self.check(exp!(OpenParen)) {
431            // possible function or tuple struct definition where `fn` or `struct` was forgotten
432            self.bump(); // `(`
433            let is_method = self.recover_self_param();
434
435            self.consume_block(exp!(OpenParen), exp!(CloseParen), ConsumeClosingDelim::Yes);
436
437            let err = if self.check(exp!(RArrow)) || self.check(exp!(OpenBrace)) {
438                self.eat_to_tokens(&[exp!(OpenBrace)]);
439                self.bump(); // `{`
440                self.consume_block(exp!(OpenBrace), exp!(CloseBrace), ConsumeClosingDelim::Yes);
441                if is_method {
442                    errors::MissingKeywordForItemDefinition::Method { span, insert_span, ident }
443                } else {
444                    errors::MissingKeywordForItemDefinition::Function { span, insert_span, ident }
445                }
446            } else if is_pub && self.check(exp!(Semi)) {
447                errors::MissingKeywordForItemDefinition::Struct { span, insert_span, ident }
448            } else {
449                errors::MissingKeywordForItemDefinition::Ambiguous {
450                    span,
451                    subdiag: if found_generics {
452                        None
453                    } else if let Ok(snippet) = self.span_to_snippet(ident_span) {
454                        Some(errors::AmbiguousMissingKwForItemSub::SuggestMacro {
455                            span: ident_span,
456                            snippet,
457                        })
458                    } else {
459                        Some(errors::AmbiguousMissingKwForItemSub::HelpMacro)
460                    },
461                }
462            };
463            Some(err)
464        } else if found_generics {
465            Some(errors::MissingKeywordForItemDefinition::Ambiguous { span, subdiag: None })
466        } else {
467            None
468        };
469
470        if let Some(err) = err { Err(self.dcx().create_err(err)) } else { Ok(()) }
471    }

478    /// Parses an item macro, e.g., `item!();`.
479    fn parse_item_macro(&mut self, vis: &Visibility) -> PResult<'a, MacCall> {
480        let path = self.parse_path(PathStyle::Mod)?; // `foo::bar`
481        self.expect(exp!(Bang))?; // `!`
482        match self.parse_delim_args() {
483            // `( .. )` or `[ .. ]` (followed by `;`), or `{ .. }`.
484            Ok(args) => {
485                self.eat_semi_for_macro_if_needed(&args);
486                self.complain_if_pub_macro(vis, false);
487                Ok(MacCall { path, args })
488            }
489
490            Err(mut err) => {
491                // Maybe the user misspelled `macro_rules` (issue #91227)
492                if self.token.is_ident()
493                    && let [segment] = path.segments.as_slice()
494                    && edit_distance("macro_rules", &segment.ident.to_string(), 2).is_some()
495                {
496                    err.span_suggestion(
497                        path.span,
498                        "perhaps you meant to define a macro",
499                        "macro_rules",
500                        Applicability::MachineApplicable,
501                    );
502                }
503                Err(err)
504            }
505        }
506    }

508    /// Recover if we parsed attributes and expected an item but there was none.
509    fn recover_attrs_no_item(&mut self, attrs: &[Attribute]) -> PResult<'a, ()> {
510        let ([start @ end] | [start, .., end]) = attrs else {
511            return Ok(());
512        };
513        let msg = if end.is_doc_comment() {
514            "expected item after doc comment"
515        } else {
516            "expected item after attributes"
517        };
518        let mut err = self.dcx().struct_span_err(end.span, msg);
519        if end.is_doc_comment() {
520            err.span_label(end.span, "this doc comment doesn't document anything");
521        } else if self.token == TokenKind::Semi {
522            err.span_suggestion_verbose(
523                self.token.span,
524                "consider removing this semicolon",
525                "",
526                Applicability::MaybeIncorrect,
527            );
528        }
529        if let [.., penultimate, _] = attrs {
530            err.span_label(start.span.to(penultimate.span), "other attributes here");
531        }
532        Err(err)
533    }

549    /// Parses an implementation item.
550    ///
551    /// ```ignore (illustrative)
552    /// impl<'a, T> TYPE { /* impl items */ }
553    /// impl<'a, T> TRAIT for TYPE { /* impl items */ }
554    /// impl<'a, T> !TRAIT for TYPE { /* impl items */ }
555    /// impl<'a, T> const TRAIT for TYPE { /* impl items */ }
556    /// ```
557    ///
558    /// We actually parse slightly more relaxed grammar for better error reporting and recovery.
559    /// ```ebnf
560    /// "impl" GENERICS "const"? "!"? TYPE "for"? (TYPE | "..") ("where" PREDICATES)? "{" BODY "}"
561    /// "impl" GENERICS "const"? "!"? TYPE ("where" PREDICATES)? "{" BODY "}"
562    /// ```
563    fn parse_item_impl(
564        &mut self,
565        attrs: &mut AttrVec,
566        defaultness: Defaultness,
567    ) -> PResult<'a, ItemKind> {
568        let safety = self.parse_safety(Case::Sensitive);
569        self.expect_keyword(exp!(Impl))?;
570
571        // First, parse generic parameters if necessary.
572        let mut generics = if self.choose_generics_over_qpath(0) {
573            self.parse_generics()?
574        } else {
575            let mut generics = Generics::default();
576            // impl A for B {}
577            //    /\ this is where `generics.span` should point when there are no type params.
578            generics.span = self.prev_token.span.shrink_to_hi();
579            generics
580        };
581
582        let constness = self.parse_constness(Case::Sensitive);
583        if let Const::Yes(span) = constness {
584            self.psess.gated_spans.gate(sym::const_trait_impl, span);
585        }
586
587        // Parse stray `impl async Trait`
588        if (self.token_uninterpolated_span().at_least_rust_2018()
589            && self.token.is_keyword(kw::Async))
590            || self.is_kw_followed_by_ident(kw::Async)
591        {
592            self.bump();
593            self.dcx().emit_err(errors::AsyncImpl { span: self.prev_token.span });
594        }
595
596        let polarity = self.parse_polarity();
597
598        // Parse both types and traits as a type, then reinterpret if necessary.
599        let ty_first = if self.token.is_keyword(kw::For) && self.look_ahead(1, |t| t != &token::Lt)
600        {
601            let span = self.prev_token.span.between(self.token.span);
602            return Err(self.dcx().create_err(errors::MissingTraitInTraitImpl {
603                span,
604                for_span: span.to(self.token.span),
605            }));
606        } else {
607            self.parse_ty_with_generics_recovery(&generics)?
608        };
609
610        // If `for` is missing we try to recover.
611        let has_for = self.eat_keyword(exp!(For));
612        let missing_for_span = self.prev_token.span.between(self.token.span);
613
614        let ty_second = if self.token == token::DotDot {
615            // We need to report this error after `cfg` expansion for compatibility reasons
616            self.bump(); // `..`, do not add it to expected tokens
617
618            // AST validation later detects this `TyKind::Dummy` and emits an
619            // error. (#121072 will hopefully remove all this special handling
620            // of the obsolete `impl Trait for ..` and then this can go away.)
621            Some(self.mk_ty(self.prev_token.span, TyKind::Dummy))
622        } else if has_for || self.token.can_begin_type() {
623            Some(self.parse_ty()?)
624        } else {
625            None
626        };
627
628        generics.where_clause = self.parse_where_clause()?;
629
630        let impl_items = self.parse_item_list(attrs, |p| p.parse_impl_item(ForceCollect::No))?;
631
632        let (of_trait, self_ty) = match ty_second {
633            Some(ty_second) => {
634                // impl Trait for Type
635                if !has_for {
636                    self.dcx().emit_err(errors::MissingForInTraitImpl { span: missing_for_span });
637                }
638
639                let ty_first = *ty_first;
640                let path = match ty_first.kind {
641                    // This notably includes paths passed through `ty` macro fragments (#46438).
642                    TyKind::Path(None, path) => path,
643                    other => {
644                        if let TyKind::ImplTrait(_, bounds) = other
645                            && let [bound] = bounds.as_slice()
646                            && let GenericBound::Trait(poly_trait_ref) = bound
647                        {
648                            // Suggest removing extra `impl` keyword:
649                            // `impl<T: Default> impl Default for Wrapper<T>`
650                            //                   ^^^^^
651                            let extra_impl_kw = ty_first.span.until(bound.span());
652                            self.dcx().emit_err(errors::ExtraImplKeywordInTraitImpl {
653                                extra_impl_kw,
654                                impl_trait_span: ty_first.span,
655                            });
656                            poly_trait_ref.trait_ref.path.clone()
657                        } else {
658                            return Err(self.dcx().create_err(
659                                errors::ExpectedTraitInTraitImplFoundType { span: ty_first.span },
660                            ));
661                        }
662                    }
663                };
664                let trait_ref = TraitRef { path, ref_id: ty_first.id };
665
666                (Some(trait_ref), ty_second)
667            }
668            None => (None, ty_first), // impl Type
669        };
670        Ok(ItemKind::Impl(Box::new(Impl {
671            safety,
672            polarity,
673            defaultness,
674            constness,
675            generics,
676            of_trait,
677            self_ty,
678            items: impl_items,
679        })))
680    }

682    fn parse_item_delegation(&mut self) -> PResult<'a, ItemKind> {
683        let span = self.token.span;
684        self.expect_keyword(exp!(Reuse))?;
685
686        let (qself, path) = if self.eat_lt() {
687            let (qself, path) = self.parse_qpath(PathStyle::Expr)?;
688            (Some(qself), path)
689        } else {
690            (None, self.parse_path(PathStyle::Expr)?)
691        };
692
693        let rename = |this: &mut Self| {
694            Ok(if this.eat_keyword(exp!(As)) { Some(this.parse_ident()?) } else { None })
695        };
696        let body = |this: &mut Self| {
697            Ok(if this.check(exp!(OpenBrace)) {
698                Some(this.parse_block()?)
699            } else {
700                this.expect(exp!(Semi))?;
701                None
702            })
703        };
704
705        let item_kind = if self.eat_path_sep() {
706            let suffixes = if self.eat(exp!(Star)) {
707                None
708            } else {
709                let parse_suffix = |p: &mut Self| Ok((p.parse_path_segment_ident()?, rename(p)?));
710                Some(self.parse_delim_comma_seq(exp!(OpenBrace), exp!(CloseBrace), parse_suffix)?.0)
711            };
712            let deleg = DelegationMac { qself, prefix: path, suffixes, body: body(self)? };
713            ItemKind::DelegationMac(Box::new(deleg))
714        } else {
715            let rename = rename(self)?;
716            let ident = rename.unwrap_or_else(|| path.segments.last().unwrap().ident);
717            let deleg = Delegation {
718                id: DUMMY_NODE_ID,
719                qself,
720                path,
721                ident,
722                rename,
723                body: body(self)?,
724                from_glob: false,
725            };
726            ItemKind::Delegation(Box::new(deleg))
727        };
728
729        let span = span.to(self.prev_token.span);
730        self.psess.gated_spans.gate(sym::fn_delegation, span);
731
732        Ok(item_kind)
733    }

735    fn parse_item_list<T>(
736        &mut self,
737        attrs: &mut AttrVec,
738        mut parse_item: impl FnMut(&mut Parser<'a>) -> PResult<'a, Option<Option<T>>>,
739    ) -> PResult<'a, ThinVec<T>> {
740        let open_brace_span = self.token.span;
741
742        // Recover `impl Ty;` instead of `impl Ty {}`
743        if self.token == TokenKind::Semi {
744            self.dcx().emit_err(errors::UseEmptyBlockNotSemi { span: self.token.span });
745            self.bump();
746            return Ok(ThinVec::new());
747        }
748
749        self.expect(exp!(OpenBrace))?;
750        attrs.extend(self.parse_inner_attributes()?);
751
752        let mut items = ThinVec::new();
753        while !self.eat(exp!(CloseBrace)) {
754            if self.recover_doc_comment_before_brace() {
755                continue;
756            }
757            self.recover_vcs_conflict_marker();
758            match parse_item(self) {
759                Ok(None) => {
760                    let mut is_unnecessary_semicolon = !items.is_empty()
761                        // When the close delim is `)` in a case like the following, `token.kind`
762                        // is expected to be `token::CloseParen`, but the actual `token.kind` is
763                        // `token::CloseBrace`. This is because the `token.kind` of the close delim
764                        // is treated as the same as that of the open delim in
765                        // `TokenTreesReader::parse_token_tree`, even if the delimiters of them are
766                        // different. Therefore, `token.kind` should not be compared here.
767                        //
768                        // issue-60075.rs
769                        // ```
770                        // trait T {
771                        //     fn qux() -> Option<usize> {
772                        //         let _ = if true {
773                        //         });
774                        //          ^ this close delim
775                        //         Some(4)
776                        //     }
777                        // ```
778                        && self
779                            .span_to_snippet(self.prev_token.span)
780                            .is_ok_and(|snippet| snippet == "}")
781                        && self.token == token::Semi;
782                    let mut semicolon_span = self.token.span;
783                    if !is_unnecessary_semicolon {
784                        // #105369, Detect spurious `;` before assoc fn body
785                        is_unnecessary_semicolon =
786                            self.token == token::OpenBrace && self.prev_token == token::Semi;
787                        semicolon_span = self.prev_token.span;
788                    }
789                    // We have to bail or we'll potentially never make progress.
790                    let non_item_span = self.token.span;
791                    let is_let = self.token.is_keyword(kw::Let);
792
793                    let mut err =
794                        self.dcx().struct_span_err(non_item_span, "non-item in item list");
795                    self.consume_block(exp!(OpenBrace), exp!(CloseBrace), ConsumeClosingDelim::Yes);
796                    if is_let {
797                        err.span_suggestion_verbose(
798                            non_item_span,
799                            "consider using `const` instead of `let` for associated const",
800                            "const",
801                            Applicability::MachineApplicable,
802                        );
803                    } else {
804                        err.span_label(open_brace_span, "item list starts here")
805                            .span_label(non_item_span, "non-item starts here")
806                            .span_label(self.prev_token.span, "item list ends here");
807                    }
808                    if is_unnecessary_semicolon {
809                        err.span_suggestion(
810                            semicolon_span,
811                            "consider removing this semicolon",
812                            "",
813                            Applicability::MaybeIncorrect,
814                        );
815                    }
816                    err.emit();
817                    break;
818                }
819                Ok(Some(item)) => items.extend(item),
820                Err(err) => {
821                    self.consume_block(exp!(OpenBrace), exp!(CloseBrace), ConsumeClosingDelim::Yes);
822                    err.with_span_label(
823                        open_brace_span,
824                        "while parsing this item list starting here",
825                    )
826                    .with_span_label(self.prev_token.span, "the item list ends here")
827                    .emit();
828                    break;
829                }
830            }
831        }
832        Ok(items)
833    }

860    fn parse_defaultness(&mut self) -> Defaultness {
861        // We are interested in `default` followed by another identifier.
862        // However, we must avoid keywords that occur as binary operators.
863        // Currently, the only applicable keyword is `as` (`default as Ty`).
864        if self.check_keyword(exp!(Default))
865            && self.look_ahead(1, |t| t.is_non_raw_ident_where(|i| i.name != kw::As))
866        {
867            self.bump(); // `default`
868            Defaultness::Default(self.prev_token_uninterpolated_span())
869        } else {
870            Defaultness::Final
871        }

884    /// Parses `unsafe? auto? trait Foo { ... }` or `trait Foo = Bar;`.
885    fn parse_item_trait(&mut self, attrs: &mut AttrVec, lo: Span) -> PResult<'a, ItemKind> {
886        let constness = self.parse_constness(Case::Sensitive);
887        if let Const::Yes(span) = constness {
888            self.psess.gated_spans.gate(sym::const_trait_impl, span);
889        }
890        let safety = self.parse_safety(Case::Sensitive);
891        // Parse optional `auto` prefix.
892        let is_auto = if self.eat_keyword(exp!(Auto)) {
893            self.psess.gated_spans.gate(sym::auto_traits, self.prev_token.span);
894            IsAuto::Yes
895        } else {
896            IsAuto::No
897        };
898
899        self.expect_keyword(exp!(Trait))?;
900        let ident = self.parse_ident()?;
901        let mut generics = self.parse_generics()?;
902
903        // Parse optional colon and supertrait bounds.
904        let had_colon = self.eat(exp!(Colon));
905        let span_at_colon = self.prev_token.span;
906        let bounds = if had_colon { self.parse_generic_bounds()? } else { Vec::new() };
907
908        let span_before_eq = self.prev_token.span;
909        if self.eat(exp!(Eq)) {
910            // It's a trait alias.
911            if had_colon {
912                let span = span_at_colon.to(span_before_eq);
913                self.dcx().emit_err(errors::BoundsNotAllowedOnTraitAliases { span });
914            }
915
916            let bounds = self.parse_generic_bounds()?;
917            generics.where_clause = self.parse_where_clause()?;
918            self.expect_semi()?;
919
920            let whole_span = lo.to(self.prev_token.span);
921            if let Const::Yes(_) = constness {
922                self.dcx().emit_err(errors::TraitAliasCannotBeConst { span: whole_span });
923            }
924            if is_auto == IsAuto::Yes {
925                self.dcx().emit_err(errors::TraitAliasCannotBeAuto { span: whole_span });
926            }
927            if let Safety::Unsafe(_) = safety {
928                self.dcx().emit_err(errors::TraitAliasCannotBeUnsafe { span: whole_span });
929            }
930
931            self.psess.gated_spans.gate(sym::trait_alias, whole_span);
932
933            Ok(ItemKind::TraitAlias(ident, generics, bounds))
934        } else {
935            // It's a normal trait.
936            generics.where_clause = self.parse_where_clause()?;
937            let items = self.parse_item_list(attrs, |p| p.parse_trait_item(ForceCollect::No))?;
938            Ok(ItemKind::Trait(Box::new(Trait {
939                constness,
940                is_auto,
941                safety,
942                ident,
943                generics,
944                bounds,
945                items,
946            })))
947        }
948    }

1004    /// Parses a `type` alias with the following grammar:
1005    /// ```ebnf
1006    /// TypeAlias = "type" Ident Generics (":" GenericBounds)? WhereClause ("=" Ty)? WhereClause ";" ;
1007    /// ```
1008    /// The `"type"` has already been eaten.
1009    fn parse_type_alias(&mut self, defaultness: Defaultness) -> PResult<'a, ItemKind> {
1010        let ident = self.parse_ident()?;
1011        let mut generics = self.parse_generics()?;
1012
1013        // Parse optional colon and param bounds.
1014        let bounds = if self.eat(exp!(Colon)) { self.parse_generic_bounds()? } else { Vec::new() };
1015        let before_where_clause = self.parse_where_clause()?;
1016
1017        let ty = if self.eat(exp!(Eq)) { Some(self.parse_ty()?) } else { None };
1018
1019        let after_where_clause = self.parse_where_clause()?;
1020
1021        let where_clauses = TyAliasWhereClauses {
1022            before: TyAliasWhereClause {
1023                has_where_token: before_where_clause.has_where_token,
1024                span: before_where_clause.span,
1025            },
1026            after: TyAliasWhereClause {
1027                has_where_token: after_where_clause.has_where_token,
1028                span: after_where_clause.span,
1029            },
1030            split: before_where_clause.predicates.len(),
1031        };
1032        let mut predicates = before_where_clause.predicates;
1033        predicates.extend(after_where_clause.predicates);
1034        let where_clause = WhereClause {
1035            has_where_token: before_where_clause.has_where_token
1036                || after_where_clause.has_where_token,
1037            predicates,
1038            span: DUMMY_SP,
1039        };
1040        generics.where_clause = where_clause;
1041
1042        self.expect_semi()?;
1043
1044        Ok(ItemKind::TyAlias(Box::new(TyAlias {
1045            defaultness,
1046            ident,
1047            generics,
1048            where_clauses,
1049            bounds,
1050            ty,
1051        })))
1052    }

1054    /// Parses a `UseTree`.
1055    ///
1056    /// ```text
1057    /// USE_TREE = [`::`] `*` |
1058    ///            [`::`] `{` USE_TREE_LIST `}` |
1059    ///            PATH `::` `*` |
1060    ///            PATH `::` `{` USE_TREE_LIST `}` |
1061    ///            PATH [`as` IDENT]
1062    /// ```
1063    fn parse_use_tree(&mut self) -> PResult<'a, UseTree> {
1064        let lo = self.token.span;
1065
1066        let mut prefix =
1067            ast::Path { segments: ThinVec::new(), span: lo.shrink_to_lo(), tokens: None };
1068        let kind =
1069            if self.check(exp!(OpenBrace)) || self.check(exp!(Star)) || self.is_import_coupler() {
1070                // `use *;` or `use ::*;` or `use {...};` or `use ::{...};`
1071                let mod_sep_ctxt = self.token.span.ctxt();
1072                if self.eat_path_sep() {
1073                    prefix
1074                        .segments
1075                        .push(PathSegment::path_root(lo.shrink_to_lo().with_ctxt(mod_sep_ctxt)));
1076                }
1077
1078                self.parse_use_tree_glob_or_nested()?
1079            } else {
1080                // `use path::*;` or `use path::{...};` or `use path;` or `use path as bar;`
1081                prefix = self.parse_path(PathStyle::Mod)?;
1082
1083                if self.eat_path_sep() {
1084                    self.parse_use_tree_glob_or_nested()?
1085                } else {
1086                    // Recover from using a colon as path separator.
1087                    while self.eat_noexpect(&token::Colon) {
1088                        self.dcx()
1089                            .emit_err(errors::SingleColonImportPath { span: self.prev_token.span });
1090
1091                        // We parse the rest of the path and append it to the original prefix.
1092                        self.parse_path_segments(&mut prefix.segments, PathStyle::Mod, None)?;
1093                        prefix.span = lo.to(self.prev_token.span);
1094                    }
1095
1096                    UseTreeKind::Simple(self.parse_rename()?)
1097                }
1098            };
1099
1100        Ok(UseTree { prefix, kind, span: lo.to(self.prev_token.span) })
1101    }

1386    /// Parse a static item with the prefix `"static" "mut"?` already parsed and stored in
1387    /// `mutability`.
1388    ///
1389    /// ```ebnf
1390    /// Static = "static" "mut"? $ident ":" $ty (= $expr)? ";" ;
1391    /// ```
1392    fn parse_static_item(
1393        &mut self,
1394        safety: Safety,
1395        mutability: Mutability,
1396    ) -> PResult<'a, ItemKind> {
1397        let ident = self.parse_ident()?;
1398
1399        if self.token == TokenKind::Lt && self.may_recover() {
1400            let generics = self.parse_generics()?;
1401            self.dcx().emit_err(errors::StaticWithGenerics { span: generics.span });

1405        // FIXME: This could maybe benefit from `.may_recover()`?

1421    /// Parse a constant item with the prefix `"const"` already parsed.
1422    ///
1423    /// ```ebnf
1424    /// Const = "const" ($ident | "_") Generics ":" $ty (= $expr)? WhereClause ";" ;
1425    /// ```
1426    fn parse_const_item(
1427        &mut self,
1428    ) -> PResult<'a, (Ident, Generics, Box<Ty>, Option<Box<ast::Expr>>)> {
1429        let ident = self.parse_ident_or_underscore()?;
1430
1431        let mut generics = self.parse_generics()?;
1432
1433        // Check the span for emptiness instead of the list of parameters in order to correctly
1434        // recognize and subsequently flag empty parameter lists (`<>`) as unstable.
1435        if !generics.span.is_empty() {
1436            self.psess.gated_spans.gate(sym::generic_const_items, generics.span);
1437        }
1438
1439        // Parse the type of a constant item. That is, the `":" $ty` fragment.
1440        // FIXME: This could maybe benefit from `.may_recover()`?
1441        let ty = match (
1442            self.eat(exp!(Colon)),
1443            self.check(exp!(Eq)) | self.check(exp!(Semi)) | self.check_keyword(exp!(Where)),
1444        ) {
1445            (true, false) => self.parse_ty()?,
1446            // If there wasn't a `:` or the colon was followed by a `=`, `;` or `where`, recover a missing type.
1447            (colon, _) => self.recover_missing_global_item_type(colon, None),
1448        };
1449
1450        // Proactively parse a where-clause to be able to provide a good error message in case we
1451        // encounter the item body following it.
1452        let before_where_clause =
1453            if self.may_recover() { self.parse_where_clause()? } else { WhereClause::default() };
1454
1455        let expr = if self.eat(exp!(Eq)) { Some(self.parse_expr()?) } else { None };
1456
1457        let after_where_clause = self.parse_where_clause()?;
1458
1459        // Provide a nice error message if the user placed a where-clause before the item body.
1460        // Users may be tempted to write such code if they are still used to the deprecated
1461        // where-clause location on type aliases and associated types. See also #89122.
1462        if before_where_clause.has_where_token
1463            && let Some(expr) = &expr
1464        {
1465            self.dcx().emit_err(errors::WhereClauseBeforeConstBody {
1466                span: before_where_clause.span,
1467                name: ident.span,
1468                body: expr.span,
1469                sugg: if !after_where_clause.has_where_token {
1470                    self.psess.source_map().span_to_snippet(expr.span).ok().map(|body| {
1471                        errors::WhereClauseBeforeConstBodySugg {
1472                            left: before_where_clause.span.shrink_to_lo(),
1473                            snippet: body,
1474                            right: before_where_clause.span.shrink_to_hi().to(expr.span),
1475                        }
1476                    })
1477                } else {
1478                    // FIXME(generic_const_items): Provide a structured suggestion to merge the first
1479                    // where-clause into the second one.
1480                    None
1481                },
1482            });
1483        }
1484
1485        // Merge the predicates of both where-clauses since either one can be relevant.
1486        // If we didn't parse a body (which is valid for associated consts in traits) and we were
1487        // allowed to recover, `before_where_clause` contains the predicates, otherwise they are
1488        // in `after_where_clause`. Further, both of them might contain predicates iff two
1489        // where-clauses were provided which is syntactically ill-formed but we want to recover from
1490        // it and treat them as one large where-clause.
1491        let mut predicates = before_where_clause.predicates;
1492        predicates.extend(after_where_clause.predicates);
1493        let where_clause = WhereClause {
1494            has_where_token: before_where_clause.has_where_token
1495                || after_where_clause.has_where_token,
1496            predicates,
1497            span: if after_where_clause.has_where_token {
1498                after_where_clause.span
1499            } else {
1500                before_where_clause.span
1501            },
1502        };
1503
1504        if where_clause.has_where_token {
1505            self.psess.gated_spans.gate(sym::generic_const_items, where_clause.span);
1506        }
1507
1508        generics.where_clause = where_clause;
1509
1510        self.expect_semi()?;
1511
1512        Ok((ident, generics, ty, expr))
1513    }

1690    /// Parses `struct Foo { ... }`.
1691    fn parse_item_struct(&mut self) -> PResult<'a, ItemKind> {
1692        let ident = self.parse_ident()?;
1693
1694        let mut generics = self.parse_generics()?;
1695
1696        // There is a special case worth noting here, as reported in issue #17904.
1697        // If we are parsing a tuple struct it is the case that the where clause
1698        // should follow the field list. Like so:
1699        //
1700        // struct Foo<T>(T) where T: Copy;
1701        //
1702        // If we are parsing a normal record-style struct it is the case
1703        // that the where clause comes before the body, and after the generics.
1704        // So if we look ahead and see a brace or a where-clause we begin
1705        // parsing a record style struct.
1706        //
1707        // Otherwise if we look ahead and see a paren we parse a tuple-style
1708        // struct.
1709
1710        let vdata = if self.token.is_keyword(kw::Where) {
1711            let tuple_struct_body;
1712            (generics.where_clause, tuple_struct_body) =
1713                self.parse_struct_where_clause(ident, generics.span)?;
1714
1715            if let Some(body) = tuple_struct_body {
1716                // If we see a misplaced tuple struct body: `struct Foo<T> where T: Copy, (T);`
1717                let body = VariantData::Tuple(body, DUMMY_NODE_ID);
1718                self.expect_semi()?;
1719                body
1720            } else if self.eat(exp!(Semi)) {
1721                // If we see a: `struct Foo<T> where T: Copy;` style decl.
1722                VariantData::Unit(DUMMY_NODE_ID)
1723            } else {
1724                // If we see: `struct Foo<T> where T: Copy { ... }`
1725                let (fields, recovered) = self.parse_record_struct_body(
1726                    "struct",
1727                    ident.span,
1728                    generics.where_clause.has_where_token,
1729                )?;
1730                VariantData::Struct { fields, recovered }
1731            }
1732        // No `where` so: `struct Foo<T>;`
1733        } else if self.eat(exp!(Semi)) {
1734            VariantData::Unit(DUMMY_NODE_ID)
1735        // Record-style struct definition
1736        } else if self.token == token::OpenBrace {
1737            let (fields, recovered) = self.parse_record_struct_body(
1738                "struct",
1739                ident.span,
1740                generics.where_clause.has_where_token,
1741            )?;
1742            VariantData::Struct { fields, recovered }
1743        // Tuple-style struct definition with optional where-clause.
1744        } else if self.token == token::OpenParen {
1745            let body = VariantData::Tuple(self.parse_tuple_struct_body()?, DUMMY_NODE_ID);
1746            generics.where_clause = self.parse_where_clause()?;
1747            self.expect_semi()?;
1748            body
1749        } else {
1750            let err = errors::UnexpectedTokenAfterStructName::new(self.token.span, self.token);
1751            return Err(self.dcx().create_err(err));
1752        };
1753
1754        Ok(ItemKind::Struct(ident, generics, vdata))
1755    }

1789    /// This function parses the fields of record structs:
1790    ///
1791    ///   - `struct S { ... }`
1792    ///   - `enum E { Variant { ... } }`
1793    pub(crate) fn parse_record_struct_body(
1794        &mut self,
1795        adt_ty: &str,
1796        ident_span: Span,
1797        parsed_where: bool,
1798    ) -> PResult<'a, (ThinVec<FieldDef>, Recovered)> {
1799        let mut fields = ThinVec::new();
1800        let mut recovered = Recovered::No;
1801        if self.eat(exp!(OpenBrace)) {
1802            while self.token != token::CloseBrace {
1803                match self.parse_field_def(adt_ty, ident_span) {
1804                    Ok(field) => {
1805                        fields.push(field);
1806                    }
1807                    Err(mut err) => {
1808                        self.consume_block(
1809                            exp!(OpenBrace),
1810                            exp!(CloseBrace),
1811                            ConsumeClosingDelim::No,
1812                        );
1813                        err.span_label(ident_span, format!("while parsing this {adt_ty}"));
1814                        let guar = err.emit();
1815                        recovered = Recovered::Yes(guar);
1816                        break;
1817                    }
1818                }
1819            }
1820            self.expect(exp!(CloseBrace))?;
1821        } else {
1822            let token_str = super::token_descr(&self.token);
1823            let where_str = if parsed_where { "" } else { "`where`, or " };
1824            let msg = format!("expected {where_str}`{{` after struct name, found {token_str}");
1825            let mut err = self.dcx().struct_span_err(self.token.span, msg);
1826            err.span_label(self.token.span, format!("expected {where_str}`{{` after struct name",));
1827            return Err(err);
1828        }

1830        Ok((fields, recovered))
1831    }

1844    pub(super) fn parse_tuple_struct_body(&mut self) -> PResult<'a, ThinVec<FieldDef>> {
1845        // This is the case where we find `struct Foo<T>(T) where T: Copy;`
1846        // Unit like structs are handled in parse_item_struct function
1847        self.parse_paren_comma_seq(|p| {
1848            let attrs = p.parse_outer_attributes()?;
1849            p.collect_tokens(None, attrs, ForceCollect::No, |p, attrs| {
1850                let mut snapshot = None;
1851                if p.is_vcs_conflict_marker(&TokenKind::Shl, &TokenKind::Lt) {
1852                    // Account for `<<<<<<<` diff markers. We can't proactively error here because
1853                    // that can be a valid type start, so we snapshot and reparse only we've
1854                    // encountered another parse error.
1855                    snapshot = Some(p.create_snapshot_for_diagnostic());
1856                }
1857                let lo = p.token.span;
1858                let vis = match p.parse_visibility(FollowedByType::Yes) {
1859                    Ok(vis) => vis,
1860                    Err(err) => {
1861                        if let Some(ref mut snapshot) = snapshot {
1862                            snapshot.recover_vcs_conflict_marker();
1863                        }
1864                        return Err(err);
1865                    }
1866                };
1867                // Unsafe fields are not supported in tuple structs, as doing so would result in a
1868                // parsing ambiguity for `struct X(unsafe fn())`.
1869                let ty = match p.parse_ty() {
1870                    Ok(ty) => ty,
1871                    Err(err) => {
1872                        if let Some(ref mut snapshot) = snapshot {
1873                            snapshot.recover_vcs_conflict_marker();
1874                        }
1875                        return Err(err);
1876                    }
1877                };
1878                let mut default = None;
1879                if p.token == token::Eq {
1880                    let mut snapshot = p.create_snapshot_for_diagnostic();
1881                    snapshot.bump();
1882                    match snapshot.parse_expr_anon_const() {
1883                        Ok(const_expr) => {
1884                            let sp = ty.span.shrink_to_hi().to(const_expr.value.span);
1885                            p.psess.gated_spans.gate(sym::default_field_values, sp);
1886                            p.restore_snapshot(snapshot);
1887                            default = Some(const_expr);
1888                        }
1889                        Err(err) => {
1890                            err.cancel();
1891                        }
1892                    }
1893                }
1894
1895                Ok((
1896                    FieldDef {
1897                        span: lo.to(ty.span),
1898                        vis,
1899                        safety: Safety::Default,
1900                        ident: None,
1901                        id: DUMMY_NODE_ID,
1902                        ty,
1903                        default,
1904                        attrs,
1905                        is_placeholder: false,
1906                    },
1907                    Trailing::from(p.token == token::Comma),
1908                    UsePreAttrPos::No,
1909                ))
1910            })
1911        })
1912        .map(|(r, _)| r)
1913    }

1929    /// Parses a structure field declaration.
1930    fn parse_single_struct_field(
1931        &mut self,
1932        adt_ty: &str,
1933        lo: Span,
1934        vis: Visibility,
1935        safety: Safety,
1936        attrs: AttrVec,
1937        ident_span: Span,
1938    ) -> PResult<'a, FieldDef> {
1939        let a_var = self.parse_name_and_ty(adt_ty, lo, vis, safety, attrs)?;
1940        match self.token.kind {
1941            token::Comma => {
1942                self.bump();
1943            }
1944            token::Semi => {
1945                self.bump();
1946                let sp = self.prev_token.span;
1947                let mut err =
1948                    self.dcx().struct_span_err(sp, format!("{adt_ty} fields are separated by `,`"));
1949                err.span_suggestion_short(
1950                    sp,
1951                    "replace `;` with `,`",
1952                    ",",
1953                    Applicability::MachineApplicable,
1954                );
1955                err.span_label(ident_span, format!("while parsing this {adt_ty}"));
1956                err.emit();
1957            }
1958            token::CloseBrace => {}
1959            token::DocComment(..) => {
1960                let previous_span = self.prev_token.span;
1961                let mut err = errors::DocCommentDoesNotDocumentAnything {
1962                    span: self.token.span,
1963                    missing_comma: None,
1964                };
1965                self.bump(); // consume the doc comment
1966                if self.eat(exp!(Comma)) || self.token == token::CloseBrace {
1967                    self.dcx().emit_err(err);
1968                } else {
1969                    let sp = previous_span.shrink_to_hi();
1970                    err.missing_comma = Some(sp);
1971                    return Err(self.dcx().create_err(err));
1972                }
1973            }
1974            _ => {
1975                let sp = self.prev_token.span.shrink_to_hi();
1976                let msg =
1977                    format!("expected `,`, or `}}`, found {}", super::token_descr(&self.token));
1978
1979                // Try to recover extra trailing angle brackets
1980                if let TyKind::Path(_, Path { segments, .. }) = &a_var.ty.kind
1981                    && let Some(last_segment) = segments.last()
1982                {
1983                    let guar = self.check_trailing_angle_brackets(
1984                        last_segment,
1985                        &[exp!(Comma), exp!(CloseBrace)],
1986                    );
1987                    if let Some(_guar) = guar {
1988                        // Handle a case like `Vec<u8>>,` where we can continue parsing fields
1989                        // after the comma
1990                        let _ = self.eat(exp!(Comma));
1991
1992                        // `check_trailing_angle_brackets` already emitted a nicer error, as
1993                        // proven by the presence of `_guar`. We can continue parsing.
1994                        return Ok(a_var);
1995                    }
1996                }
1997
1998                let mut err = self.dcx().struct_span_err(sp, msg);
1999
2000                if self.token.is_ident()
2001                    || (self.token == TokenKind::Pound
2002                        && (self.look_ahead(1, |t| t == &token::OpenBracket)))
2003                {
2004                    // This is likely another field, TokenKind::Pound is used for `#[..]`
2005                    // attribute for next field. Emit the diagnostic and continue parsing.
2006                    err.span_suggestion(
2007                        sp,
2008                        "try adding a comma",
2009                        ",",
2010                        Applicability::MachineApplicable,
2011                    );
2012                    err.emit();
2013                } else {
2014                    return Err(err);
2015                }
2016            }
2017        }
2018        Ok(a_var)
2019    }

2021    fn expect_field_ty_separator(&mut self) -> PResult<'a, ()> {
2022        if let Err(err) = self.expect(exp!(Colon)) {
2023            let sm = self.psess.source_map();
2024            let eq_typo = self.token == token::Eq && self.look_ahead(1, |t| t.is_path_start());
2025            let semi_typo = self.token == token::Semi
2026                && self.look_ahead(1, |t| {
2027                    t.is_path_start()
2028                    // We check that we are in a situation like `foo; bar` to avoid bad suggestions
2029                    // when there's no type and `;` was used instead of a comma.
2030                    && match (sm.lookup_line(self.token.span.hi()), sm.lookup_line(t.span.lo())) {
2031                        (Ok(l), Ok(r)) => l.line == r.line,
2032                        _ => true,
2033                    }
2034                });
2035            if eq_typo || semi_typo {
2036                self.bump();
2037                // Gracefully handle small typos.
2038                err.with_span_suggestion_short(
2039                    self.prev_token.span,
2040                    "field names and their types are separated with `:`",
2041                    ":",
2042                    Applicability::MachineApplicable,
2043                )
2044                .emit();
2045            } else {
2046                return Err(err);
2047            }
2048        }
2049        Ok(())
2050    }

2188    /// Parses a declarative macro 2.0 definition.
2189    /// The `macro` keyword has already been parsed.
2190    /// ```ebnf
2191    /// MacBody = "{" TOKEN_STREAM "}" ;
2192    /// MacParams = "(" TOKEN_STREAM ")" ;

2195    fn parse_item_decl_macro(&mut self, lo: Span) -> PResult<'a, ItemKind> {
2196        let ident = self.parse_ident()?;
2197        let body = if self.check(exp!(OpenBrace)) {
2198            self.parse_delim_args()? // `MacBody`
2199        } else if self.check(exp!(OpenParen)) {

2207            let bspan = body.span();

2210            let dspan = DelimSpan::from_pair(pspan.shrink_to_lo(), bspan.shrink_to_hi());

2212        } else {
2213            self.unexpected_any()?
2214        };
2215
2216        self.psess.gated_spans.gate(sym::decl_macro, lo.to(self.prev_token.span));

2218    }

2306                "add a semicolon",
2307                ';',
2308                Applicability::MaybeIncorrect,
2309            );
2310        }
2311        err.emit();
2312    }

2314    /// Checks if current token is one of tokens which cannot be nested like `kw::Enum`. In case
2315    /// it is, we try to parse the item and report error about nested types.
2316    fn recover_nested_adt_item(&mut self, keyword: Symbol) -> PResult<'a, bool> {
2317        if (self.token.is_keyword(kw::Enum)
2318            || self.token.is_keyword(kw::Struct)
2319            || self.token.is_keyword(kw::Union))
2320            && self.look_ahead(1, |t| t.is_ident())
2321        {
2322            let kw_token = self.token;
2323            let kw_str = pprust::token_to_string(&kw_token);
2324            let item = self.parse_item(ForceCollect::No)?;
2325            let mut item = item.unwrap().span;
2326            if self.token == token::Comma {
2327                item = item.to(self.token.span);
2328            }
2329            self.dcx().emit_err(errors::NestedAdt {
2330                span: kw_token.span,
2331                item,
2332                kw_str,
2333                keyword: keyword.as_str(),
2334            });
2335            // We successfully parsed the item but we must inform the caller about nested problem.
2336            return Ok(false);
2337        }
2338        Ok(true)
2339    }

2342/// The parsing configuration used to parse a parameter list (see `parse_fn_params`).
2343///
2344/// The function decides if, per-parameter `p`, `p` must have a pattern or just a type.
2345///
2346/// This function pointer accepts an edition, because in edition 2015, trait declarations
2347/// were allowed to omit parameter names. In 2018, they became required.
2348type ReqName = fn(Edition) -> bool;
2349
2350/// Parsing configuration for functions.
2351///
2352/// The syntax of function items is slightly different within trait definitions,
2353/// impl blocks, and modules. It is still parsed using the same code, just with
2354/// different flags set, so that even when the input is wrong and produces a parse
2355/// error, it still gets into the AST and the rest of the parser and
2356/// type checker can run.
2357#[derive(Clone, Copy)]
2358pub(crate) struct FnParseMode {
2359    /// A function pointer that decides if, per-parameter `p`, `p` must have a
2360    /// pattern or just a type. This field affects parsing of the parameters list.
2361    ///
2362    /// ```text
2363    /// fn foo(alef: A) -> X { X::new() }
2364    ///        -----^^ affects parsing this part of the function signature
2365    ///        |
2366    ///        if req_name returns false, then this name is optional
2367    ///
2368    /// fn bar(A) -> X;
2369    ///        ^
2370    ///        |
2371    ///        if req_name returns true, this is an error
2372    /// ```
2373    ///
2374    /// Calling this function pointer should only return false if:
2375    ///
2376    ///   * The item is being parsed inside of a trait definition.
2377    ///     Within an impl block or a module, it should always evaluate
2378    ///     to true.
2379    ///   * The span is from Edition 2015. In particular, you can get a
2380    ///     2015 span inside a 2021 crate using macros.
2381    pub(super) req_name: ReqName,
2382    /// If this flag is set to `true`, then plain, semicolon-terminated function
2383    /// prototypes are not allowed here.
2384    ///
2385    /// ```text
2386    /// fn foo(alef: A) -> X { X::new() }
2387    ///                      ^^^^^^^^^^^^
2388    ///                      |
2389    ///                      this is always allowed
2390    ///
2391    /// fn bar(alef: A, bet: B) -> X;
2392    ///                             ^
2393    ///                             |
2394    ///                             if req_body is set to true, this is an error
2395    /// ```
2396    ///
2397    /// This field should only be set to false if the item is inside of a trait
2398    /// definition or extern block. Within an impl block or a module, it should
2399    /// always be set to true.
2400    pub(super) req_body: bool,
2401}

2403/// Parsing of functions and methods.
2404impl<'a> Parser<'a> {
2405    /// Parse a function starting from the front matter (`const ...`) to the body `{ ... }` or `;`.
2406    fn parse_fn(
2407        &mut self,
2408        attrs: &mut AttrVec,
2409        fn_parse_mode: FnParseMode,
2410        sig_lo: Span,
2411        vis: &Visibility,
2412        case: Case,
2413    ) -> PResult<'a, (Ident, FnSig, Generics, Option<Box<FnContract>>, Option<Box<Block>>)> {
2414        let fn_span = self.token.span;
2415        let header = self.parse_fn_front_matter(vis, case, FrontMatterParsingMode::Function)?; // `const ... fn`
2416        let ident = self.parse_ident()?; // `foo`
2417        let mut generics = self.parse_generics()?; // `<'a, T, ...>`
2418        let decl = match self.parse_fn_decl(
2419            fn_parse_mode.req_name,
2420            AllowPlus::Yes,
2421            RecoverReturnSign::Yes,
2422        ) {
2423            Ok(decl) => decl,
2424            Err(old_err) => {
2425                // If we see `for Ty ...` then user probably meant `impl` item.
2426                if self.token.is_keyword(kw::For) {
2427                    old_err.cancel();
2428                    return Err(self.dcx().create_err(errors::FnTypoWithImpl { fn_span }));
2429                } else {
2430                    return Err(old_err);
2431                }
2432            }
2433        };
2434
2435        // Store the end of function parameters to give better diagnostics
2436        // inside `parse_fn_body()`.
2437        let fn_params_end = self.prev_token.span.shrink_to_hi();
2438
2439        let contract = self.parse_contract()?;
2440
2441        generics.where_clause = self.parse_where_clause()?; // `where T: Ord`
2442
2443        // `fn_params_end` is needed only when it's followed by a where clause.
2444        let fn_params_end =
2445            if generics.where_clause.has_where_token { Some(fn_params_end) } else { None };
2446
2447        let mut sig_hi = self.prev_token.span;
2448        // Either `;` or `{ ... }`.
2449        let body =
2450            self.parse_fn_body(attrs, &ident, &mut sig_hi, fn_parse_mode.req_body, fn_params_end)?;
2451        let fn_sig_span = sig_lo.to(sig_hi);
2452        Ok((ident, FnSig { header, decl, span: fn_sig_span }, generics, contract, body))
2453    }
2454
2455    /// Provide diagnostics when function body is not found
2456    fn error_fn_body_not_found(
2457        &mut self,
2458        ident_span: Span,
2459        req_body: bool,
2460        fn_params_end: Option<Span>,
2461    ) -> PResult<'a, ErrorGuaranteed> {
2462        let expected: &[_] =
2463            if req_body { &[exp!(OpenBrace)] } else { &[exp!(Semi), exp!(OpenBrace)] };
2464        match self.expected_one_of_not_found(&[], expected) {
2465            Ok(error_guaranteed) => Ok(error_guaranteed),
2466            Err(mut err) => {
2467                if self.token == token::CloseBrace {
2468                    // The enclosing `mod`, `trait` or `impl` is being closed, so keep the `fn` in
2469                    // the AST for typechecking.
2470                    err.span_label(ident_span, "while parsing this `fn`");
2471                    Ok(err.emit())
2472                } else if self.token == token::RArrow
2473                    && let Some(fn_params_end) = fn_params_end
2474                {
2475                    // Instead of a function body, the parser has encountered a right arrow
2476                    // preceded by a where clause.
2477
2478                    // Find whether token behind the right arrow is a function trait and
2479                    // store its span.
2480                    let fn_trait_span =
2481                        [sym::FnOnce, sym::FnMut, sym::Fn].into_iter().find_map(|symbol| {
2482                            if self.prev_token.is_ident_named(symbol) {
2483                                Some(self.prev_token.span)
2484                            } else {
2485                                None
2486                            }
2487                        });
2488
2489                    // Parse the return type (along with the right arrow) and store its span.
2490                    // If there's a parse error, cancel it and return the existing error
2491                    // as we are primarily concerned with the
2492                    // expected-function-body-but-found-something-else error here.
2493                    let arrow_span = self.token.span;
2494                    let ty_span = match self.parse_ret_ty(
2495                        AllowPlus::Yes,
2496                        RecoverQPath::Yes,
2497                        RecoverReturnSign::Yes,
2498                    ) {
2499                        Ok(ty_span) => ty_span.span().shrink_to_hi(),
2500                        Err(parse_error) => {
2501                            parse_error.cancel();
2502                            return Err(err);
2503                        }
2504                    };
2505                    let ret_ty_span = arrow_span.to(ty_span);
2506
2507                    if let Some(fn_trait_span) = fn_trait_span {
2508                        // Typo'd Fn* trait bounds such as
2509                        // fn foo<F>() where F: FnOnce -> () {}
2510                        err.subdiagnostic(errors::FnTraitMissingParen { span: fn_trait_span });
2511                    } else if let Ok(snippet) = self.psess.source_map().span_to_snippet(ret_ty_span)
2512                    {
2513                        // If token behind right arrow is not a Fn* trait, the programmer
2514                        // probably misplaced the return type after the where clause like
2515                        // `fn foo<T>() where T: Default -> u8 {}`
2516                        err.primary_message(
2517                            "return type should be specified after the function parameters",
2518                        );
2519                        err.subdiagnostic(errors::MisplacedReturnType {
2520                            fn_params_end,
2521                            snippet,
2522                            ret_ty_span,
2523                        });
2524                    }
2525                    Err(err)
2526                } else {
2527                    Err(err)
2528                }
2529            }
2530        }
2531    }

2533    /// Parse the "body" of a function.
2534    /// This can either be `;` when there's no body,
2535    /// or e.g. a block when the function is a provided one.
2536    fn parse_fn_body(
2537        &mut self,
2538        attrs: &mut AttrVec,
2539        ident: &Ident,
2540        sig_hi: &mut Span,
2541        req_body: bool,
2542        fn_params_end: Option<Span>,
2543    ) -> PResult<'a, Option<Box<Block>>> {
2544        let has_semi = if req_body {
2545            self.token == TokenKind::Semi
2546        } else {
2547            // Only include `;` in list of expected tokens if body is not required
2548            self.check(exp!(Semi))
2549        };
2550        let (inner_attrs, body) = if has_semi {
2551            // Include the trailing semicolon in the span of the signature
2552            self.expect_semi()?;
2553            *sig_hi = self.prev_token.span;
2554            (AttrVec::new(), None)
2555        } else if self.check(exp!(OpenBrace)) || self.token.is_metavar_block() {
2556            self.parse_block_common(self.token.span, BlockCheckMode::Default, None)
2557                .map(|(attrs, body)| (attrs, Some(body)))?
2558        } else if self.token == token::Eq {
2559            // Recover `fn foo() = $expr;`.
2560            self.bump(); // `=`
2561            let eq_sp = self.prev_token.span;
2562            let _ = self.parse_expr()?;
2563            self.expect_semi()?; // `;`
2564            let span = eq_sp.to(self.prev_token.span);
2565            let guar = self.dcx().emit_err(errors::FunctionBodyEqualsExpr {
2566                span,
2567                sugg: errors::FunctionBodyEqualsExprSugg { eq: eq_sp, semi: self.prev_token.span },
2568            });
2569            (AttrVec::new(), Some(self.mk_block_err(span, guar)))
2570        } else {
2571            self.error_fn_body_not_found(ident.span, req_body, fn_params_end)?;
2572            (AttrVec::new(), None)
2573        };
2574        attrs.extend(inner_attrs);
2575        Ok(body)
2576    }

2578    /// Is the current token the start of an `FnHeader` / not a valid parse?
2579    ///
2580    /// `check_pub` adds additional `pub` to the checks in case users place it
2581    /// wrongly, can be used to ensure `pub` never comes after `default`.
2582    pub(super) fn check_fn_front_matter(&mut self, check_pub: bool, case: Case) -> bool {
2583        const ALL_QUALS: &[ExpKeywordPair] = &[
2584            exp!(Pub),
2585            exp!(Gen),
2586            exp!(Const),
2587            exp!(Async),
2588            exp!(Unsafe),
2589            exp!(Safe),
2590            exp!(Extern),
2591        ];
2592
2593        // We use an over-approximation here.
2594        // `const const`, `fn const` won't parse, but we're not stepping over other syntax either.
2595        // `pub` is added in case users got confused with the ordering like `async pub fn`,
2596        // only if it wasn't preceded by `default` as `default pub` is invalid.
2597        let quals: &[_] = if check_pub {
2598            ALL_QUALS
2599        } else {
2600            &[exp!(Gen), exp!(Const), exp!(Async), exp!(Unsafe), exp!(Safe), exp!(Extern)]
2601        };
2602        self.check_keyword_case(exp!(Fn), case) // Definitely an `fn`.
2603            // `$qual fn` or `$qual $qual`:
2604            || quals.iter().any(|&exp| self.check_keyword_case(exp, case))
2605                && self.look_ahead(1, |t| {
2606                    // `$qual fn`, e.g. `const fn` or `async fn`.
2607                    t.is_keyword_case(kw::Fn, case)
2608                    // Two qualifiers `$qual $qual` is enough, e.g. `async unsafe`.
2609                    || (
2610                        (
2611                            t.is_non_raw_ident_where(|i|
2612                                quals.iter().any(|exp| exp.kw == i.name)
2613                                    // Rule out 2015 `const async: T = val`.
2614                                    && i.is_reserved()
2615                            )
2616                            || case == Case::Insensitive
2617                                && t.is_non_raw_ident_where(|i| quals.iter().any(|exp| {
2618                                    exp.kw.as_str() == i.name.as_str().to_lowercase()
2619                                }))
2620                        )
2621                        // Rule out `unsafe extern {`.
2622                        && !self.is_unsafe_foreign_mod()
2623                        // Rule out `async gen {` and `async gen move {`
2624                        && !self.is_async_gen_block())
2625                })
2626            // `extern ABI fn`
2627            || self.check_keyword_case(exp!(Extern), case)
2628                // Use `tree_look_ahead` because `ABI` might be a metavariable,
2629                // i.e. an invisible-delimited sequence, and `tree_look_ahead`
2630                // will consider that a single element when looking ahead.
2631                && self.look_ahead(1, |t| t.can_begin_string_literal())
2632                && (self.tree_look_ahead(2, |tt| {
2633                    match tt {
2634                        TokenTree::Token(t, _) => t.is_keyword_case(kw::Fn, case),
2635                        TokenTree::Delimited(..) => false,
2636                    }
2637                }) == Some(true) ||
2638                    // This branch is only for better diagnostics; `pub`, `unsafe`, etc. are not
2639                    // allowed here.
2640                    (self.may_recover()
2641                        && self.tree_look_ahead(2, |tt| {
2642                            match tt {
2643                                TokenTree::Token(t, _) =>
2644                                    ALL_QUALS.iter().any(|exp| {
2645                                        t.is_keyword(exp.kw)
2646                                    }),
2647                                TokenTree::Delimited(..) => false,
2648                            }
2649                        }) == Some(true)
2650                        && self.tree_look_ahead(3, |tt| {
2651                            match tt {
2652                                TokenTree::Token(t, _) => t.is_keyword_case(kw::Fn, case),
2653                                TokenTree::Delimited(..) => false,
2654                            }
2655                        }) == Some(true)
2656                    )
2657                )
2658    }

2660    /// Parses all the "front matter" (or "qualifiers") for a `fn` declaration,
2661    /// up to and including the `fn` keyword. The formal grammar is:
2662    ///
2663    /// ```text
2664    /// Extern = "extern" StringLit? ;
2665    /// FnQual = "const"? "async"? "unsafe"? Extern? ;
2666    /// FnFrontMatter = FnQual "fn" ;
2667    /// ```
2668    ///
2669    /// `vis` represents the visibility that was already parsed, if any. Use
2670    /// `Visibility::Inherited` when no visibility is known.
2671    ///
2672    /// If `parsing_mode` is `FrontMatterParsingMode::FunctionPtrType`, we error on `const` and `async` qualifiers,
2673    /// which are not allowed in function pointer types.
2674    pub(super) fn parse_fn_front_matter(
2675        &mut self,
2676        orig_vis: &Visibility,
2677        case: Case,
2678        parsing_mode: FrontMatterParsingMode,
2679    ) -> PResult<'a, FnHeader> {
2680        let sp_start = self.token.span;
2681        let constness = self.parse_constness(case);
2682        if parsing_mode == FrontMatterParsingMode::FunctionPtrType
2683            && let Const::Yes(const_span) = constness
2684        {
2685            self.dcx().emit_err(FnPointerCannotBeConst {
2686                span: const_span,
2687                suggestion: const_span.until(self.token.span),
2688            });
2689        }
2690
2691        let async_start_sp = self.token.span;
2692        let coroutine_kind = self.parse_coroutine_kind(case);
2693        if parsing_mode == FrontMatterParsingMode::FunctionPtrType
2694            && let Some(ast::CoroutineKind::Async { span: async_span, .. }) = coroutine_kind
2695        {
2696            self.dcx().emit_err(FnPointerCannotBeAsync {
2697                span: async_span,
2698                suggestion: async_span.until(self.token.span),
2699            });
2700        }
2701        // FIXME(gen_blocks): emit a similar error for `gen fn()`
2702
2703        let unsafe_start_sp = self.token.span;
2704        let safety = self.parse_safety(case);
2705
2706        let ext_start_sp = self.token.span;
2707        let ext = self.parse_extern(case);
2708
2709        if let Some(CoroutineKind::Async { span, .. }) = coroutine_kind {
2710            if span.is_rust_2015() {
2711                self.dcx().emit_err(errors::AsyncFnIn2015 {
2712                    span,
2713                    help: errors::HelpUseLatestEdition::new(),
2714                });
2715            }
2716        }
2717
2718        match coroutine_kind {
2719            Some(CoroutineKind::Gen { span, .. }) | Some(CoroutineKind::AsyncGen { span, .. }) => {
2720                self.psess.gated_spans.gate(sym::gen_blocks, span);
2721            }
2722            Some(CoroutineKind::Async { .. }) | None => {}
2723        }
2724
2725        if !self.eat_keyword_case(exp!(Fn), case) {
2726            // It is possible for `expect_one_of` to recover given the contents of
2727            // `self.expected_token_types`, therefore, do not use `self.unexpected()` which doesn't
2728            // account for this.
2729            match self.expect_one_of(&[], &[]) {
2730                Ok(Recovered::Yes(_)) => {}
2731                Ok(Recovered::No) => unreachable!(),
2732                Err(mut err) => {
2733                    // Qualifier keywords ordering check
2734                    enum WrongKw {
2735                        Duplicated(Span),
2736                        Misplaced(Span),
2737                        /// `MisplacedDisallowedQualifier` is only used instead of `Misplaced`,
2738                        /// when the misplaced keyword is disallowed by the current `FrontMatterParsingMode`.
2739                        /// In this case, we avoid generating the suggestion to swap around the keywords,
2740                        /// as we already generated a suggestion to remove the keyword earlier.
2741                        MisplacedDisallowedQualifier,
2742                    }
2743
2744                    // We may be able to recover
2745                    let mut recover_constness = constness;
2746                    let mut recover_coroutine_kind = coroutine_kind;
2747                    let mut recover_safety = safety;
2748                    // This will allow the machine fix to directly place the keyword in the correct place or to indicate
2749                    // that the keyword is already present and the second instance should be removed.
2750                    let wrong_kw = if self.check_keyword(exp!(Const)) {
2751                        match constness {
2752                            Const::Yes(sp) => Some(WrongKw::Duplicated(sp)),
2753                            Const::No => {
2754                                recover_constness = Const::Yes(self.token.span);
2755                                match parsing_mode {
2756                                    FrontMatterParsingMode::Function => {
2757                                        Some(WrongKw::Misplaced(async_start_sp))
2758                                    }
2759                                    FrontMatterParsingMode::FunctionPtrType => {
2760                                        self.dcx().emit_err(FnPointerCannotBeConst {
2761                                            span: self.token.span,
2762                                            suggestion: self
2763                                                .token
2764                                                .span
2765                                                .with_lo(self.prev_token.span.hi()),
2766                                        });
2767                                        Some(WrongKw::MisplacedDisallowedQualifier)
2768                                    }
2769                                }
2770                            }
2771                        }
2772                    } else if self.check_keyword(exp!(Async)) {
2773                        match coroutine_kind {
2774                            Some(CoroutineKind::Async { span, .. }) => {
2775                                Some(WrongKw::Duplicated(span))
2776                            }
2777                            Some(CoroutineKind::AsyncGen { span, .. }) => {
2778                                Some(WrongKw::Duplicated(span))
2779                            }
2780                            Some(CoroutineKind::Gen { .. }) => {
2781                                recover_coroutine_kind = Some(CoroutineKind::AsyncGen {
2782                                    span: self.token.span,
2783                                    closure_id: DUMMY_NODE_ID,
2784                                    return_impl_trait_id: DUMMY_NODE_ID,
2785                                });
2786                                // FIXME(gen_blocks): This span is wrong, didn't want to think about it.
2787                                Some(WrongKw::Misplaced(unsafe_start_sp))
2788                            }
2789                            None => {
2790                                recover_coroutine_kind = Some(CoroutineKind::Async {
2791                                    span: self.token.span,
2792                                    closure_id: DUMMY_NODE_ID,
2793                                    return_impl_trait_id: DUMMY_NODE_ID,
2794                                });
2795                                match parsing_mode {
2796                                    FrontMatterParsingMode::Function => {
2797                                        Some(WrongKw::Misplaced(async_start_sp))
2798                                    }
2799                                    FrontMatterParsingMode::FunctionPtrType => {
2800                                        self.dcx().emit_err(FnPointerCannotBeAsync {
2801                                            span: self.token.span,
2802                                            suggestion: self
2803                                                .token
2804                                                .span
2805                                                .with_lo(self.prev_token.span.hi()),
2806                                        });
2807                                        Some(WrongKw::MisplacedDisallowedQualifier)
2808                                    }
2809                                }
2810                            }
2811                        }
2812                    } else if self.check_keyword(exp!(Unsafe)) {
2813                        match safety {
2814                            Safety::Unsafe(sp) => Some(WrongKw::Duplicated(sp)),
2815                            Safety::Safe(sp) => {
2816                                recover_safety = Safety::Unsafe(self.token.span);
2817                                Some(WrongKw::Misplaced(sp))
2818                            }
2819                            Safety::Default => {
2820                                recover_safety = Safety::Unsafe(self.token.span);
2821                                Some(WrongKw::Misplaced(ext_start_sp))
2822                            }
2823                        }
2824                    } else if self.check_keyword(exp!(Safe)) {
2825                        match safety {
2826                            Safety::Safe(sp) => Some(WrongKw::Duplicated(sp)),
2827                            Safety::Unsafe(sp) => {
2828                                recover_safety = Safety::Safe(self.token.span);
2829                                Some(WrongKw::Misplaced(sp))
2830                            }
2831                            Safety::Default => {
2832                                recover_safety = Safety::Safe(self.token.span);
2833                                Some(WrongKw::Misplaced(ext_start_sp))
2834                            }
2835                        }
2836                    } else {
2837                        None
2838                    };
2839
2840                    // The keyword is already present, suggest removal of the second instance
2841                    if let Some(WrongKw::Duplicated(original_sp)) = wrong_kw {
2842                        let original_kw = self
2843                            .span_to_snippet(original_sp)
2844                            .expect("Span extracted directly from keyword should always work");
2845
2846                        err.span_suggestion(
2847                            self.token_uninterpolated_span(),
2848                            format!("`{original_kw}` already used earlier, remove this one"),
2849                            "",
2850                            Applicability::MachineApplicable,
2851                        )
2852                        .span_note(original_sp, format!("`{original_kw}` first seen here"));
2853                    }
2854                    // The keyword has not been seen yet, suggest correct placement in the function front matter

2866                                ).note("keyword order for functions declaration is `pub`, `default`, `const`, `async`, `unsafe`, `extern`");
2867                        }
2868                    }

2870                    else if self.check_keyword(exp!(Pub)) {
2871                        let sp = sp_start.to(self.prev_token.span);
2872                        if let Ok(snippet) = self.span_to_snippet(sp) {
2873                            let current_vis = match self.parse_visibility(FollowedByType::No) {
2874                                Ok(v) => v,
2875                                Err(d) => {
2876                                    d.cancel();
2877                                    return Err(err);
2878                                }
2879                            };
2880                            let vs = pprust::vis_to_string(&current_vis);
2881                            let vs = vs.trim_end();

2884                            if matches!(orig_vis.kind, VisibilityKind::Inherited) {
2885                                err.span_suggestion(
2886                                    sp_start.to(self.prev_token.span),
2887                                    format!("visibility `{vs}` must come before `{snippet}`"),
2888                                    format!("{vs} {snippet}"),
2889                                    Applicability::MachineApplicable,
2890                                );
2891                            }
2892                            // There was an explicit visibility
2893                            else {
2894                                err.span_suggestion(
2895                                    current_vis.span,
2896                                    "there is already a visibility modifier, remove one",
2897                                    "",
2898                                    Applicability::MachineApplicable,
2899                                )
2900                                .span_note(orig_vis.span, "explicit visibility first seen here");
2901                            }
2902                        }
2903                    }
2904
2905                    // FIXME(gen_blocks): add keyword recovery logic for genness
2906
2907                    if let Some(wrong_kw) = wrong_kw
2908                        && self.may_recover()
2909                        && self.look_ahead(1, |tok| tok.is_keyword_case(kw::Fn, case))
2910                    {
2911                        // Advance past the misplaced keyword and `fn`
2912                        self.bump();
2913                        self.bump();
2914                        // When we recover from a `MisplacedDisallowedQualifier`, we already emitted an error for the disallowed qualifier
2915                        // So we don't emit another error that the qualifier is unexpected.
2916                        if matches!(wrong_kw, WrongKw::MisplacedDisallowedQualifier) {
2917                            err.cancel();
2918                        } else {
2919                            err.emit();
2920                        }
2921                        return Ok(FnHeader {
2922                            constness: recover_constness,
2923                            safety: recover_safety,
2924                            coroutine_kind: recover_coroutine_kind,
2925                            ext,
2926                        });
2927                    }
2928
2929                    return Err(err);
2930                }
2931            }
2932        }
2933
2934        Ok(FnHeader { constness, safety, coroutine_kind, ext })

3087    /// Returns the parsed optional self parameter and whether a self shortcut was used.
3088    fn parse_self_param(&mut self) -> PResult<'a, Option<Param>> {
3089        // Extract an identifier *after* having confirmed that the token is one.
3090        let expect_self_ident = |this: &mut Self| match this.token.ident() {
3091            Some((ident, IdentIsRaw::No)) => {
3092                this.bump();
3093                ident
3094            }
3095            _ => unreachable!(),
3096        };
3097        // is lifetime `n` tokens ahead?
3098        let is_lifetime = |this: &Self, n| this.look_ahead(n, |t| t.is_lifetime());
3099        // Is `self` `n` tokens ahead?
3100        let is_isolated_self = |this: &Self, n| {
3101            this.is_keyword_ahead(n, &[kw::SelfLower])
3102                && this.look_ahead(n + 1, |t| t != &token::PathSep)
3103        };
3104        // Is `pin const self` `n` tokens ahead?
3105        let is_isolated_pin_const_self = |this: &Self, n| {
3106            this.look_ahead(n, |token| token.is_ident_named(sym::pin))
3107                && this.is_keyword_ahead(n + 1, &[kw::Const])
3108                && is_isolated_self(this, n + 2)
3109        };
3110        // Is `mut self` `n` tokens ahead?
3111        let is_isolated_mut_self =
3112            |this: &Self, n| this.is_keyword_ahead(n, &[kw::Mut]) && is_isolated_self(this, n + 1);
3113        // Is `pin mut self` `n` tokens ahead?
3114        let is_isolated_pin_mut_self = |this: &Self, n| {
3115            this.look_ahead(n, |token| token.is_ident_named(sym::pin))
3116                && is_isolated_mut_self(this, n + 1)
3117        };
3118        // Parse `self` or `self: TYPE`. We already know the current token is `self`.
3119        let parse_self_possibly_typed = |this: &mut Self, m| {
3120            let eself_ident = expect_self_ident(this);
3121            let eself_hi = this.prev_token.span;
3122            let eself = if this.eat(exp!(Colon)) {
3123                SelfKind::Explicit(this.parse_ty()?, m)
3124            } else {
3125                SelfKind::Value(m)
3126            };
3127            Ok((eself, eself_ident, eself_hi))
3128        };
3129        let expect_self_ident_not_typed =
3130            |this: &mut Self, modifier: &SelfKind, modifier_span: Span| {
3131                let eself_ident = expect_self_ident(this);
3132
3133                // Recover `: Type` after a qualified self
3134                if this.may_recover() && this.eat_noexpect(&token::Colon) {
3135                    let snap = this.create_snapshot_for_diagnostic();
3136                    match this.parse_ty() {
3137                        Ok(ty) => {
3138                            this.dcx().emit_err(errors::IncorrectTypeOnSelf {
3139                                span: ty.span,
3140                                move_self_modifier: errors::MoveSelfModifier {
3141                                    removal_span: modifier_span,
3142                                    insertion_span: ty.span.shrink_to_lo(),
3143                                    modifier: modifier.to_ref_suggestion(),
3144                                },
3145                            });
3146                        }
3147                        Err(diag) => {
3148                            diag.cancel();
3149                            this.restore_snapshot(snap);
3150                        }
3151                    }
3152                }
3153                eself_ident
3154            };
3155        // Recover for the grammar `*self`, `*const self`, and `*mut self`.
3156        let recover_self_ptr = |this: &mut Self| {
3157            this.dcx().emit_err(errors::SelfArgumentPointer { span: this.token.span });
3158
3159            Ok((SelfKind::Value(Mutability::Not), expect_self_ident(this), this.prev_token.span))
3160        };
3161
3162        // Parse optional `self` parameter of a method.
3163        // Only a limited set of initial token sequences is considered `self` parameters; anything
3164        // else is parsed as a normal function parameter list, so some lookahead is required.
3165        let eself_lo = self.token.span;
3166        let (eself, eself_ident, eself_hi) = match self.token.uninterpolate().kind {
3167            token::And => {

3169                let skip_lifetime_count = has_lifetime as usize;
3170                let eself = if is_isolated_self(self, skip_lifetime_count + 1) {
3171                    // `&{'lt} self`
3172                    self.bump(); // &
3173                    let lifetime = has_lifetime.then(|| self.expect_lifetime());
3174                    SelfKind::Region(lifetime, Mutability::Not)
3175                } else if is_isolated_mut_self(self, skip_lifetime_count + 1) {
3176                    // `&{'lt} mut self`
3177                    self.bump(); // &
3178                    let lifetime = has_lifetime.then(|| self.expect_lifetime());
3179                    self.bump(); // mut
3180                    SelfKind::Region(lifetime, Mutability::Mut)
3181                } else if is_isolated_pin_const_self(self, skip_lifetime_count + 1) {
3182                    // `&{'lt} pin const self`
3183                    self.bump(); // &
3184                    let lifetime = has_lifetime.then(|| self.expect_lifetime());
3185                    self.psess.gated_spans.gate(sym::pin_ergonomics, self.token.span);
3186                    self.bump(); // pin
3187                    self.bump(); // const
3188                    SelfKind::Pinned(lifetime, Mutability::Not)
3189                } else if is_isolated_pin_mut_self(self, skip_lifetime_count + 1) {
3190                    // `&{'lt} pin mut self`
3191                    self.bump(); // &
3192                    let lifetime = has_lifetime.then(|| self.expect_lifetime());
3193                    self.psess.gated_spans.gate(sym::pin_ergonomics, self.token.span);
3194                    self.bump(); // pin
3195                    self.bump(); // mut
3196                    SelfKind::Pinned(lifetime, Mutability::Mut)
3197                } else {
3198                    // `&not_self`

3213            {
3214                self.bump();
3215                self.bump();
3216                recover_self_ptr(self)?
3217            }
3218            // `self` and `self: TYPE`
3219            token::Ident(..) if is_isolated_self(self, 0) => {
3220                parse_self_possibly_typed(self, Mutability::Not)?
3221            }
3222            // `mut self` and `mut self: TYPE`
3223            token::Ident(..) if is_isolated_mut_self(self, 0) => {
3224                self.bump();
3225                parse_self_possibly_typed(self, Mutability::Mut)?
3226            }
3227            _ => return Ok(None),
3228        };
