44    /// A generic scope visitor.
45    /// Visits scopes in order to resolve some identifier in them or perform other actions.
46    /// If the callback returns `Some` result, we stop visiting scopes and return it.
47    pub(crate) fn visit_scopes<'r, T>(
48        mut self: CmResolver<'r, 'ra, 'tcx>,
49        scope_set: ScopeSet<'ra>,
50        parent_scope: &ParentScope<'ra>,
51        ctxt: SyntaxContext,
52        mut visitor: impl FnMut(
53            &mut CmResolver<'r, 'ra, 'tcx>,
54            Scope<'ra>,
55            UsePrelude,
56            SyntaxContext,
57        ) -> Option<T>,
58    ) -> Option<T> {
59        // General principles:
60        // 1. Not controlled (user-defined) names should have higher priority than controlled names
61        //    built into the language or standard library. This way we can add new names into the
62        //    language or standard library without breaking user code.
63        // 2. "Closed set" below means new names cannot appear after the current resolution attempt.
64        // Places to search (in order of decreasing priority):
65        // (Type NS)
66        // 1. FIXME: Ribs (type parameters), there's no necessary infrastructure yet
67        //    (open set, not controlled).
68        // 2. Names in modules (both normal `mod`ules and blocks), loop through hygienic parents
69        //    (open, not controlled).
70        // 3. Extern prelude (open, the open part is from macro expansions, not controlled).
71        // 4. Tool modules (closed, controlled right now, but not in the future).
72        // 5. Standard library prelude (de-facto closed, controlled).
73        // 6. Language prelude (closed, controlled).
74        // (Value NS)
75        // 1. FIXME: Ribs (local variables), there's no necessary infrastructure yet
76        //    (open set, not controlled).
77        // 2. Names in modules (both normal `mod`ules and blocks), loop through hygienic parents
78        //    (open, not controlled).
79        // 3. Standard library prelude (de-facto closed, controlled).
80        // (Macro NS)
81        // 1-3. Derive helpers (open, not controlled). All ambiguities with other names
82        //    are currently reported as errors. They should be higher in priority than preludes
83        //    and probably even names in modules according to the "general principles" above. They
84        //    also should be subject to restricted shadowing because are effectively produced by
85        //    derives (you need to resolve the derive first to add helpers into scope), but they
86        //    should be available before the derive is expanded for compatibility.
87        //    It's mess in general, so we are being conservative for now.
88        // 1-3. `macro_rules` (open, not controlled), loop through `macro_rules` scopes. Have higher
89        //    priority than prelude macros, but create ambiguities with macros in modules.
90        // 1-3. Names in modules (both normal `mod`ules and blocks), loop through hygienic parents
91        //    (open, not controlled). Have higher priority than prelude macros, but create
92        //    ambiguities with `macro_rules`.
93        // 4. `macro_use` prelude (open, the open part is from macro expansions, not controlled).
94        // 4a. User-defined prelude from macro-use
95        //    (open, the open part is from macro expansions, not controlled).
96        // 4b. "Standard library prelude" part implemented through `macro-use` (closed, controlled).
97        // 4c. Standard library prelude (de-facto closed, controlled).
98        // 6. Language prelude: builtin attributes (closed, controlled).
99
100        let rust_2015 = ctxt.edition().is_rust_2015();
101        let (ns, macro_kind) = match scope_set {
102            ScopeSet::All(ns)
103            | ScopeSet::ModuleAndExternPrelude(ns, _)
104            | ScopeSet::Late(ns, ..) => (ns, None),
105            ScopeSet::Macro(macro_kind) => (MacroNS, Some(macro_kind)),
106        };
107        let module = match scope_set {
108            // Start with the specified module.
109            ScopeSet::Late(_, module, _) | ScopeSet::ModuleAndExternPrelude(_, module) => module,
110            // Jump out of trait or enum modules, they do not act as scopes.
111            _ => parent_scope.module.nearest_item_scope(),
112        };
113        let module_and_extern_prelude = matches!(scope_set, ScopeSet::ModuleAndExternPrelude(..));
114        let mut scope = match ns {
115            _ if module_and_extern_prelude => Scope::Module(module, None),
116            TypeNS | ValueNS => Scope::Module(module, None),
117            MacroNS => Scope::DeriveHelpers(parent_scope.expansion),
118        };
119        let mut ctxt = ctxt.normalize_to_macros_2_0();
120        let mut use_prelude = !module.no_implicit_prelude;
121
122        loop {
123            let visit = match scope {
124                // Derive helpers are not in scope when resolving derives in the same container.
125                Scope::DeriveHelpers(expn_id) => {
126                    !(expn_id == parent_scope.expansion && macro_kind == Some(MacroKind::Derive))
127                }
128                Scope::DeriveHelpersCompat => true,
129                Scope::MacroRules(macro_rules_scope) => {
130                    // Use "path compression" on `macro_rules` scope chains. This is an optimization
131                    // used to avoid long scope chains, see the comments on `MacroRulesScopeRef`.
132                    // As another consequence of this optimization visitors never observe invocation
133                    // scopes for macros that were already expanded.
134                    while let MacroRulesScope::Invocation(invoc_id) = macro_rules_scope.get() {
135                        if let Some(next_scope) = self.output_macro_rules_scopes.get(&invoc_id) {
136                            macro_rules_scope.set(next_scope.get());
137                        } else {
138                            break;
139                        }
140                    }
141                    true
142                }
143                Scope::Module(..) => true,
144                Scope::MacroUsePrelude => use_prelude || rust_2015,
145                Scope::BuiltinAttrs => true,
146                Scope::ExternPrelude => use_prelude || module_and_extern_prelude,
147                Scope::ToolPrelude => use_prelude,
148                Scope::StdLibPrelude => use_prelude || ns == MacroNS,
149                Scope::BuiltinTypes => true,
150            };
151
152            if visit {
153                let use_prelude = if use_prelude { UsePrelude::Yes } else { UsePrelude::No };
154                if let break_result @ Some(..) = visitor(&mut self, scope, use_prelude, ctxt) {
155                    return break_result;
156                }
157            }
158
159            scope = match scope {
160                Scope::DeriveHelpers(LocalExpnId::ROOT) => Scope::DeriveHelpersCompat,
161                Scope::DeriveHelpers(expn_id) => {
162                    // Derive helpers are not visible to code generated by bang or derive macros.
163                    let expn_data = expn_id.expn_data();
164                    match expn_data.kind {
165                        ExpnKind::Root
166                        | ExpnKind::Macro(MacroKind::Bang | MacroKind::Derive, _) => {
167                            Scope::DeriveHelpersCompat
168                        }
169                        _ => Scope::DeriveHelpers(expn_data.parent.expect_local()),
170                    }
171                }
172                Scope::DeriveHelpersCompat => Scope::MacroRules(parent_scope.macro_rules),
173                Scope::MacroRules(macro_rules_scope) => match macro_rules_scope.get() {
174                    MacroRulesScope::Binding(binding) => {
175                        Scope::MacroRules(binding.parent_macro_rules_scope)
176                    }
177                    MacroRulesScope::Invocation(invoc_id) => {
178                        Scope::MacroRules(self.invocation_parent_scopes[&invoc_id].macro_rules)
179                    }
180                    MacroRulesScope::Empty => Scope::Module(module, None),
181                },
182                Scope::Module(..) if module_and_extern_prelude => match ns {
183                    TypeNS => {
184                        ctxt.adjust(ExpnId::root());
185                        Scope::ExternPrelude
186                    }
187                    ValueNS | MacroNS => break,
188                },
189                Scope::Module(module, prev_lint_id) => {
190                    use_prelude = !module.no_implicit_prelude;
191                    let derive_fallback_lint_id = match scope_set {
192                        ScopeSet::Late(.., lint_id) => lint_id,
193                        _ => None,
194                    };
195                    match self.hygienic_lexical_parent(module, &mut ctxt, derive_fallback_lint_id) {
196                        Some((parent_module, lint_id)) => {
197                            Scope::Module(parent_module, lint_id.or(prev_lint_id))
198                        }
199                        None => {
200                            ctxt.adjust(ExpnId::root());
201                            match ns {
202                                TypeNS => Scope::ExternPrelude,
203                                ValueNS => Scope::StdLibPrelude,
204                                MacroNS => Scope::MacroUsePrelude,
205                            }
206                        }
207                    }
208                }
209                Scope::MacroUsePrelude => Scope::StdLibPrelude,
210                Scope::BuiltinAttrs => break, // nowhere else to search
211                Scope::ExternPrelude if module_and_extern_prelude => break,
212                Scope::ExternPrelude => Scope::ToolPrelude,
213                Scope::ToolPrelude => Scope::StdLibPrelude,
214                Scope::StdLibPrelude => match ns {
215                    TypeNS => Scope::BuiltinTypes,
216                    ValueNS => break, // nowhere else to search
217                    MacroNS => Scope::BuiltinAttrs,
218                },
219                Scope::BuiltinTypes => break, // nowhere else to search
220            };
221        }
222
223        None
224    }

226    fn hygienic_lexical_parent(
227        &self,
228        module: Module<'ra>,
229        ctxt: &mut SyntaxContext,
230        derive_fallback_lint_id: Option<NodeId>,
231    ) -> Option<(Module<'ra>, Option<NodeId>)> {
232        if !module.expansion.outer_expn_is_descendant_of(*ctxt) {
233            return Some((self.expn_def_scope(ctxt.remove_mark()), None));
234        }
235
236        if let ModuleKind::Block = module.kind {
237            return Some((module.parent.unwrap().nearest_item_scope(), None));
238        }
239
240        // We need to support the next case under a deprecation warning
241        // ```
242        // struct MyStruct;
243        // ---- begin: this comes from a proc macro derive
244        // mod implementation_details {
245        //     // Note that `MyStruct` is not in scope here.
246        //     impl SomeTrait for MyStruct { ... }
247        // }
248        // ---- end
249        // ```
250        // So we have to fall back to the module's parent during lexical resolution in this case.
251        if derive_fallback_lint_id.is_some()
252            && let Some(parent) = module.parent
253            // Inner module is inside the macro
254            && module.expansion != parent.expansion
255            // Parent module is outside of the macro
256            && module.expansion.is_descendant_of(parent.expansion)
257            // The macro is a proc macro derive
258            && let Some(def_id) = module.expansion.expn_data().macro_def_id
259        {
260            let ext = &self.get_macro_by_def_id(def_id).ext;
261            if ext.builtin_name.is_none()
262                && ext.macro_kind() == MacroKind::Derive
263                && parent.expansion.outer_expn_is_descendant_of(*ctxt)
264            {
265                return Some((parent, derive_fallback_lint_id));
266            }
267        }
268
269        None
270    }

272    /// This resolves the identifier `ident` in the namespace `ns` in the current lexical scope.
273    /// More specifically, we proceed up the hierarchy of scopes and return the binding for
274    /// `ident` in the first scope that defines it (or None if no scopes define it).
275    ///
276    /// A block's items are above its local variables in the scope hierarchy, regardless of where
277    /// the items are defined in the block. For example,
278    /// ```rust
279    /// fn f() {
280    ///    g(); // Since there are no local variables in scope yet, this resolves to the item.
281    ///    let g = || {};
282    ///    fn g() {}
283    ///    g(); // This resolves to the local variable `g` since it shadows the item.
284    /// }
285    /// ```
286    ///
287    /// Invariant: This must only be called during main resolution, not during
288    /// import resolution.
289    #[instrument(level = "debug", skip(self, ribs))]
290    pub(crate) fn resolve_ident_in_lexical_scope(
291        &mut self,
292        mut ident: Ident,
293        ns: Namespace,
294        parent_scope: &ParentScope<'ra>,
295        finalize: Option<Finalize>,
296        ribs: &[Rib<'ra>],
297        ignore_binding: Option<NameBinding<'ra>>,
298    ) -> Option<LexicalScopeBinding<'ra>> {
299        assert!(ns == TypeNS || ns == ValueNS);
300        let orig_ident = ident;
301        let (general_span, normalized_span) = if ident.name == kw::SelfUpper {
302            // FIXME(jseyfried) improve `Self` hygiene
303            let empty_span = ident.span.with_ctxt(SyntaxContext::root());
304            (empty_span, empty_span)
305        } else if ns == TypeNS {
306            let normalized_span = ident.span.normalize_to_macros_2_0();
307            (normalized_span, normalized_span)
308        } else {
309            (ident.span.normalize_to_macro_rules(), ident.span.normalize_to_macros_2_0())
310        };
311        ident.span = general_span;
312        let normalized_ident = Ident { span: normalized_span, ..ident };
313
314        // Walk backwards up the ribs in scope.
315        let mut module = self.graph_root;
316        for (i, rib) in ribs.iter().enumerate().rev() {
317            debug!("walk rib\n{:?}", rib.bindings);
318            // Use the rib kind to determine whether we are resolving parameters
319            // (macro 2.0 hygiene) or local variables (`macro_rules` hygiene).
320            let rib_ident = if rib.kind.contains_params() { normalized_ident } else { ident };
321            if let Some((original_rib_ident_def, res)) = rib.bindings.get_key_value(&rib_ident) {
322                // The ident resolves to a type parameter or local variable.
323                return Some(LexicalScopeBinding::Res(self.validate_res_from_ribs(
324                    i,
325                    rib_ident,
326                    *res,
327                    finalize.map(|finalize| finalize.path_span),
328                    *original_rib_ident_def,
329                    ribs,
330                )));
331            }
332
333            module = match rib.kind {
334                RibKind::Module(module) => module,
335                RibKind::MacroDefinition(def) if def == self.macro_def(ident.span.ctxt()) => {
336                    // If an invocation of this macro created `ident`, give up on `ident`
337                    // and switch to `ident`'s source from the macro definition.
338                    ident.span.remove_mark();
339                    continue;
340                }
341                _ => continue,
342            };
343
344            match module.kind {
345                ModuleKind::Block => {} // We can see through blocks
346                _ => break,
347            }
348
349            let item = self.cm().resolve_ident_in_module_unadjusted(
350                ModuleOrUniformRoot::Module(module),
351                ident,
352                ns,
353                parent_scope,
354                Shadowing::Unrestricted,
355                finalize.map(|finalize| Finalize { used: Used::Scope, ..finalize }),
356                ignore_binding,
357                None,
358            );
359            if let Ok(binding) = item {
360                // The ident resolves to an item.
361                return Some(LexicalScopeBinding::Item(binding));
362            }
363        }
364        self.cm()
365            .early_resolve_ident_in_lexical_scope(
366                orig_ident,
367                ScopeSet::Late(ns, module, finalize.map(|finalize| finalize.node_id)),
368                parent_scope,
369                finalize,
370                finalize.is_some(),
371                ignore_binding,
372                None,
373            )
374            .ok()
375            .map(LexicalScopeBinding::Item)
376    }

378    /// Resolve an identifier in lexical scope.
379    /// This is a variation of `fn resolve_ident_in_lexical_scope` that can be run during
380    /// expansion and import resolution (perhaps they can be merged in the future).
381    /// The function is used for resolving initial segments of macro paths (e.g., `foo` in
382    /// `foo::bar!();` or `foo!();`) and also for import paths on 2018 edition.
383    #[instrument(level = "debug", skip(self))]
384    pub(crate) fn early_resolve_ident_in_lexical_scope<'r>(
385        self: CmResolver<'r, 'ra, 'tcx>,
386        orig_ident: Ident,
387        scope_set: ScopeSet<'ra>,
388        parent_scope: &ParentScope<'ra>,
389        finalize: Option<Finalize>,
390        force: bool,
391        ignore_binding: Option<NameBinding<'ra>>,
392        ignore_import: Option<Import<'ra>>,
393    ) -> Result<NameBinding<'ra>, Determinacy> {
394        bitflags::bitflags! {
395            #[derive(Clone, Copy)]
396            struct Flags: u8 {
397                const MACRO_RULES          = 1 << 0;
398                const MODULE               = 1 << 1;
399                const MISC_SUGGEST_CRATE   = 1 << 2;
400                const MISC_SUGGEST_SELF    = 1 << 3;
401                const MISC_FROM_PRELUDE    = 1 << 4;
402            }
403        }
404
405        assert!(force || finalize.is_none()); // `finalize` implies `force`
406
407        // Make sure `self`, `super` etc produce an error when passed to here.
408        if orig_ident.is_path_segment_keyword() {
409            return Err(Determinacy::Determined);
410        }
411
412        let (ns, macro_kind) = match scope_set {
413            ScopeSet::All(ns)
414            | ScopeSet::ModuleAndExternPrelude(ns, _)
415            | ScopeSet::Late(ns, ..) => (ns, None),
416            ScopeSet::Macro(macro_kind) => (MacroNS, Some(macro_kind)),
417        };
418
419        // This is *the* result, resolution from the scope closest to the resolved identifier.
420        // However, sometimes this result is "weak" because it comes from a glob import or
421        // a macro expansion, and in this case it cannot shadow names from outer scopes, e.g.
422        // mod m { ... } // solution in outer scope
423        // {
424        //     use prefix::*; // imports another `m` - innermost solution
425        //                    // weak, cannot shadow the outer `m`, need to report ambiguity error
426        //     m::mac!();
427        // }
428        // So we have to save the innermost solution and continue searching in outer scopes
429        // to detect potential ambiguities.
430        let mut innermost_result: Option<(NameBinding<'_>, Flags)> = None;
431        let mut determinacy = Determinacy::Determined;
432
433        // Go through all the scopes and try to resolve the name.
434        let break_result = self.visit_scopes(
435            scope_set,
436            parent_scope,
437            orig_ident.span.ctxt(),
438            |this, scope, use_prelude, ctxt| {
439                let ident = Ident::new(orig_ident.name, orig_ident.span.with_ctxt(ctxt));
440                let result = match scope {
441                    Scope::DeriveHelpers(expn_id) => {
442                        if let Some(binding) = this.helper_attrs.get(&expn_id).and_then(|attrs| {
443                            attrs.iter().rfind(|(i, _)| ident == *i).map(|(_, binding)| *binding)
444                        }) {
445                            Ok((binding, Flags::empty()))
446                        } else {
447                            Err(Determinacy::Determined)
448                        }
449                    }
450                    Scope::DeriveHelpersCompat => {
451                        // FIXME: Try running this logic earlier, to allocate name bindings for
452                        // legacy derive helpers when creating an attribute invocation with
453                        // following derives. Legacy derive helpers are not common, so it shouldn't
454                        // affect performance. It should also allow to remove the `derives`
455                        // component from `ParentScope`.
456                        let mut result = Err(Determinacy::Determined);
457                        for derive in parent_scope.derives {
458                            let parent_scope = &ParentScope { derives: &[], ..*parent_scope };
459                            match this.reborrow().resolve_macro_path(
460                                derive,
461                                Some(MacroKind::Derive),
462                                parent_scope,
463                                true,
464                                force,
465                                ignore_import,
466                                None,
467                            ) {
468                                Ok((Some(ext), _)) => {
469                                    if ext.helper_attrs.contains(&ident.name) {
470                                        let binding = this.arenas.new_pub_res_binding(
471                                            Res::NonMacroAttr(NonMacroAttrKind::DeriveHelperCompat),
472                                            derive.span,
473                                            LocalExpnId::ROOT,
474                                        );
475                                        result = Ok((binding, Flags::empty()));
476                                        break;
477                                    }
478                                }
479                                Ok(_) | Err(Determinacy::Determined) => {}
480                                Err(Determinacy::Undetermined) => {
481                                    result = Err(Determinacy::Undetermined)
482                                }
483                            }
484                        }
485                        result
486                    }
487                    Scope::MacroRules(macro_rules_scope) => match macro_rules_scope.get() {
488                        MacroRulesScope::Binding(macro_rules_binding)
489                            if ident == macro_rules_binding.ident =>
490                        {
491                            Ok((macro_rules_binding.binding, Flags::MACRO_RULES))
492                        }
493                        MacroRulesScope::Invocation(_) => Err(Determinacy::Undetermined),
494                        _ => Err(Determinacy::Determined),
495                    },
496                    Scope::Module(module, derive_fallback_lint_id) => {
497                        let (adjusted_parent_scope, finalize) =
498                            if matches!(scope_set, ScopeSet::ModuleAndExternPrelude(..)) {
499                                (parent_scope, finalize)
500                            } else {
501                                (
502                                    &ParentScope { module, ..*parent_scope },
503                                    finalize.map(|f| Finalize { used: Used::Scope, ..f }),
504                                )
505                            };
506                        let binding = this.reborrow().resolve_ident_in_module_unadjusted(
507                            ModuleOrUniformRoot::Module(module),
508                            ident,
509                            ns,
510                            adjusted_parent_scope,
511                            if matches!(scope_set, ScopeSet::Late(..)) {
512                                Shadowing::Unrestricted
513                            } else {
514                                Shadowing::Restricted
515                            },
516                            finalize,
517                            ignore_binding,
518                            ignore_import,
519                        );
520                        match binding {
521                            Ok(binding) => {
522                                if let Some(lint_id) = derive_fallback_lint_id {
523                                    this.get_mut().lint_buffer.buffer_lint(
524                                        PROC_MACRO_DERIVE_RESOLUTION_FALLBACK,
525                                        lint_id,
526                                        orig_ident.span,
527                                        BuiltinLintDiag::ProcMacroDeriveResolutionFallback {
528                                            span: orig_ident.span,
529                                            ns,
530                                            ident,
531                                        },
532                                    );
533                                }
534                                let misc_flags = if module == this.graph_root {
535                                    Flags::MISC_SUGGEST_CRATE
536                                } else if module.is_normal() {
537                                    Flags::MISC_SUGGEST_SELF
538                                } else {
539                                    Flags::empty()
540                                };
541                                Ok((binding, Flags::MODULE | misc_flags))
542                            }
543                            Err((Determinacy::Undetermined, Weak::No)) => {
544                                return Some(Err(Determinacy::determined(force)));
545                            }
546                            Err((Determinacy::Undetermined, Weak::Yes)) => {
547                                Err(Determinacy::Undetermined)
548                            }
549                            Err((Determinacy::Determined, _)) => Err(Determinacy::Determined),
550                        }
551                    }
552                    Scope::MacroUsePrelude => {
553                        match this.macro_use_prelude.get(&ident.name).cloned() {
554                            Some(binding) => Ok((binding, Flags::MISC_FROM_PRELUDE)),
555                            None => Err(Determinacy::determined(
556                                this.graph_root.unexpanded_invocations.borrow().is_empty(),
557                            )),
558                        }
559                    }
560                    Scope::BuiltinAttrs => match this.builtin_attrs_bindings.get(&ident.name) {
561                        Some(binding) => Ok((*binding, Flags::empty())),
562                        None => Err(Determinacy::Determined),
563                    },
564                    Scope::ExternPrelude => {
565                        match this.reborrow().extern_prelude_get(ident, finalize.is_some()) {
566                            Some(binding) => Ok((binding, Flags::empty())),
567                            None => Err(Determinacy::determined(
568                                this.graph_root.unexpanded_invocations.borrow().is_empty(),
569                            )),
570                        }
571                    }
572                    Scope::ToolPrelude => match this.registered_tool_bindings.get(&ident) {
573                        Some(binding) => Ok((*binding, Flags::empty())),
574                        None => Err(Determinacy::Determined),
575                    },
576                    Scope::StdLibPrelude => {
577                        let mut result = Err(Determinacy::Determined);
578                        if let Some(prelude) = this.prelude
579                            && let Ok(binding) = this.reborrow().resolve_ident_in_module_unadjusted(
580                                ModuleOrUniformRoot::Module(prelude),
581                                ident,
582                                ns,
583                                parent_scope,
584                                Shadowing::Unrestricted,
585                                None,
586                                ignore_binding,
587                                ignore_import,
588                            )
589                            && (matches!(use_prelude, UsePrelude::Yes)
590                                || this.is_builtin_macro(binding.res()))
591                        {
592                            result = Ok((binding, Flags::MISC_FROM_PRELUDE));
593                        }
594
595                        result
596                    }
597                    Scope::BuiltinTypes => match this.builtin_types_bindings.get(&ident.name) {
598                        Some(binding) => {
599                            if matches!(ident.name, sym::f16)
600                                && !this.tcx.features().f16()
601                                && !ident.span.allows_unstable(sym::f16)
602                                && finalize.is_some()
603                                && innermost_result.is_none()
604                            {
605                                feature_err(
606                                    this.tcx.sess,
607                                    sym::f16,
608                                    ident.span,
609                                    "the type `f16` is unstable",
610                                )
611                                .emit();
612                            }
613                            if matches!(ident.name, sym::f128)
614                                && !this.tcx.features().f128()
615                                && !ident.span.allows_unstable(sym::f128)
616                                && finalize.is_some()
617                                && innermost_result.is_none()
618                            {
619                                feature_err(
620                                    this.tcx.sess,
621                                    sym::f128,
622                                    ident.span,
623                                    "the type `f128` is unstable",
624                                )
625                                .emit();
626                            }
627                            Ok((*binding, Flags::empty()))
628                        }
629                        None => Err(Determinacy::Determined),
630                    },
631                };
632
633                match result {
634                    Ok((binding, flags)) => {
635                        let binding_macro_kind = binding.macro_kind();
636                        // If we're looking for an attribute, that might be supported by a
637                        // `macro_rules!` macro.
638                        // FIXME: Replace this with tracking multiple macro kinds for one Def.
639                        if !(sub_namespace_match(binding_macro_kind, macro_kind)
640                            || (binding_macro_kind == Some(MacroKind::Bang)
641                                && macro_kind == Some(MacroKind::Attr)
642                                && this
643                                    .get_macro(binding.res())
644                                    .is_some_and(|macro_data| macro_data.attr_ext.is_some())))
645                        {
646                            return None;
647                        }
648
649                        if finalize.is_none() || matches!(scope_set, ScopeSet::Late(..)) {
650                            return Some(Ok(binding));
651                        }
652
653                        if let Some((innermost_binding, innermost_flags)) = innermost_result {
654                            // Found another solution, if the first one was "weak", report an error.
655                            let (res, innermost_res) = (binding.res(), innermost_binding.res());
656                            if res != innermost_res {
657                                let is_builtin = |res| {
658                                    matches!(res, Res::NonMacroAttr(NonMacroAttrKind::Builtin(..)))
659                                };
660                                let derive_helper =
661                                    Res::NonMacroAttr(NonMacroAttrKind::DeriveHelper);
662                                let derive_helper_compat =
663                                    Res::NonMacroAttr(NonMacroAttrKind::DeriveHelperCompat);
664
665                                let ambiguity_error_kind = if is_builtin(innermost_res)
666                                    || is_builtin(res)
667                                {
668                                    Some(AmbiguityKind::BuiltinAttr)
669                                } else if innermost_res == derive_helper_compat
670                                    || res == derive_helper_compat && innermost_res != derive_helper
671                                {
672                                    Some(AmbiguityKind::DeriveHelper)
673                                } else if innermost_flags.contains(Flags::MACRO_RULES)
674                                    && flags.contains(Flags::MODULE)
675                                    && !this.disambiguate_macro_rules_vs_modularized(
676                                        innermost_binding,
677                                        binding,
678                                    )
679                                    || flags.contains(Flags::MACRO_RULES)
680                                        && innermost_flags.contains(Flags::MODULE)
681                                        && !this.disambiguate_macro_rules_vs_modularized(
682                                            binding,
683                                            innermost_binding,
684                                        )
685                                {
686                                    Some(AmbiguityKind::MacroRulesVsModularized)
687                                } else if innermost_binding.is_glob_import() {
688                                    Some(AmbiguityKind::GlobVsOuter)
689                                } else if innermost_binding
690                                    .may_appear_after(parent_scope.expansion, binding)
691                                {
692                                    Some(AmbiguityKind::MoreExpandedVsOuter)
693                                } else {
694                                    None
695                                };
696                                if let Some(kind) = ambiguity_error_kind {
697                                    let misc = |f: Flags| {
698                                        if f.contains(Flags::MISC_SUGGEST_CRATE) {
699                                            AmbiguityErrorMisc::SuggestCrate
700                                        } else if f.contains(Flags::MISC_SUGGEST_SELF) {
701                                            AmbiguityErrorMisc::SuggestSelf
702                                        } else if f.contains(Flags::MISC_FROM_PRELUDE) {
703                                            AmbiguityErrorMisc::FromPrelude
704                                        } else {
705                                            AmbiguityErrorMisc::None
706                                        }
707                                    };
708                                    this.get_mut().ambiguity_errors.push(AmbiguityError {
709                                        kind,
710                                        ident: orig_ident,
711                                        b1: innermost_binding,
712                                        b2: binding,
713                                        warning: false,
714                                        misc1: misc(innermost_flags),
715                                        misc2: misc(flags),
716                                    });
717                                    return Some(Ok(innermost_binding));
718                                }
719                            }
720                        } else {
721                            // Found the first solution.
722                            innermost_result = Some((binding, flags));
723                        }
724                    }
725                    Err(Determinacy::Determined) => {}
726                    Err(Determinacy::Undetermined) => determinacy = Determinacy::Undetermined,
727                }
728
729                None
730            },
731        );
732
733        if let Some(break_result) = break_result {
734            return break_result;
735        }
736
737        // The first found solution was the only one, return it.
738        if let Some((binding, _)) = innermost_result {
739            return Ok(binding);
740        }
741
742        Err(Determinacy::determined(determinacy == Determinacy::Determined || force))
743    }

759    pub(crate) fn resolve_ident_in_module<'r>(
760        self: CmResolver<'r, 'ra, 'tcx>,
761        module: ModuleOrUniformRoot<'ra>,
762        mut ident: Ident,
763        ns: Namespace,
764        parent_scope: &ParentScope<'ra>,
765        finalize: Option<Finalize>,
766        ignore_binding: Option<NameBinding<'ra>>,
767        ignore_import: Option<Import<'ra>>,
768    ) -> Result<NameBinding<'ra>, (Determinacy, Weak)> {
769        let tmp_parent_scope;
770        let mut adjusted_parent_scope = parent_scope;
771        match module {
772            ModuleOrUniformRoot::Module(m) => {
773                if let Some(def) = ident.span.normalize_to_macros_2_0_and_adjust(m.expansion) {

775                        ParentScope { module: self.expn_def_scope(def), ..*parent_scope };
776                    adjusted_parent_scope = &tmp_parent_scope;

781            }
782            ModuleOrUniformRoot::ModuleAndExternPrelude(..) | ModuleOrUniformRoot::CurrentScope => {
783                // No adjustments
784            }
785        }
786        self.resolve_ident_in_module_unadjusted(
787            module,
788            ident,
789            ns,
790            adjusted_parent_scope,
791            Shadowing::Unrestricted,
792            finalize,
793            ignore_binding,
794            ignore_import,
795        )

797    /// Attempts to resolve `ident` in namespaces `ns` of `module`.
798    /// Invariant: if `finalize` is `Some`, expansion and import resolution must be complete.
799    #[instrument(level = "debug", skip(self))]
800    fn resolve_ident_in_module_unadjusted<'r>(
801        mut self: CmResolver<'r, 'ra, 'tcx>,
802        module: ModuleOrUniformRoot<'ra>,
803        ident: Ident,
804        ns: Namespace,
805        parent_scope: &ParentScope<'ra>,
806        shadowing: Shadowing,
807        finalize: Option<Finalize>,
808        // This binding should be ignored during in-module resolution, so that we don't get
809        // "self-confirming" import resolutions during import validation and checking.
810        ignore_binding: Option<NameBinding<'ra>>,
811        ignore_import: Option<Import<'ra>>,
812    ) -> Result<NameBinding<'ra>, (Determinacy, Weak)> {
813        let module = match module {
814            ModuleOrUniformRoot::Module(module) => module,
815            ModuleOrUniformRoot::ModuleAndExternPrelude(module) => {
816                assert_eq!(shadowing, Shadowing::Unrestricted);
817                let binding = self.early_resolve_ident_in_lexical_scope(
818                    ident,
819                    ScopeSet::ModuleAndExternPrelude(ns, module),
820                    parent_scope,
821                    finalize,
822                    finalize.is_some(),
823                    ignore_binding,
824                    ignore_import,
825                );
826                return binding.map_err(|determinacy| (determinacy, Weak::No));
827            }
828            ModuleOrUniformRoot::ExternPrelude => {
829                assert_eq!(shadowing, Shadowing::Unrestricted);
830                return if ns != TypeNS {
831                    Err((Determined, Weak::No))
832                } else if let Some(binding) =
833                    self.reborrow().extern_prelude_get(ident, finalize.is_some())
834                {
835                    Ok(binding)
836                } else if !self.graph_root.unexpanded_invocations.borrow().is_empty() {
837                    // Macro-expanded `extern crate` items can add names to extern prelude.
838                    Err((Undetermined, Weak::No))
839                } else {
840                    Err((Determined, Weak::No))
841                };
842            }
843            ModuleOrUniformRoot::CurrentScope => {
844                assert_eq!(shadowing, Shadowing::Unrestricted);
845                if ns == TypeNS {
846                    if ident.name == kw::Crate || ident.name == kw::DollarCrate {
847                        let module = self.resolve_crate_root(ident);
848                        return Ok(module.self_binding.unwrap());
849                    } else if ident.name == kw::Super || ident.name == kw::SelfLower {
850                        // FIXME: Implement these with renaming requirements so that e.g.
851                        // `use super;` doesn't work, but `use super as name;` does.
852                        // Fall through here to get an error from `early_resolve_...`.
853                    }
854                }
855
856                let binding = self.early_resolve_ident_in_lexical_scope(
857                    ident,
858                    ScopeSet::All(ns),
859                    parent_scope,
860                    finalize,
861                    finalize.is_some(),
862                    ignore_binding,
863                    ignore_import,
864                );
865                return binding.map_err(|determinacy| (determinacy, Weak::No));
866            }
867        };
868
869        let key = BindingKey::new(ident, ns);
870        // `try_borrow_mut` is required to ensure exclusive access, even if the resulting binding
871        // doesn't need to be mutable. It will fail when there is a cycle of imports, and without
872        // the exclusive access infinite recursion will crash the compiler with stack overflow.
873        let resolution = &*self
874            .resolution_or_default(module, key)
875            .try_borrow_mut()
876            .map_err(|_| (Determined, Weak::No))?;
877
878        // If the primary binding is unusable, search further and return the shadowed glob
879        // binding if it exists. What we really want here is having two separate scopes in
880        // a module - one for non-globs and one for globs, but until that's done use this
881        // hack to avoid inconsistent resolution ICEs during import validation.
882        let binding = [resolution.non_glob_binding, resolution.glob_binding]
883            .into_iter()
884            .find_map(|binding| if binding == ignore_binding { None } else { binding });
885
886        if let Some(finalize) = finalize {
887            return self.get_mut().finalize_module_binding(
888                ident,
889                binding,
890                if resolution.non_glob_binding.is_some() { resolution.glob_binding } else { None },
891                parent_scope,
892                finalize,
893                shadowing,
894            );
895        }
896
897        let check_usable = |this: CmResolver<'r, 'ra, 'tcx>, binding: NameBinding<'ra>| {
898            let usable = this.is_accessible_from(binding.vis, parent_scope.module);
899            if usable { Ok(binding) } else { Err((Determined, Weak::No)) }
900        };
901
902        // Items and single imports are not shadowable, if we have one, then it's determined.
903        if let Some(binding) = binding
904            && !binding.is_glob_import()
905        {
906            return check_usable(self, binding);
907        }
908
909        // --- From now on we either have a glob resolution or no resolution. ---
910
911        // Check if one of single imports can still define the name,
912        // if it can then our result is not determined and can be invalidated.
913        if self.reborrow().single_import_can_define_name(
914            &resolution,
915            binding,
916            ns,
917            ignore_import,
918            ignore_binding,
919            parent_scope,
920        ) {
921            return Err((Undetermined, Weak::No));
922        }
923
924        // So we have a resolution that's from a glob import. This resolution is determined
925        // if it cannot be shadowed by some new item/import expanded from a macro.
926        // This happens either if there are no unexpanded macros, or expanded names cannot
927        // shadow globs (that happens in macro namespace or with restricted shadowing).
928        //
929        // Additionally, any macro in any module can plant names in the root module if it creates
930        // `macro_export` macros, so the root module effectively has unresolved invocations if any
931        // module has unresolved invocations.
932        // However, it causes resolution/expansion to stuck too often (#53144), so, to make
933        // progress, we have to ignore those potential unresolved invocations from other modules
934        // and prohibit access to macro-expanded `macro_export` macros instead (unless restricted
935        // shadowing is enabled, see `macro_expanded_macro_export_errors`).
936        if let Some(binding) = binding {
937            if binding.determined() || ns == MacroNS || shadowing == Shadowing::Restricted {
938                return check_usable(self, binding);
939            } else {
940                return Err((Undetermined, Weak::No));
941            }
942        }
943
944        // --- From now on we have no resolution. ---
945
946        // Now we are in situation when new item/import can appear only from a glob or a macro
947        // expansion. With restricted shadowing names from globs and macro expansions cannot
948        // shadow names from outer scopes, so we can freely fallback from module search to search
949        // in outer scopes. For `early_resolve_ident_in_lexical_scope` to continue search in outer
950        // scopes we return `Undetermined` with `Weak::Yes`.
951
952        // Check if one of unexpanded macros can still define the name,
953        // if it can then our "no resolution" result is not determined and can be invalidated.
954        if !module.unexpanded_invocations.borrow().is_empty() {
955            return Err((Undetermined, Weak::Yes));
956        }
957
958        // Check if one of glob imports can still define the name,
959        // if it can then our "no resolution" result is not determined and can be invalidated.
960        for glob_import in module.globs.borrow().iter() {
961            if ignore_import == Some(*glob_import) {
962                continue;
963            }
964            if !self.is_accessible_from(glob_import.vis, parent_scope.module) {
965                continue;
966            }
967            let module = match glob_import.imported_module.get() {
968                Some(ModuleOrUniformRoot::Module(module)) => module,
969                Some(_) => continue,
970                None => return Err((Undetermined, Weak::Yes)),
971            };
972            let tmp_parent_scope;
973            let (mut adjusted_parent_scope, mut ident) =
974                (parent_scope, ident.normalize_to_macros_2_0());
975            match ident.span.glob_adjust(module.expansion, glob_import.span) {
976                Some(Some(def)) => {
977                    tmp_parent_scope =
978                        ParentScope { module: self.expn_def_scope(def), ..*parent_scope };
979                    adjusted_parent_scope = &tmp_parent_scope;
980                }
981                Some(None) => {}
982                None => continue,
983            };
984            let result = self.reborrow().resolve_ident_in_module_unadjusted(
985                ModuleOrUniformRoot::Module(module),
986                ident,
987                ns,
988                adjusted_parent_scope,
989                Shadowing::Unrestricted,
990                None,
991                ignore_binding,
992                ignore_import,
993            );
994
995            match result {
996                Err((Determined, _)) => continue,
997                Ok(binding)
998                    if !self.is_accessible_from(binding.vis, glob_import.parent_scope.module) =>
999                {
1000                    continue;
1001                }
1002                Ok(_) | Err((Undetermined, _)) => return Err((Undetermined, Weak::Yes)),
1003            }
1004        }
1005
1006        // No resolution and no one else can define the name - determinate error.
1007        Err((Determined, Weak::No))
1008    }

1010    fn finalize_module_binding(
1011        &mut self,
1012        ident: Ident,
1013        binding: Option<NameBinding<'ra>>,
1014        shadowed_glob: Option<NameBinding<'ra>>,
1015        parent_scope: &ParentScope<'ra>,
1016        finalize: Finalize,
1017        shadowing: Shadowing,
1018    ) -> Result<NameBinding<'ra>, (Determinacy, Weak)> {
1019        let Finalize { path_span, report_private, used, root_span, .. } = finalize;
1020
1021        let Some(binding) = binding else {
1022            return Err((Determined, Weak::No));
1023        };
1024
1025        if !self.is_accessible_from(binding.vis, parent_scope.module) {
1026            if report_private {
1027                self.privacy_errors.push(PrivacyError {
1028                    ident,
1029                    binding,
1030                    dedup_span: path_span,
1031                    outermost_res: None,
1032                    source: None,
1033                    parent_scope: *parent_scope,
1034                    single_nested: path_span != root_span,
1035                });
1036            } else {
1037                return Err((Determined, Weak::No));
1038            }
1039        }
1040
1041        // Forbid expanded shadowing to avoid time travel.
1042        if let Some(shadowed_glob) = shadowed_glob
1043            && shadowing == Shadowing::Restricted
1044            && binding.expansion != LocalExpnId::ROOT
1045            && binding.res() != shadowed_glob.res()
1046        {
1047            self.ambiguity_errors.push(AmbiguityError {
1048                kind: AmbiguityKind::GlobVsExpanded,
1049                ident,
1050                b1: binding,
1051                b2: shadowed_glob,
1052                warning: false,
1053                misc1: AmbiguityErrorMisc::None,
1054                misc2: AmbiguityErrorMisc::None,
1055            });
1056        }
1057
1058        if shadowing == Shadowing::Unrestricted
1059            && binding.expansion != LocalExpnId::ROOT
1060            && let NameBindingKind::Import { import, .. } = binding.kind
1061            && matches!(import.kind, ImportKind::MacroExport)
1062        {
1063            self.macro_expanded_macro_export_errors.insert((path_span, binding.span));
1064        }
1065
1066        self.record_use(ident, binding, used);
1067        return Ok(binding);
1068    }

1070    // Checks if a single import can define the `Ident` corresponding to `binding`.
1071    // This is used to check whether we can definitively accept a glob as a resolution.
1072    fn single_import_can_define_name<'r>(
1073        mut self: CmResolver<'r, 'ra, 'tcx>,
1074        resolution: &NameResolution<'ra>,
1075        binding: Option<NameBinding<'ra>>,
1076        ns: Namespace,
1077        ignore_import: Option<Import<'ra>>,
1078        ignore_binding: Option<NameBinding<'ra>>,
1079        parent_scope: &ParentScope<'ra>,
1080    ) -> bool {
1081        for single_import in &resolution.single_imports {
1082            if ignore_import == Some(*single_import) {
1083                continue;
1084            }
1085            if !self.is_accessible_from(single_import.vis, parent_scope.module) {
1086                continue;
1087            }
1088            if let Some(ignored) = ignore_binding
1089                && let NameBindingKind::Import { import, .. } = ignored.kind
1090                && import == *single_import
1091            {
1092                continue;
1093            }
1094
1095            let Some(module) = single_import.imported_module.get() else {
1096                return true;
1097            };
1098            let ImportKind::Single { source, target, bindings, .. } = &single_import.kind else {
1099                unreachable!();
1100            };
1101            if source != target {
1102                if bindings.iter().all(|binding| binding.get().binding().is_none()) {
1103                    return true;
1104                } else if bindings[ns].get().binding().is_none() && binding.is_some() {
1105                    return true;
1106                }
1107            }
1108
1109            match self.reborrow().resolve_ident_in_module(
1110                module,
1111                *source,
1112                ns,
1113                &single_import.parent_scope,
1114                None,
1115                ignore_binding,
1116                ignore_import,
1117            ) {
1118                Err((Determined, _)) => continue,
1119                Ok(binding)
1120                    if !self.is_accessible_from(binding.vis, single_import.parent_scope.module) =>
1121                {
1122                    continue;
1123                }
1124                Ok(_) | Err((Undetermined, _)) => {
1125                    return true;
1126                }
1127            }
1128        }
1129
1130        false
1131    }

1133    /// Validate a local resolution (from ribs).
1134    #[instrument(level = "debug", skip(self, all_ribs))]
1135    fn validate_res_from_ribs(
1136        &mut self,
1137        rib_index: usize,
1138        rib_ident: Ident,
1139        mut res: Res,
1140        finalize: Option<Span>,
1141        original_rib_ident_def: Ident,
1142        all_ribs: &[Rib<'ra>],
1143    ) -> Res {
1144        debug!("validate_res_from_ribs({:?})", res);
1145        let ribs = &all_ribs[rib_index + 1..];
1146
1147        // An invalid forward use of a generic parameter from a previous default
1148        // or in a const param ty.
1149        if let RibKind::ForwardGenericParamBan(reason) = all_ribs[rib_index].kind {
1150            if let Some(span) = finalize {
1151                let res_error = if rib_ident.name == kw::SelfUpper {
1152                    ResolutionError::ForwardDeclaredSelf(reason)
1153                } else {
1154                    ResolutionError::ForwardDeclaredGenericParam(rib_ident.name, reason)
1155                };
1156                self.report_error(span, res_error);
1157            }
1158            assert_eq!(res, Res::Err);
1159            return Res::Err;
1160        }
1161
1162        match res {
1163            Res::Local(_) => {
1164                use ResolutionError::*;
1165                let mut res_err = None;
1166
1167                for rib in ribs {
1168                    match rib.kind {
1169                        RibKind::Normal
1170                        | RibKind::FnOrCoroutine
1171                        | RibKind::Module(..)
1172                        | RibKind::MacroDefinition(..)
1173                        | RibKind::ForwardGenericParamBan(_) => {
1174                            // Nothing to do. Continue.
1175                        }
1176                        RibKind::Item(..) | RibKind::AssocItem => {
1177                            // This was an attempt to access an upvar inside a
1178                            // named function item. This is not allowed, so we
1179                            // report an error.
1180                            if let Some(span) = finalize {
1181                                // We don't immediately trigger a resolve error, because
1182                                // we want certain other resolution errors (namely those
1183                                // emitted for `ConstantItemRibKind` below) to take
1184                                // precedence.
1185                                res_err = Some((span, CannotCaptureDynamicEnvironmentInFnItem));
1186                            }
1187                        }
1188                        RibKind::ConstantItem(_, item) => {
1189                            // Still doesn't deal with upvars
1190                            if let Some(span) = finalize {
1191                                let (span, resolution_error) = match item {
1192                                    None if rib_ident.name == kw::SelfLower => {
1193                                        (span, LowercaseSelf)
1194                                    }
1195                                    None => {
1196                                        // If we have a `let name = expr;`, we have the span for
1197                                        // `name` and use that to see if it is followed by a type
1198                                        // specifier. If not, then we know we need to suggest
1199                                        // `const name: Ty = expr;`. This is a heuristic, it will
1200                                        // break down in the presence of macros.
1201                                        let sm = self.tcx.sess.source_map();
1202                                        let type_span = match sm.span_look_ahead(
1203                                            original_rib_ident_def.span,
1204                                            ":",
1205                                            None,
1206                                        ) {
1207                                            None => {
1208                                                Some(original_rib_ident_def.span.shrink_to_hi())
1209                                            }
1210                                            Some(_) => None,
1211                                        };
1212                                        (
1213                                            rib_ident.span,
1214                                            AttemptToUseNonConstantValueInConstant {
1215                                                ident: original_rib_ident_def,
1216                                                suggestion: "const",
1217                                                current: "let",
1218                                                type_span,
1219                                            },
1220                                        )
1221                                    }
1222                                    Some((ident, kind)) => (
1223                                        span,
1224                                        AttemptToUseNonConstantValueInConstant {
1225                                            ident,
1226                                            suggestion: "let",
1227                                            current: kind.as_str(),
1228                                            type_span: None,
1229                                        },
1230                                    ),
1231                                };
1232                                self.report_error(span, resolution_error);
1233                            }
1234                            return Res::Err;
1235                        }
1236                        RibKind::ConstParamTy => {
1237                            if let Some(span) = finalize {
1238                                self.report_error(
1239                                    span,
1240                                    ParamInTyOfConstParam { name: rib_ident.name },
1241                                );
1242                            }
1243                            return Res::Err;
1244                        }
1245                        RibKind::InlineAsmSym => {
1246                            if let Some(span) = finalize {
1247                                self.report_error(span, InvalidAsmSym);
1248                            }
1249                            return Res::Err;
1250                        }
1251                    }
1252                }
1253                if let Some((span, res_err)) = res_err {
1254                    self.report_error(span, res_err);
1255                    return Res::Err;
1256                }
1257            }
1258            Res::Def(DefKind::TyParam, _) | Res::SelfTyParam { .. } | Res::SelfTyAlias { .. } => {
1259                for rib in ribs {
1260                    let (has_generic_params, def_kind) = match rib.kind {
1261                        RibKind::Normal
1262                        | RibKind::FnOrCoroutine
1263                        | RibKind::Module(..)
1264                        | RibKind::MacroDefinition(..)
1265                        | RibKind::InlineAsmSym
1266                        | RibKind::AssocItem
1267                        | RibKind::ForwardGenericParamBan(_) => {
1268                            // Nothing to do. Continue.
1269                            continue;
1270                        }
1271
1272                        RibKind::ConstParamTy => {
1273                            if !self.tcx.features().generic_const_parameter_types() {
1274                                if let Some(span) = finalize {
1275                                    self.report_error(
1276                                        span,
1277                                        ResolutionError::ParamInTyOfConstParam {
1278                                            name: rib_ident.name,
1279                                        },
1280                                    );
1281                                }
1282                                return Res::Err;
1283                            } else {
1284                                continue;
1285                            }
1286                        }
1287
1288                        RibKind::ConstantItem(trivial, _) => {
1289                            if let ConstantHasGenerics::No(cause) = trivial {
1290                                // HACK(min_const_generics): If we encounter `Self` in an anonymous
1291                                // constant we can't easily tell if it's generic at this stage, so
1292                                // we instead remember this and then enforce the self type to be
1293                                // concrete later on.
1294                                if let Res::SelfTyAlias {
1295                                    alias_to: def,
1296                                    forbid_generic: _,
1297                                    is_trait_impl,
1298                                } = res
1299                                {
1300                                    res = Res::SelfTyAlias {
1301                                        alias_to: def,
1302                                        forbid_generic: true,
1303                                        is_trait_impl,
1304                                    }
1305                                } else {
1306                                    if let Some(span) = finalize {
1307                                        let error = match cause {
1308                                            NoConstantGenericsReason::IsEnumDiscriminant => {
1309                                                ResolutionError::ParamInEnumDiscriminant {
1310                                                    name: rib_ident.name,
1311                                                    param_kind: ParamKindInEnumDiscriminant::Type,
1312                                                }
1313                                            }
1314                                            NoConstantGenericsReason::NonTrivialConstArg => {
1315                                                ResolutionError::ParamInNonTrivialAnonConst {
1316                                                    name: rib_ident.name,
1317                                                    param_kind:
1318                                                        ParamKindInNonTrivialAnonConst::Type,
1319                                                }
1320                                            }
1321                                        };
1322                                        let _: ErrorGuaranteed = self.report_error(span, error);
1323                                    }
1324
1325                                    return Res::Err;
1326                                }
1327                            }
1328
1329                            continue;
1330                        }
1331
1332                        // This was an attempt to use a type parameter outside its scope.
1333                        RibKind::Item(has_generic_params, def_kind) => {
1334                            (has_generic_params, def_kind)
1335                        }
1336                    };
1337
1338                    if let Some(span) = finalize {
1339                        self.report_error(
1340                            span,
1341                            ResolutionError::GenericParamsFromOuterItem(
1342                                res,
1343                                has_generic_params,
1344                                def_kind,
1345                            ),
1346                        );
1347                    }
1348                    return Res::Err;
1349                }
1350            }
1351            Res::Def(DefKind::ConstParam, _) => {
1352                for rib in ribs {
1353                    let (has_generic_params, def_kind) = match rib.kind {
1354                        RibKind::Normal
1355                        | RibKind::FnOrCoroutine
1356                        | RibKind::Module(..)
1357                        | RibKind::MacroDefinition(..)
1358                        | RibKind::InlineAsmSym
1359                        | RibKind::AssocItem
1360                        | RibKind::ForwardGenericParamBan(_) => continue,
1361
1362                        RibKind::ConstParamTy => {
1363                            if !self.tcx.features().generic_const_parameter_types() {
1364                                if let Some(span) = finalize {
1365                                    self.report_error(
1366                                        span,
1367                                        ResolutionError::ParamInTyOfConstParam {
1368                                            name: rib_ident.name,
1369                                        },
1370                                    );
1371                                }
1372                                return Res::Err;
1373                            } else {
1374                                continue;
1375                            }
1376                        }
1377
1378                        RibKind::ConstantItem(trivial, _) => {
1379                            if let ConstantHasGenerics::No(cause) = trivial {
1380                                if let Some(span) = finalize {
1381                                    let error = match cause {
1382                                        NoConstantGenericsReason::IsEnumDiscriminant => {
1383                                            ResolutionError::ParamInEnumDiscriminant {
1384                                                name: rib_ident.name,
1385                                                param_kind: ParamKindInEnumDiscriminant::Const,
1386                                            }
1387                                        }
1388                                        NoConstantGenericsReason::NonTrivialConstArg => {
1389                                            ResolutionError::ParamInNonTrivialAnonConst {
1390                                                name: rib_ident.name,
1391                                                param_kind: ParamKindInNonTrivialAnonConst::Const {
1392                                                    name: rib_ident.name,
1393                                                },
1394                                            }
1395                                        }
1396                                    };
1397                                    self.report_error(span, error);
1398                                }
1399
1400                                return Res::Err;
1401                            }
1402
1403                            continue;
1404                        }
1405
1406                        RibKind::Item(has_generic_params, def_kind) => {
1407                            (has_generic_params, def_kind)
1408                        }
1409                    };
1410
1411                    // This was an attempt to use a const parameter outside its scope.
1412                    if let Some(span) = finalize {
1413                        self.report_error(
1414                            span,
1415                            ResolutionError::GenericParamsFromOuterItem(
1416                                res,
1417                                has_generic_params,
1418                                def_kind,
1419                            ),
1420                        );
1421                    }
1422                    return Res::Err;
1423                }
1424            }
1425            _ => {}
1426        }
1427
1428        res
1429    }

1472    pub(crate) fn resolve_path_with_ribs<'r>(
1473        mut self: CmResolver<'r, 'ra, 'tcx>,
1474        path: &[Segment],
1475        opt_ns: Option<Namespace>, // `None` indicates a module path in import
1476        parent_scope: &ParentScope<'ra>,
1477        source: Option<PathSource<'_, '_, '_>>,
1478        finalize: Option<Finalize>,
1479        ribs: Option<&PerNS<Vec<Rib<'ra>>>>,
1480        ignore_binding: Option<NameBinding<'ra>>,
1481        ignore_import: Option<Import<'ra>>,
1482    ) -> PathResult<'ra> {
1483        let mut module = None;
1484        let mut module_had_parse_errors = false;
1485        let mut allow_super = true;
1486        let mut second_binding = None;
1487
1488        // We'll provide more context to the privacy errors later, up to `len`.
1489        let privacy_errors_len = self.privacy_errors.len();
1490        fn record_segment_res<'r, 'ra, 'tcx>(
1491            mut this: CmResolver<'r, 'ra, 'tcx>,
1492            finalize: Option<Finalize>,
1493            res: Res,
1494            id: Option<NodeId>,
1495        ) {
1496            if finalize.is_some()
1497                && let Some(id) = id
1498                && !this.partial_res_map.contains_key(&id)
1499            {
1500                assert!(id != ast::DUMMY_NODE_ID, "Trying to resolve dummy id");
1501                this.get_mut().record_partial_res(id, PartialRes::new(res));
1502            }
1503        }
1504
1505        for (segment_idx, &Segment { ident, id, .. }) in path.iter().enumerate() {
1506            debug!("resolve_path ident {} {:?} {:?}", segment_idx, ident, id);
1507
1508            let is_last = segment_idx + 1 == path.len();
1509            let ns = if is_last { opt_ns.unwrap_or(TypeNS) } else { TypeNS };
1510            let name = ident.name;
1511
1512            allow_super &= ns == TypeNS && (name == kw::SelfLower || name == kw::Super);
1513
1514            if ns == TypeNS {
1515                if allow_super && name == kw::Super {
1516                    let mut ctxt = ident.span.ctxt().normalize_to_macros_2_0();
1517                    let self_module = match segment_idx {
1518                        0 => Some(self.resolve_self(&mut ctxt, parent_scope.module)),
1519                        _ => match module {
1520                            Some(ModuleOrUniformRoot::Module(module)) => Some(module),
1521                            _ => None,
1522                        },
1523                    };
1524                    if let Some(self_module) = self_module
1525                        && let Some(parent) = self_module.parent
1526                    {
1527                        module =
1528                            Some(ModuleOrUniformRoot::Module(self.resolve_self(&mut ctxt, parent)));
1529                        continue;
1530                    }
1531                    return PathResult::failed(
1532                        ident,
1533                        false,
1534                        finalize.is_some(),
1535                        module_had_parse_errors,
1536                        module,
1537                        || ("there are too many leading `super` keywords".to_string(), None),
1538                    );
1539                }
1540                if segment_idx == 0 {
1541                    if name == kw::SelfLower {
1542                        let mut ctxt = ident.span.ctxt().normalize_to_macros_2_0();
1543                        let self_mod = self.resolve_self(&mut ctxt, parent_scope.module);
1544                        if let Some(res) = self_mod.res() {
1545                            record_segment_res(self.reborrow(), finalize, res, id);
1546                        }
1547                        module = Some(ModuleOrUniformRoot::Module(self_mod));
1548                        continue;
1549                    }
1550                    if name == kw::PathRoot && ident.span.at_least_rust_2018() {
1551                        module = Some(ModuleOrUniformRoot::ExternPrelude);
1552                        continue;
1553                    }
1554                    if name == kw::PathRoot
1555                        && ident.span.is_rust_2015()
1556                        && self.tcx.sess.at_least_rust_2018()
1557                    {
1558                        // `::a::b` from 2015 macro on 2018 global edition
1559                        let crate_root = self.resolve_crate_root(ident);
1560                        module = Some(ModuleOrUniformRoot::ModuleAndExternPrelude(crate_root));
1561                        continue;
1562                    }
1563                    if name == kw::PathRoot || name == kw::Crate || name == kw::DollarCrate {
1564                        // `::a::b`, `crate::a::b` or `$crate::a::b`
1565                        let crate_root = self.resolve_crate_root(ident);
1566                        if let Some(res) = crate_root.res() {
1567                            record_segment_res(self.reborrow(), finalize, res, id);
1568                        }
1569                        module = Some(ModuleOrUniformRoot::Module(crate_root));
1570                        continue;
1571                    }
1572                }
1573            }
1574
1575            // Report special messages for path segment keywords in wrong positions.
1576            if ident.is_path_segment_keyword() && segment_idx != 0 {
1577                return PathResult::failed(
1578                    ident,
1579                    false,
1580                    finalize.is_some(),
1581                    module_had_parse_errors,
1582                    module,
1583                    || {
1584                        let name_str = if name == kw::PathRoot {
1585                            "crate root".to_string()
1586                        } else {
1587                            format!("`{name}`")
1588                        };
1589                        let label = if segment_idx == 1 && path[0].ident.name == kw::PathRoot {
1590                            format!("global paths cannot start with {name_str}")
1591                        } else {
1592                            format!("{name_str} in paths can only be used in start position")
1593                        };
1594                        (label, None)
1595                    },
1596                );
1597            }
1598
1599            let binding = if let Some(module) = module {
1600                self.reborrow()
1601                    .resolve_ident_in_module(
1602                        module,
1603                        ident,
1604                        ns,
1605                        parent_scope,
1606                        finalize,
1607                        ignore_binding,
1608                        ignore_import,
1609                    )
1610                    .map_err(|(determinacy, _)| determinacy)
1611            } else if let Some(ribs) = ribs
1612                && let Some(TypeNS | ValueNS) = opt_ns
1613            {
1614                assert!(ignore_import.is_none());
1615                match self.get_mut().resolve_ident_in_lexical_scope(
1616                    ident,
1617                    ns,
1618                    parent_scope,
1619                    finalize,
1620                    &ribs[ns],
1621                    ignore_binding,
1622                ) {
1623                    // we found a locally-imported or available item/module
1624                    Some(LexicalScopeBinding::Item(binding)) => Ok(binding),
1625                    // we found a local variable or type param
1626                    Some(LexicalScopeBinding::Res(res)) => {
1627                        record_segment_res(self.reborrow(), finalize, res, id);
1628                        return PathResult::NonModule(PartialRes::with_unresolved_segments(
1629                            res,
1630                            path.len() - 1,
1631                        ));
1632                    }
1633                    _ => Err(Determinacy::determined(finalize.is_some())),
1634                }
1635            } else {
1636                self.reborrow().early_resolve_ident_in_lexical_scope(
1637                    ident,
1638                    ScopeSet::All(ns),
1639                    parent_scope,
1640                    finalize,
1641                    finalize.is_some(),
1642                    ignore_binding,
1643                    ignore_import,
1644                )
1645            };
1646
1647            match binding {
1648                Ok(binding) => {
1649                    if segment_idx == 1 {
1650                        second_binding = Some(binding);
1651                    }
1652                    let res = binding.res();
1653
1654                    // Mark every privacy error in this path with the res to the last element. This allows us
1655                    // to detect the item the user cares about and either find an alternative import, or tell
1656                    // the user it is not accessible.
1657                    if finalize.is_some() {
1658                        for error in &mut self.get_mut().privacy_errors[privacy_errors_len..] {
1659                            error.outermost_res = Some((res, ident));
1660                            error.source = match source {
1661                                Some(PathSource::Struct(Some(expr)))
1662                                | Some(PathSource::Expr(Some(expr))) => Some(expr.clone()),
1663                                _ => None,
1664                            };
1665                        }
1666                    }
1667
1668                    let maybe_assoc = opt_ns != Some(MacroNS) && PathSource::Type.is_expected(res);
1669                    if let Some(def_id) = binding.res().module_like_def_id() {
1670                        if self.mods_with_parse_errors.contains(&def_id) {
1671                            module_had_parse_errors = true;
1672                        }
1673                        module = Some(ModuleOrUniformRoot::Module(self.expect_module(def_id)));
1674                        record_segment_res(self.reborrow(), finalize, res, id);
1675                    } else if res == Res::ToolMod && !is_last && opt_ns.is_some() {
1676                        if binding.is_import() {
1677                            self.dcx().emit_err(errors::ToolModuleImported {
1678                                span: ident.span,
1679                                import: binding.span,
1680                            });
1681                        }
1682                        let res = Res::NonMacroAttr(NonMacroAttrKind::Tool);
1683                        return PathResult::NonModule(PartialRes::new(res));
1684                    } else if res == Res::Err {
1685                        return PathResult::NonModule(PartialRes::new(Res::Err));
1686                    } else if opt_ns.is_some() && (is_last || maybe_assoc) {
1687                        if let Some(finalize) = finalize {
1688                            self.get_mut().lint_if_path_starts_with_module(
1689                                finalize,
1690                                path,
1691                                second_binding,
1692                            );
1693                        }
1694                        record_segment_res(self.reborrow(), finalize, res, id);
1695                        return PathResult::NonModule(PartialRes::with_unresolved_segments(
1696                            res,
1697                            path.len() - segment_idx - 1,
1698                        ));
1699                    } else {
1700                        return PathResult::failed(
1701                            ident,
1702                            is_last,
1703                            finalize.is_some(),
1704                            module_had_parse_errors,
1705                            module,
1706                            || {
1707                                let label = format!(
1708                                    "`{ident}` is {} {}, not a module",
1709                                    res.article(),
1710                                    res.descr()
1711                                );
1712                                (label, None)
1713                            },
1714                        );
1715                    }
1716                }
1717                Err(Undetermined) => return PathResult::Indeterminate,
1718                Err(Determined) => {
1719                    if let Some(ModuleOrUniformRoot::Module(module)) = module
1720                        && opt_ns.is_some()
1721                        && !module.is_normal()
1722                    {
1723                        return PathResult::NonModule(PartialRes::with_unresolved_segments(
1724                            module.res().unwrap(),
1725                            path.len() - segment_idx,
1726                        ));
1727                    }
1728
1729                    let mut this = self.reborrow();
1730                    return PathResult::failed(
1731                        ident,
1732                        is_last,
1733                        finalize.is_some(),
1734                        module_had_parse_errors,
1735                        module,
1736                        || {
1737                            this.get_mut().report_path_resolution_error(
1738                                path,
1739                                opt_ns,
1740                                parent_scope,
1741                                ribs,
1742                                ignore_binding,
1743                                ignore_import,
1744                                module,
1745                                segment_idx,
1746                                ident,
1747                            )
1748                        },
1749                    );
1750                }
1751            }
1752        }
1753
1754        if let Some(finalize) = finalize {
1755            self.get_mut().lint_if_path_starts_with_module(finalize, path, second_binding);
1756        }
1757
1758        PathResult::Module(match module {
1759            Some(module) => module,
1760            None if path.is_empty() => ModuleOrUniformRoot::CurrentScope,
1761            _ => bug!("resolve_path: non-empty path `{:?}` has no module", path),
1762        })
1763    }
