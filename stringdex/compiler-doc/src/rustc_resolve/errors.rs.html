1219        let mut multispan: MultiSpan = self.span.into();
1220        match self.item_was {

1222                let key = "feature".into();

1224                let msg = diag.dcx.eagerly_translate_to_string(
1225                    fluent::resolve_item_was_behind_feature,
1226                    [(&key, &value)].into_iter(),
1227                );
1228                multispan.push_span_label(span, msg);
1229            }
1230            ItemWas::CfgOut { span } => {
1231                multispan.push_span_label(span, fluent::resolve_item_was_cfg_out);
1232            }
1233        }
1234        diag.span_note(multispan, fluent::resolve_found_an_item_configured_out);
