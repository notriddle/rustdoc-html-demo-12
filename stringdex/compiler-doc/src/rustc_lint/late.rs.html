24/// Extract the [`LintStore`] from [`Session`].
25///
26/// This function exists because [`Session::lint_store`] is type-erased.
27pub fn unerased_lint_store(sess: &Session) -> &LintStore {
28    let store: &dyn Any = sess.lint_store.as_deref().unwrap();
29    store.downcast_ref().unwrap()
30}

32macro_rules! lint_callback { ($cx:expr, $f:ident, $($args:expr),*) => ({
33    $cx.pass.$f(&$cx.context, $($args),*);
34}) }

36/// Implements the AST traversal for late lint passes. `T` provides the
37/// `check_*` methods.
38struct LateContextAndPass<'tcx, T: LateLintPass<'tcx>> {
39    context: LateContext<'tcx>,
40    pass: T,
41}
42
43impl<'tcx, T: LateLintPass<'tcx>> LateContextAndPass<'tcx, T> {
44    /// Merge the lints specified by any lint attributes into the
45    /// current lint context, call the provided function, then reset the
46    /// lints in effect to their previous state.
47    fn with_lint_attrs<F>(&mut self, id: HirId, f: F)
48    where
49        F: FnOnce(&mut Self),
50    {
51        let attrs = self.context.tcx.hir_attrs(id);
52        let prev = self.context.last_node_with_lint_attrs;
53        self.context.last_node_with_lint_attrs = id;
54        debug!("late context: enter_attrs({:?})", attrs);
55        lint_callback!(self, check_attributes, attrs);
56        for attr in attrs {
57            lint_callback!(self, check_attribute, attr);
58        }
59        f(self);
60        debug!("late context: exit_attrs({:?})", attrs);
61        lint_callback!(self, check_attributes_post, attrs);
62        self.context.last_node_with_lint_attrs = prev;
63    }
64
65    fn with_param_env<F>(&mut self, id: hir::OwnerId, f: F)
66    where
67        F: FnOnce(&mut Self),
68    {
69        let old_param_env = self.context.param_env;
70        self.context.param_env = self.context.tcx.param_env(id);
71        f(self);
72        self.context.param_env = old_param_env;
73    }
74
75    fn process_mod(&mut self, m: &'tcx hir::Mod<'tcx>, n: HirId) {
76        lint_callback!(self, check_mod, m, n);
77        hir_visit::walk_mod(self, m);
78    }
79}

81impl<'tcx, T: LateLintPass<'tcx>> hir_visit::Visitor<'tcx> for LateContextAndPass<'tcx, T> {
82    type NestedFilter = nested_filter::All;

84    /// Because lints are scoped lexically, we want to walk nested
85    /// items in the context of the outer item, so enable
86    /// deep-walking.
87    fn maybe_tcx(&mut self) -> Self::MaybeTyCtxt {
88        self.context.tcx
89    }
90
91    fn visit_nested_body(&mut self, body_id: hir::BodyId) {
92        let old_enclosing_body = self.context.enclosing_body.replace(body_id);
93        let old_cached_typeck_results = self.context.cached_typeck_results.get();
94
95        // HACK(eddyb) avoid trashing `cached_typeck_results` when we're
96        // nested in `visit_fn`, which may have already resulted in them
97        // being queried.
98        if old_enclosing_body != Some(body_id) {
99            self.context.cached_typeck_results.set(None);
100        }
101
102        let body = self.context.tcx.hir_body(body_id);
103        self.visit_body(body);
104        self.context.enclosing_body = old_enclosing_body;
105
106        // See HACK comment above.
107        if old_enclosing_body != Some(body_id) {
108            self.context.cached_typeck_results.set(old_cached_typeck_results);
109        }
110    }
111
112    fn visit_param(&mut self, param: &'tcx hir::Param<'tcx>) {
113        self.with_lint_attrs(param.hir_id, |cx| {
114            hir_visit::walk_param(cx, param);
115        });
116    }
117
118    fn visit_body(&mut self, body: &hir::Body<'tcx>) {
119        lint_callback!(self, check_body, body);
120        hir_visit::walk_body(self, body);
121        lint_callback!(self, check_body_post, body);
122    }
123
124    fn visit_item(&mut self, it: &'tcx hir::Item<'tcx>) {
125        let generics = self.context.generics.take();
126        self.context.generics = it.kind.generics();
127        let old_cached_typeck_results = self.context.cached_typeck_results.take();
128        let old_enclosing_body = self.context.enclosing_body.take();
129        self.with_lint_attrs(it.hir_id(), |cx| {
130            cx.with_param_env(it.owner_id, |cx| {
131                lint_callback!(cx, check_item, it);
132                hir_visit::walk_item(cx, it);
133                lint_callback!(cx, check_item_post, it);
134            });
135        });
136        self.context.enclosing_body = old_enclosing_body;
137        self.context.cached_typeck_results.set(old_cached_typeck_results);
138        self.context.generics = generics;
139    }
140
141    fn visit_foreign_item(&mut self, it: &'tcx hir::ForeignItem<'tcx>) {
142        self.with_lint_attrs(it.hir_id(), |cx| {
143            cx.with_param_env(it.owner_id, |cx| {
144                lint_callback!(cx, check_foreign_item, it);
145                hir_visit::walk_foreign_item(cx, it);

147        })

157    }

159    fn visit_expr_field(&mut self, field: &'tcx hir::ExprField<'tcx>) {
160        self.with_lint_attrs(field.hir_id, |cx| hir_visit::walk_expr_field(cx, field))
161    }

163    fn visit_expr(&mut self, e: &'tcx hir::Expr<'tcx>) {
164        ensure_sufficient_stack(|| {
165            self.with_lint_attrs(e.hir_id, |cx| {
166                lint_callback!(cx, check_expr, e);
167                hir_visit::walk_expr(cx, e);
168                lint_callback!(cx, check_expr_post, e);
169            })
170        })
171    }
172
173    fn visit_stmt(&mut self, s: &'tcx hir::Stmt<'tcx>) {
174        // See `EarlyContextAndPass::visit_stmt` for an explanation
175        // of why we call `walk_stmt` outside of `with_lint_attrs`
176        self.with_lint_attrs(s.hir_id, |cx| {
177            lint_callback!(cx, check_stmt, s);
178        });
179        hir_visit::walk_stmt(self, s);
180    }
181
182    fn visit_fn(
183        &mut self,
184        fk: hir_visit::FnKind<'tcx>,
185        decl: &'tcx hir::FnDecl<'tcx>,
186        body_id: hir::BodyId,
187        span: Span,
188        id: LocalDefId,
189    ) {
190        // Wrap in typeck results here, not just in visit_nested_body,
191        // in order for `check_fn` to be able to use them.
192        let old_enclosing_body = self.context.enclosing_body.replace(body_id);
193        let old_cached_typeck_results = self.context.cached_typeck_results.take();
194        let body = self.context.tcx.hir_body(body_id);
195        lint_callback!(self, check_fn, fk, decl, body, span, id);
196        hir_visit::walk_fn(self, fk, decl, body_id, id);
197        self.context.enclosing_body = old_enclosing_body;
198        self.context.cached_typeck_results.set(old_cached_typeck_results);
199    }

305// Combines multiple lint passes into a single pass, at runtime. Each
306// `check_foo` method in `$methods` within this pass simply calls `check_foo`
307// once per `$pass`. Compare with `declare_combined_late_lint_pass`, which is
308// similar, but combines lint passes at compile time.
309struct RuntimeCombinedLateLintPass<'a, 'tcx> {
310    passes: &'a mut [LateLintPassObject<'tcx>],
311}
312
313#[allow(rustc::lint_pass_impl_without_macro)]
314impl LintPass for RuntimeCombinedLateLintPass<'_, '_> {
315    fn name(&self) -> &'static str {
316        panic!()
317    }
318    fn get_lints(&self) -> crate::LintVec {
319        panic!()
320    }
321}
322
323macro_rules! impl_late_lint_pass {
324    ([], [$($(#[$attr:meta])* fn $f:ident($($param:ident: $arg:ty),*);)*]) => {
325        impl<'tcx> LateLintPass<'tcx> for RuntimeCombinedLateLintPass<'_, 'tcx> {
326            $(fn $f(&mut self, context: &LateContext<'tcx>, $($param: $arg),*) {
327                for pass in self.passes.iter_mut() {
328                    pass.$f(context, $($param),*);
329                }
330            })*
331        }
332    };
333}

337pub fn late_lint_mod<'tcx, T: LateLintPass<'tcx> + 'tcx>(
338    tcx: TyCtxt<'tcx>,
339    module_def_id: LocalModDefId,
340    builtin_lints: T,
341) {
342    let context = LateContext {
343        tcx,
344        enclosing_body: None,
345        cached_typeck_results: Cell::new(None),
346        param_env: ty::ParamEnv::empty(),
347        effective_visibilities: tcx.effective_visibilities(()),
348        last_node_with_lint_attrs: tcx.local_def_id_to_hir_id(module_def_id),
349        generics: None,
350        only_module: true,
351    };
352
353    // Note: `passes` is often empty. In that case, it's faster to run
354    // `builtin_lints` directly rather than bundling it up into the
355    // `RuntimeCombinedLateLintPass`.
356    let store = unerased_lint_store(tcx.sess);
357
358    if store.late_module_passes.is_empty() {
359        // If all builtin lints can be skipped, there is no point in running `late_lint_mod_inner`
360        // at all. This happens often for dependencies built with `--cap-lints=allow`.
361        let dont_need_to_run = tcx.lints_that_dont_need_to_run(());
362        let can_skip_lints = builtin_lints
363            .get_lints()
364            .iter()
365            .all(|lint| dont_need_to_run.contains(&LintId::of(lint)));
366        if !can_skip_lints {
367            late_lint_mod_inner(tcx, module_def_id, context, builtin_lints);
368        }
369    } else {
370        let builtin_lints = Box::new(builtin_lints) as Box<dyn LateLintPass<'tcx>>;
371        let mut binding = store
372            .late_module_passes
373            .iter()
374            .map(|mk_pass| (mk_pass)(tcx))
375            .chain(std::iter::once(builtin_lints))
376            .collect::<Vec<_>>();
377
378        let pass = RuntimeCombinedLateLintPass { passes: binding.as_mut_slice() };
379        late_lint_mod_inner(tcx, module_def_id, context, pass);
380    }
381}

383fn late_lint_mod_inner<'tcx, T: LateLintPass<'tcx>>(
384    tcx: TyCtxt<'tcx>,
385    module_def_id: LocalModDefId,
386    context: LateContext<'tcx>,
387    pass: T,
388) {
389    let mut cx = LateContextAndPass { context, pass };
390
391    let (module, _span, hir_id) = tcx.hir_get_module(module_def_id);
392
393    cx.with_lint_attrs(hir_id, |cx| {
394        // There is no module lint that will have the crate itself as an item, so check it here.
395        if hir_id == hir::CRATE_HIR_ID {
396            lint_callback!(cx, check_crate,);
397        }
398
399        cx.process_mod(module, hir_id);
400
401        if hir_id == hir::CRATE_HIR_ID {
402            lint_callback!(cx, check_crate_post,);
403        }
404    });
405}

407fn late_lint_crate<'tcx>(tcx: TyCtxt<'tcx>) {
408    // Note: `passes` is often empty.
409    let passes: Vec<_> =
410        unerased_lint_store(tcx.sess).late_passes.iter().map(|mk_pass| (mk_pass)(tcx)).collect();
411
412    if passes.is_empty() {
413        return;
414    }
415
416    let context = LateContext {
417        tcx,
418        enclosing_body: None,
419        cached_typeck_results: Cell::new(None),
420        param_env: ty::ParamEnv::empty(),
421        effective_visibilities: tcx.effective_visibilities(()),
422        last_node_with_lint_attrs: hir::CRATE_HIR_ID,
423        generics: None,
424        only_module: false,
425    };
426
427    let lints_that_dont_need_to_run = tcx.lints_that_dont_need_to_run(());
428
429    let mut filtered_passes: Vec<Box<dyn LateLintPass<'tcx>>> = passes
430        .into_iter()
431        .filter(|pass| {
432            let lints = (**pass).get_lints();
433            // Lintless passes are always in
434            lints.is_empty() ||
435            // If the pass doesn't have a single needed lint, omit it
436            !lints.iter().all(|lint| lints_that_dont_need_to_run.contains(&LintId::of(lint)))
437        })
438        .collect();
439
440    filtered_passes.push(Box::new(HardwiredLints));
441    let pass = RuntimeCombinedLateLintPass { passes: &mut filtered_passes[..] };
442    late_lint_crate_inner(tcx, context, pass);
443}
444
445fn late_lint_crate_inner<'tcx, T: LateLintPass<'tcx>>(
446    tcx: TyCtxt<'tcx>,
447    context: LateContext<'tcx>,
448    pass: T,
449) {
450    let mut cx = LateContextAndPass { context, pass };
451
452    // Visit the whole crate.
453    cx.with_lint_attrs(hir::CRATE_HIR_ID, |cx| {
454        // Since the root module isn't visited as an item (because it isn't an
455        // item), warn for it here.
456        lint_callback!(cx, check_crate,);
457        tcx.hir_walk_toplevel_module(cx);
458        lint_callback!(cx, check_crate_post,);
459    })
460}

462/// Performs lint checking on a crate.
463pub fn check_crate<'tcx>(tcx: TyCtxt<'tcx>) {
464    join(
465        || {
466            tcx.sess.time("crate_lints", || {
467                // Run whole crate non-incremental lints
468                late_lint_crate(tcx);
469            });
470        },
471        || {
472            tcx.sess.time("module_lints", || {
473                // Run per-module lints
474                tcx.par_hir_for_each_module(|module| tcx.ensure_ok().lint_mod(module));
475            });
476        },
477    );
