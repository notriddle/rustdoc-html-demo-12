147/// Test if it is valid for a MIR assignment to assign `src`-typed place to `dest`-typed value.
148/// This test should be symmetric, as it is primarily about layout compatibility.
149pub(super) fn mir_assign_valid_types<'tcx>(
150    tcx: TyCtxt<'tcx>,
151    typing_env: TypingEnv<'tcx>,
152    src: TyAndLayout<'tcx>,
153    dest: TyAndLayout<'tcx>,
154) -> bool {
155    // Type-changing assignments can happen when subtyping is used. While
156    // all normal lifetimes are erased, higher-ranked types with their
157    // late-bound lifetimes are still around and can lead to type
158    // differences.
159    if util::relate_types(tcx, typing_env, Variance::Covariant, src.ty, dest.ty) {
160        // Make sure the layout is equal, too -- just to be safe. Miri really
161        // needs layout equality. For performance reason we skip this check when
162        // the types are equal. Equal types *can* have different layouts when
163        // enum downcast is involved (as enum variants carry the type of the
164        // enum), but those should never occur in assignments.
165        if cfg!(debug_assertions) || src.ty != dest.ty {
166            assert_eq!(src.layout, dest.layout);
167        }
168        true
169    } else {
170        false
171    }
172}

174/// Use the already known layout if given (but sanity check in debug mode),
175/// or compute the layout.
176#[cfg_attr(not(debug_assertions), inline(always))]
177pub(super) fn from_known_layout<'tcx>(
178    tcx: TyCtxtAt<'tcx>,
179    typing_env: TypingEnv<'tcx>,
180    known_layout: Option<TyAndLayout<'tcx>>,
181    compute: impl FnOnce() -> InterpResult<'tcx, TyAndLayout<'tcx>>,
182) -> InterpResult<'tcx, TyAndLayout<'tcx>> {
183    match known_layout {
184        None => compute(),
185        Some(known_layout) => {
186            if cfg!(debug_assertions) {
187                let check_layout = compute()?;
188                if !mir_assign_valid_types(tcx.tcx, typing_env, check_layout, known_layout) {
189                    span_bug!(
190                        tcx.span,
191                        "expected type differs from actual type.\nexpected: {}\nactual: {}",
192                        known_layout.ty,
193                        check_layout.ty,
194                    );
195                }
196            }
197            interp_ok(known_layout)
198        }
199    }
200}

202/// Turn the given error into a human-readable string. Expects the string to be printed, so if
203/// `RUSTC_CTFE_BACKTRACE` is set this will show a backtrace of the rustc internals that
204/// triggered the error.
205///
206/// This is NOT the preferred way to render an error; use `report` from `const_eval` instead.
207/// However, this is useful when error messages appear in ICEs.
208pub fn format_interp_error<'tcx>(dcx: DiagCtxtHandle<'_>, e: InterpErrorInfo<'tcx>) -> String {
209    let (e, backtrace) = e.into_parts();
210    backtrace.print_backtrace();
211    // FIXME(fee1-dead), HACK: we want to use the error as title therefore we can just extract the
212    // label and arguments from the InterpError.
213    #[allow(rustc::untranslatable_diagnostic)]
214    let mut diag = dcx.struct_allow("");
215    let msg = e.diagnostic_message();
216    e.add_args(&mut diag);
217    let s = dcx.eagerly_translate_to_string(msg, diag.args.iter());
218    diag.cancel();
219    s
220}

362    /// Walks up the callstack from the intrinsic's callsite, searching for the first callsite in a
363    /// frame which is not `#[track_caller]`. This matches the `caller_location` intrinsic,
364    /// and is primarily intended for the panic machinery.
365    pub(crate) fn find_closest_untracked_caller_location(&self) -> Span {
366        for frame in self.stack().iter().rev() {
367            debug!("find_closest_untracked_caller_location: checking frame {:?}", frame.instance);
368
369            // Assert that the frame we look at is actually executing code currently
370            // (`loc` is `Right` when we are unwinding and the frame does not require cleanup).
371            let loc = frame.loc.left().unwrap();
372
373            // This could be a non-`Call` terminator (such as `Drop`), or not a terminator at all
374            // (such as `box`). Use the normal span by default.
375            let mut source_info = *frame.body.source_info(loc);
376
377            // If this is a `Call` terminator, use the `fn_span` instead.
378            let block = &frame.body.basic_blocks[loc.block];
379            if loc.statement_index == block.statements.len() {
380                debug!(
381                    "find_closest_untracked_caller_location: got terminator {:?} ({:?})",
382                    block.terminator(),
383                    block.terminator().kind,
384                );
385                if let mir::TerminatorKind::Call { fn_span, .. } = block.terminator().kind {
386                    source_info.span = fn_span;
387                }
388            }
389
390            let caller_location = if frame.instance.def.requires_caller_location(*self.tcx) {
391                // We use `Err(())` as indication that we should continue up the call stack since
392                // this is a `#[track_caller]` function.
393                Some(Err(()))
394            } else {
395                None
396            };
397            if let Ok(span) =
398                frame.body.caller_location_span(source_info, caller_location, *self.tcx, Ok)
399            {
400                return span;
401            }
402        }
403
404        span_bug!(self.cur_span(), "no non-`#[track_caller]` frame found")
405    }

407    /// Returns the actual dynamic size and alignment of the place at the given type.
408    /// Only the "meta" (metadata) part of the place matters.
409    /// This can fail to provide an answer for extern types.
410    pub(super) fn size_and_align_from_meta(
411        &self,
412        metadata: &MemPlaceMeta<M::Provenance>,
413        layout: &TyAndLayout<'tcx>,
414    ) -> InterpResult<'tcx, Option<(Size, Align)>> {
415        if layout.is_sized() {
416            return interp_ok(Some((layout.size, layout.align.abi)));
417        }
418        match layout.ty.kind() {
419            ty::Adt(..) | ty::Tuple(..) => {
420                // First get the size of all statically known fields.
421                // Don't use type_of::sizing_type_of because that expects t to be sized,
422                // and it also rounds up to alignment, which we want to avoid,
423                // as the unsized field's alignment could be smaller.
424                assert!(!layout.ty.is_simd());
425                assert!(layout.fields.count() > 0);
426                trace!("DST layout: {:?}", layout);
427
428                let unsized_offset_unadjusted = layout.fields.offset(layout.fields.count() - 1);
429                let sized_align = layout.align.abi;
430
431                // Recurse to get the size of the dynamically sized field (must be
432                // the last field). Can't have foreign types here, how would we
433                // adjust alignment and size for them?
434                let field = layout.field(self, layout.fields.count() - 1);
435                let Some((unsized_size, mut unsized_align)) =
436                    self.size_and_align_from_meta(metadata, &field)?
437                else {
438                    // A field with an extern type. We don't know the actual dynamic size
439                    // or the alignment.
440                    return interp_ok(None);
441                };
442
443                // # First compute the dynamic alignment
444
445                // Packed type alignment needs to be capped.
446                if let ty::Adt(def, _) = layout.ty.kind()
447                    && let Some(packed) = def.repr().pack
448                {
449                    unsized_align = unsized_align.min(packed);
450                }
451
452                // Choose max of two known alignments (combined value must
453                // be aligned according to more restrictive of the two).
454                let full_align = sized_align.max(unsized_align);
455
456                // # Then compute the dynamic size
457
458                let unsized_offset_adjusted = unsized_offset_unadjusted.align_to(unsized_align);
459                let full_size = (unsized_offset_adjusted + unsized_size).align_to(full_align);
460
461                // Just for our sanitiy's sake, assert that this is equal to what codegen would compute.
462                assert_eq!(
463                    full_size,
464                    (unsized_offset_unadjusted + unsized_size).align_to(full_align)
465                );
466
467                // Check if this brought us over the size limit.
468                if full_size > self.max_size_of_val() {
469                    throw_ub!(InvalidMeta(InvalidMetaKind::TooBig));
470                }
471                interp_ok(Some((full_size, full_align)))
472            }
473            ty::Dynamic(expected_trait, _, ty::Dyn) => {
474                let vtable = metadata.unwrap_meta().to_pointer(self)?;
475                // Read size and align from vtable (already checks size).
476                interp_ok(Some(self.get_vtable_size_and_align(vtable, Some(expected_trait))?))
477            }
478
479            ty::Slice(_) | ty::Str => {
480                let len = metadata.unwrap_meta().to_target_usize(self)?;
481                let elem = layout.field(self, 0);
482
483                // Make sure the slice is not too big.
484                let size = elem.size.bytes().saturating_mul(len); // we rely on `max_size_of_val` being smaller than `u64::MAX`.
485                let size = Size::from_bytes(size);
486                if size > self.max_size_of_val() {
487                    throw_ub!(InvalidMeta(InvalidMetaKind::SliceTooBig));
488                }
489                interp_ok(Some((size, elem.align.abi)))
490            }
491
492            ty::Foreign(_) => interp_ok(None),
493
494            _ => span_bug!(self.cur_span(), "size_and_align_of::<{}> not supported", layout.ty),
495        }
496    }

511    /// *Return* to the given `target` basic block.
512    /// Do *not* use for unwinding! Use `unwind_to_block` instead.
513    ///
514    /// If `target` is `None`, that indicates the function cannot return, so we raise UB.
515    pub fn return_to_block(&mut self, target: Option<mir::BasicBlock>) -> InterpResult<'tcx> {
516        if let Some(target) = target {
517            self.go_to_block(target);
518            interp_ok(())
519        } else {
520            throw_ub!(Unreachable)
521        }
522    }

524    /// *Unwind* to the given `target` basic block.
525    /// Do *not* use for returning! Use `return_to_block` instead.
526    ///
527    /// If `target` is `UnwindAction::Continue`, that indicates the function does not need cleanup
528    /// during unwinding, and we will just keep propagating that upwards.
529    ///
530    /// If `target` is `UnwindAction::Unreachable`, that indicates the function does not allow
531    /// unwinding, and doing so is UB.
532    #[cold] // usually we have normal returns, not unwinding
533    pub fn unwind_to_block(&mut self, target: mir::UnwindAction) -> InterpResult<'tcx> {
534        self.frame_mut().loc = match target {
535            mir::UnwindAction::Cleanup(block) => Left(mir::Location { block, statement_index: 0 }),
536            mir::UnwindAction::Continue => Right(self.frame_mut().body.span),
537            mir::UnwindAction::Unreachable => {
538                throw_ub_custom!(fluent::const_eval_unreachable_unwind);
539            }
540            mir::UnwindAction::Terminate(reason) => {
541                self.frame_mut().loc = Right(self.frame_mut().body.span);
542                M::unwind_terminate(self, reason)?;
543                // This might have pushed a new stack frame, or it terminated execution.
544                // Either way, `loc` will not be updated.
545                return interp_ok(());
546            }
547        };
548        interp_ok(())
549    }

551    /// Call a query that can return `ErrorHandled`. Should be used for statics and other globals.
552    /// (`mir::Const`/`ty::Const` have `eval` methods that can be used directly instead.)
553    pub fn ctfe_query<T>(
554        &self,
555        query: impl FnOnce(TyCtxtAt<'tcx>) -> Result<T, ErrorHandled>,
556    ) -> Result<T, ErrorHandled> {
557        // Use a precise span for better cycle errors.
558        query(self.tcx.at(self.cur_span())).map_err(|err| {
559            err.emit_note(*self.tcx);
560            err
561        })
562    }

565        &self,
566        instance: ty::Instance<'tcx>,
567    ) -> InterpResult<'tcx, MPlaceTy<'tcx, M::Provenance>> {
568        let gid = GlobalId { instance, promoted: None };
569        let val = if self.tcx.is_static(gid.instance.def_id()) {
570            let alloc_id = self.tcx.reserve_and_set_static_alloc(gid.instance.def_id());
571
572            let ty = instance.ty(self.tcx.tcx, self.typing_env);
573            mir::ConstAlloc { alloc_id, ty }
574        } else {
575            self.ctfe_query(|tcx| tcx.eval_to_allocation_raw(self.typing_env.as_query_input(gid)))?
576        };
577        self.raw_const_to_mplace(val)
578    }
579
580    pub fn eval_mir_constant(
581        &self,
582        val: &mir::Const<'tcx>,
583        span: Span,
584        layout: Option<TyAndLayout<'tcx>>,
585    ) -> InterpResult<'tcx, OpTy<'tcx, M::Provenance>> {
586        let const_val = val.eval(*self.tcx, self.typing_env, span).map_err(|err| {
587                if M::ALL_CONSTS_ARE_PRECHECKED {
588                    match err {
589                        ErrorHandled::TooGeneric(..) => {},
590                        ErrorHandled::Reported(reported, span) => {
591                            if reported.is_allowed_in_infallible() {
592                                // These errors can just sometimes happen, even when the expression
593                                // is nominally "infallible", e.g. when running out of memory
594                                // or when some layout could not be computed.
595                            } else {
596                                // Looks like the const is not captured by `required_consts`, that's bad.
597                                span_bug!(span, "interpret const eval failure of {val:?} which is not in required_consts");
598                            }
599                        }
600                    }
601                }
602                err.emit_note(*self.tcx);
603                err
604            })?;
605        self.const_val_to_op(const_val, val.ty(), layout)
606    }

619    where
620        F1: rustc_apfloat::Float + rustc_apfloat::FloatConvert<F2>,
621        F2: rustc_apfloat::Float,

627#[doc(hidden)]
628/// Helper struct for the `dump_place` function.
629pub struct PlacePrinter<'a, 'tcx, M: Machine<'tcx>> {
630    ecx: &'a InterpCx<'tcx, M>,
631    place: Place<M::Provenance>,
632}
633
634impl<'a, 'tcx, M: Machine<'tcx>> std::fmt::Debug for PlacePrinter<'a, 'tcx, M> {
635    fn fmt(&self, fmt: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
636        match self.place {
637            Place::Local { local, offset, locals_addr } => {
638                debug_assert_eq!(locals_addr, self.ecx.frame().locals_addr());
639                let mut allocs = Vec::new();
640                write!(fmt, "{local:?}")?;
641                if let Some(offset) = offset {
642                    write!(fmt, "+{:#x}", offset.bytes())?;
643                }
644                write!(fmt, ":")?;
645
646                self.ecx.frame().locals[local].print(&mut allocs, fmt)?;
647
648                write!(fmt, ": {:?}", self.ecx.dump_allocs(allocs.into_iter().flatten().collect()))
649            }
650            Place::Ptr(mplace) => match mplace.ptr.provenance.and_then(Provenance::get_alloc_id) {
651                Some(alloc_id) => {
652                    write!(fmt, "by ref {:?}: {:?}", mplace.ptr, self.ecx.dump_alloc(alloc_id))
653                }
654                ptr => write!(fmt, " integral by ref: {ptr:?}"),
655            },
656        }
657    }
658}
