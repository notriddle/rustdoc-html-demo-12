561/// Dependency checks.
562///
563/// `root` is path to the directory with the root `Cargo.toml` (for the workspace). `cargo` is path
564/// to the cargo executable.
565pub fn check(root: &Path, cargo: &Path, bless: bool, bad: &mut bool) {
566    let mut checked_runtime_licenses = false;
567
568    check_proc_macro_dep_list(root, cargo, bless, bad);
569
570    for &(workspace, exceptions, permitted_deps, submodules) in WORKSPACES {
571        if has_missing_submodule(root, submodules) {
572            continue;
573        }
574
575        if !root.join(workspace).join("Cargo.lock").exists() {
576            tidy_error!(bad, "the `{workspace}` workspace doesn't have a Cargo.lock");
577            continue;
578        }
579
580        let mut cmd = cargo_metadata::MetadataCommand::new();
581        cmd.cargo_path(cargo)
582            .manifest_path(root.join(workspace).join("Cargo.toml"))
583            .features(cargo_metadata::CargoOpt::AllFeatures)
584            .other_options(vec!["--locked".to_owned()]);
585        let metadata = t!(cmd.exec());
586
587        check_license_exceptions(&metadata, workspace, exceptions, bad);
588        if let Some((crates, permitted_deps)) = permitted_deps {
589            check_permitted_dependencies(&metadata, workspace, permitted_deps, crates, bad);
590        }
591
592        if workspace == "library" {
593            check_runtime_license_exceptions(&metadata, bad);
594            check_runtime_no_duplicate_dependencies(&metadata, bad);
595            check_runtime_no_proc_macros(&metadata, bad);
596            checked_runtime_licenses = true;
597        }
598    }
599
600    // Sanity check to ensure we don't accidentally remove the workspace containing the runtime
601    // crates.
602    assert!(checked_runtime_licenses);
603}

605/// Ensure the list of proc-macro crate transitive dependencies is up to date
606fn check_proc_macro_dep_list(root: &Path, cargo: &Path, bless: bool, bad: &mut bool) {
607    let mut cmd = cargo_metadata::MetadataCommand::new();
608    cmd.cargo_path(cargo)
609        .manifest_path(root.join("Cargo.toml"))
610        .features(cargo_metadata::CargoOpt::AllFeatures)
611        .other_options(vec!["--locked".to_owned()]);
612    let metadata = t!(cmd.exec());
613    let is_proc_macro_pkg = |pkg: &Package| pkg.targets.iter().any(|target| target.is_proc_macro());
614
615    let mut proc_macro_deps = HashSet::new();
616    for pkg in metadata.packages.iter().filter(|pkg| is_proc_macro_pkg(pkg)) {
617        deps_of(&metadata, &pkg.id, &mut proc_macro_deps);
618    }
619    // Remove the proc-macro crates themselves
620    proc_macro_deps.retain(|pkg| !is_proc_macro_pkg(&metadata[pkg]));
621
622    let proc_macro_deps: HashSet<_> =
623        proc_macro_deps.into_iter().map(|dep| metadata[dep].name.as_ref()).collect();
624    let expected = proc_macro_deps::CRATES.iter().copied().collect::<HashSet<_>>();
625
626    let needs_blessing = proc_macro_deps.difference(&expected).next().is_some()
627        || expected.difference(&proc_macro_deps).next().is_some();
628
629    if needs_blessing && bless {
630        let mut proc_macro_deps: Vec<_> = proc_macro_deps.into_iter().collect();
631        proc_macro_deps.sort();
632        let mut file = File::create(root.join("src/bootstrap/src/utils/proc_macro_deps.rs"))
633            .expect("`proc_macro_deps` should exist");
634        writeln!(
635            &mut file,
636            "/// Do not update manually - use `./x.py test tidy --bless`
637/// Holds all direct and indirect dependencies of proc-macro crates in tree.
638/// See <https://github.com/rust-lang/rust/issues/134863>
639pub static CRATES: &[&str] = &[
640    // tidy-alphabetical-start"
641        )
642        .unwrap();
643        for dep in proc_macro_deps {
644            writeln!(&mut file, "    {dep:?},").unwrap();
645        }
646        writeln!(
647            &mut file,
648            "    // tidy-alphabetical-end
649];"
650        )
651        .unwrap();
652    } else {
653        let old_bad = *bad;
654
655        for missing in proc_macro_deps.difference(&expected) {
656            tidy_error!(
657                bad,
658                "proc-macro crate dependency `{missing}` is not registered in `src/bootstrap/src/utils/proc_macro_deps.rs`",
659            );
660        }
661        for extra in expected.difference(&proc_macro_deps) {
662            tidy_error!(
663                bad,
664                "`{extra}` is registered in `src/bootstrap/src/utils/proc_macro_deps.rs`, but is not a proc-macro crate dependency",
665            );
666        }
667        if *bad != old_bad {
668            eprintln!("Run `./x.py test tidy --bless` to regenerate the list");
669        }
670    }
671}

673/// Used to skip a check if a submodule is not checked out, and not in a CI environment.
674///
675/// This helps prevent enforcing developers to fetch submodules for tidy.
676pub fn has_missing_submodule(root: &Path, submodules: &[&str]) -> bool {
677    !CiEnv::is_ci()
678        && submodules.iter().any(|submodule| {
679            let path = root.join(submodule);
680            !path.exists()
681            // If the directory is empty, we can consider it as an uninitialized submodule.
682            || read_dir(path).unwrap().next().is_none()
683        })
684}

686/// Check that all licenses of runtime dependencies are in the valid list in `LICENSES`.
687///
688/// Unlike for tools we don't allow exceptions to the `LICENSES` list for the runtime with the sole
689/// exception of `fortanix-sgx-abi` which is only used on x86_64-fortanix-unknown-sgx.
690fn check_runtime_license_exceptions(metadata: &Metadata, bad: &mut bool) {
691    for pkg in &metadata.packages {
692        if pkg.source.is_none() {
693            // No need to check local packages.
694            continue;
695        }
696        let license = match &pkg.license {
697            Some(license) => license,
698            None => {
699                tidy_error!(bad, "dependency `{}` does not define a license expression", pkg.id);
700                continue;
701            }
702        };
703        if !LICENSES.contains(&license.as_str()) {
704            // This is a specific exception because SGX is considered "third party".
705            // See https://github.com/rust-lang/rust/issues/62620 for more.
706            // In general, these should never be added and this exception
707            // should not be taken as precedent for any new target.
708            if *pkg.name == "fortanix-sgx-abi" && pkg.license.as_deref() == Some("MPL-2.0") {
709                continue;
710            }
711
712            tidy_error!(bad, "invalid license `{}` in `{}`", license, pkg.id);
713        }
714    }
715}

717/// Check that all licenses of tool dependencies are in the valid list in `LICENSES`.
718///
719/// Packages listed in `exceptions` are allowed for tools.
720fn check_license_exceptions(
721    metadata: &Metadata,
722    workspace: &str,
723    exceptions: &[(&str, &str)],
724    bad: &mut bool,
725) {
726    // Validate the EXCEPTIONS list hasn't changed.
727    for (name, license) in exceptions {
728        // Check that the package actually exists.
729        if !metadata.packages.iter().any(|p| *p.name == *name) {
730            tidy_error!(
731                bad,
732                "could not find exception package `{}` in workspace `{workspace}`\n\
733                Remove from EXCEPTIONS list if it is no longer used.",
734                name
735            );
736        }
737        // Check that the license hasn't changed.
738        for pkg in metadata.packages.iter().filter(|p| *p.name == *name) {
739            match &pkg.license {
740                None => {
741                    tidy_error!(
742                        bad,
743                        "dependency exception `{}` in workspace `{workspace}` does not declare a license expression",
744                        pkg.id
745                    );
746                }
747                Some(pkg_license) => {
748                    if pkg_license.as_str() != *license {
749                        println!(
750                            "dependency exception `{name}` license in workspace `{workspace}` has changed"
751                        );
752                        println!("    previously `{license}` now `{pkg_license}`");
753                        println!("    update EXCEPTIONS for the new license");
754                        *bad = true;
755                    }
756                }
757            }
758        }
759    }
760
761    let exception_names: Vec<_> = exceptions.iter().map(|(name, _license)| *name).collect();
762
763    // Check if any package does not have a valid license.
764    for pkg in &metadata.packages {
765        if pkg.source.is_none() {
766            // No need to check local packages.
767            continue;
768        }
769        if exception_names.contains(&pkg.name.as_str()) {
770            continue;
771        }
772        let license = match &pkg.license {
773            Some(license) => license,
774            None => {
775                tidy_error!(
776                    bad,
777                    "dependency `{}` in workspace `{workspace}` does not define a license expression",
778                    pkg.id
779                );
780                continue;
781            }
782        };
783        if !LICENSES.contains(&license.as_str()) {
784            tidy_error!(
785                bad,
786                "invalid license `{}` for package `{}` in workspace `{workspace}`",
787                license,
788                pkg.id
789            );
790        }
791    }
792}

794fn check_runtime_no_duplicate_dependencies(metadata: &Metadata, bad: &mut bool) {
795    let mut seen_pkgs = HashSet::new();
796    for pkg in &metadata.packages {
797        if pkg.source.is_none() {
798            continue;
799        }
800
801        if !seen_pkgs.insert(&*pkg.name) {
802            tidy_error!(
803                bad,
804                "duplicate package `{}` is not allowed for the standard library",
805                pkg.name
806            );
807        }
808    }
809}

821        }

825/// Checks the dependency of `restricted_dependency_crates` at the given path. Changes `bad` to
826/// `true` if a check failed.
827///
828/// Specifically, this checks that the dependencies are on the `permitted_dependencies`.
829fn check_permitted_dependencies(
830    metadata: &Metadata,
831    descr: &str,
832    permitted_dependencies: &[&'static str],
833    restricted_dependency_crates: &[&'static str],
834    bad: &mut bool,
835) {
836    let mut has_permitted_dep_error = false;
837    let mut deps = HashSet::new();
838    for to_check in restricted_dependency_crates {
839        let to_check = pkg_from_name(metadata, to_check);
840        deps_of(metadata, &to_check.id, &mut deps);
841    }
842
843    // Check that the PERMITTED_DEPENDENCIES does not have unused entries.
844    for permitted in permitted_dependencies {
845        fn compare(pkg: &Package, permitted: &str) -> bool {
846            if let Some((name, version)) = permitted.split_once("@") {
847                let Ok(version) = Version::parse(version) else {
848                    return false;
849                };
850                *pkg.name == name && pkg.version == version
851            } else {
852                *pkg.name == permitted
853            }
854        }
855        if !deps.iter().any(|dep_id| compare(pkg_from_id(metadata, dep_id), permitted)) {
856            tidy_error!(
857                bad,
858                "could not find allowed package `{permitted}`\n\
859                Remove from PERMITTED_DEPENDENCIES list if it is no longer used.",
860            );
861            has_permitted_dep_error = true;
862        }
863    }
864
865    // Get in a convenient form.
866    let permitted_dependencies: HashMap<_, _> = permitted_dependencies
867        .iter()
868        .map(|s| {
869            if let Some((name, version)) = s.split_once('@') {
870                (name, Version::parse(version).ok())
871            } else {
872                (*s, None)
873            }
874        })
875        .collect();
876
877    for dep in deps {
878        let dep = pkg_from_id(metadata, dep);
879        // If this path is in-tree, we don't require it to be explicitly permitted.
880        if dep.source.is_some() {
881            let is_eq = if let Some(version) = permitted_dependencies.get(dep.name.as_str()) {
882                if let Some(version) = version { version == &dep.version } else { true }
883            } else {
884                false
885            };
886            if !is_eq {
887                tidy_error!(bad, "Dependency for {descr} not explicitly permitted: {}", dep.id);
888                has_permitted_dep_error = true;
889            }
890        }
891    }
892
893    if has_permitted_dep_error {
894        eprintln!("Go to `{PERMITTED_DEPS_LOCATION}` for the list.");
895    }
896}

898/// Finds a package with the given name.
899fn pkg_from_name<'a>(metadata: &'a Metadata, name: &'static str) -> &'a Package {
900    let mut i = metadata.packages.iter().filter(|p| *p.name == name);
901    let result =
902        i.next().unwrap_or_else(|| panic!("could not find package `{name}` in package list"));
903    assert!(i.next().is_none(), "more than one package found for `{name}`");
904    result
905}
906
907fn pkg_from_id<'a>(metadata: &'a Metadata, id: &PackageId) -> &'a Package {
908    metadata.packages.iter().find(|p| &p.id == id).unwrap()
909}
910
911/// Recursively find all dependencies.
912fn deps_of<'a>(metadata: &'a Metadata, pkg_id: &'a PackageId, result: &mut HashSet<&'a PackageId>) {
913    if !result.insert(pkg_id) {
914        return;
915    }
916    let node = metadata
917        .resolve
918        .as_ref()
919        .unwrap()
920        .nodes
921        .iter()
922        .find(|n| &n.id == pkg_id)
923        .unwrap_or_else(|| panic!("could not find `{pkg_id}` in resolve"));
924    for dep in &node.deps {
925        deps_of(metadata, &dep.pkg, result);
926    }
927}
