1use crate::spec::{
2    RustcAbi, SanitizerSet, StackProbeType, Target, TargetMetadata, TargetOptions, base,

4
5// See https://developer.android.com/ndk/guides/abis.html#x86
6// for target ABI requirements.
7
8pub(crate) fn target() -> Target {
9    let mut base = base::android::opts();
10
11    base.max_atomic_width = Some(64);
12
13    base.rustc_abi = Some(RustcAbi::X86Sse2);
14    // https://developer.android.com/ndk/guides/abis.html#x86
15    base.cpu = "pentium4".into();
16    base.features = "+mmx,+sse,+sse2,+sse3,+ssse3".into();
17    base.stack_probes = StackProbeType::Inline;
18
19    Target {
20        llvm_target: "i686-linux-android".into(),
21        metadata: TargetMetadata {
22            description: Some("32-bit x86 Android".into()),
23            tier: Some(2),
24            host_tools: Some(false),
25            std: Some(true),
26        },
27        pointer_width: 32,
28        data_layout: "e-m:e-p:32:32-p270:32:32-p271:32:32-p272:64:64-\
29            i128:128-f64:32:64-f80:32-n8:16:32-S128"
30            .into(),
31        arch: "x86".into(),
32        options: TargetOptions { supported_sanitizers: SanitizerSet::ADDRESS, ..base },
33    }
34}
