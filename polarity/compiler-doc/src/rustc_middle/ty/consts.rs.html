148    pub fn is_trivially_wf(self) -> bool {
149        match self.kind() {
150            ty::ConstKind::Param(_) | ty::ConstKind::Placeholder(_) | ty::ConstKind::Bound(..) => {
151                true
152            }
153            ty::ConstKind::Infer(_)
154            | ty::ConstKind::Unevaluated(..)
155            | ty::ConstKind::Value(_)
156            | ty::ConstKind::Error(_)
157            | ty::ConstKind::Expr(_) => false,
158        }
159    }

200impl<'tcx> Const<'tcx> {
201    /// Creates a constant with the given integer value and interns it.
202    #[inline]
203    pub fn from_bits(
204        tcx: TyCtxt<'tcx>,
205        bits: u128,
206        typing_env: ty::TypingEnv<'tcx>,
207        ty: Ty<'tcx>,
208    ) -> Self {
209        let size = tcx
210            .layout_of(typing_env.as_query_input(ty))
211            .unwrap_or_else(|e| panic!("could not compute layout for {ty:?}: {e:?}"))
212            .size;
213        ty::Const::new_value(
214            tcx,
215            ty::ValTree::from_scalar_int(tcx, ScalarInt::try_from_uint(bits, size).unwrap()),
216            ty,
217        )
218    }
219
220    #[inline]
221    /// Creates an interned zst constant.
222    pub fn zero_sized(tcx: TyCtxt<'tcx>, ty: Ty<'tcx>) -> Self {
223        ty::Const::new_value(tcx, ty::ValTree::zst(tcx), ty)
224    }
225
226    #[inline]
227    /// Creates an interned bool constant.
228    pub fn from_bool(tcx: TyCtxt<'tcx>, v: bool) -> Self {
229        Self::from_bits(tcx, v as u128, ty::TypingEnv::fully_monomorphized(), tcx.types.bool)
230    }
231
232    #[inline]
233    /// Creates an interned usize constant.
234    pub fn from_target_usize(tcx: TyCtxt<'tcx>, n: u64) -> Self {
235        Self::from_bits(tcx, n as u128, ty::TypingEnv::fully_monomorphized(), tcx.types.usize)
236    }
237
238    /// Panics if `self.kind != ty::ConstKind::Value`.
239    pub fn to_value(self) -> ty::Value<'tcx> {
240        match self.kind() {
241            ty::ConstKind::Value(cv) => cv,
242            _ => bug!("expected ConstKind::Value, got {:?}", self.kind()),
243        }
244    }
245
246    /// Attempts to convert to a value.
247    ///
248    /// Note that this does not evaluate the constant.
249    pub fn try_to_value(self) -> Option<ty::Value<'tcx>> {
250        match self.kind() {
251            ty::ConstKind::Value(cv) => Some(cv),
252            _ => None,
253        }
254    }
255
256    /// Convenience method to extract the value of a usize constant,
257    /// useful to get the length of an array type.
258    ///
259    /// Note that this does not evaluate the constant.
260    #[inline]
261    pub fn try_to_target_usize(self, tcx: TyCtxt<'tcx>) -> Option<u64> {
262        self.try_to_value()?.try_to_target_usize(tcx)
263    }
264
265    pub fn is_ct_infer(self) -> bool {
266        matches!(self.kind(), ty::ConstKind::Infer(_))
267    }
268
269    /// Iterator that walks `self` and any types reachable from
270    /// `self`, in depth-first order. Note that just walks the types
271    /// that appear in `self`, it does not descend into the fields of
272    /// structs or variants. For example:
273    ///
274    /// ```text
275    /// isize => { isize }
276    /// Foo<Bar<isize>> => { Foo<Bar<isize>>, Bar<isize>, isize }
277    /// [isize] => { [isize], isize }
278    /// ```
279    pub fn walk(self) -> TypeWalker<TyCtxt<'tcx>> {
280        TypeWalker::new(self.into())
281    }
282}
